56  /**
57   * Default Jersey client {@link org.glassfish.jersey.client.spi.Connector connector} provider
58   * that provides connector instances which delegate HTTP requests to {@link java.net.HttpURLConnection}
59   * for processing.
60   * <p>
61   * The provided connector instances override default behaviour of the property
62   * {@link ClientProperties#REQUEST_ENTITY_PROCESSING} and use {@link RequestEntityProcessing#BUFFERED}
63   * request entity processing by default.
64   * </p>
65   * <p>
66   * Due to a bug in the chunked transport coding support of {@code HttpURLConnection} that causes
67   * requests to fail unpredictably, this connector provider allows to configure the provided connector
68   * instances to use {@code HttpURLConnection}'s fixed-length streaming mode as a workaround. This
69   * workaround can be enabled via {@link #useFixedLengthStreaming()} method or via
70   * {@link #USE_FIXED_LENGTH_STREAMING} Jersey client configuration property.
71   * </p>
72   *
73   * @author Marek Potociar (marek.potociar at oracle.com)
74   */
75  public class HttpUrlConnectorProvider implements ConnectorProvider {
76      /**
77       * If {@code true}, the {@link HttpUrlConnector} (if used) will assume the content length
78       * from the value of {@value javax.ws.rs.core.HttpHeaders#CONTENT_LENGTH} request
79       * header (if present).
80       * <p>
81       * When this property is enabled and the request has a valid non-zero content length
82       * value specified in its {@value javax.ws.rs.core.HttpHeaders#CONTENT_LENGTH} request
83       * header, that this value will be used as an input to the
84       * {@link java.net.HttpURLConnection#setFixedLengthStreamingMode(int)} method call
85       * invoked on the underlying {@link java.net.HttpURLConnection connection}.
86       * This will also suppress the entity buffering in the @{code HttpURLConnection},
87       * which is undesirable in certain scenarios, e.g. when streaming large entities.
88       * </p>
89       * <p>
90       * Note that the content length value defined in the request header must exactly match
91       * the real size of the entity. If the {@link javax.ws.rs.core.HttpHeaders#CONTENT_LENGTH} header
92       * is explicitly specified in a request, this property will be ignored and the
93       * request entity will be still buffered by the underlying @{code HttpURLConnection} infrastructure.
94       * </p>
95       * <p>
96       * This property also overrides the behaviour enabled by the
97       * {@link org.glassfish.jersey.client.ClientProperties#CHUNKED_ENCODING_SIZE} property.
98       * Chunked encoding will only be used, if the size is not specified in the header of the request.
99       * </p>
100      * <p>
101      * Note that this property only applies to client run-times that are configured to use the default
102      * {@link HttpUrlConnector} as the client connector. The property is ignored by other connectors.
103      * </p>
104      * <p>
105      * The default value is {@code false}.
106      * </p>
107      * <p>
108      * The name of the configuration property is <tt>{@value}</tt>.
109      * </p>
110      *
111      * @since 2.5
112      */
113     public static final String USE_FIXED_LENGTH_STREAMING =
114             "jersey.config.client.httpUrlConnector.useFixedLengthStreaming";
115

224     @Override
225     public Connector getConnector(final Client client, final Configuration config) {
226         final Map<String, Object> properties = config.getProperties();
227 
228         int computedChunkSize = ClientProperties.getValue(properties,
229                 ClientProperties.CHUNKED_ENCODING_SIZE, chunkSize, Integer.class);
230         if (computedChunkSize < 0) {
231             LOGGER.warning(LocalizationMessages.NEGATIVE_CHUNK_SIZE(computedChunkSize, chunkSize));
232             computedChunkSize = chunkSize;
233         }
234 
235         final boolean computedUseFixedLengthStreaming = ClientProperties.getValue(properties,
236                 USE_FIXED_LENGTH_STREAMING, useFixedLengthStreaming, Boolean.class);
237         final boolean computedUseSetMethodWorkaround = ClientProperties.getValue(properties,
238                 SET_METHOD_WORKAROUND, useSetMethodWorkaround, Boolean.class);
239 
240         return createHttpUrlConnector(client, connectionFactory, computedChunkSize, computedUseFixedLengthStreaming,
241                                       computedUseSetMethodWorkaround);
242     }

244     /**
245      * Create {@link HttpUrlConnector}.
246      *
247      * @param client              JAX-RS client instance for which the connector is being created.
248      * @param connectionFactory   {@link javax.net.ssl.HttpsURLConnection} factory to be used when creating
249      *                            connections.
250      * @param chunkSize           chunk size to use when using HTTP chunked transfer coding.
251      * @param fixLengthStreaming  specify if the the {@link java.net.HttpURLConnection#setFixedLengthStreamingMode(int)
252      *                            fixed-length streaming mode} on the underlying HTTP URL connection instances should
253      *                            be used when sending requests.
254      * @param setMethodWorkaround specify if the reflection workaround should be used to set HTTP URL connection method

257      */
258     protected Connector createHttpUrlConnector(Client client, ConnectionFactory connectionFactory,
259                                                int chunkSize, boolean fixLengthStreaming,
260                                                boolean setMethodWorkaround) {
261         return new HttpUrlConnector(
262                 client,
263                 connectionFactory,
264                 chunkSize,
265                 fixLengthStreaming,
266                 setMethodWorkaround);
267     }

269     /**
270      * A factory for {@link java.net.HttpURLConnection} instances.
271      * <p>
272      * A factory may be used to create a {@link java.net.HttpURLConnection} and configure
273      * it in a custom manner that is not possible using the Client API.
274      * <p>
275      * A custom factory instance may be registered in the {@code HttpUrlConnectorProvider} instance
276      * via {@link #connectionFactory(ConnectionFactory)} method.
277      */
278     public interface ConnectionFactory {
279 
280         /**
281          * Get a {@link java.net.HttpURLConnection} for a given URL.
282          * <p>
283          * Implementation of the method MUST be thread-safe and MUST ensure that
284          * a dedicated {@link java.net.HttpURLConnection} instance is returned for concurrent
285          * requests.
286          * </p>
287          *
288          * @param url the endpoint URL.
289          * @return the {@link java.net.HttpURLConnection}.
290          * @throws java.io.IOException in case the connection cannot be provided.
291          */
292         public HttpURLConnection getConnection(URL url) throws IOException;
293     }

295     private static class DefaultConnectionFactory implements ConnectionFactory {
296 
297         @Override
298         public HttpURLConnection getConnection(final URL url) throws IOException {
299             return (HttpURLConnection) url.openConnection();
300         }
301     }

303     @Override
304     public boolean equals(final Object o) {
305         if (this == o) {
306             return true;
307         }
308         if (o == null || getClass() != o.getClass()) {
309             return false;
310         }
311 
312         final HttpUrlConnectorProvider that = (HttpUrlConnectorProvider) o;
313 
314         if (chunkSize != that.chunkSize) {
315             return false;
316         }
317         if (useFixedLengthStreaming != that.useFixedLengthStreaming) {
318             return false;
319         }
320 
321         return connectionFactory.equals(that.connectionFactory);
322     }
323 
324     @Override
325     public int hashCode() {
326         int result = connectionFactory.hashCode();
327         result = 31 * result + chunkSize;
328         result = 31 * result + (useFixedLengthStreaming ? 1 : 0);
329         return result;
330     }
