42  import java.text.ParseException;
43  import java.util.Arrays;
44  import java.util.List;
45  import java.util.Map;
46  import java.util.regex.Matcher;
47  import java.util.regex.Pattern;

49  import javax.ws.rs.core.CacheControl;

51  import javax.inject.Singleton;
52  
53  import org.glassfish.jersey.internal.LocalizationMessages;
54  import org.glassfish.jersey.spi.HeaderDelegateProvider;
55  import static org.glassfish.jersey.message.internal.Utils.throwIllegalArgumentExceptionIfNull;

57  /**
58   * {@code Cache-Control} {@link HeaderDelegateProvider header delegate provider}.
59   *
60   * @author Paul Sandoz
61   * @author Marek Potociar (marek.potociar at oracle.com)
62   * @author hubick@java.net
63   */
64  @Singleton
65  public final class CacheControlProvider implements HeaderDelegateProvider<CacheControl> {
66  
67      private static final Pattern WHITESPACE = Pattern.compile("\\s");
68      private static final Pattern COMMA_SEPARATED_LIST = Pattern.compile("[\\s]*,[\\s]*");
69  
70      @Override
71      public boolean supports(Class<?> type) {
72          return type == CacheControl.class;
73      }
74  
75      @Override
76      public String toString(CacheControl header) {
77  
78          throwIllegalArgumentExceptionIfNull(header, LocalizationMessages.CACHE_CONTROL_IS_NULL());
79  
80          StringBuilder b = new StringBuilder();
81          if (header.isPrivate()) {
82              appendQuotedWithSeparator(b, "private", buildListValue(header.getPrivateFields()));
83          }
84          if (header.isNoCache()) {
85              appendQuotedWithSeparator(b, "no-cache", buildListValue(header.getNoCacheFields()));
86          }
87          if (header.isNoStore()) {
88              appendWithSeparator(b, "no-store");
89          }
90          if (header.isNoTransform()) {
91              appendWithSeparator(b, "no-transform");
92          }
93          if (header.isMustRevalidate()) {
94              appendWithSeparator(b, "must-revalidate");
95          }
96          if (header.isProxyRevalidate()) {
97              appendWithSeparator(b, "proxy-revalidate");
98          }
99          if (header.getMaxAge() != -1) {
100             appendWithSeparator(b, "max-age", header.getMaxAge());
101         }
102         if (header.getSMaxAge() != -1) {
103             appendWithSeparator(b, "s-maxage", header.getSMaxAge());
104         }
105 
106         for (Map.Entry<String, String> e : header.getCacheExtension().entrySet()) {
107             appendWithSeparator(b, e.getKey(), quoteIfWhitespace(e.getValue()));
108         }
109 
110         return b.toString();
111     }
112 
113     private void readFieldNames(List<String> fieldNames, HttpHeaderReader reader)
114             throws ParseException {
115         if (!reader.hasNextSeparator('=', false)) {
116             return;
117         }
118         reader.nextSeparator('=');
119         fieldNames.addAll(Arrays.asList(COMMA_SEPARATED_LIST.split(reader.nextQuotedString())));
120     }
121 
122     private int readIntValue(HttpHeaderReader reader, String directiveName)
123             throws ParseException {
124         reader.nextSeparator('=');
125         int index = reader.getIndex();
126         try {
127             return Integer.parseInt(reader.nextToken().toString());
128         } catch (NumberFormatException nfe) {
129             ParseException pe = new ParseException(
130                     "Error parsing integer value for " + directiveName + " directive", index);
131             pe.initCause(nfe);
132             throw pe;
133         }
134     }
135 
136     private void readDirective(CacheControl cacheControl,
137                                HttpHeaderReader reader) throws ParseException {
138 
139         final String directiveName = reader.nextToken().toString().toLowerCase();
140         if ("private".equals(directiveName)) {
141             cacheControl.setPrivate(true);
142             readFieldNames(cacheControl.getPrivateFields(), reader);
143         } else if ("public".equals(directiveName)) {
144             // CacheControl API doesn't support 'public' for some reason.
145             cacheControl.getCacheExtension().put(directiveName, null);
146         } else if ("no-cache".equals(directiveName)) {
147             cacheControl.setNoCache(true);
148             readFieldNames(cacheControl.getNoCacheFields(), reader);
149         } else if ("no-store".equals(directiveName)) {
150             cacheControl.setNoStore(true);
151         } else if ("no-transform".equals(directiveName)) {
152             cacheControl.setNoTransform(true);
153         } else if ("must-revalidate".equals(directiveName)) {
154             cacheControl.setMustRevalidate(true);
155         } else if ("proxy-revalidate".equals(directiveName)) {
156             cacheControl.setProxyRevalidate(true);
157         } else if ("max-age".equals(directiveName)) {
158             cacheControl.setMaxAge(readIntValue(reader, directiveName));
159         } else if ("s-maxage".equals(directiveName)) {
160             cacheControl.setSMaxAge(readIntValue(reader, directiveName));
161         } else {
162             String value = null;
163             if (reader.hasNextSeparator('=', false)) {
164                 reader.nextSeparator('=');
165                 value = reader.nextTokenOrQuotedString().toString();
166             }
167             cacheControl.getCacheExtension().put(directiveName, value);
168         }
169     }
170 
171     @Override
172     public CacheControl fromString(String header) {
173 
174         throwIllegalArgumentExceptionIfNull(header, LocalizationMessages.CACHE_CONTROL_IS_NULL());
175 
176         try {
177             HttpHeaderReader reader = HttpHeaderReader.newInstance(header);
178             CacheControl cacheControl = new CacheControl();
179             cacheControl.setNoTransform(false); // defaults to true
180             while (reader.hasNext()) {
181                 readDirective(cacheControl, reader);
182                 if (reader.hasNextSeparator(',', true)) {
183                     reader.nextSeparator(',');
184                 }
185             }
186             return cacheControl;
187         } catch (ParseException pe) {
188             throw new IllegalArgumentException(
189                     "Error parsing cache control '" + header + "'", pe);
190         }
191     }
192 
193     private void appendWithSeparator(StringBuilder b, String field) {
194         if (b.length() > 0) {
195             b.append(", ");
196         }
197         b.append(field);
198     }
199 
200     private void appendQuotedWithSeparator(StringBuilder b, String field, String value) {
201         appendWithSeparator(b, field);
202         if (value != null && !value.isEmpty()) {
203             b.append("=\"");
204             b.append(value);
205             b.append("\"");
206         }
207     }
208 
209     private void appendWithSeparator(StringBuilder b, String field, String value) {
210         appendWithSeparator(b, field);
211         if (value != null && !value.isEmpty()) {
212             b.append("=");
213             b.append(value);
214         }
215     }
216 
217     private void appendWithSeparator(StringBuilder b, String field, int value) {
218         appendWithSeparator(b, field);
219         b.append("=");
220         b.append(value);
221     }
222 
223     private String buildListValue(List<String> values) {
224         StringBuilder b = new StringBuilder();
225         for (String value : values) {
226             appendWithSeparator(b, value);
227         }
228         return b.toString();
229     }
230 
231     private String quoteIfWhitespace(String value) {
232         if (value == null) {
233             return null;
234         }
235         Matcher m = WHITESPACE.matcher(value);
236         if (m.find()) {
237             return "\"" + value + "\"";
238         }
239         return value;
240     }
