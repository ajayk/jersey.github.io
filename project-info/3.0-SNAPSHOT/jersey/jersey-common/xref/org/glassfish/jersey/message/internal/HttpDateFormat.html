40  package org.glassfish.jersey.message.internal;
41  
42  import java.text.ParseException;
43  import java.text.SimpleDateFormat;
44  import java.util.Arrays;
45  import java.util.Collections;
46  import java.util.Date;
47  import java.util.List;
48  import java.util.Locale;
49  import java.util.TimeZone;
50  
51  /**
52   * Helper class for HTTP specified date formats.
53   *
54   * @author Paul Sandoz
55   * @author Marek Potociar (marek.potociar at oracle.com)
56   */
57  public final class HttpDateFormat {
58  
59      private HttpDateFormat() {
60      }
61      /**
62       * The date format pattern for RFC 1123.
63       */
64      private static final String RFC1123_DATE_FORMAT_PATTERN = "EEE, dd MMM yyyy HH:mm:ss zzz";
65      /**
66       * The date format pattern for RFC 1036.
67       */
68      private static final String RFC1036_DATE_FORMAT_PATTERN = "EEEE, dd-MMM-yy HH:mm:ss zzz";
69      /**
70       * The date format pattern for ANSI C asctime().
71       */
72      private static final String ANSI_C_ASCTIME_DATE_FORMAT_PATTERN = "EEE MMM d HH:mm:ss yyyy";
73  
74      private static final TimeZone GMT_TIME_ZONE = TimeZone.getTimeZone("GMT");
75  
76      private static final ThreadLocal<List<SimpleDateFormat>> dateFormats = new ThreadLocal<List<SimpleDateFormat>>() {
77  
78          @Override
79          protected synchronized List<SimpleDateFormat> initialValue() {
80              return createDateFormats();
81          }
82      };
83  
84      private static List<SimpleDateFormat> createDateFormats() {
85          final SimpleDateFormat[] formats = new SimpleDateFormat[]{
86              new SimpleDateFormat(RFC1123_DATE_FORMAT_PATTERN, Locale.US),
87              new SimpleDateFormat(RFC1036_DATE_FORMAT_PATTERN, Locale.US),
88              new SimpleDateFormat(ANSI_C_ASCTIME_DATE_FORMAT_PATTERN, Locale.US)
89          };
90          formats[0].setTimeZone(GMT_TIME_ZONE);
91          formats[1].setTimeZone(GMT_TIME_ZONE);
92          formats[2].setTimeZone(GMT_TIME_ZONE);
93  
94          return Collections.unmodifiableList(Arrays.asList(formats));
95      }
96  
97      /**
98       * Return an unmodifiable list of HTTP specified date formats to use for
99       * parsing or formatting {@link Date}.
100      * <p>
101      * The list of date formats are scoped to the current thread and may be
102      * used without requiring to synchronize access to the instances when
103      * parsing or formatting.
104      *
105      * @return the list of data formats.
106      */
107     private static List<SimpleDateFormat> getDateFormats() {
108         return dateFormats.get();
109     }
110 
111     /**
112      * Get the preferred HTTP specified date format (RFC 1123).
113      * <p>
114      * The date format is scoped to the current thread and may be
115      * used without requiring to synchronize access to the instance when
116      * parsing or formatting.
117      *
118      * @return the preferred of data format.
119      */
120     public static SimpleDateFormat getPreferredDateFormat() {
121         // returns clone because calling SDF.parse(...) can change time zone
122         return (SimpleDateFormat) dateFormats.get().get(0).clone();
123     }
124 
125     /**
126      * Read a date.
127      *
128      * @param date the date as a string.
129      *
130      * @return the date
131      * @throws java.text.ParseException in case the date string cannot be parsed.
132      */
133     public static Date readDate(final String date) throws ParseException {
134         ParseException pe = null;
135         for (final SimpleDateFormat f : HttpDateFormat.getDateFormats()) {
136             try {
137                 Date result = f.parse(date);
138                 // parse can change time zone -> set it back to GMT
139                 f.setTimeZone(GMT_TIME_ZONE);
140                 return result;
141             } catch (final ParseException e) {
142                 pe = (pe == null) ? e : pe;
143             }
144         }
145 
146         throw pe;
147     }
148 }
