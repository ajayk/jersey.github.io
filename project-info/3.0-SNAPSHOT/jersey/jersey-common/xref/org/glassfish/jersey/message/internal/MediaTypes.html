70      /**
71       * A comparator for media types, that implements the "partial order" defined in the resource matching algorithm
72       * section of the JAX-RS specification, except that this comparator is "inverted" so that it allows for natural
73       * sorting in Java collections, where "lower" values are put to the front of a collection.
74       *
75       * IOW, when used to sort a collection, the resulting collection can be iterated in a way that the more specific
76       * media types are preferred over the less specific ones:
77       *
78       * <pre>
79       *   m/n &lt; m/&#42; &lt; &#42;/&#42;</pre>
80       *
81       * The actual media type values are ignored, i.e. the different media types are considered equal if they are
82       * comparably specific:
83       *
84       * <pre>
85       *   compare(m/n, x/y) == 0
86       *   compare(m/&#42;, x/&#42;) == 0</pre>
87       */
88      public static final Comparator<MediaType> PARTIAL_ORDER_COMPARATOR = new Comparator<MediaType>() {
89  
90          private int rank(final MediaType type) {
91              // "m/n" = 0; "m/*" = 1; "*/*" = 3; ...also "*/n" = 2, but that's just nonsense...
92              return ((type.isWildcardType() ? 1 : 0) << 1) | (type.isWildcardSubtype() ? 1 : 0);
93          }
94  
95          @Override
96          public int compare(MediaType typeA, MediaType typeB) {
97              return rank(typeA) - rank(typeB);
98          }
99      };

101     /**
102      * Comparator for lists of media types.
103      * <p>
104      * The least specific content type of each list is obtained and then compared
105      * using {@link #PARTIAL_ORDER_COMPARATOR}.
106      * <p>
107      * Assumes each list is already ordered according to {@link #PARTIAL_ORDER_COMPARATOR}
108      * and therefore the least specific media type is at the end of the list.
109      */
110     public static final Comparator<List<? extends MediaType>> MEDIA_TYPE_LIST_COMPARATOR =
111             new Comparator<List<? extends MediaType>>() {
112 
113                 @Override
114                 public int compare(List<? extends MediaType> o1, List<? extends MediaType> o2) {
115                     return PARTIAL_ORDER_COMPARATOR.compare(getLeastSpecific(o1), getLeastSpecific(o2));
116                 }
117 
118                 private MediaType getLeastSpecific(List<? extends MediaType> l) {
119                     // FIXME: really comparing just last one?
120                     return l.get(l.size() - 1);
121                 }
122             };

173     /**
174      * Determine if the two media types are type-equal (their {@link MediaType#getType() type}
175      * and {@link MediaType#getSubtype() subtype} are equal). For example:
176      * <pre>
177      *  m/n == m/n
178      *  m/n;p1 == m/n;p2
179      *
180      *  m/n != m/y
181      *  m/n != x/n
182      *  m/n != x/y</pre>
183      *
184      * @param m1 first media type.
185      * @param m2 second media type.
186      * @return {@code true} if the two media types are of the same type and subtype,
187      * {@code false} otherwise.
188      */
189     public static boolean typeEqual(MediaType m1, MediaType m2) {
190         if (m1 == null || m2 == null) {
191             return false;
192         }
193 
194         return m1.getSubtype().equalsIgnoreCase(m2.getSubtype()) && m1.getType().equalsIgnoreCase(m2.getType());
195     }

197     /**
198      * Determine if the two list of media types share a common
199      * {@link #typeEqual(javax.ws.rs.core.MediaType, javax.ws.rs.core.MediaType) type-equal}
200      * sub-list.
201      *
202      * @param ml1 first media type list.
203      * @param ml2 second media type list.
204      * @return {@code true} if the two media type lists intersect by sharing a
205      * common type-equal sub-list, {@code false} otherwise.
206      */
207     public static boolean intersect(List<? extends MediaType> ml1, List<? extends MediaType> ml2) {
208         for (MediaType m1 : ml1) {
209             for (MediaType m2 : ml2) {
210                 if (MediaTypes.typeEqual(m1, m2)) {
211                     return true;
212                 }
213             }
214         }
215         return false;
216     }

218     /**
219      * Get the most specific media type from a pair of media types. The most
220      * specific media type is the media type from the pair that has least
221      * wild cards present, or has more parameters specified.
222      *
223      * @param m1 the first media type.
224      * @param m2 the second media type.
225      * @return the most specific media type. If the media types are equally
226      * specific then the first media type is returned.
227      */
228     public static MediaType mostSpecific(MediaType m1, MediaType m2) {
229         if (m1.isWildcardType() && !m2.isWildcardType()) {
230             return m2;
231         }
232         if (m1.isWildcardSubtype() && !m2.isWildcardSubtype()) {
233             return m2;
234         }
235         if (m2.getParameters().size() > m1.getParameters().size()) {
236             return m2;
237         }
238 
239         return m1;
240     }

272     /**
273      * Create an unmodifiable list of media type from a string array of media types.
274      *
275      * @param mediaTypes the string array of media types.
276      * @return the list of {@link MediaType}, ordered according to {@link #PARTIAL_ORDER_COMPARATOR}.
277      */
278     public static List<MediaType> createFrom(String[] mediaTypes) {
279         List<MediaType> result = new ArrayList<MediaType>();
280 
281         try {
282             for (String mediaType : mediaTypes) {
283                 HttpHeaderReader.readMediaTypes(result, mediaType);
284             }
285         } catch (ParseException ex) {
286             throw new IllegalArgumentException(ex);
287         }
288 
289         Collections.sort(result, PARTIAL_ORDER_COMPARATOR);
290 
291         return Collections.unmodifiableList(result);
292     }

351     /**
352      * Strips any quality parameters, i.e. q and qs from given media type.
353      *
354      * @param mediaType type to strip quality parameters from
355      * @return media type instance corresponding to the given one with quality parameters stripped off
356      * or the original instance if no such parameters are present
357      */
358     public static MediaType stripQualityParams(MediaType mediaType) {
359         final Map<String, String> oldParameters = mediaType.getParameters();
360         if (oldParameters.isEmpty()
361                 || (!oldParameters.containsKey(Quality.QUALITY_SOURCE_PARAMETER_NAME)
362                             && !oldParameters.containsKey(Quality.QUALITY_PARAMETER_NAME))) {
363             return mediaType;
364         }
365 
366         return new MediaType(mediaType.getType(), mediaType.getSubtype(),
367                 Maps.filterKeys(oldParameters, QUALITY_PARAM_FILTERING_PREDICATE));
368     }

386     /**
387      * Convert media types into {@link java.lang.String}. The result string contains
388      * media types in the same order, separated by comma ',' and enclosed into quotes.
389      * For example for input media types
390      * {@link MediaType#TEXT_PLAIN_TYPE}, {@link MediaType#TEXT_PLAIN_TYPE} and
391      * {@link MediaType#APPLICATION_JSON_TYPE} the result will be
392      * "text/plain", "application/json", "text/html".
393      *
394      * @param mediaTypes {@link Iterable iterable} with {@link MediaType media types}.
395      * @return Media types converted into String.
396      */
397     public static String convertToString(Iterable<MediaType> mediaTypes) {
398         StringBuilder sb = new StringBuilder();
399         boolean isFirst = true;
400         for (MediaType mediaType : mediaTypes) {
401             if (!isFirst) {
402                 sb.append(", ");
403             } else {
404                 isFirst = false;
405             }
406             sb.append("\"").append(mediaType.toString()).append("\"");
407         }
408         return sb.toString();
409 
410     }

417      * @param mediaType media type.
