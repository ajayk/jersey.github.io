155     /**
156      * Log supplied errors and return a status flag indicating whether a fatal issue has been found
157      * in the error collection.
158      *
159      * @param errors a collection of errors to be logged.
160      * @return {@code true} if there are any fatal issues present in the collection, {@code false}
161      *         otherwise.
162      */
163     private static boolean logErrors(final Collection<ErrorMessage> errors) {
164         boolean isFatal = false;
165 
166         if (!errors.isEmpty()) {
167             StringBuilder fatals = new StringBuilder("\n");
168             StringBuilder warnings = new StringBuilder();
169             StringBuilder hints = new StringBuilder();
170 
171             for (final ErrorMessage error : errors) {
172                 switch (error.getSeverity()) {
173                     case FATAL:
174                         isFatal = true;
175                         fatals.append(LocalizationMessages.ERROR_MSG(error.getMessage())).append('\n');
176                         break;
177                     case WARNING:
178                         warnings.append(LocalizationMessages.WARNING_MSG(error.getMessage())).append('\n');
179                         break;
180                     case HINT:
181                         warnings.append(LocalizationMessages.HINT_MSG(error.getMessage())).append('\n');
182                         break;
183                 }
184             }
185 
186             if (isFatal) {
187                 LOGGER.severe(LocalizationMessages.ERRORS_AND_WARNINGS_DETECTED(fatals.append(warnings)
188                         .append(hints).toString()));
189             } else {
190                 if (warnings.length() > 0) {
191                     LOGGER.warning(LocalizationMessages.WARNINGS_DETECTED(warnings.toString()));
192                 }
193 
194                 if (hints.length() > 0) {
195                     LOGGER.config(LocalizationMessages.HINTS_DETECTED(hints.toString()));
196                 }
197             }
198         }
199 
200         return isFatal;
201     }

295     private static <T> T process(final Producer<T> task, final boolean throwException) {
296         try {
297             return process((Callable<T>) task, throwException);
298         } catch (RuntimeException ex) {
299             throw ex;
300         } catch (Exception ex) {
301             throw new RuntimeException(ex);
302         }
303     }

305     private static <T> T process(final Callable<T> task, final boolean throwException) throws Exception {
306         Errors instance = errors.get();
307         if (instance == null) {
308             instance = new Errors();
309             errors.set(instance);
310         }
311         instance.preProcess();
312 
313         Exception caught = null;
314         try {
315             return task.call();
316         } catch (Exception re) {
317             // If a runtime exception is caught then report errors and rethrow.
318             caught = re;
319         } finally {
320             instance.postProcess(throwException && caught == null);
321         }
322 
323         throw caught;
324     }

326     private static Errors getInstance() {
327         final Errors instance = errors.get();
328         // No error processing in scope
329         if (instance == null) {
330             throw new IllegalStateException(LocalizationMessages.NO_ERROR_PROCESSING_IN_SCOPE());
331         }
332         // The following should not be necessary but given the fragile nature of
333         // static thread local probably best to add it in case some internals of
334         // this class change
335         if (instance.stack == 0) {
336             errors.remove();
337             throw new IllegalStateException(LocalizationMessages.NO_ERROR_PROCESSING_IN_SCOPE());
338         }
339         return instance;
340     }

388     private final ArrayList<ErrorMessage> issues = new ArrayList<ErrorMessage>(0);
389 
390     private Errors() {
391     }
392 
393     private Deque<Integer> mark = new ArrayDeque<Integer>(4);
394     private int stack = 0;
395 
396     private void _mark() {
397         mark.addLast(issues.size());
398     }
399 
400     private void _unmark() {
401         mark.pollLast();
402     }
403 
404     private void _reset() {
405         final Integer _pos = mark.pollLast(); // also performs "unmark" functionality
406         final int markedPos = (_pos == null) ? -1 : _pos;
407 
408         if (markedPos >= 0 && markedPos < issues.size()) {
409             issues.subList(markedPos, issues.size()).clear();
410         }
411     }
412 
413     private void preProcess() {
414         stack++;
415     }
416 
417     private void postProcess(boolean throwException) {
418         stack--;
419 
420         if (stack == 0) {
421             try {
422                 if (!issues.isEmpty()) {
423                     processErrors(throwException);
424                 }
425             } finally {
426                 errors.remove();
427             }
428         }
429     }
430 
431     private List<ErrorMessage> _getErrorMessages(final boolean afterMark) {
432         if (afterMark) {
433             final Integer _pos = mark.peekLast();
434             final int markedPos = (_pos == null) ? -1 : _pos;
435 
436             if (markedPos >= 0 && markedPos < issues.size()) {
437                 return Collections.unmodifiableList(new ArrayList<ErrorMessage>(issues.subList(markedPos, issues.size())));
438             } // else return all errors
439         }
440 
441         return Collections.unmodifiableList(new ArrayList<ErrorMessage>(issues));
442     }

509         @Override
510         public boolean equals(Object o) {
511             if (this == o) {
512                 return true;
513             }
514             if (o == null || getClass() != o.getClass()) {
515                 return false;
516             }
517 
518             ErrorMessage that = (ErrorMessage) o;
519 
520             if (message != null ? !message.equals(that.message) : that.message != null) {
521                 return false;
522             }
523             if (severity != that.severity) {
524                 return false;
525             }
526             if (source != null ? !source.equals(that.source) : that.source != null) {
527                 return false;
528             }
529 
530             return true;
531         }
532 
533         @Override
534         public int hashCode() {
535             int result = source != null ? source.hashCode() : 0;
536             result = 31 * result + (message != null ? message.hashCode() : 0);
537             result = 31 * result + (severity != null ? severity.hashCode() : 0);
538             return result;
539         }
