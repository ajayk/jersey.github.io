40  package org.glassfish.jersey.spi;
41  
42  import java.util.concurrent.ExecutorService;
43  
44  /**
45   * An extension contract for providing pluggable executor service providers to be used by
46   * Jersey client or server runtime whenever a specific executor service is needed to execute a Jersey runtime processing task.
47   * <p>
48   * This mechanism allows Jersey to run in environments that have specific thread management and provisioning requirements,
49   * such as application servers, cloud environments etc.
50   * Dedicated Jersey extension modules or applications running in such environment may provide a custom
51   * implementation of the {@code ExecutorServiceProvider} interface to customize the default
52   * Jersey runtime thread management & provisioning strategy in order to comply with the threading requirements,
53   * models and policies specific to each particular environment.
54   * </p>
55   * <p>
56   * When Jersey runtime no longer requires the use of a provided executor service instance, it invokes the provider's
57   * {@link #dispose} method to signal the provider that the executor service instance can be disposed of. In this method,
58   * provider is free to implement the proper shut-down logic for the disposed executor service instance and perform other
59   * necessary cleanup. Yet, some providers may wish to implement a shared executor service strategy. In such case,
60   * it may not be desirable to shut down the released executor service in the {@link #dispose} method. Instead, to perform the
61   * eventual shut-down procedure, the provider may either rely on an explicit invocation of it's specific clean-up method.
62   * Since all Jersey providers operate in a <em>container</em> environment, a good clean-up strategy for a shared executor
63   * service provider implementation is to expose a {@link javax.annotation.PreDestroy &#64;PreDestroy}-annotated method
64   * that will be invoked for all instances managed by the container, before the container shuts down.
65   * </p>
66   * <p>
67   * IMPORTANT: Please note that any pre-destroy methods may not be invoked for instances created outside of the container
68   * and later registered within the container. Pre-destroy methods are only guaranteed to be invoked for those instances
69   * that are created and managed by the container.
70   * </p>
71   * <p>
72   * Jersey runtime expects that a concrete executor service provider implementation class is annotated with a
73   * {@link javax.inject.Qualifier qualifier} annotation. This qualifier is then used to create a qualified injection point
74   * for injecting the executor service instance provided by the annotated provider. {@link javax.inject.Named Named} providers
75   * are also supported. For example:
76   * </p>
77   * <pre>
78   * &#64;Named("my-executor")
79   * public MyExecutorProvider implements ExecutorServiceProvider {
80   *     ...
81   * }
82   *
83   * ...
84   *
85   * // Injecting ExecutorService provided by the MyExecutorProvider
86   * &#64;Inject &#64;Named("my-executor") ExecutorService myExecutor;
87   * </pre>
88   *
89   * @author Marek Potociar (marek.potociar at oracle.com)
90   * @see ScheduledExecutorServiceProvider
91   * @see ThreadPoolExecutorProvider
92   * @since 2.18
93   */
94  @Contract
95  public interface ExecutorServiceProvider {
96  
97      /**
98       * Get an executor service to be used by Jersey client or server runtime to execute specific tasks.
99       * <p>
100      * This method is <em>usually</em> invoked just once at either Jersey client or server application runtime initialization,
101      * it <em>may</em> however be invoked multiple times. Once the instance of the provided executor service is not
102      * needed anymore by Jersey application runtime, it will be {@link #dispose disposed}.
103      * This typically happens in one of the following situations:
104      * </p>
105      * <ul>
106      * <li>Jersey client instance is closed (client runtime is shut down).</li>
107      * <li>Jersey container running a server-side Jersey application is shut down.</li>
108      * <li>Jersey server-side application is un-deployed.</li>
109      * </ul>
110      *
111      * @return an executor service. Must not return {@code null}.
112      */
113     public ExecutorService getExecutorService();
114 
115     /**
116      * Invoked when Jersey runtime no longer requires use of the provided executor service.
117      *
118      * @param executorService executor service to be disposed.
119      */
120     public void dispose(ExecutorService executorService);
121 }
