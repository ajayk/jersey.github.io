231     @Override
232     public boolean configure(FeatureContext context) {
233         boolean enabled = false;
234 
235         if (context.getConfiguration().getRuntimeType() == RuntimeType.CLIENT) {
236             ClientLoggingFilter clientLoggingFilter = (ClientLoggingFilter) createLoggingFilter(context, RuntimeType.CLIENT);
237             context.register(clientLoggingFilter);
238             enabled = true;
239         }
240         if (context.getConfiguration().getRuntimeType() == RuntimeType.SERVER) {
241             ServerLoggingFilter serverClientFilter = (ServerLoggingFilter) createLoggingFilter(context, RuntimeType.SERVER);
242             context.register(serverClientFilter);
243             enabled = true;
244         }
245         return enabled;
246     }

248     private LoggingInterceptor createLoggingFilter(FeatureContext context, RuntimeType runtimeType) {
249         Map properties = context.getConfiguration().getProperties();
250         String filterLoggerName = CommonProperties.getValue(
251                 properties,
252                 runtimeType == RuntimeType.SERVER ? LOGGING_FEATURE_LOGGER_NAME_SERVER : LOGGING_FEATURE_LOGGER_NAME_CLIENT,
253                 CommonProperties.getValue(
254                         properties,
255                         LOGGING_FEATURE_LOGGER_NAME,
256                         DEFAULT_LOGGER_NAME
257                 ));
258         String filterLevel = CommonProperties.getValue(
259                 properties,
260                 runtimeType == RuntimeType.SERVER ? LOGGING_FEATURE_LOGGER_LEVEL_SERVER : LOGGING_FEATURE_LOGGER_LEVEL_CLIENT,
261                 CommonProperties.getValue(
262                         context.getConfiguration().getProperties(),
263                         LOGGING_FEATURE_LOGGER_LEVEL,
264                         DEFAULT_LOGGER_LEVEL));
265         Verbosity filterVerbosity = CommonProperties.getValue(
266                 properties,
267                 runtimeType == RuntimeType.SERVER ? LOGGING_FEATURE_VERBOSITY_SERVER : LOGGING_FEATURE_VERBOSITY_CLIENT,
268                 CommonProperties.getValue(
269                         properties,
270                         LOGGING_FEATURE_VERBOSITY,
271                         DEFAULT_VERBOSITY
272                 ));
273         int filterMaxEntitySize = CommonProperties.getValue(
274                 properties,
275                 runtimeType == RuntimeType.SERVER ? LOGGING_FEATURE_MAX_ENTITY_SIZE_SERVER
276                         : LOGGING_FEATURE_MAX_ENTITY_SIZE_CLIENT,
277                 CommonProperties.getValue(
278                         properties,
279                         LOGGING_FEATURE_MAX_ENTITY_SIZE,
280                         DEFAULT_MAX_ENTITY_SIZE
281                 ));
282 
283         Level loggerLevel = Level.parse(filterLevel);
284 
285         if (runtimeType == RuntimeType.SERVER) {
286             return new ServerLoggingFilter(filterLogger != null ? filterLogger : Logger.getLogger(filterLoggerName),
287                     level != null ? level : loggerLevel,
288                     verbosity != null ? verbosity : filterVerbosity,
289                     maxEntitySize != null ? maxEntitySize : filterMaxEntitySize);
290         } else {
291             return new ClientLoggingFilter(filterLogger != null ? filterLogger : Logger.getLogger(filterLoggerName),
292                     level != null ? level : loggerLevel,
293                     verbosity != null ? verbosity : filterVerbosity,
294                     maxEntitySize != null ? maxEntitySize : filterMaxEntitySize);
295         }
296     }

298     /**
299      * {@code Verbosity} determines how detailed message will be logged.
300      * <p>
301      * <ul>
302      * <li>The lowest verbosity ({@link #HEADERS_ONLY}) will log only request/response headers.</li>
303      * <li>
304      * The medium verbosity will log request/response headers, as well as an entity if considered a readable text. See {@link
305      * #PAYLOAD_TEXT}.
306      * </li>
307      * <li>The highest verbosity will log all types of an entity (besides the request/response headers.</li>
308      * </ul>
309      * <p>
310      * Note that the entity is logged up to the maximum number specified in any of the following constructors {@link
311      * LoggingFeature#LoggingFeature(Logger, Integer)}, {@link LoggingFeature#LoggingFeature(Logger, Level, Verbosity, Integer)}
312      * or by some of the feature's properties (see {@link #LOGGING_FEATURE_MAX_ENTITY_SIZE}, {@link
313      * #LOGGING_FEATURE_MAX_ENTITY_SIZE_CLIENT}, {@link #LOGGING_FEATURE_MAX_ENTITY_SIZE_SERVER}.
314      */
315     public enum Verbosity {
316         /**
317          * Only content of HTTP headers is logged. No message payload data are logged.
318          */
319         HEADERS_ONLY,
320         /**
321          * Content of HTTP headers as well as entity content of textual media types is logged. Following is the list of media
322          * types that are considered textual for the logging purposes:
323          * <ul>
324          * <li>{@code text/*}</li>
325          * <li>{@code application/atom+xml}</li>
326          * <li>{@code application/json}</li>
327          * <li>{@code application/svg+xml}</li>
328          * <li>{@code application/x-www-form-urlencoded}</li>
329          * <li>{@code application/xhtml+xml}</li>
330          * <li>{@code application/xml}</li>
331          * </ul>
332          */
333         PAYLOAD_TEXT,
334         /**
335          * Full verbose logging. Content of HTTP headers as well as any message payload content will be logged.
336          */
337         PAYLOAD_ANY
338     }
