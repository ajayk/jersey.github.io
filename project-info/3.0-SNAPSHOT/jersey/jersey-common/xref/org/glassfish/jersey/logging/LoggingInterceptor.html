69  /**
70   * An interceptor that logs an entity if configured so and provides a common logic for {@link ClientLoggingFilter}

96      private static final Set<MediaType> READABLE_APP_MEDIA_TYPES = new HashSet<MediaType>() {{
97          add(TEXT_MEDIA_TYPE);
98          add(MediaType.APPLICATION_ATOM_XML_TYPE);
99          add(MediaType.APPLICATION_FORM_URLENCODED_TYPE);

106     private static final Comparator<Map.Entry<String, List<String>>> COMPARATOR =
107             new Comparator<Map.Entry<String, List<String>>>() {

109                 @Override

122     /**
123      * Creates a logging filter with custom logger and entity logging turned on, but potentially limiting the size
124      * of entity to be buffered and logged.
125      *
126      * @param logger        the logger to log messages to.
127      * @param level         level at which the messages will be logged.
128      * @param verbosity     verbosity of the logged messages. See {@link Verbosity}.
129      * @param maxEntitySize maximum number of entity bytes to be logged (and buffered) - if the entity is larger,
130      *                      logging filter will print (and buffer in memory) only the specified number of bytes
131      *                      and print "...more..." string at the end. Negative values are interpreted as zero.
132      */
133     LoggingInterceptor(final Logger logger, final Level level, final Verbosity verbosity, final int maxEntitySize) {
134         this.logger = logger;
135         this.level = level;
136         this.verbosity = verbosity;
137         this.maxEntitySize = Math.max(0, maxEntitySize);
138     }

140     /**
141      * Logs a {@link StringBuilder} parameter at required level.
142      *
143      * @param b message to log
144      */
145     void log(final StringBuilder b) {
146         if (logger != null && logger.isLoggable(level)) {
147             logger.log(level, b.toString());
148         }

151     private StringBuilder prefixId(final StringBuilder b, final long id) {
152         b.append(Long.toString(id)).append(" ");
153         return b;
154     }
155 
156     void printRequestLine(final StringBuilder b, final String note, final long id, final String method, final URI uri) {
157         prefixId(b, id).append(NOTIFICATION_PREFIX)
158                 .append(note)
159                 .append(" on thread ").append(Thread.currentThread().getName())
160                 .append("\n");
161         prefixId(b, id).append(REQUEST_PREFIX).append(method).append(" ")
162                 .append(uri.toASCIIString()).append("\n");
163     }
164 
165     void printResponseLine(final StringBuilder b, final String note, final long id, final int status) {
166         prefixId(b, id).append(NOTIFICATION_PREFIX)
167                 .append(note)
168                 .append(" on thread ").append(Thread.currentThread().getName()).append("\n");
169         prefixId(b, id).append(RESPONSE_PREFIX)
170                 .append(Integer.toString(status))
171                 .append("\n");
172     }
173 
174     void printPrefixedHeaders(final StringBuilder b,
175                               final long id,
176                               final String prefix,
177                               final MultivaluedMap<String, String> headers) {
178         for (final Map.Entry<String, List<String>> headerEntry : getSortedHeaders(headers.entrySet())) {
179             final List<?> val = headerEntry.getValue();
180             final String header = headerEntry.getKey();
181 
182             if (val.size() == 1) {
183                 prefixId(b, id).append(prefix).append(header).append(": ").append(val.get(0)).append("\n");
184             } else {
185                 final StringBuilder sb = new StringBuilder();
186                 boolean add = false;
187                 for (final Object s : val) {
188                     if (add) {
189                         sb.append(',');
190                     }
191                     add = true;
192                     sb.append(s);
193                 }
194                 prefixId(b, id).append(prefix).append(header).append(": ").append(sb.toString()).append("\n");
195             }
196         }
197     }
198 
199     Set<Map.Entry<String, List<String>>> getSortedHeaders(final Set<Map.Entry<String, List<String>>> headers) {
200         final TreeSet<Map.Entry<String, List<String>>> sortedHeaders = new TreeSet<Map.Entry<String, List<String>>>(COMPARATOR);
201         sortedHeaders.addAll(headers);
202         return sortedHeaders;
203     }

205     InputStream logInboundEntity(final StringBuilder b, InputStream stream, final Charset charset) throws IOException {
206         if (!stream.markSupported()) {
207             stream = new BufferedInputStream(stream);
208         }
209         stream.mark(maxEntitySize + 1);
210         final byte[] entity = new byte[maxEntitySize + 1];
211         final int entitySize = stream.read(entity);
212         b.append(new String(entity, 0, Math.min(entitySize, maxEntitySize), charset));
213         if (entitySize > maxEntitySize) {
214             b.append("...more...");
215         }
216         b.append('\n');
217         stream.reset();
218         return stream;
219     }

221     @Override
222     public void aroundWriteTo(final WriterInterceptorContext writerInterceptorContext)
223             throws IOException, WebApplicationException {
224         final LoggingStream stream = (LoggingStream) writerInterceptorContext.getProperty(ENTITY_LOGGER_PROPERTY);
225         writerInterceptorContext.proceed();
226         if (logger.isLoggable(level) && printEntity(verbosity, writerInterceptorContext.getMediaType())) {
227             if (stream != null) {
228                 log(stream.getStringBuilder(MessageUtils.getCharset(writerInterceptorContext.getMediaType())));
229             }
230         }
231     }

233     /**
234      * Returns {@code true} if specified {@link MediaType} is considered textual.
235      * <p>
236      * See {@link #READABLE_APP_MEDIA_TYPES}.
237      *
238      * @param mediaType the media type of the entity
239      * @return {@code true} if specified {@link MediaType} is considered textual.
240      */
241     static boolean isReadable(MediaType mediaType) {
242         if (mediaType != null) {
243             for (MediaType readableMediaType : READABLE_APP_MEDIA_TYPES) {
244                 if (readableMediaType.isCompatible(mediaType)) {
245                     return true;
246                 }
247             }
248         }
249         return false;
250     }
251 
252     /**
253      * Returns {@code true} if entity has to be printed.
254      *
255      * @param verbosity the configured verbosity .
256      * @param mediaType the media type of the payload.
257      * @return {@code true} if entity has to be printed.
258      */
259     static boolean printEntity(Verbosity verbosity, MediaType mediaType) {
260         return verbosity == Verbosity.PAYLOAD_ANY || (verbosity == Verbosity.PAYLOAD_TEXT && isReadable(mediaType));
261     }

263     /**
264      * Helper class used to log an entity to the output stream up to the specified maximum number of bytes.
265      */
266     class LoggingStream extends FilterOutputStream {
267 
268         private final StringBuilder b;
269         private final ByteArrayOutputStream baos = new ByteArrayOutputStream();
270 
271         /**
272          * Creates {@code LoggingStream} with the entity and the underlying output stream as parameters.
273          *
274          * @param b     contains the entity to log.
275          * @param inner the underlying output stream.
276          */
277         LoggingStream(final StringBuilder b, final OutputStream inner) {
278             super(inner);
279 
280             this.b = b;
281         }
282 
283         StringBuilder getStringBuilder(final Charset charset) {
284             // write entity to the builder
285             final byte[] entity = baos.toByteArray();
286 
287             b.append(new String(entity, 0, Math.min(entity.length, maxEntitySize), charset));
288             if (entity.length > maxEntitySize) {
289                 b.append("...more...");
290             }
291             b.append('\n');
292 
293             return b;
294         }
295 
296         @Override
297         public void write(final int i) throws IOException {
298             if (baos.size() <= maxEntitySize) {
299                 baos.write(i);
300             }
301             out.write(i);
302         }
303     }
