68      /**
69       * Order the templates according to JAX-RS specification.
70       * <p>
71       * Sort the set of matching resource classes using the number of
72       * characters in the regular expression not resulting from template
73       * variables as the primary key, the number of matching groups
74       * as a secondary key, and the number of explicit regular expression
75       * declarations as the tertiary key.
76       * </p>
77       */
78      public static final Comparator<UriTemplate> COMPARATOR = new Comparator<UriTemplate>() {
79  
80          @Override
81          public int compare(UriTemplate o1, UriTemplate o2) {
82              if (o1 == null && o2 == null) {
83                  return 0;
84              }
85              if (o1 == null) {
86                  return 1;
87              }
88              if (o2 == null) {
89                  return -1;
90              }
91  
92              if (o1 == EMPTY && o2 == EMPTY) {
93                  return 0;
94              }
95              if (o1 == EMPTY) {
96                  return 1;
97              }
98              if (o2 == EMPTY) {
99                  return -1;
100             }
101 
102             // Compare the number of explicit characters
103             // Note that it is important that o2 is compared against o1
104             // so that a regular expression with say 10 explicit characters
105             // is less than a regular expression with say 5 explicit characters.
106             int i = o2.getNumberOfExplicitCharacters() - o1.getNumberOfExplicitCharacters();
107             if (i != 0) {
108                 return i;
109             }
110 
111             // If the number of explicit characters is equal
112             // compare the number of template variables
113             // Note that it is important that o2 is compared against o1
114             // so that a regular expression with say 10 template variables
115             // is less than a regular expression with say 5 template variables.
116             i = o2.getNumberOfTemplateVariables() - o1.getNumberOfTemplateVariables();
117             if (i != 0) {
118                 return i;
119             }
120 
121             // If the number of template variables is equal
122             // compare the number of explicit regexes
123             i = o2.getNumberOfExplicitRegexes() - o1.getNumberOfExplicitRegexes();
124             if (i != 0) {
125                 return i;
126             }
127 
128             // If the number of explicit characters and template variables
129             // are equal then comapre the regexes
130             // The order does not matter as long as templates with different
131             // explicit characters are distinguishable
132             return o2.pattern.getRegex().compareTo(o1.pattern.getRegex());
133         }
134     };

293     /**
294      * Resolve a relative URI reference against a base URI as defined in
295      * <a href="http://tools.ietf.org/html/rfc3986#section-5.4">RFC 3986</a>.
296      *
297      * @param baseUri base URI to be used for resolution.
298      * @param refUri  reference URI to be resolved against the base URI.
299      * @return resolved URI.
300      */
301     public static URI resolve(final URI baseUri, URI refUri) {
302         Preconditions.checkNotNull(baseUri, "Input base URI parameter must not be null.");
303         Preconditions.checkNotNull(refUri, "Input reference URI parameter must not be null.");
304 
305         final String refString = refUri.toString();
306         if (refString.isEmpty()) {
307             // we need something to resolve against
308             refUri = URI.create("#");
309         } else if (refString.startsWith("?")) {
310             String baseString = baseUri.toString();
311             final int qIndex = baseString.indexOf('?');
312             baseString = qIndex > -1 ? baseString.substring(0, qIndex) : baseString;
313             return URI.create(baseString + refString);
314         }
315 
316         URI result = baseUri.resolve(refUri);
317         if (refString.isEmpty()) {
318             final String resolvedString = result.toString();
319             result = URI.create(resolvedString.substring(0, resolvedString.indexOf('#')));
320         }
321 
322         return normalize(result);
323     }

325     /**
326      * Normalize the URI by resolve the dot & dot-dot path segments as described in
327      * <a href="http://tools.ietf.org/html/rfc3986#section-5.2.4">RFC 3986</a>.
328      *
329      * This method provides a workaround for issues with {@link java.net.URI#normalize()} which
330      * is not able to properly normalize absolute paths that start with a {@code ".."} segment,
331      * e.g. {@code "/../a/b"} as required by RFC 3986 (according to RFC 3986 the path {@code "/../a/b"}
332      * should resolve to {@code "/a/b"}, while {@code URI.normalize()} keeps the {@code ".."} segment
333      * in the URI path.
334      *
335      * @param uri the original URI string.
336      * @return the URI with dot and dot-dot segments resolved.
337      *
338      * @throws IllegalArgumentException If the given string violates the URI specification RFC.
339      * @see java.net.URI#normalize()
340      */
341     public static URI normalize(final String uri) {
342         return normalize(URI.create(uri));
343     }

345     /**
346      * Normalize the URI by resolve the dot & dot-dot path segments as described in
347      * <a href="http://tools.ietf.org/html/rfc3986#section-5.2.4">RFC 3986</a>.
348      *
349      * This method provides a workaround for issues with {@link java.net.URI#normalize()} which
350      * is not able to properly normalize absolute paths that start with a {@code ".."} segment,
351      * e.g. {@code "/../a/b"} as required by RFC 3986 (according to RFC 3986 the path {@code "/../a/b"}
352      * should resolve to {@code "/a/b"}, while {@code URI.normalize()} keeps the {@code ".."} segment
353      * in the URI path.
354      *
355      * @param uri the original URI.
356      * @return the URI with dot and dot-dot segments resolved.
357      *
358      * @see java.net.URI#normalize()
359      */
360     public static URI normalize(final URI uri) {
361         Preconditions.checkNotNull(uri, "Input reference URI parameter must not be null.");
362 
363         final String path = uri.getPath();
364 
365         if (path == null || path.isEmpty() || !path.contains("/.")) {
366             return uri;
367         }
368 
369         final String[] segments = path.split("/");
370         final Deque<String> resolvedSegments = new ArrayDeque<String>(segments.length);
371 
372         for (final String segment : segments) {
373             if (segment.isEmpty() || ".".equals(segment)) {
374                 // skip
375             } else if ("..".equals(segment)) {
376                 resolvedSegments.pollLast();
377             } else {
378                 resolvedSegments.offer(segment);
379             }
380         }
381 
382         final StringBuilder pathBuilder = new StringBuilder();
383         for (final String segment : resolvedSegments) {
384             pathBuilder.append('/').append(segment);
385         }
386 
387         String resultString = createURIWithStringValues(uri.getScheme(),
388                 uri.getAuthority(),
389                 null,
390                 null,
391                 null,
392                 pathBuilder.toString(),
393                 uri.getQuery(),
394                 uri.getFragment(),
395                 EMPTY_VALUES,
396                 false,
397                 false);
398 
399         return URI.create(resultString);
400     }

598     /**
599      * Create a URI by substituting any template variables
600      * for corresponding template values.
601      * <p>
602      * A URI template variable without a value will be substituted by the
603      * empty string.
604      *
605      * @param values the array of template values. The values will be
606      *               substituted in order of occurrence of unique template variables.
607      * @param offset the offset into the template value array.
608      * @param length the length of the template value array.
609      * @return the URI.
610      */
611     public final String createURI(final String[] values, final int offset, final int length) {
612 
613         TemplateValueStrategy ns = new TemplateValueStrategy() {
614             private final int lengthPlusOffset = length + offset;
615             private int v = offset;
616             private final Map<String, String> mapValues = new HashMap<String, String>();
617 
618             @Override
619             public String valueFor(String templateVariable, String matchedGroup) {
620                 // Check if a template variable has already occurred
621                 // If so use the value to ensure that two or more declarations of
622                 // a template variable have the same value
623                 String tValue = mapValues.get(templateVariable);
624                 if (tValue == null) {
625                     if (v < lengthPlusOffset) {
626                         tValue = values[v++];
627                         if (tValue != null) {
628                             mapValues.put(templateVariable, tValue);
629                         }
630                     }
631                 }
632 
633                 return tValue;
634             }
635         };
636 
637         final StringBuilder sb = new StringBuilder();
638         resolveTemplate(normalizedTemplate, sb, ns);
639         return sb.toString();
640     }

642     /**
643      * Build a URI based on the parameters provided by the variable name strategy.
644      *
645      * @param normalizedTemplate normalized URI template. A normalized template is a template without any explicit regular
646      *                           expressions.
647      * @param builder            URI string builder to be used.
648      * @param valueStrategy      The template value producer strategy to use.
649      */
650     private static void resolveTemplate(
651             String normalizedTemplate,
652             StringBuilder builder,
653             TemplateValueStrategy valueStrategy) {
654         // Find all template variables
655         Matcher m = TEMPLATE_NAMES_PATTERN.matcher(normalizedTemplate);
656 
657         int i = 0;
658         while (m.find()) {
659             builder.append(normalizedTemplate, i, m.start());
660             String variableName = m.group(1);
661             // TODO matrix
662             char firstChar = variableName.charAt(0);
663             if (firstChar == '?' || firstChar == ';') {
664                 final char prefix;
665                 final char separator;
666                 final String emptyValueAssignment;
667                 if (firstChar == '?') {
668                     // query
669                     prefix = '?';
670                     separator = '&';
671                     emptyValueAssignment = "=";
672                 } else {
673                     // matrix
674                     prefix = ';';
675                     separator = ';';
676                     emptyValueAssignment = "";
677                 }
678 
679                 int index = builder.length();
680                 String[] variables = variableName.substring(1).split(", ?");
681                 for (String variable : variables) {
682                     try {
683                         String value = valueStrategy.valueFor(variable, m.group());
684                         if (value != null) {
685                             if (index != builder.length()) {
686                                 builder.append(separator);
687                             }
688 
689                             builder.append(variable);
690                             if (value.isEmpty()) {
691                                 builder.append(emptyValueAssignment);
692                             } else {
693                                 builder.append('=');
694                                 builder.append(value);
695                             }
696                         }
697                     } catch (IllegalArgumentException ex) {
698                         // no value found => ignore the variable
699                     }
700                 }
701 
702                 if (index != builder.length() && (index == 0 || builder.charAt(index - 1) != prefix)) {
703                     builder.insert(index, prefix);
704                 }
705             } else {
706                 String value = valueStrategy.valueFor(variableName, m.group());
707 
708                 if (value != null) {
709                     builder.append(value);
710                 }
711             }
712 
713             i = m.end();
714         }
715         builder.append(normalizedTemplate, i, normalizedTemplate.length());
716     }

910     private static String createURIWithStringValues(
911             final String scheme, final String authority, final String userInfo, final String host, final String port,
912             final String path, final String query, final String fragment, final String[] values, final boolean encode,
913             final boolean encodeSlashInPath, final Map<String, ?> mapValues) {
914 
915         final StringBuilder sb = new StringBuilder();
916         int offset = 0;
917 
918         if (scheme != null) {
919             offset = createUriComponent(UriComponent.Type.SCHEME, scheme, values,
920                     offset, false, mapValues, sb);
921             sb.append(':');
922         }
923 
924         boolean hasAuthority = false;
925         if (notEmpty(userInfo) || notEmpty(host) || notEmpty(port)) {
926             hasAuthority = true;
927             sb.append("//");
928 
929             if (notEmpty(userInfo)) {
930                 offset = createUriComponent(UriComponent.Type.USER_INFO, userInfo, values,
931                         offset, encode, mapValues, sb);
932                 sb.append('@');
933             }
934 
935             if (notEmpty(host)) {
936                 // TODO check IPv6 address
937                 offset = createUriComponent(UriComponent.Type.HOST, host, values,
938                         offset, encode, mapValues, sb);
939             }
940 
941             if (notEmpty(port)) {
942                 sb.append(':');
943                 offset = createUriComponent(UriComponent.Type.PORT, port, values,
944                         offset, false, mapValues, sb);
945             }
946         } else if (notEmpty(authority)) {
947             hasAuthority = true;
948             sb.append("//");
949 
950             offset = createUriComponent(UriComponent.Type.AUTHORITY, authority, values,
951                     offset, encode, mapValues, sb);
952         }
953 
954         if (notEmpty(path) || notEmpty(query) || notEmpty(fragment)) {
955             // make sure we append at least the root path if only query or fragment is present
956             if (hasAuthority && (path == null || path.isEmpty() || path.charAt(0) != '/')) {
957                 sb.append('/');
958             }
959 
960             if (notEmpty(path)) {
961                 // path template values are treated as path segments unless encodeSlashInPath is false.
962                 UriComponent.Type t = (encodeSlashInPath) ? UriComponent.Type.PATH_SEGMENT : UriComponent.Type.PATH;
963 
964                 offset = createUriComponent(t, path, values,
965                         offset, encode, mapValues, sb);
966             }
967 
968             if (notEmpty(query)) {
969                 sb.append('?');
970                 offset = createUriComponent(UriComponent.Type.QUERY_PARAM, query, values,
971                         offset, encode, mapValues, sb);
972             }
973 
974             if (notEmpty(fragment)) {
975                 sb.append('#');
976                 createUriComponent(UriComponent.Type.FRAGMENT, fragment, values,
977                         offset, encode, mapValues, sb);
978             }
979         }
980         return sb.toString();
981     }
