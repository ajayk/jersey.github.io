68  /**
69   * Scopes a single request/response processing execution on a single thread.
70   * <p>
71   * To execute a code inside of the request scope use one of the {@code runInScope(...)}
72   * methods and supply the task encapsulating the code that should be executed in the scope.
73   * </p>
74   * <p>
75   * Example:
76   * </p>
77   * <pre>
78   * &#064;Inject
79   * RequestScope requestScope;
80   *
81   * ...
82   *
83   * requestScope.runInScope(new Runnable() {
84   *     &#064;Override
85   *     public void run() {
86   *          System.out.println("This is executed in the request scope...");
87   *     }
88   * });
89   * </pre>
90   * <p>
91   * An instance of the request scope can be suspended and retrieved via a call to
92   * {@link RequestScope#suspendCurrent} method. This instance can be later
93   * used to resume the same request scope and run another task in the same scope:
94   * </p>
95   * <pre>
96   *  Instance requestScopeInstance =
97   *      requestScope.runInScope(new Callable&lt;Instance&gt;() {
98   *          &#064;Override
99   *          public Instance call() {
100  *              // This is executed in the new request scope.
101  *
102  *              // The following call will cause that the
103  *              // RequestScope.Instance will not be released
104  *              // automatically and we will have to release
105  *              // it explicitly at the end.
106  *              return requestScope.suspendCurrent();
107  *          }
108  *      });
109  *
110  *  requestScope.runInScope(requestScopeInstance, new Runnable() {
111  *
112  *      &#064;Override
113  *      public void run() {
114  *          // This is executed in the same request scope as code above.
115  *      }
116  *  });
117  *
118  *  // The scope instance must be explicitly released.
119  *  requestScopeInstance.release();
120  * </pre>
121  * <p>
122  * In the previous example the {@link RequestScope.Instance request scope instance}
123  * was suspended and retrieved which also informs {@code requestScope} that it
124  * should not automatically release the instance once the running task is finished.
125  * The {@code requestScopeInstance} is then used to initialize the next
126  * request-scoped execution. The second task will run in the same request scope as the
127  * first task. At the end the suspended {@code requestScopeInstance} must be
128  * manually {@link RequestScope.Instance#release released}. Not releasing the instance
129  * could cause memory leaks. Please note that calling {@link RequestScope#suspendCurrent}
130  * does not retrieve an immutable snapshot of the current request scope but
131  * a live reference to the internal {@link RequestScope.Instance request scope instance}
132  * which may change it's state during each request-scoped task execution for
133  * which this scope instance is used.
134  * </p>
135  *
136  * @author Marek Potociar (marek.potociar at oracle.com)
137  * @author Miroslav Fuksa
138  */
139 @Singleton
140 public class RequestScope implements Context<RequestScoped> {

155     @Override
156     public <U> U findOrCreate(ActiveDescriptor<U> activeDescriptor, ServiceHandle<?> root) {
157 
158         final Instance instance = current();
159 
160         U retVal = instance.get(activeDescriptor);
161         if (retVal == null) {
162             retVal = activeDescriptor.create(root);
163             instance.put(activeDescriptor, retVal);
164         }
165         return retVal;
166     }

206     /**
207      * Get a new reference for to currently running request scope instance. This call
208      * prevents automatic {@link RequestScope.Instance#release() release} of the scope
209      * instance once the task that runs in the scope has finished.
210      * <p>
211      * The returned scope instance may be used to run additional task(s) in the
212      * same request scope using one of the {@code #runInScope(Instance, ...)} methods.
213      * </p>
214      * <p>
215      * Note that the returned instance must be {@link RequestScope.Instance#release()
216      * released} manually once not needed anymore to prevent memory leaks.
217      * </p>
218      *
219      * @return currently active {@link RequestScope.Instance request scope instance}.
220      * @throws IllegalStateException in case there is no active request scope associated
221      *                               with the current thread or if the request scope has
222      *                               been already shut down.
223      * @see #suspendCurrent()
224      */
225     public Instance referenceCurrent() throws IllegalStateException {
226         return current().getReference();
227     }

229     private Instance current() {
230         checkState(isActive, "Request scope has been already shut down.");
231 
232         final Instance scopeInstance = currentScopeInstance.get();
233         checkState(scopeInstance != null, "Not inside a request scope.");
234 
235         return scopeInstance;
236     }
237 
238     private Instance retrieveCurrent() {
239         checkState(isActive, "Request scope has been already shut down.");
240         return currentScopeInstance.get();
241     }
242 
243     private void setCurrent(Instance instance) {
244         checkState(isActive, "Request scope has been already shut down.");
245         currentScopeInstance.set(instance);
246     }
247 
248     private void resumeCurrent(Instance instance) {
249         currentScopeInstance.set(instance);
250     }

252     /**
253      * Get the current {@link RequestScope.Instance request scope instance}
254      * and mark it as suspended. This call prevents automatic
255      * {@link RequestScope.Instance#release() release} of the scope instance
256      * once the task that runs in the scope has finished.
257      * <p>
258      * The returned scope instance may be used to run additional task(s) in the
259      * same request scope using one of the {@code #runInScope(Instance, ...)} methods.
260      * </p>
261      * <p>
262      * Note that the returned instance must be {@link RequestScope.Instance#release()
263      * released} manually once not needed anymore to prevent memory leaks.
264      * </p>
265      *
266      * @return currently active {@link RequestScope.Instance request scope instance}
267      *         that was suspended or {@code null} if the thread is not currently running
268      *         in an active request scope.
269      * @see #referenceCurrent()
270      */
271     public Instance suspendCurrent() {
272         final Instance scopeInstance = retrieveCurrent();
273         if (scopeInstance == null) {
274             return null;
275         }
276         try {
277             return scopeInstance.getReference();
278         } finally {
279             logger.debugLog("Returned a new reference of the request scope instance {0}", scopeInstance);
280         }
281     }

284      * Creates a new instance of the {@link RequestScope.Instance request scope instance}.
285      * This instance can be then used to run task in the request scope. Returned instance
286      * is suspended by default and must therefore be closed explicitly as it is shown in
287      * the following example:
288      * <pre>
289      * Instance instance = requestScope.createInstance();
290      * requestScope.runInScope(instance, someRunnableTask);
291      * instance.release();
292      * </pre>
293      *
294      * @return New suspended request scope instance.
295      */
296     public Instance createInstance() {
297         return new Instance();

300     /**
301      * Runs the {@link Runnable task} in the request scope initialized from the
302      * {@link RequestScope.Instance scope instance}. The {@link RequestScope.Instance
303      * scope instance} is NOT released by the method (this must be done explicitly). The
304      * current thread might be already in any request scope and in that case the scope
305      * will be changed to the scope defined by the {@link RequestScope.Instance scope
306      * instance}. At the end of the method the request scope is returned to its original
307      * state.
308      *
309      * @param scopeInstance The request scope instance from which the request scope will
310      *                      be initialized.
311      * @param task          Task to be executed.
312      */
313     public void runInScope(Instance scopeInstance, Runnable task) {
314         final Instance oldInstance = retrieveCurrent();
315         try {
316             setCurrent(scopeInstance.getReference());
317             Errors.process(task);
318         } finally {
319             scopeInstance.release();
320             resumeCurrent(oldInstance);
321         }
322     }

451     /**
452      * Implementation of the request scope instance.
453      */
454     public static final class Instance {
455         /*
456          * Scope instance UUID.
457          *
458          * For performance reasons, it's only generated if toString() method is invoked,
459          * e.g. as part of some low-level logging.
460          */
461 
462         private final LazyUid id = new LazyUid();
463         /**
464          * A map of injectable instances in this scope.
465          */
466         private final Map<ActiveDescriptor<?>, Object> store;
467         /**
468          * Holds the number of snapshots of this scope.
469          */
470         private final AtomicInteger referenceCounter;
471 
472         private Instance() {
473             this.store = new HashMap<ActiveDescriptor<?>, Object>();
474             this.referenceCounter = new AtomicInteger(1);
475         }
476 
477         /**
478          * Get a "new" reference of the scope instance. This will increase
479          * the internal reference counter which prevents the scope instance
480          * to be destroyed until a {@link #release()} method is explicitly
481          * called (once per each {@code getReference()} method call).
482          *
483          * @return referenced scope instance.
484          */
485         private Instance getReference() {
486             // TODO: replace counter with a phantom reference + reference queue-based solution
487             referenceCounter.incrementAndGet();
488             return this;
489         }
490 
491         /**
492          * Get an inhabitant stored in the scope instance that matches the active descriptor .
493          *
494          * @param <T>        inhabitant type.
495          * @param descriptor inhabitant descriptor.
496          * @return matched inhabitant stored in the scope instance or {@code null} if not matched.
497          */
498         @SuppressWarnings("unchecked")
499         <T> T get(ActiveDescriptor<T> descriptor) {
500             return (T) store.get(descriptor);
501         }
502 
503         /**
504          * Store a new inhabitant for the given descriptor.
505          *
506          * @param <T>        inhabitant type.
507          * @param descriptor inhabitant descriptor.
508          * @param value      inhabitant value.
509          * @return old inhabitant previously stored for the given descriptor or
510          *         {@code null} if none stored.
511          */
512         @SuppressWarnings("unchecked")
513         <T> T put(ActiveDescriptor<T> descriptor, T value) {
514             checkState(!store.containsKey(descriptor),
515                     "An instance for the descriptor %s was already seeded in this scope. Old instance: %s New instance: %s",
516                     descriptor,
517                     store.get(descriptor),
518                     value);
519 
520             return (T) store.put(descriptor, value);
521         }
522 
523         /**
524          * Remove a value for the descriptor if present in the scope instance store.
525          *
526          * @param descriptor key for the value to be removed.
527          */
528         @SuppressWarnings("unchecked")
529         <T> void remove(ActiveDescriptor<T> descriptor) {
530             final T removed = (T) store.remove(descriptor);
531             if (removed != null) {
532                 descriptor.dispose(removed);
533             }
534         }
535 
536         private <T> boolean contains(ActiveDescriptor<T> provider) {
537             return store.containsKey(provider);
538         }
539 
540         /**
541          * Release a single reference to the current request scope instance.
542          *
543          * Once all instance references are released, the instance will be recycled.
544          */
545         public void release() {
546             if (referenceCounter.decrementAndGet() < 1) {
547                 try {
548                     for (final ActiveDescriptor<?> descriptor : Sets.newHashSet(store.keySet())) {
549                         remove(descriptor);
550                     }
551                 } finally {
552                     logger.debugLog("Released scope instance {0}", this);
553                 }
554             }
555         }
556 
557         @Override
558         public String toString() {
559             return MoreObjects.toStringHelper(this).add("id", id.value()).add("referenceCounter", referenceCounter.get())
560                     .add("store size", store.size()).toString();
561         }
562     }
