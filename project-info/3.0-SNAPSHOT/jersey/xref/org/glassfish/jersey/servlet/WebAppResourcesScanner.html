40  package org.glassfish.jersey.servlet;
41  
42  import java.io.IOException;
43  import java.io.InputStream;
44  import java.util.Deque;
45  import java.util.LinkedList;
46  import java.util.NoSuchElementException;
47  import java.util.Set;
48  
49  import javax.servlet.ServletContext;
50  
51  import org.glassfish.jersey.server.internal.AbstractResourceFinderAdapter;
52  import org.glassfish.jersey.server.internal.scanning.JarFileScanner;
53  import org.glassfish.jersey.server.internal.scanning.ResourceFinderException;
54  import org.glassfish.jersey.server.internal.scanning.CompositeResourceFinder;
55  
56  /**
57   * A scanner that recursively scans resources within a Web application.
58   *
59   * @author Paul Sandoz
60   */
61  final class WebAppResourcesScanner extends AbstractResourceFinderAdapter {
62  
63      private static final String[] paths = new String[] {"/WEB-INF/lib/", "/WEB-INF/classes/"};
64  
65      private final ServletContext sc;
66      private CompositeResourceFinder compositeResourceFinder = new CompositeResourceFinder();
67  
68      /**
69       * Scan from a set of web resource paths.
70       * <p/>
71       *
72       * @param sc {@link ServletContext}.
73       */
74      WebAppResourcesScanner(final ServletContext sc) {
75          this.sc = sc;
76  
77          processPaths(paths);
78      }
79  
80      private void processPaths(final String... paths) {
81          for (final String path : paths) {
82  
83              final Set<String> resourcePaths = sc.getResourcePaths(path);
84              if (resourcePaths == null) {
85                  break;
86              }
87  
88              compositeResourceFinder.push(new AbstractResourceFinderAdapter() {
89  
90                  private final Deque<String> resourcePathsStack = new LinkedList<String>() {
91  
92                      private static final long serialVersionUID = 3109256773218160485L;
93  
94                      {
95                          for (final String resourcePath : resourcePaths) {
96                              push(resourcePath);
97                          }
98                      }
99                  };
100 
101                 private String current;
102                 private String next;
103 
104                 @Override
105                 public boolean hasNext() {
106                     while (next == null && !resourcePathsStack.isEmpty()) {
107                         next = resourcePathsStack.pop();
108 
109                         if (next.endsWith("/")) {
110                             processPaths(next);
111                             next = null;
112                         } else if (next.endsWith(".jar")) {
113                             try {
114                                 compositeResourceFinder.push(new JarFileScanner(sc.getResourceAsStream(next), "", true));
115                             } catch (final IOException ioe) {
116                                 throw new ResourceFinderException(ioe);
117                             }
118                             next = null;
119                         }
120                     }
121 
122                     return next != null;
123                 }
124 
125                 @Override
126                 public String next() {
127                     if (next != null || hasNext()) {
128                         current = next;
129                         next = null;
130                         return current;
131                     }
132 
133                     throw new NoSuchElementException();
134                 }
135 
136                 @Override
137                 public InputStream open() {
138                     return sc.getResourceAsStream(current);
139                 }
140 
141                 @Override
142                 public void reset() {
143                     throw new UnsupportedOperationException();
144                 }
145             });
146 
147         }
148     }
149 
150     @Override
151     public boolean hasNext() {
152         return compositeResourceFinder.hasNext();
153     }
154 
155     @Override
156     public String next() {
157         return compositeResourceFinder.next();
158     }
159 
160     @Override
161     public InputStream open() {
162         return compositeResourceFinder.open();
163     }
164 
165     @Override
166     public void close() {
167         compositeResourceFinder.close();
168     }
169 
170     @Override
171     public void reset() {
172         compositeResourceFinder = new CompositeResourceFinder();
173         processPaths(paths);
174     }
175 }
