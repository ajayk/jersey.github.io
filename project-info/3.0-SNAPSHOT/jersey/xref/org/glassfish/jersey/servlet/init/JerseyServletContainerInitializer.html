75  /*
76   It is RECOMMENDED that implementations support the Servlet 3 framework
77   pluggability mechanism to enable portability between containers and to avail
78   themselves of container-supplied class scanning facilities.
79   When using the pluggability mechanism the following conditions MUST be met:
80  
81   - If no Application subclass is present the added servlet MUST be
82   named "javax.ws.rs.core.Application" and all root resource classes and
83   providers packaged in the web application MUST be included in the published
84   JAX-RS application. The application MUST be packaged with a web.xml that
85   specifies a servlet mapping for the added servlet.
86  
87   - If an Application subclass is present and there is already a servlet defined
88   that has a servlet initialization parameter named "javax.ws.rs.Application"
89   whose value is the fully qualified name of the Application subclass then no
90   servlet should be added by the JAX-RS implementation's ContainerInitializer
91   since the application is already being handled by an existing servlet.
92  
93   - If an application subclass is present that is not being handled by an
94   existing servlet then the servlet added by the ContainerInitializer MUST be
95   named with the fully qualified name of the Application subclass.  If the
96   Application subclass is annotated with @PathPrefix and no servlet-mapping
97   exists for the added servlet then a new servlet mapping is added with the
98   value of the @PathPrefix  annotation with "/*" appended otherwise the existing
99   mapping is used. If the Application subclass is not annotated with @PathPrefix
100  then the application MUST be packaged with a web.xml that specifies a servlet
101  mapping for the added servlet. It is an error for more than one Application
102  to be deployed at the same effective servlet mapping.
103 
104  In either of the latter two cases, if both Application#getClasses and
105  Application#getSingletons return an empty list then all root resource classes
106  and providers packaged in the web application MUST be included in the
107  published JAX-RS application. If either getClasses or getSingletons return a
108  non-empty list then only those classes or singletons returned MUST be included
109  in the published JAX-RS application.
110 
111  If not using the Servlet 3 framework pluggability mechanism
112  (e.g. in a pre-Servlet 3.0 container), the servlet-class or filter-class
113  element of the web.xml descriptor SHOULD name the JAX-RS
114  implementation-supplied Servlet or Filter class respectively. The
115  application-supplied subclass of Application SHOULD be identified using an
116  init-param with a param-name of javax.ws.rs.Application.
117  */

120  * {@link ServletContainerInitializer} implementation used for Servlet 3.x deployment.
121  *
122  * @author Paul Sandoz
123  * @author Martin Matula
124  * @author Libor Kramolis (libor.kramolis at oracle.com)
125  */
126 @HandlesTypes({ Path.class, Provider.class, Application.class, ApplicationPath.class })
127 public final class JerseyServletContainerInitializer implements ServletContainerInitializer {
128 
129     private static final Logger LOGGER = Logger.getLogger(JerseyServletContainerInitializer.class.getName());
130 
131     @Override
132     public void onStartup(Set<Class<?>> classes, final ServletContext servletContext) throws ServletException {
133         final ServletContainerProvider[] allServletContainerProviders =
134                 ServletContainerProviderFactory.getAllServletContainerProviders();
135 
136         if (classes == null) {
137             classes = Collections.emptySet();
138         }
139         // PRE INIT
140         for (final ServletContainerProvider servletContainerProvider : allServletContainerProviders) {
141             servletContainerProvider.preInit(servletContext, classes);
142         }
143         // INIT IMPL
144         onStartupImpl(classes, servletContext);
145         // POST INIT
146         for (final ServletContainerProvider servletContainerProvider : allServletContainerProviders) {
147             servletContainerProvider.postInit(servletContext, classes, findJerseyServletNames(servletContext));
148         }
149         // ON REGISTER
150         for (final ServletContainerProvider servletContainerProvider : allServletContainerProviders) {
151             servletContainerProvider.onRegister(servletContext, findJerseyServletNames(servletContext));
152         }
153     }
154 
155     private void onStartupImpl(final Set<Class<?>> classes, final ServletContext servletContext) throws ServletException {
156         // first see if there are any application classes in the web app
157         for (final Class<? extends Application> applicationClass : getApplicationClasses(classes)) {
158             final ServletRegistration servletRegistration = servletContext.getServletRegistration(applicationClass.getName());
159 
160             if (servletRegistration != null) {
161                 addServletWithExistingRegistration(servletContext, servletRegistration, applicationClass, classes);
162             } else {
163                 // Servlet is not registered with app name or the app name is used to register a different servlet
164                 // check if some servlet defines the app in init params
165                 final List<Registration> srs = getInitParamDeclaredRegistrations(servletContext, applicationClass);
166                 if (!srs.isEmpty()) {
167                     // app handled by at least one servlet or filter
168                     // fix the registrations if needed (i.e. add servlet class)
169                     for (final Registration sr : srs) {
170                         if (sr instanceof ServletRegistration) {
171                             addServletWithExistingRegistration(servletContext, (ServletRegistration) sr,
172                                     applicationClass, classes);
173                         }
174                     }
175                 } else {
176                     // app not handled by any servlet/filter -> add it
177                     addServletWithApplication(servletContext, applicationClass, classes);
178                 }
179             }
180         }
181 
182         // check for javax.ws.rs.core.Application registration
183         addServletWithDefaultConfiguration(servletContext, classes);
184     }

186     /**
187      * Returns names of all registered Jersey servlets.
188      *
189      * Servlets are configured in {@code web.xml} or managed via Servlet API.
190      *
191      * @param servletContext the {@link ServletContext} of the web application that is being started
192      * @return list of Jersey servlet names or empty array, never returns {@code null}
193      */
194     private static Set<String> findJerseyServletNames(final ServletContext servletContext) {
195         final Set<String> jerseyServletNames = new HashSet<>();
196 
197         for (final ServletRegistration servletRegistration : servletContext.getServletRegistrations().values()) {
198             if (isJerseyServlet(servletRegistration.getClassName())) {
199                 jerseyServletNames.add(servletRegistration.getName());
200             }
201         }
202         return Collections.unmodifiableSet(jerseyServletNames);
203     }

205     /**
206      * Check if the {@code className} is an implementation of a Jersey Servlet container.
207      *
208      * @return {@code true} if the class is a Jersey servlet container class, {@code false} otherwise.
209      */
210     private static boolean isJerseyServlet(final String className) {
211         return ServletContainer.class.getName().equals(className)
212                 || "org.glassfish.jersey.servlet.portability.PortableServletContainer".equals(className);
213     }

215     private static List<Registration> getInitParamDeclaredRegistrations(final ServletContext context,
216                                                                         final Class<? extends Application> clazz) {
217         final List<Registration> registrations = Lists.newArrayList();
218         collectJaxRsRegistrations(context.getServletRegistrations(), registrations, clazz);
219         collectJaxRsRegistrations(context.getFilterRegistrations(), registrations, clazz);
220         return registrations;
221     }
222 
223     private static void collectJaxRsRegistrations(final Map<String, ? extends Registration> registrations,
224                                                   final List<Registration> collected, final Class<? extends Application> a) {
225         for (final Registration sr : registrations.values()) {
226             final Map<String, String> ips = sr.getInitParameters();
227             if (ips.containsKey(ServletProperties.JAXRS_APPLICATION_CLASS)) {
228                 if (ips.get(ServletProperties.JAXRS_APPLICATION_CLASS).equals(a.getName())) {
229                     collected.add(sr);
230                 }
231             }
232         }
233     }

235     /**
236      * Enhance default servlet (named {@link Application}) configuration.
237      */
238     private static void addServletWithDefaultConfiguration(final ServletContext context,
239                                                            final Set<Class<?>> classes) throws ServletException {
240 
241         ServletRegistration registration = context.getServletRegistration(Application.class.getName());
242 
243         if (registration != null) {
244             final Set<Class<?>> appClasses = getRootResourceAndProviderClasses(classes);
245             final ResourceConfig resourceConfig = ResourceConfig.forApplicationClass(ResourceConfig.class, appClasses)
246                     .addProperties(getInitParams(registration))
247                     .addProperties(Utils.getContextParams(context));
248 
249             if (registration.getClassName() != null) {
250                 // class name present - complete servlet registration from container point of view
251                 Utils.store(resourceConfig, context, registration.getName());
252             } else {
253                 // no class name - no complete servlet registration from container point of view
254                 final ServletContainer servlet = new ServletContainer(resourceConfig);
255                 registration = context.addServlet(registration.getName(), servlet);
256                 ((ServletRegistration.Dynamic) registration).setLoadOnStartup(1);
257 
258                 if (registration.getMappings().isEmpty()) {
259                     // Error
260                     LOGGER.log(Level.WARNING, LocalizationMessages.JERSEY_APP_NO_MAPPING(registration.getName()));
261                 } else {
262                     LOGGER.log(Level.CONFIG,
263                             LocalizationMessages.JERSEY_APP_REGISTERED_CLASSES(registration.getName(), appClasses));
264                 }
265             }
266         }
267     }

269     /**
270      * Add new servlet according to {@link Application} subclass with {@link ApplicationPath} annotation or existing
271      * {@code servlet-mapping}.
272      */
273     private static void addServletWithApplication(final ServletContext context,
274                                                   final Class<? extends Application> clazz,
275                                                   final Set<Class<?>> defaultClasses) throws ServletException {
276         final ApplicationPath ap = clazz.getAnnotation(ApplicationPath.class);
277         if (ap != null) {
278             // App is annotated with ApplicationPath
279             final ResourceConfig resourceConfig = ResourceConfig.forApplicationClass(clazz, defaultClasses)
280                     .addProperties(Utils.getContextParams(context));
281             final ServletContainer s = new ServletContainer(resourceConfig);
282             final ServletRegistration.Dynamic dsr = context.addServlet(clazz.getName(), s);
283             dsr.setAsyncSupported(true);
284             dsr.setLoadOnStartup(1);
285 
286             final String mapping = createMappingPath(ap);
287             if (!mappingExists(context, mapping)) {
288                 dsr.addMapping(mapping);
289 
290                 LOGGER.log(Level.CONFIG, LocalizationMessages.JERSEY_APP_REGISTERED_MAPPING(clazz.getName(), mapping));
291             } else {
292                 LOGGER.log(Level.WARNING, LocalizationMessages.JERSEY_APP_MAPPING_CONFLICT(clazz.getName(), mapping));
293             }
294         }
295     }

297     /**
298      * Enhance existing servlet configuration.
299      */
300     private static void addServletWithExistingRegistration(final ServletContext context,
301                                                            ServletRegistration registration,
302                                                            final Class<? extends Application> clazz,
303                                                            final Set<Class<?>> classes) throws ServletException {
304         // create a new servlet container for a given app.
305         final ResourceConfig resourceConfig = ResourceConfig.forApplicationClass(clazz, classes)
306                 .addProperties(getInitParams(registration))
307                 .addProperties(Utils.getContextParams(context));
308 
309         if (registration.getClassName() != null) {
310             // class name present - complete servlet registration from container point of view
311             Utils.store(resourceConfig, context, registration.getName());
312         } else {
313             // no class name - no complete servlet registration from container point of view
314             final ServletContainer servlet = new ServletContainer(resourceConfig);
315             final ServletRegistration.Dynamic dynamicRegistration = context.addServlet(clazz.getName(), servlet);
316             dynamicRegistration.setAsyncSupported(true);
317             dynamicRegistration.setLoadOnStartup(1);
318             registration = dynamicRegistration;
319         }
320         if (registration.getMappings().isEmpty()) {
321             final ApplicationPath ap = clazz.getAnnotation(ApplicationPath.class);
322             if (ap != null) {
323                 final String mapping = createMappingPath(ap);
324                 if (!mappingExists(context, mapping)) {
325                     registration.addMapping(mapping);
326 
327                     LOGGER.log(Level.CONFIG, LocalizationMessages.JERSEY_APP_REGISTERED_MAPPING(clazz.getName(), mapping));
328                 } else {
329                     LOGGER.log(Level.WARNING, LocalizationMessages.JERSEY_APP_MAPPING_CONFLICT(clazz.getName(), mapping));
330                 }
331             } else {
332                 // Error
333                 LOGGER.log(Level.WARNING, LocalizationMessages.JERSEY_APP_NO_MAPPING_OR_ANNOTATION(clazz.getName(),
334                         ApplicationPath.class.getSimpleName()));
335             }
336         } else {
337             LOGGER.log(Level.CONFIG, LocalizationMessages.JERSEY_APP_REGISTERED_APPLICATION(clazz.getName()));
338         }
339     }

341     private static Map<String, Object> getInitParams(final ServletRegistration sr) {
342         final Map<String, Object> initParams = Maps.newHashMap();
343         for (final Map.Entry<String, String> entry : sr.getInitParameters().entrySet()) {
344             initParams.put(entry.getKey(), entry.getValue());
345         }
346         return initParams;
347     }

349     private static boolean mappingExists(final ServletContext sc, final String mapping) {
350         for (final ServletRegistration sr : sc.getServletRegistrations().values()) {
351             for (final String declaredMapping : sr.getMappings()) {
352                 if (mapping.equals(declaredMapping)) {
353                     return true;
354                 }
355             }
356         }
357 
358         return false;
359     }

363     private static String createMappingPath(final ApplicationPath ap) {
364         String path = ap.value();
365         if (!path.startsWith("/")) {
366             path = "/" + path;
367         }
368 
369         if (!path.endsWith("/*")) {
370             if (path.endsWith("/")) {
371                 path += "*";
372             } else {
373                 path += "/*";
374             }
375         }
376 
377         return path;
378     }

380     private static Set<Class<? extends Application>> getApplicationClasses(final Set<Class<?>> classes) {
381         final Set<Class<? extends Application>> s = new LinkedHashSet<>();
382         for (final Class<?> c : classes) {
383             if (Application.class != c && Application.class.isAssignableFrom(c)) {
384                 s.add(c.asSubclass(Application.class));
385             }
386         }
387 
388         return s;
389     }
390 
391     private static Set<Class<?>> getRootResourceAndProviderClasses(final Set<Class<?>> classes) {
392         // TODO filter out any classes from the Jersey jars
393         final Set<Class<?>> s = new LinkedHashSet<>();
394         for (final Class<?> c : classes) {
395             if (c.isAnnotationPresent(Path.class) || c.isAnnotationPresent(Provider.class)) {
396                 s.add(c);
397             }
398         }
399 
400         return s;
401     }
