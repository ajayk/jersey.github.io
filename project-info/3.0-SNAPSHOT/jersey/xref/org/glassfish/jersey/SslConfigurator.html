66  /**
67   * Utility class, which helps to configure {@link SSLContext} instances.
68   *
69   * For example:
70   * <pre>
71   * SslConfigurator sslConfig = SslConfigurator.newInstance()
72   *    .trustStoreFile("truststore.jks")
73   *    .trustStorePassword("asdfgh")
74   *    .trustStoreType("JKS")
75   *    .trustManagerFactoryAlgorithm("PKIX")
76   *
77   *    .keyStoreFile("keystore.jks")
78   *    .keyPassword("asdfgh")
79   *    .keyStoreType("JKS")
80   *    .keyManagerFactoryAlgorithm("SunX509")
81   *    .keyStoreProvider("SunJSSE")
82   *
83   *    .securityProtocol("SSL");
84   *
85   * SSLContext sslContext = sslConfig.createSSLContext();
86   * </pre>
87   *
88   * @author Alexey Stashok
89   * @author Hubert Iwaniuk
90   * @author Bruno Harbulot
91   * @author Marek Potociar (marek.potociar at oracle.com)
92   */

641     /**
642      * Create new SSL context instance using the current SSL context configuration.
643      *
644      * @return newly configured SSL context instance.
645      */
646     public SSLContext createSSLContext() {
647         TrustManagerFactory trustManagerFactory = null;
648         KeyManagerFactory keyManagerFactory = null;
649 
650         KeyStore _keyStore = keyStore;
651         if (_keyStore == null && (keyStoreBytes != null || keyStoreFile != null)) {
652             try {
653                 if (keyStoreProvider != null) {
654                     _keyStore = KeyStore.getInstance(
655                             keyStoreType != null ? keyStoreType : KeyStore.getDefaultType(), keyStoreProvider);
656                 } else {
657                     _keyStore = KeyStore.getInstance(keyStoreType != null ? keyStoreType : KeyStore.getDefaultType());
658                 }
659                 InputStream keyStoreInputStream = null;
660                 try {
661                     if (keyStoreBytes != null) {
662                         keyStoreInputStream = new ByteArrayInputStream(keyStoreBytes);
663                     } else if (!keyStoreFile.equals("NONE")) {
664                         keyStoreInputStream = new FileInputStream(keyStoreFile);
665                     }
666                     _keyStore.load(keyStoreInputStream, keyStorePass);
667                 } finally {
668                     try {
669                         if (keyStoreInputStream != null) {
670                             keyStoreInputStream.close();
671                         }
672                     } catch (IOException ignored) {
673                     }
674                 }
675             } catch (KeyStoreException e) {
676                 throw new IllegalStateException(LocalizationMessages.SSL_KS_IMPL_NOT_FOUND(), e);
677             } catch (CertificateException e) {
678                 throw new IllegalStateException(LocalizationMessages.SSL_KS_CERT_LOAD_ERROR(), e);
