43  import java.io.IOException;
44  import java.net.URI;
45  import java.net.URISyntaxException;
46  import java.util.Arrays;
47  import java.util.Collections;
48  import java.util.HashSet;
49  import java.util.List;
50  import java.util.Map;
51  import java.util.Set;

57  class RedirectHandler {
58  
59      private static final Set<Integer> REDIRECT_STATUS_CODES = Collections
60              .unmodifiableSet(new HashSet<>(Arrays.asList(301, 302, 303, 307, 308)));
61  
62      private final int maxRedirects;
63      private final boolean followRedirects;
64      private final Set<URI> redirectUriHistory;
65      private final HttpConnectionPool httpConnectionPool;
66      private final HttpRequest originalHttpRequest;
67  
68      private volatile URI lastRequestUri = null;
69  
70      RedirectHandler(HttpConnectionPool httpConnectionPool, HttpRequest originalHttpRequest,
71                      ConnectorConfiguration connectorConfiguration) {
72          this.followRedirects = connectorConfiguration.getFollowRedirects();
73          this.maxRedirects = connectorConfiguration.getMaxRedirects();
74          this.httpConnectionPool = httpConnectionPool;
75          this.originalHttpRequest = originalHttpRequest;
76          this.redirectUriHistory = new HashSet<>(maxRedirects);
77          this.lastRequestUri = originalHttpRequest.getUri();
78      }
79  
80      void handleRedirects(final HttpResponse httpResponse, final CompletionHandler<HttpResponse> completionHandler) {
81          if (!followRedirects) {
82              completionHandler.completed(httpResponse);
83              return;
84          }
85  
86          if (!REDIRECT_STATUS_CODES.contains(httpResponse.getStatusCode())) {
87              completionHandler.completed(httpResponse);
88              return;
89          }
90  
91          if (httpResponse.getStatusCode() != 303) {
92              // we support other methods than GET and HEAD only with 303
93              if (!Constants.HEAD.equals(originalHttpRequest.getMethod()) && !Constants.GET
94                      .equals(originalHttpRequest.getMethod())) {
95                  completionHandler.completed(httpResponse);
96                  return;
97              }
98          }
99  
100         // reading the body is not necessary, but if we wait until the entire body has arrived, we can reuse the same connection
101         consumeBodyIfPresent(httpResponse, new CompletionHandler<Void>() {
102             @Override
103             public void failed(Throwable throwable) {
104                 completionHandler.failed(throwable);
105             }
106 
107             @Override
108             public void completed(Void r) {
109                 doRedirect(httpResponse, new CompletionHandler<HttpResponse>() {
110                     @Override
111                     public void failed(Throwable throwable) {
112                         completionHandler.failed(throwable);
113                     }
114 
115                     @Override
116                     public void completed(HttpResponse result) {
117                         handleRedirects(result, completionHandler);
118                     }
119                 });
120             }
121         });
122     }

124     private void doRedirect(final HttpResponse httpResponse, final CompletionHandler<HttpResponse> completionHandler) {
125 
126         // get location header
127         String locationString = null;
128         final List<String> locationHeader = httpResponse.getHeader("Location");
129         if (locationHeader != null && !locationHeader.isEmpty()) {
130             locationString = locationHeader.get(0);
131         }
132 
133         if (locationString == null || locationString.isEmpty()) {
134             completionHandler.failed(new RedirectException(LocalizationMessages.REDIRECT_NO_LOCATION()));
135             return;
136         }
137 
138         URI location;
139         try {
140             location = new URI(locationString);
141 
142             if (!location.isAbsolute()) {
143                 // location is not absolute, we need to resolve it.
144                 URI baseUri = lastRequestUri;
145                 location = baseUri.resolve(location.normalize());
146             }
147         } catch (URISyntaxException e) {
148             completionHandler.failed(new RedirectException(LocalizationMessages.REDIRECT_ERROR_DETERMINING_LOCATION(), e));
149             return;
150         }
151 
152         // infinite loop detection
153         boolean alreadyRequested = !redirectUriHistory.add(location);
154         if (alreadyRequested) {
155             completionHandler.failed(new RedirectException(LocalizationMessages.REDIRECT_INFINITE_LOOP()));
156             return;
157         }
158 
159         // maximal number of redirection
160         if (redirectUriHistory.size() > maxRedirects) {
161             completionHandler.failed(new RedirectException(LocalizationMessages.REDIRECT_LIMIT_REACHED(maxRedirects)));
162             return;
163         }
164 
165         String method = originalHttpRequest.getMethod();
166         Map<String, List<String>> headers = originalHttpRequest.getHeaders();
167         if (httpResponse.getStatusCode() == 303 && !method.equals(Constants.HEAD)) {
168             // in case of 303 we rewrite every method except HEAD to GET
169             method = Constants.GET;
170             // remove entity-transport headers if present
171             headers.remove(Constants.CONTENT_LENGTH);
172             headers.remove(Constants.TRANSFER_ENCODING_HEADER);
173         }
174 
175         HttpRequest httpRequest = HttpRequest.createBodyless(method, location);
176         httpRequest.getHeaders().putAll(headers);
177         lastRequestUri = location;
178 
179         httpConnectionPool.send(httpRequest, completionHandler);
180     }
181 
182     private void consumeBodyIfPresent(HttpResponse response, final CompletionHandler<Void> completionHandler) {
183         final AsynchronousBodyInputStream bodyStream = response.getBodyStream();
184         bodyStream.setReadListener(new ReadListener() {
185             @Override
186             public void onDataAvailable() {
187                 while (bodyStream.isReady()) {
188                     try {
189                         bodyStream.read();
190                     } catch (IOException e) {
191                         completionHandler.failed(e);
192                     }
193                 }
194             }
195 
196             @Override
197             public void onAllDataRead() {
198                 completionHandler.completed(null);
199             }
200 
201             @Override
202             public void onError(Throwable t) {
203                 completionHandler.failed(t);
204             }
205         });
206     }
207 
208     URI getLastRequestUri() {
209         return lastRequestUri;
210     }
