463     @Override
464     public int hashCode() {
465         int result = poolName != null ? poolName.hashCode() : 0;
466         result = 31 * result + corePoolSize;
467         result = 31 * result + maxPoolSize;
468         result = 31 * result + (queue != null ? queue.hashCode() : 0);
469         result = 31 * result + queueLimit;
470         result = 31 * result + (int) (keepAliveTimeMillis ^ (keepAliveTimeMillis >>> 32));
471         result = 31 * result + (threadFactory != null ? threadFactory.hashCode() : 0);
472         result = 31 * result + priority;
473         result = 31 * result + (isDaemon ? 1 : 0);
474         result = 31 * result + (initialClassLoader != null ? initialClassLoader.hashCode() : 0);
475         return result;
476     }
