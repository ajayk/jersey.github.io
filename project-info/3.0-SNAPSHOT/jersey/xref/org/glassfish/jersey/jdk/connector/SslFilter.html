57  class SslFilter extends Filter<ByteBuffer, ByteBuffer, ByteBuffer, ByteBuffer> {

59  /* SYNCHRONIZATION NOTE: SSLEngine#wrap and SSLEngine#unwrap can be done concurrently (one thread doing wrap
60      and another doing unwrap). The same operation cannot be done concurrently (2 threads doing wrap).
61  
62      Method doHandshakeStep must be synchronized, because it might be entered both by writing and reading thread
63      during re-handshake. Write, close and re-handshake cannot be done concurrently, because all those operations might
64      do SSLEngine#wrap. Read can be be done concurrently with any other operation, because even thought re-handshake
65      can do SSLEngine#unwrap, it won't do so if it was entered from write operation.
66  
67      Operations upstreamFilter#onRead cannot be done while holding a lock of this class. Doing so might lead to a deadlock. An
68      example of deadlock would be if a thread holding a lock in upstreamFilter#onRead writes a response synchronously (blocks
69      and waits for write completion handler). The write completion handler might be executed by another thread which will not be
70      able to obtain a lock for this class.*/
71  
72      /* Some operations on SSL engine require a buffer as a parameter even if they don't need any data.
73      This buffer is for that purpose. */
74      private static final ByteBuffer emptyBuffer = ByteBuffer.allocate(0);
75  
76      // buffer for passing data to the upper filter
77      private final ByteBuffer applicationInputBuffer;
78      // buffer for passing data to the transport filter
79      private final ByteBuffer networkOutputBuffer;
80      private final SSLEngine sslEngine;
81      private final HostnameVerifier customHostnameVerifier;
82      private final String serverHost;
83      private final WriteQueue writeQueue = new WriteQueue();
84  
85      private volatile State state = State.NOT_STARTED;
86      /*
87       * Pending write operation stored when writing data was not possible. It will be resumed when write operation is
88       * available again. Only one write operation can be in progress at a time. Trying to store more than one pending
89       * application write indicates that an upper stack called write without waiting for the completion handler
90       * of the previous write.
91       * Currently this is used only during re-handshake.
92       */
93      private Runnable pendingApplicationWrite = null;

95      /**
96       * SSL Filter constructor, takes upstream filter as a parameter.
97       *
98       * @param downstreamFilter       a filter that is positioned under the SSL filter.
99       * @param sslContext             configuration of SSL engine.
100      * @param serverHost             server host (hostname or IP address), which will be used to verify authenticity of
101      *                               the server (the provided host will be compared against the host in the certificate
102      *                               provided by the server). IP address and hostname cannot be used interchangeably -
103      *                               if a certificate contains hostname and an IP address of the server is provided here,
104      *                               the verification will fail.
105      * @param customHostnameVerifier hostname verifier that will be used instead of the default one.
106      */
107     SslFilter(Filter<ByteBuffer, ByteBuffer, ?, ?> downstreamFilter,
108               SSLContext sslContext,
109               String serverHost,
110               HostnameVerifier customHostnameVerifier) {
111         super(downstreamFilter);
112         this.serverHost = serverHost;
113         sslEngine = sslContext.createSSLEngine(serverHost, -1);
114         sslEngine.setUseClientMode(true);
115         this.customHostnameVerifier = customHostnameVerifier;
116 
117         /**
118          * Enable server host verification.
119          * This can be moved to {@link SslEngineConfigurator} with the rest of {@link SSLEngine} configuration
120          * when {@link SslEngineConfigurator} supports Java 7.
121          */
122         if (customHostnameVerifier == null) {
123             SSLParameters sslParameters = sslEngine.getSSLParameters();
124             sslParameters.setEndpointIdentificationAlgorithm("HTTPS");
125             sslEngine.setSSLParameters(sslParameters);
126         }
127 
128         applicationInputBuffer = ByteBuffer.allocate(sslEngine.getSession().getApplicationBufferSize());
129         networkOutputBuffer = ByteBuffer.allocate(sslEngine.getSession().getPacketBufferSize());
130     }

132     @Override
133     synchronized void write(final ByteBuffer applicationData, final CompletionHandler<ByteBuffer> completionHandler) {
134         switch (state) {
135             // before SSL is started, write just passes through
136             case NOT_STARTED: {
137                 writeQueue.write(applicationData, completionHandler);
138                 return;
139             }
140 
141             /* TODO:
142              The current model does not permit calling write before SSL handshake has completed, if we allow this
143              we could easily get rid of the onSslHandshakeCompleted event. The SSL filter can simply store the write until
144              the handshake has completed like during re-handshake. With such a change HANDSHAKING and REHANDSHAKING could
145              be collapsed into one state. */
146             case HANDSHAKING: {
147                 completionHandler.failed(new IllegalStateException("Cannot write until SSL handshake has been completed"));
148                 break;
149             }
150 
151             /* Suspend all writes until the re-handshaking is done. Data are permitted during re-handshake in SSL, but this
152              would only complicate things */
153             case REHANDSHAKING: {
154                 storePendingApplicationWrite(applicationData, completionHandler);
155                 break;
156             }
157 
158             case DATA: {
159                 handleWrite(applicationData, completionHandler);
160                 break;
161             }
162 
163             case CLOSED: {
164                 // the engine is closed just abort with failure
165                 completionHandler.failed(new IllegalStateException(LocalizationMessages.SSL_SESSION_CLOSED()));
166                 break;
167             }
168         }
169     }

171     private void handleWrite(final ByteBuffer applicationData, final CompletionHandler<ByteBuffer> completionHandler) {
172         try {
173             // transport buffer always writes all data, so there are not leftovers in the networkOutputBuffer
174             networkOutputBuffer.clear();
175             SSLEngineResult result = sslEngine.wrap(applicationData, networkOutputBuffer);
176 
177             switch (result.getStatus()) {
178                 case BUFFER_OVERFLOW: {
179                     /* this means that the content of the ssl packet (max 16kB) did not fit into
180                        networkOutputBuffer, we make sure to set networkOutputBuffer > max 16kB + SSL headers
181                        when initializing this filter. This indicates a bug. */
182                     throw new IllegalStateException("SSL packet does not fit into the network buffer: "
183                             + networkOutputBuffer + "\n" + getDebugState());
184                 }
185 
186                 case BUFFER_UNDERFLOW: {
187                     /* This basically says that there is not enough data to create an SSL packet. Javadoc suggests that
188                     BUFFER_UNDERFLOW can occur only after unwrap(), but to be 100% sure we handle all possible error states: */
189                     throw new IllegalStateException("SSL engine underflow with the following application input: "
190                             + applicationData + "\n" + getDebugState());
191                 }
192 
193                 case CLOSED: {
194                     state = State.CLOSED;
195                     break;
196                 }
197 
198                 case OK: {
199                     // check if we started re-handshaking
200                     if (result.getHandshakeStatus() != SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING) {
201                         state = State.REHANDSHAKING;
202                     }
203 
204                     networkOutputBuffer.flip();
205                     // write only if something was written to the output buffer
206                     if (networkOutputBuffer.hasRemaining()) {
207                         writeQueue.write(networkOutputBuffer, new CompletionHandler<ByteBuffer>() {
208                             @Override
209                             public void completed(ByteBuffer result) {
210                                 handlePostWrite(applicationData, completionHandler);
211                             }
212 
213                             @Override
214                             public void failed(Throwable throwable) {
215                                 completionHandler.failed(throwable);
216                             }
217                         });
218                     } else {
219                         handlePostWrite(applicationData, completionHandler);
220                     }
221                     break;
222                 }
223             }
224 
225         } catch (SSLException e) {
226             handleSslError(e);
227         }
228     }

230     private synchronized void handlePostWrite(final ByteBuffer applicationData,
231                                               final CompletionHandler<ByteBuffer> completionHandler) {
232         if (state == State.REHANDSHAKING) {
233             if (applicationData.hasRemaining()) {
234                 // the remaining data will be sent after re-handshake
235                 storePendingApplicationWrite(applicationData, completionHandler);
236                 // start re-handshaking
237                 doHandshakeStep(emptyBuffer);
238             }
239         } else {
240             if (applicationData.hasRemaining()) {
241                 // make sure to empty the application output buffer
242                 handleWrite(applicationData, completionHandler);
243             } else {
244                 completionHandler.completed(applicationData);
245             }
246         }
247     }
248 
249     private void storePendingApplicationWrite(final ByteBuffer applicationData,
250                                               final CompletionHandler<ByteBuffer> completionHandler) {
251         // store the write until re-handshaking is completed
252         if (pendingApplicationWrite != null) {
253             /* If this happens it means a bug in this class or upper layer called another write() without waiting
254              for a completion handler of the previous one. */
255             throw new IllegalStateException("Only one write operation can be in progress\n" + getDebugState());
256         }
257 
258         pendingApplicationWrite = () -> {
259             // go again through the entire write procedure like this data came directly from the application
260             write(applicationData, completionHandler);
261         };
262     }

264     @Override
265     synchronized void close() {
266         if (state == State.NOT_STARTED) {
267             downstreamFilter.close();
268             return;
269         }
270 
271         sslEngine.closeOutbound();
272         try {
273             LazyBuffer lazyBuffer = new LazyBuffer();
274 
275             while (sslEngine.getHandshakeStatus() == SSLEngineResult.HandshakeStatus.NEED_WRAP) {
276                 ByteBuffer buffer = lazyBuffer.get();
277                 SSLEngineResult result = sslEngine.wrap(emptyBuffer, buffer);
278 
279                 switch (result.getStatus()) {
280                     case BUFFER_OVERFLOW: {
281                         lazyBuffer.resize();
282                         break;
283                     }
284 
285                     case BUFFER_UNDERFLOW: {
286                         /* This basically says that there is not enough data to create an SSL packet. Javadoc suggests that
287                         BUFFER_UNDERFLOW can occur only after unwrap(), but to be 100% sure we handle all possible error
288                         states: */
289                         throw new IllegalStateException("SSL engine underflow while close operation \n" + getDebugState());
290                     }
291 
292                     // CLOSE or OK are expected outcomes
293                 }
294 
295             }
296 
297             if (lazyBuffer.isAllocated()) {
298                 ByteBuffer buffer = lazyBuffer.get();
299                 buffer.flip();
300                 writeQueue.write(buffer, new CompletionHandler<ByteBuffer>() {
301 
302                     @Override
303                     public void completed(ByteBuffer result) {
304                         downstreamFilter.close();
305                     }
306 
307                     @Override
308                     public void failed(Throwable throwable) {
309                         downstreamFilter.close();
310                     }
311                 });
312             } else {
313                 // make sure we close even if SSL had nothing to send
314                 downstreamFilter.close();
315             }
316         } catch (Exception e) {
317             handleSslError(e);
318         }
319     }

321     @Override
322     boolean processRead(ByteBuffer networkData) {
323         /* A flag indicating if we should keep reading from the network buffer.
324         If false, the buffer contains an uncompleted packet -> stop reading, SSL engine accepts only whole packets */
325         boolean readMore = true;
326 
327         while (networkData.hasRemaining() && readMore) {
328             switch (state) {
329                 // before SSL is started write just passes through
330                 case NOT_STARTED: {
331                     return true;
332                 }
333 
334                 case HANDSHAKING:
335                 case REHANDSHAKING: {
336                     readMore = doHandshakeStep(networkData);
337                     break;
338                 }
339 
340                 case DATA: {
341                     readMore = handleRead(networkData);
342                     break;
343                 }
344 
345                 case CLOSED: {
346                     // drop any data that arrive after the SSL has been closed
347                     networkData.clear();
348                     readMore = false;
349                 }
350             }
351         }
352 
353         return false;
354     }

356     private boolean handleRead(ByteBuffer networkData) {
357         try {
358             applicationInputBuffer.clear();
359             SSLEngineResult result = sslEngine.unwrap(networkData, applicationInputBuffer);
360 
361             switch (result.getStatus()) {
362                 case BUFFER_OVERFLOW: {
363                     /* This means that the content of the ssl packet (max 16kB) did not fit into
364                        applicationInputBuffer, but we make sure to set applicationInputBuffer > max 16kB
365                        when initializing this filter. This indicates a bug.*/
366                     throw new IllegalStateException("Contents of a SSL packet did not fit into buffer: "
367                             + applicationInputBuffer + "\n" + getDebugState());
368                 }
369 
370                 case BUFFER_UNDERFLOW: {
371                     // the ssl packet is not full, return and indicate that we won't get more from this buffer
372                     return false;
373                 }
374 
375                 case CLOSED:
376                 case OK: {
377                     if (result.bytesProduced() > 0) {
378                         applicationInputBuffer.flip();
379                         upstreamFilter.onRead(applicationInputBuffer);
380                         applicationInputBuffer.compact();
381                     }
382 
383                     if (sslEngine.isInboundDone()) {
384                         /* we have just received a close alert from our peer, so we are done. If there is something
385                         remaining in the input buffer, just drop it. */
386 
387                         // signal that there is nothing useful left in this buffer
388                         return false;
389                     }
390 
391                     // we started re-handshaking
392                     if (result.getHandshakeStatus() != SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING
393                             // make sure we don't confuse re-handshake with closing handshake
394                             && !sslEngine.isOutboundDone()) {
395                         state = State.REHANDSHAKING;
396                         return doHandshakeStep(networkData);
397                     }
398 
399                     break;
400                 }
401             }
402         } catch (SSLException e) {
403             handleSslError(e);
404         }
405 
406         return true;
407     }

409     private boolean doHandshakeStep(ByteBuffer networkData) {
410         /* Buffer used to store application data read during this handshake step.
411         Application data can be interleaved with handshake messages only during re-handshake.
412         We don't use applicationInputBuffer, because we might want to store more than one packet */
413         LazyBuffer inputBuffer = new LazyBuffer();
414         boolean handshakeFinished = false;
415 
416         synchronized (this) {
417             if (SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING.equals(sslEngine.getHandshakeStatus())) {
418                 // we stopped handshaking while waiting for the lock
419                 return true;
420             }
421 
422             try {
423                 /* we don't use networkOutputBuffer, because there might be a write operation still in progress ->
424                 we don't want to corrupt the buffer it is using */
425                 LazyBuffer outputBuffer = new LazyBuffer();
426                 boolean stepFinished = false;
427                 while (!stepFinished) {
428                     SSLEngineResult.HandshakeStatus hs = sslEngine.getHandshakeStatus();
429 
430                     switch (hs) {
431                         case NOT_HANDSHAKING: {
432                             /* This should never happen. If we are here and not handshaking, it means a bug
433                             in the state machine of this class, because we stopped handshaking and did not exit this while loop.
434                             The could be caused either by overlooking FINISHED state or incorrectly treating an error. */
435 
436                             throw new IllegalStateException("Trying to handshake, but SSL engine not in HANDSHAKING state."
437                                     + "SSL filter state: \n" + getDebugState());
438                         }
439 
440                         case FINISHED: {
441                             /* According to SSLEngine javadoc FINISHED status can be returned only in SSLEngineResult,
442                             but just to make sure we don't end up in an infinite loop when presented with an SSLEngine
443                             implementation that does not respect this:*/
444                             stepFinished = true;
445                             handshakeFinished = true;
446                             break;
447                         }
448                         // needs to write data to the network
449                         case NEED_WRAP: {
450                             ByteBuffer byteBuffer = outputBuffer.get();
451                             SSLEngineResult result = sslEngine.wrap(emptyBuffer, byteBuffer);
452 
453                             if (result.getHandshakeStatus() == SSLEngineResult.HandshakeStatus.FINISHED) {
454                                 stepFinished = true;
455                                 handshakeFinished = true;
456                             }
457 
458                             switch (result.getStatus()) {
459                                 case BUFFER_OVERFLOW: {
460                                     outputBuffer.resize();
461                                     break;
462                                 }
463 
464                                 case BUFFER_UNDERFLOW: {
465                                     /* This basically says that there is not enough data to create an SSL packet. Javadoc suggests
466                                     that BUFFER_UNDERFLOW can occur only after unwrap(), but to be 100% sure we handle all
467                                     possible error states: */
468                                     throw new IllegalStateException("SSL engine underflow with the following SSL filter "
469                                             + "state: \n" + getDebugState());
470                                 }
471 
472                                 case CLOSED: {
473                                     stepFinished = true;
474                                     state = State.CLOSED;
475                                     break;
476                                 }
477                             }
478 
479                             break;
480                         }
481 
482                         case NEED_UNWRAP: {
483 
484                             SSLEngineResult result = sslEngine.unwrap(networkData, applicationInputBuffer);
485 
486                             applicationInputBuffer.flip();
487                             if (applicationInputBuffer.hasRemaining()) {
488                                 // data can flow during re-handshake
489                                 inputBuffer.append(applicationInputBuffer);
490                             }
491                             applicationInputBuffer.compact();
492 
493                             if (result.getHandshakeStatus() == SSLEngineResult.HandshakeStatus.FINISHED) {
494                                 stepFinished = true;
495                                 handshakeFinished = true;
496                             }
497 
498                             switch (result.getStatus()) {
499                                 case BUFFER_OVERFLOW: {
500                                     /* This means that the content of the ssl packet (max 16kB) did not fit into
501                                     applicationInputBuffer, but we make sure to set applicationInputBuffer > max 16kB
502                                     when initializing this filter. This indicates a bug. */
503                                     throw new IllegalStateException("SSL packet does not fit into the network buffer: "
504                                             + getDebugState());
505                                 }
506 
507                                 case BUFFER_UNDERFLOW: {
508                                     // indicate that we won't get more from this buffer
509                                     stepFinished = true;
510                                     break;
511                                 }
512 
513                                 case CLOSED: {
514                                     stepFinished = true;
515                                     state = State.CLOSED;
516                                     break;
517                                 }
518                             }
519 
520                             break;
521                         }
522                         // needs to execute long running task (for instance validating certificates)
523                         case NEED_TASK: {
524                             Runnable delegatedTask;
525                             while ((delegatedTask = sslEngine.getDelegatedTask()) != null) {
526                                 delegatedTask.run();
527                             }
528                             break;
529                         }
530                     }
531                 }
532 
533                 // now write the stored wrap() results
534                 if (outputBuffer.isAllocated()) {
535                     ByteBuffer buffer = outputBuffer.get();
536                     buffer.flip();
537                     writeQueue.write(buffer, null);
538                 }
539 
540             } catch (Exception e) {
541                 handleSslError(e);
542             }
543         }
544 
545         /* Handle any read data.
546         We have to execute upstreamFilter.onRead after releasing the lock. See the synchronization note on top.
547         Only one read operation can be in progress at a time, so even though we have released the lock, no other
548         SSlEngine#unwrap can be performed until this method returns. So there is no chance of the read data
549         being mixed up */
550         if (inputBuffer.isAllocated()) {
551             ByteBuffer buffer = inputBuffer.get();
552             upstreamFilter.onRead(buffer);
553         }
554 
555         if (handshakeFinished) {
556             handleHandshakeFinished();
557             // indicate that there still might be usable data in the input buffer
558             return true;
559         }
560 
561         /* if we are here, it means that we are waiting for more data -> indicate that there is nothing usable in the
562         input buffer left */
563         return false;
564     }

566     private void handleHandshakeFinished() {
567         // Apply a custom host verifier if present. Do it for both handshaking and re-handshaking.
568         if (customHostnameVerifier != null && !customHostnameVerifier.verify(serverHost, sslEngine.getSession())) {
569             handleSslError(new SSLException("Server host name verification using " + customHostnameVerifier
570                     .getClass() + " has failed"));
571             return;
572         }
573 
574         if (state == State.HANDSHAKING) {
575             state = State.DATA;
576             upstreamFilter.onSslHandshakeCompleted();
577         } else if (state == State.REHANDSHAKING) {
578             state = State.DATA;
579             if (pendingApplicationWrite != null) {
580                 Runnable write = pendingApplicationWrite;
581                 // set pending write to null to cover the extremely improbable case that we start re-handshaking again
582                 pendingApplicationWrite = null;
583 
584                 write.run();
585             }
586         }
587     }
588 
589     private void handleSslError(Throwable t) {
590         onError(t);
591     }

593     @Override
594     void startSsl() {
595         try {
596             state = State.HANDSHAKING;
597             sslEngine.beginHandshake();
598             doHandshakeStep(emptyBuffer);
599         } catch (SSLException e) {
600             handleSslError(e);
601         }
602     }

618     private String getDebugState() {
619         return "SslFilter{"
620                 + "\napplicationInputBuffer=" + applicationInputBuffer
621                 + ",\nnetworkOutputBuffer=" + networkOutputBuffer
622                 + ",\nsslEngineStatus=" + sslEngine.getHandshakeStatus()
623                 + ",\nsslSession=" + sslEngine.getSession()

625                 + ",\npendingApplicationWrite=" + pendingApplicationWrite
626                 + ",\npendingWritesSize=" + writeQueue
627                 + '}';

630     private enum State {
631         NOT_STARTED,
632         HANDSHAKING,
633         REHANDSHAKING,
634         DATA,
635         CLOSED
636     }

638     private class LazyBuffer {
639 
640         private ByteBuffer buffer = null;
641 
642         ByteBuffer get() {
643             if (buffer == null) {
644                 buffer = ByteBuffer.allocate(sslEngine.getSession().getPacketBufferSize());
645             }
646 
647             return buffer;
648         }
649 
650         boolean isAllocated() {
651             return buffer != null;
652         }
653 
654         void resize() {
655             int increment = sslEngine.getSession().getPacketBufferSize();
656             int newSize = buffer.position() + increment;
657             ByteBuffer newBuffer = ByteBuffer.allocate(newSize);
658             buffer.flip();
659             newBuffer.flip();
660             buffer = Utils.appendBuffers(newBuffer, buffer, newBuffer.limit(), 50);
661             buffer.compact();
662         }
663 
664         void append(ByteBuffer b) {
665             if (buffer == null) {
666                 buffer = ByteBuffer.allocate(b.remaining());
667                 buffer.flip();
668             }
669             int newSize = buffer.limit() + b.remaining();
670             buffer = Utils.appendBuffers(buffer, b, newSize, 50);
671         }
672     }

674     // synchronized on the outer class, because there is a danger of deadlock if this has its own lock
675     private class WriteQueue {
676 
677         private final Queue<Runnable> pendingWrites = new LinkedList<>();
678 
679         void write(final ByteBuffer data, final CompletionHandler<ByteBuffer> completionHandler) {
680             synchronized (SslFilter.this) {
681                 Runnable r = () -> downstreamFilter.write(data, new CompletionHandler<ByteBuffer>() {
682 
683                     @Override
684                     public void completed(ByteBuffer result) {
685                         if (completionHandler != null) {
686                             completionHandler.completed(result);
687                         }
688 
689                         onWriteCompleted();
690                     }
691 
692                     @Override
693                     public void failed(Throwable throwable) {
694                         if (completionHandler != null) {
695                             completionHandler.failed(throwable);
696                         }
697 
698                         onWriteCompleted();
699                     }
700                 });
701 
702                 pendingWrites.offer(r);
703                 // if our task is the first one in the queue, there is no other write task in progress -> process it
704                 if (pendingWrites.peek() == r) {
705                     r.run();
706                 }
