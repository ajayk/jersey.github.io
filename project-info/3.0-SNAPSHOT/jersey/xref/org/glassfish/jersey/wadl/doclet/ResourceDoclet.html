115     /**
116      * Start the doclet.
117      *
118      * @param root the root JavaDoc document.
119      * @return true if no exception is thrown.
120      */
121     public static boolean start(final RootDoc root) {
122         final String output = getOptionArg(root.options(), OPTION_OUTPUT);
123 
124         final String classpath = getOptionArg(root.options(), OPTION_CLASSPATH);
125         // LOG.info( "Have classpath: " + classpath );
126         final String[] classpathElements = classpath.split(File.pathSeparator);
127 
128         final ClassLoader cl = Thread.currentThread().getContextClassLoader();
129         final ClassLoader ncl = new Loader(classpathElements,
130                 ResourceDoclet.class.getClassLoader());
131         Thread.currentThread().setContextClassLoader(ncl);
132 
133         final String docProcessorOption = getOptionArg(root.options(), OPTION_DOC_PROCESSORS);
134         final String[] docProcessors = docProcessorOption != null ? docProcessorOption.split(":") : null;
135         final DocProcessorWrapper docProcessor = new DocProcessorWrapper();
136         try {
137             if (docProcessors != null && docProcessors.length > 0) {
138                 final Class<?> clazz = Class.forName(docProcessors[0], true, Thread.currentThread().getContextClassLoader());
139                 final Class<? extends DocProcessor> dpClazz = clazz.asSubclass(DocProcessor.class);
140                 docProcessor.add(dpClazz.newInstance());
141             }
142         } catch (final Exception e) {
143             LOG.log(Level.SEVERE, "Could not load docProcessors " + docProcessorOption, e);
144         }
145 
146         try {
147             final ResourceDocType result = new ResourceDocType();
148 
149             final ClassDoc[] classes = root.classes();
150             for (final ClassDoc classDoc : classes) {
151                 LOG.fine("Writing class " + classDoc.qualifiedTypeName());
152                 final ClassDocType classDocType = new ClassDocType();
153                 classDocType.setClassName(classDoc.qualifiedTypeName());
154                 classDocType.setCommentText(classDoc.commentText());
155                 docProcessor.processClassDoc(classDoc, classDocType);
156 
157                 for (final MethodDoc methodDoc : classDoc.methods()) {
158 
159                     final MethodDocType methodDocType = new MethodDocType();
160                     methodDocType.setMethodName(methodDoc.name());
161                     methodDocType.setMethodSignature(methodDoc.signature());
162                     methodDocType.setCommentText(methodDoc.commentText());
163                     docProcessor.processMethodDoc(methodDoc, methodDocType);
164 
165                     addParamDocs(methodDoc, methodDocType, docProcessor);
166 
167                     addRequestRepresentationDoc(methodDoc, methodDocType);
168 
169                     addResponseDoc(methodDoc, methodDocType);
170 
171                     classDocType.getMethodDocs().add(methodDocType);
172                 }
173 
174                 result.getDocs().add(classDocType);
175             }
176 
177             try {
178                 final Class<?>[] clazzes = getJAXBContextClasses(result, docProcessor);
179                 final JAXBContext c = JAXBContext.newInstance(clazzes);
180                 final Marshaller m = c.createMarshaller();
181                 m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
182                 final OutputStream out = new BufferedOutputStream(new FileOutputStream(output));
183 
184                 final String[] cdataElements = getCDataElements(docProcessor);
185                 final XMLSerializer serializer = getXMLSerializer(out, cdataElements);
186 
187                 m.marshal(result, serializer);
188                 out.close();
189 
190                 LOG.info("Wrote " + output);
191 
192             } catch (final Exception e) {
193                 LOG.log(Level.SEVERE, "Could not serialize ResourceDoc.", e);
194                 return false;
195             }
196         } finally {
197             Thread.currentThread().setContextClassLoader(cl);
198         }
199 
200         return true;
201     }

203     private static String[] getCDataElements(final DocProcessor docProcessor) {
204         final String[] original = new String[] {"ns1^commentText", "ns2^commentText", "^commentText"};
205         if (docProcessor == null) {
206             return original;
207         } else {
208             final String[] cdataElements = docProcessor.getCDataElements();
209             if (cdataElements == null || cdataElements.length == 0) {
210                 return original;
211             } else {
212 
213                 final String[] result = copyOf(original, original.length + cdataElements.length);
214                 for (int i = 0; i < cdataElements.length; i++) {
215                     result[original.length + i] = cdataElements[i];
216                 }
217                 return result;
218             }
219         }
220     }

222     @SuppressWarnings("unchecked")
223     private static <T, U> T[] copyOf(final U[] original, final int newLength) {
224         final T[] copy = (original.getClass() == Object[].class)
225                 ? (T[]) new Object[newLength]
226                 : (T[]) Array.newInstance(original.getClass().getComponentType(), newLength);
227         System.arraycopy(original, 0, copy, 0,
228                 Math.min(original.length, newLength));
229         return copy;
230     }

232     private static Class<?>[] getJAXBContextClasses(
233             final ResourceDocType result, final DocProcessor docProcessor) {
234         final Class<?>[] clazzes;
235         if (docProcessor == null) {
236             clazzes = new Class<?>[1];
237         } else {
238             final Class<?>[] requiredJaxbContextClasses = docProcessor.getRequiredJaxbContextClasses();
239             if (requiredJaxbContextClasses != null) {
240                 clazzes = new Class<?>[1 + requiredJaxbContextClasses.length];
241                 for (int i = 0; i < requiredJaxbContextClasses.length; i++) {
242                     clazzes[i + 1] = requiredJaxbContextClasses[i];
243                 }
244             } else {
245                 clazzes = new Class<?>[1];
246             }
247         }
248         clazzes[0] = result.getClass();
249         return clazzes;
250     }

252     private static XMLSerializer getXMLSerializer(final OutputStream os, final String[] cdataElements)
253             throws InstantiationException,
254             IllegalAccessException, ClassNotFoundException {
255         // configure an OutputFormat to handle CDATA
256         final OutputFormat of = new OutputFormat();
257 
258         // specify which of your elements you want to be handled as CDATA.
259         // The use of the '^' between the namespaceURI and the localname
260         // seems to be an implementation detail of the xerces code.
261         // When processing xml that doesn't use namespaces, simply omit the
262         // namespace prefix as shown in the third CDataElement below.
263         of.setCDataElements(cdataElements);
264 
265         // set any other options you'd like
266         of.setPreserveSpace(true);
267         of.setIndenting(true);
268 
269         // create the serializer
270         final XMLSerializer serializer = new XMLSerializer(of);
271 
272         serializer.setOutputByteStream(os);
273 
274         return serializer;
275     }

277     private static void addResponseDoc(final MethodDoc methodDoc,
278                                        final MethodDocType methodDocType) {
279 
280         final ResponseDocType responseDoc = new ResponseDocType();
281 
282         final Tag returnTag = getSingleTagOrNull(methodDoc, "return");
283         if (returnTag != null) {
284             responseDoc.setReturnDoc(returnTag.text());
285         }
286 
287         final Tag[] responseParamTags = methodDoc.tags("response.param");
288         for (final Tag responseParamTag : responseParamTags) {
289             // LOG.info( "Have responseparam tag: " + print( responseParamTag ) );
290             final WadlParamType wadlParam = new WadlParamType();
291             for (final Tag inlineTag : responseParamTag.inlineTags()) {
292                 final String tagName = inlineTag.name();
293                 final String tagText = inlineTag.text();
294                 /* skip empty tags
295                  */
296                 if (isEmpty(tagText)) {
297                     if (LOG.isLoggable(Level.FINE)) {
298                         LOG.fine("Skipping empty inline tag of @response.param in method "
299                                 + methodDoc.qualifiedName() + ": " + tagName);
300                     }
301                     continue;
302                 }
303                 switch (tagName) {
304                     case "@name":
305                         wadlParam.setName(tagText);
306                         break;
307                     case "@style":
308                         wadlParam.setStyle(tagText);
309                         break;
310                     case "@type":
311                         wadlParam.setType(QName.valueOf(tagText));
312                         break;
313                     case "@doc":
314                         wadlParam.setDoc(tagText);
315                         break;
316                     default:
317                         LOG.warning("Unknown inline tag of @response.param in method "
318                                 + methodDoc.qualifiedName() + ": " + tagName
319                                 + " (value: " + tagText + ")");
320                         break;
321                 }
322             }
323             responseDoc.getWadlParams().add(wadlParam);
324         }
325 
326         final Map<String, List<Tag>> tagsByStatus = getResponseRepresentationTags(methodDoc);
327         for (final Entry<String, List<Tag>> entry : tagsByStatus.entrySet()) {
328             final RepresentationDocType representationDoc = new RepresentationDocType();
329             representationDoc.setStatus(Long.valueOf(entry.getKey()));
330             for (final Tag tag : entry.getValue()) {
331                 if (tag.name().endsWith(".qname")) {
332                     representationDoc.setElement(QName.valueOf(tag.text()));
333                 } else if (tag.name().endsWith(".mediaType")) {
334                     representationDoc.setMediaType(tag.text());
335                 } else if (tag.name().endsWith(".example")) {
336                     representationDoc.setExample(getSerializedExample(tag));
337                 } else if (tag.name().endsWith(".doc")) {
338                     representationDoc.setDoc(tag.text());
339                 } else {
340                     LOG.warning("Unknown response representation tag " + tag.name());
341                 }
342             }
343             responseDoc.getRepresentations().add(representationDoc);
344         }
345 
346         methodDocType.setResponseDoc(responseDoc);
347     }

353     private static void addRequestRepresentationDoc(final MethodDoc methodDoc,
354                                                     final MethodDocType methodDocType) {
355         final Tag requestElement = getSingleTagOrNull(methodDoc, "request.representation.qname");
356         final Tag requestExample = getSingleTagOrNull(methodDoc, "request.representation.example");
357         if (requestElement != null || requestExample != null) {
358             final RequestDocType requestDoc = new RequestDocType();
359             final RepresentationDocType representationDoc = new RepresentationDocType();
360 
361             /* requestElement exists
362              */
363             if (requestElement != null) {
364                 representationDoc.setElement(QName.valueOf(requestElement.text()));
365             }
366 
367             /* requestExample exists
368              */
369             if (requestExample != null) {
370                 final String example = getSerializedExample(requestExample);
371                 if (!isEmpty(example)) {
372                     representationDoc.setExample(example);
373                 } else {
374                     LOG.warning("Could not get serialized example for method " + methodDoc.qualifiedName());
375                 }
376             }
377 
378             requestDoc.setRepresentationDoc(representationDoc);
379             methodDocType.setRequestDoc(requestDoc);
380         }
381     }

383     private static Map<String, List<Tag>> getResponseRepresentationTags(final MethodDoc methodDoc) {
384         final Map<String, List<Tag>> tagsByStatus = new HashMap<>();
385         for (final Tag tag : methodDoc.tags()) {
386             final Matcher matcher = PATTERN_RESPONSE_REPRESENTATION.matcher(tag.name());
387             if (matcher.matches()) {
388                 final String status = matcher.group(1);
389                 List<Tag> tags = tagsByStatus.get(status);
390                 if (tags == null) {
391                     tags = new ArrayList<>();
392                     tagsByStatus.put(status, tags);
393                 }
394                 tags.add(tag);
395             }
396         }
397         return tagsByStatus;
398     }

400     /**
401      * Searches an <code>@link</code> tag within the inline tags of the specified tag
402      * and serializes the referenced instance.
403      *
404      * @param tag the tag containing the inline tags to be searched.
405      * @return the {@code String} representation of the {@link com.sun.javadoc.Tag} or null if the parameter is null.
406      */
407     private static String getSerializedExample(final Tag tag) {
408         if (tag != null) {
409             final Tag[] inlineTags = tag.inlineTags();
410             if (inlineTags != null && inlineTags.length > 0) {
411                 for (final Tag inlineTag : inlineTags) {
412                     if (LOG.isLoggable(Level.FINE)) {
413                         LOG.fine("Have inline tag: " + print(inlineTag));
414                     }
415                     if ("@link".equals(inlineTag.name())) {
416                         if (LOG.isLoggable(Level.FINE)) {
417                             LOG.fine("Have link: " + print(inlineTag));
418                         }
419                         final SeeTag linkTag = (SeeTag) inlineTag;
420                         return getSerializedLinkFromTag(linkTag);
421                     } else if (!isEmpty(inlineTag.text())) {
422                         return inlineTag.text();
423                     }
424                 }
425             } else {
426                 LOG.fine("Have example: " + print(tag));
427                 return tag.text();
428             }
429         }
430         return null;
431     }

433     private static Tag getSingleTagOrNull(final MethodDoc methodDoc, final String tagName) {
434         final Tag[] tags = methodDoc.tags(tagName);
435         if (tags != null && tags.length == 1) {
436             return tags[0];
437         }
438         return null;
439     }

441     private static void addParamDocs(final MethodDoc methodDoc,
442                                      final MethodDocType methodDocType,
443                                      final DocProcessor docProcessor) {
444         final Parameter[] parameters = methodDoc.parameters();
445         final ParamTag[] paramTags = methodDoc.paramTags();
446 
447         /* only use both javadoc and reflection information when the number
448          * of params are the same
449          */
450         if (parameters != null && paramTags != null
451                 && parameters.length == paramTags.length) {
452 
453             for (int i = 0; i < parameters.length; i++) {
454                 final Parameter parameter = parameters[i];
455 
456                 /* TODO: this only works if the params and tags are in the same
457                  * order. If the param tags are mixed up, the comments for parameters
458                  * will be wrong.
459                  */
460                 final ParamTag paramTag = paramTags[i];
461 
462                 final ParamDocType paramDocType = new ParamDocType();
463                 paramDocType.setParamName(paramTag.parameterName());
464                 paramDocType.setCommentText(paramTag.parameterComment());
465                 docProcessor.processParamTag(paramTag, parameter, paramDocType);
466 
467                 final AnnotationDesc[] annotations = parameter.annotations();
468                 if (annotations != null) {
469                     for (final AnnotationDesc annotationDesc : annotations) {
470                         final AnnotationDocType annotationDocType = new AnnotationDocType();
471                         final String typeName = annotationDesc.annotationType().qualifiedName();
472                         annotationDocType.setAnnotationTypeName(typeName);
473                         for (final ElementValuePair elementValuePair : annotationDesc.elementValues()) {
474                             final NamedValueType namedValueType = new NamedValueType();
475                             namedValueType.setName(elementValuePair.element().name());
476                             namedValueType.setValue(elementValuePair.value().value().toString());
477                             annotationDocType.getAttributeDocs().add(namedValueType);
478                         }
479                         paramDocType.getAnnotationDocs().add(annotationDocType);
480                     }
481                 }
482 
483                 methodDocType.getParamDocs().add(paramDocType);
484 
485             }
486 
487         }
488     }

490     private static String getSerializedLinkFromTag(final SeeTag linkTag) {
491         final MemberDoc referencedMember = linkTag.referencedMember();
492 
493         if (referencedMember == null) {
494             throw new NullPointerException("Referenced member of @link " + print(linkTag) + " cannot be resolved.");
495         }
496 
497         if (!referencedMember.isStatic()) {
498             LOG.warning("Referenced member of @link " + print(linkTag) + " is not static."
499                     + " Right now only references to static members are supported.");
500             return null;
501         }
502 
503         /* Get referenced example bean
504          */
505         final ClassDoc containingClass = referencedMember.containingClass();
506         final Object object;
507         try {
508             final Field declaredField = Class.forName(containingClass.qualifiedName(), false, Thread.currentThread()
509                     .getContextClassLoader()).getDeclaredField(referencedMember.name());
510             if (referencedMember.isFinal()) {
511                 declaredField.setAccessible(true);
512             }
513             object = declaredField.get(null);
514             LOG.log(Level.FINE, "Got object " + object);
515         } catch (final Exception e) {
516             LOG.info("Have classloader: " + ResourceDoclet.class.getClassLoader().getClass());
517             LOG.info("Have thread classloader " + Thread.currentThread().getContextClassLoader().getClass());
518             LOG.info("Have system classloader " + ClassLoader.getSystemClassLoader().getClass());
519             LOG.log(Level.SEVERE, "Could not get field " + referencedMember.qualifiedName(), e);
520             return null;
521         }
522 
523         /* marshal the bean to xml
524          */
525         try {
526             final JAXBContext jaxbContext = JAXBContext.newInstance(object.getClass());
527             final StringWriter stringWriter = new StringWriter();
528             final Marshaller marshaller = jaxbContext.createMarshaller();
529             marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
530             marshaller.marshal(object, stringWriter);
531             final String result = stringWriter.getBuffer().toString();
532             LOG.log(Level.FINE, "Got marshalled output:\n" + result);
533             return result;
534         } catch (final Exception e) {
535             LOG.log(Level.SEVERE, "Could serialize bean to xml: " + object, e);
536             return null;
537         }
538     }

540     private static String print(final Tag tag) {
541         return String.valueOf(tag.getClass()) + "["
542                 + "firstSentenceTags=" + toCSV(tag.firstSentenceTags())
543                 + ", inlineTags=" + toCSV(tag.inlineTags())
544                 + ", kind=" + tag.kind()
545                 + ", name=" + tag.name()
546                 + ", text=" + tag.text()
547                 + "]";
548     }

550     static String toCSV(final Tag[] items) {
551         if (items == null) {
552             return null;
553         }
554         return toCSV(Arrays.asList(items));
555     }
556 
557     static String toCSV(final Collection<Tag> items) {
558         return toCSV(items, ", ", null);
559     }
560 
561     static String toCSV(final Collection<Tag> items, final String separator, final String delimiter) {
562         if (items == null) {
563             return null;
564         }
565         if (items.isEmpty()) {
566             return "";
567         }
568         final StringBuilder sb = new StringBuilder();
569         for (final Iterator<Tag> iter = items.iterator(); iter.hasNext(); ) {
570             if (delimiter != null) {
571                 sb.append(delimiter);
572             }
573             final Tag item = iter.next();
574             sb.append(item.name());
575             if (delimiter != null) {
576                 sb.append(delimiter);
577             }
578             if (iter.hasNext()) {
579                 sb.append(separator);
580             }
581         }
582         return sb.toString();
583     }

586      * Return array length for given option: 1 + the number of arguments that
587      * the option takes.
588      *
589      * @param option option
590      * @return the number of args for the specified option
591      */
592     public static int optionLength(final String option) {
593         LOG.fine("Invoked with option " + option);
594 
595         if (OPTION_OUTPUT.equals(option)
596                 || OPTION_CLASSPATH.equals(option)
597                 || OPTION_DOC_PROCESSORS.equals(option)) {
598             return 2;
599         }
600 
601         return 0;
602     }

604     /**
605      * Validate options.
606      *
607      * @param options  options to be validated
608      * @param reporter {@link com.sun.javadoc.DocErrorReporter} for collecting eventual errors
609      * @return if the specified options are valid
610      */
611     public static boolean validOptions(final String[][] options, final DocErrorReporter reporter) {
612         return validOption(OPTION_OUTPUT, "<path-to-file>", options, reporter)
613                 && validOption(OPTION_CLASSPATH, "<path>", options, reporter);
614     }

616     private static boolean validOption(final String optionName,
617                                        final String reportOptionName,
618                                        final String[][] options,
619                                        final DocErrorReporter reporter) {
620         final String option = getOptionArg(options, optionName);
621 
622         final boolean foundOption = option != null && !option.trim().isEmpty();
623         if (!foundOption) {
624             reporter.printError(optionName + " " + reportOptionName + " must be specified.");
625         }
626         return foundOption;
627     }

629     private static String getOptionArg(final String[][] options, final String option) {
630 
631         for (final String[] opt : options) {
632             if (opt[0].equals(option)) {
633                 return opt[1];
634             }
635         }
636 
637         return null;
638     }
