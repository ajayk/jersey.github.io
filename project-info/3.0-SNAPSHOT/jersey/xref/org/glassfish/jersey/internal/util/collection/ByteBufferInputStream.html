42  import java.io.IOException;
43  import java.nio.ByteBuffer;
44  import java.util.concurrent.BlockingQueue;
45  import java.util.concurrent.atomic.AtomicBoolean;
46  import java.util.concurrent.atomic.AtomicReference;

48  import org.glassfish.jersey.internal.LocalizationMessages;

50  /**
51   * An {@link java.io.InputStream} backed by a queue of {@link java.nio.ByteBuffer byte buffers}
52   * to be read.
53   * <p>
54   * This input stream serves as a bridging inbound I/O component between a blocking upper I/O layer
55   * and an underlying non-blocking I/O layer. In addition to the blocking {@code InputStream.read} operations,
56   * this input stream provides the non-blocking {@code tryRead} counterparts.
57   * </p>
58   *
59   * @author Marek Potociar (marek.potociar at oracle.com)
60   */
61  public final class ByteBufferInputStream extends NonBlockingInputStream {
62  
63      /**
64       * Constant buffer indicating EOF.
65       */
66      private static final ByteBuffer EOF = ByteBuffer.wrap(new byte[] {});
67  
68      /**
69       * Read-side EOF flag. Does not have to be volatile, it is transient and only accessed from the reader thread.
70       */
71      private boolean eof = false;
72      /**
73       * Currently read byte buffer.
74       */
75      private ByteBuffer current;
76      /**
77       * Queue of byte buffers to be read.
78       */
79      private final BlockingQueue<ByteBuffer> buffers;
80      /**
81       * Content represents the buffer queue status.
82       *
83       * null = open, POISON_PILL = closed, Throwable = closed with error
84       */
85      private final AtomicReference<Object> queueStatus = new AtomicReference<Object>(null);
86      /**
87       * Closed flag.
88       */
89      private final AtomicBoolean closed = new AtomicBoolean(false);
90  
91      /**
92       * Create a new input stream that is backed by a a queue of {@link java.nio.ByteBuffer byte buffers}
93       * to be read.
94       */
95      public ByteBufferInputStream() {
96          this.buffers = DataStructures.createLinkedTransferQueue();
97          this.current = null;
98      }

100     private boolean fetchChunk(final boolean block) throws InterruptedException {
101         if (eof) {
102             return false;
103         }
104 
105         // Read until no buffers available (poll returned null)
106         // or until a non-empty buffer or EOF is reached.
107         do {
108             if (closed.get()) {
109                 current = EOF;
110                 break;
111             }
112 
113             current = (block) ? buffers.take() : buffers.poll();
114         } while (current != null && current != EOF && !current.hasRemaining());
115 
116         eof = current == EOF;
117         return !eof;
118     }
119 
120     private void checkNotClosed() throws IOException {
121         if (closed.get()) {
122             throw new IOException(LocalizationMessages.INPUT_STREAM_CLOSED());
123         }
124     }
125 
126     private void checkThrowable() throws IOException {
127         final Object o = queueStatus.get();
128         if (o != null && o != EOF) { // should be faster than instanceof
129             // if not null or EOF, then it must be Throwable
130             if (queueStatus.compareAndSet(o, EOF)) {
131                 // clear throwable flag and throw exception
132                 try {
133                     throw new IOException((Throwable) o);
134                 } finally {
135                     close();
136                 }
137             }
138         }
139     }

141     @Override
142     public int available() throws IOException {
143         if (eof || closed.get()) {
144             checkThrowable();
145             return 0;
146         }
147 
148         int available = 0;
149         if (current != null && current.hasRemaining()) {
150             available = current.remaining();
151         }
152         for (final ByteBuffer buffer : buffers) {
153             if (buffer == EOF) {
154                 break;
155             }
156             available += buffer.remaining();
157         }
158 
159         checkThrowable();
160         return closed.get() ? 0 : available;
161     }
162 
163     @Override
164     public int read() throws IOException {
165         return tryRead(true);
166     }
167 
168     @Override
169     public int read(byte[] b, int off, int len) throws IOException {
170         return tryRead(b, off, len, true);
171     }

173     public int tryRead() throws IOException {
174         return tryRead(false);
175     }
176 
177     @Override
178     public int tryRead(final byte[] b) throws IOException {
179         return tryRead(b, 0, b.length);
180     }
181 
182     @Override
183     public int tryRead(final byte[] b, final int off, final int len) throws IOException {
184         return tryRead(b, off, len, false);
185     }

187     @Override
188     public void close() throws IOException {
189         if (closed.compareAndSet(false, true)) {
190             closeQueue();
191             // we can now safely clear the queue - any blocking read waiting for a buffer
192             // has been resumed by the EOF buffer
193             buffers.clear();
194         }
195         checkThrowable();
196     }

198     /**
199      * Put the {@code ByteBuffer} to the internal queue to be available for reading from the stream.
200      *
201      * <p>
202      * If the sink is open, the method {@link BlockingQueue#put(Object) puts the buffer} into an internal
203      * byte buffer read queue , waiting if necessary for space to become available. Then the method returns
204      * {@code true} to indicate the buffer has been successfully queued. In case the internal read queue has been
205      * {@link #closeQueue() closed} already, the method simply returns {@code false} without registering
206      * the buffer in the closed queue.
207      * </p>
208      *
209      * @param src the source buffer to be registered in the byte buffer read queue.
210      * @return {@code true} if the byte buffer has been successfully put in the read queue,
211      *         {@code false} if the read queue has been closed.
212      * @throws InterruptedException in case the put operation has been interrupted.
213      */
214     public boolean put(final ByteBuffer src) throws InterruptedException {
215         if (queueStatus.get() == null) {
216             buffers.put(src);
217             return true;
218         }
219         return false;
220     }
221 
222     /**
223      * Closes the byte buffer sink of this input stream to indicate that writing to the stream
224      * has been finished.
225      * <p>
226      * If the sink has already been closed then this method returns immediately. Otherwise the
227      * sink is marked as closed and no more data can be written to it.
228      * </p>
229      */
230     public void closeQueue() {
231         if (queueStatus.compareAndSet(null, EOF)) {
232             try {
233                 buffers.put(EOF);
234             } catch (final InterruptedException e) {
235                 Thread.currentThread().interrupt();
236             }
237         }
238     }
239 
240     /**
241      * Closes the byte buffer sink of this input stream to indicate that writing to the stream
242      * has been finished due to a throwable.
243      *
244      * The throwable set by this method will be propagated to the reader thread when a new attempt
245      * to read bytes is made.
246      * <p>
247      * If the sink has already been closed then this method only sets the throwable in the stream and
248      * then returns immediately. Otherwise the sink is also marked as closed and no more data can be
249      * written to it.
250      * </p>
251      *
252      * @param throwable throwable that is set in the stream. It will be thrown by the stream in case
253      *                  an attempt to read more data or check available bytes is made.
254      */
255     public void closeQueue(final Throwable throwable) {
256         if (queueStatus.compareAndSet(null, throwable)) {
257             try {
258                 buffers.put(EOF);
259             } catch (final InterruptedException e) {
260                 Thread.currentThread().interrupt();
261             }
262         }
263     }

265     private int tryRead(final byte[] b, final int off, final int len, boolean block) throws IOException {
266         checkThrowable();
267         checkNotClosed();
268 
269         if (b == null) {
270             throw new NullPointerException();
271         } else if (off < 0 || len < 0 || len > b.length - off) {
272             throw new IndexOutOfBoundsException();
273         } else if (len == 0) {
274             return 0;
275         }
276 
277         if (eof) {
278             return -1;
279         }
280 
281         int i = 0;
282         while (i < len) {
283             if (current != null && current.hasRemaining()) {
284                 final int available = current.remaining();
285                 if (available < len - i) {
286                     current.get(b, off + i, available);
287                     i += available;
288                 } else {
289                     current.get(b, off + i, len - i);
290                     return len;
291                 }
292             } else {
293                 try {
294                     if (!fetchChunk(block) || current == null) {
295                         break;  // eof or no data
296                     }
297                 } catch (final InterruptedException e) {
298                     Thread.currentThread().interrupt();
299                     if (block) {
300                         throw new IOException(e);
301                     }
302                 }
303             }
304         }
305 
306         return i == 0 && eof ? -1 : i;
307     }
308 
309     private int tryRead(boolean block) throws IOException {
310         checkThrowable();
311         checkNotClosed();
312 
313         if (eof) {
314             return -1;
315         }
316 
317         if (current != null && current.hasRemaining()) {
318             return current.get() & 0xFF;
319         }
320 
321         try {
322             // try to fetch, but don't block && check if something has been fetched
323             if (fetchChunk(block) && current != null) {
324                 return current.get() & 0xFF;
325             } else if (block) {
326                 return -1;
327             }
328         } catch (final InterruptedException e) {
329             Thread.currentThread().interrupt();
330             if (block) {
331                 throw new IOException(e);
332             }
333         }
334 
335         return (eof) ? -1 : NOTHING;
336     }
