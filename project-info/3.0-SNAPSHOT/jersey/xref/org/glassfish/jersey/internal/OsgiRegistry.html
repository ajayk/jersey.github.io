81  /**
82   * Utility class to deal with OSGi runtime specific behavior.
83   * This is mainly to handle META-INF/services lookup
84   * and generic/application class lookup issue in OSGi.
85   *
86   * When OSGi runtime is detected by the {@link ServiceFinder} class,
87   * an instance of OsgiRegistry is created and associated with given
88   * OSGi BundleContext. META-INF/services entries are then being accessed
89   * via the OSGi Bundle API as direct ClassLoader#getResource() method invocation
90   * does not work in this case within OSGi.
91   *
92   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
93   * @author Adam Lindenthal (adam.lindenthal at oracle.com)
94   */
95  public final class OsgiRegistry implements SynchronousBundleListener {

110     /**
111      * Returns an {@code OsgiRegistry} instance. Call this method only if sure that the application is running in OSGi
112      * environment, otherwise a call to this method can lead to an {@link ClassNotFoundException}.
113      *
114      * @return an {@code OsgiRegistry} instance.
115      */
116     public static synchronized OsgiRegistry getInstance() {
117         if (instance == null) {
118             final ClassLoader classLoader = AccessController
119                     .doPrivileged(ReflectionHelper.getClassLoaderPA(ReflectionHelper.class));
120             if (classLoader instanceof BundleReference) {
121                 final BundleContext context = FrameworkUtil.getBundle(OsgiRegistry.class).getBundleContext();
122                 if (context != null) { // context could be still null if the current bundle has not been started
123                     instance = new OsgiRegistry(context);
124                 }
125             }
126         }
127         return instance;
128     }

130     private final class OsgiServiceFinder extends ServiceFinder.ServiceIteratorProvider {
131 
132         final ServiceFinder.ServiceIteratorProvider defaultIterator = new ServiceFinder.DefaultServiceIteratorProvider();
133 
134         @Override
135         public <T> Iterator<T> createIterator(
136                 final Class<T> serviceClass,
137                 final String serviceName,
138                 final ClassLoader loader,
139                 final boolean ignoreOnClassNotFound) {
140 
141             final List<Class<?>> providerClasses = locateAllProviders(serviceName);
142             if (!providerClasses.isEmpty()) {
143                 return new Iterator<T>() {
144 
145                     Iterator<Class<?>> it = providerClasses.iterator();
146 
147                     @Override
148                     public boolean hasNext() {
149                         return it.hasNext();
150                     }
151 
152                     @SuppressWarnings("unchecked")
153                     @Override
154                     public T next() {
155                         final Class<T> nextClass = (Class<T>) it.next();
156                         try {
157                             return nextClass.newInstance();
158                         } catch (final Exception ex) {
159                             final ServiceConfigurationError sce = new ServiceConfigurationError(serviceName + ": "
160                                     + LocalizationMessages.PROVIDER_COULD_NOT_BE_CREATED(
161                                     nextClass.getName(), serviceClass, ex.getLocalizedMessage()));
162                             sce.initCause(ex);
163                             throw sce;
164                         }
165                     }
166 
167                     @Override
168                     public void remove() {
169                         throw new UnsupportedOperationException();
170                     }
171                 };
172             }
173             return defaultIterator.createIterator(serviceClass, serviceName, loader, ignoreOnClassNotFound);
174         }
175 
176         @Override
177         public <T> Iterator<Class<T>> createClassIterator(
178                 final Class<T> service, final String serviceName, final ClassLoader loader, final boolean ignoreOnClassNotFound) {
179             final List<Class<?>> providerClasses = locateAllProviders(serviceName);
180             if (!providerClasses.isEmpty()) {
181                 return new Iterator<Class<T>>() {
182 
183                     Iterator<Class<?>> it = providerClasses.iterator();
184 
185                     @Override
186                     public boolean hasNext() {
187                         return it.hasNext();
188                     }
189 
190                     @SuppressWarnings("unchecked")
191                     @Override
192                     public Class<T> next() {
193                         return (Class<T>) it.next();
194                     }
195 
196                     @Override
197                     public void remove() {
198                         throw new UnsupportedOperationException();
199                     }
200                 };
201             }
202             return defaultIterator.createClassIterator(service, serviceName, loader, ignoreOnClassNotFound);
203         }

206     private static class BundleSpiProvidersLoader implements Callable<List<Class<?>>> {
207 
208         private final String spi;
209         private final URL spiRegistryUrl;
210         private final String spiRegistryUrlString;
211         private final Bundle bundle;
212 
213         BundleSpiProvidersLoader(final String spi, final URL spiRegistryUrl, final Bundle bundle) {
214             this.spi = spi;
215             this.spiRegistryUrl = spiRegistryUrl;
216             this.spiRegistryUrlString = spiRegistryUrl.toExternalForm();
217             this.bundle = bundle;
218         }
219 
220         @Override
221         public List<Class<?>> call() throws Exception {
222             BufferedReader reader = null;
223 
224             try {
225                 if (LOGGER.isLoggable(Level.FINEST)) {
226                     LOGGER.log(Level.FINEST, "Loading providers for SPI: {0}", spi);
227                 }
228                 reader = new BufferedReader(new InputStreamReader(spiRegistryUrl.openStream(), "UTF-8"));
229                 String providerClassName;
230 
231                 final List<Class<?>> providerClasses = new ArrayList<Class<?>>();
232                 while ((providerClassName = reader.readLine()) != null) {
233                     if (providerClassName.trim().length() == 0) {
234                         continue;
235                     }
236                     if (LOGGER.isLoggable(Level.FINEST)) {
237                         LOGGER.log(Level.FINEST, "SPI provider: {0}", providerClassName);
238                     }
239                     providerClasses.add(loadClass(bundle, providerClassName));
240                 }
241 
242                 return providerClasses;
243             } catch (final Exception e) {
244                 LOGGER.log(Level.WARNING, LocalizationMessages.EXCEPTION_CAUGHT_WHILE_LOADING_SPI_PROVIDERS(), e);
245                 throw e;
246             } catch (final Error e) {
247                 LOGGER.log(Level.WARNING, LocalizationMessages.ERROR_CAUGHT_WHILE_LOADING_SPI_PROVIDERS(), e);
248                 throw e;
249             } finally {
250                 if (reader != null) {
251                     try {
252                         reader.close();
253                     } catch (final IOException ioe) {
254                         LOGGER.log(Level.FINE, "Error closing SPI registry stream:" + spiRegistryUrl, ioe);
255                     }
256                 }
257             }
258         }
259 
260         @Override
261         public String toString() {
262             return spiRegistryUrlString;
263         }
264 
265         @Override
266         public int hashCode() {
267             return spiRegistryUrlString.hashCode();
268         }
269 
270         @Override
271         public boolean equals(final Object obj) {
272             if (obj instanceof BundleSpiProvidersLoader) {
273                 return spiRegistryUrlString.equals(((BundleSpiProvidersLoader) obj).spiRegistryUrlString);
274             } else {
275                 return false;
276             }
277         }

280     @Override
281     public void bundleChanged(final BundleEvent event) {
282 
283         if (event.getType() == BundleEvent.RESOLVED) {
284             register(event.getBundle());
285         } else if (event.getType() == BundleEvent.UNRESOLVED || event.getType() == BundleEvent.UNINSTALLED) {
286 
287             final Bundle unregisteredBundle = event.getBundle();
288 
289             lock.writeLock().lock();
290             try {
291                 factories.remove(unregisteredBundle.getBundleId());
292 
293                 if (unregisteredBundle.getSymbolicName().equals(CoreBundleSymbolicNAME)) {
294                     bundleContext.removeBundleListener(this);
295                     factories.clear();
296                 }
297             } finally {
298                 lock.writeLock().unlock();
299             }
300         }

303     /**
304      * Translates bundle entry path as returned from {@link org.osgi.framework.Bundle#findEntries(String, String, boolean)} to
305      * fully qualified class name that resides in given package path (directly or indirectly in its subpackages).
306      *
307      * @param packagePath     The package path where the class is located (even recursively)
308      * @param bundleEntryPath The bundle path to translate.
309      * @return Fully qualified class name.
310      */
311     public static String bundleEntryPathToClassName(String packagePath, String bundleEntryPath) {
312         // normalize packagePath
313         packagePath = normalizedPackagePath(packagePath);
314 
315         // remove WEB-INF/classes from bundle entry path
316         if (bundleEntryPath.contains(WEB_INF_CLASSES)) {
317             bundleEntryPath = bundleEntryPath.substring(bundleEntryPath.indexOf(WEB_INF_CLASSES) + WEB_INF_CLASSES.length());
318         }
319 
320         final int packageIndex = bundleEntryPath.indexOf(packagePath);
321 
322         String normalizedClassNamePath = packageIndex > -1
323                 // the package path was found in the bundle path
324                 ? bundleEntryPath.substring(packageIndex)
325                 // the package path is not included in the bundle entry path
326                 // fall back to the original implementation of the translation which does not consider recursion
327                 : packagePath + bundleEntryPath.substring(bundleEntryPath.lastIndexOf('/') + 1);
328 
329         return (normalizedClassNamePath.startsWith("/") ? normalizedClassNamePath.substring(1) : normalizedClassNamePath)
330                 .replace('/', '.').replace(".class", "");
331     }

333     /**
334      * Returns whether the given entry path is located directly in the provided package path. That is,
335      * if the entry is located in a sub-package, then {@code false} is returned.
336      *
337      * @param packagePath Package path which the entry is compared to
338      * @param entryPath Entry path
339      * @return Whether the given entry path is located directly in the provided package path.
340      */
341     public static boolean isPackageLevelEntry(String packagePath, final String entryPath) {
342         // normalize packagePath
343         packagePath = normalizedPackagePath(packagePath);
344 
345         // if the package path is contained in the jar entry name, subtract it
346         String entryWithoutPackagePath = entryPath.contains(packagePath)
347                 ? entryPath.substring(entryPath.indexOf(packagePath) + packagePath.length())
348                 : entryPath;
349 
350         return !(entryWithoutPackagePath.startsWith("/") ? entryWithoutPackagePath.substring(1)
351                          : entryWithoutPackagePath)
352                 .contains("/");
353     }
354 
355     /**
356      * Normalized package returns path that does not start with '/' character and ends with '/' character.
357      * If the argument is '/' then returned value is empty string "".
358      *
359      * @param packagePath package path to normalize.
360      * @return Normalized package path.
361      */
362     public static String normalizedPackagePath(String packagePath) {
363         packagePath = packagePath.startsWith("/") ? packagePath.substring(1) : packagePath;
364         packagePath = packagePath.endsWith("/") ? packagePath : packagePath + "/";
365         packagePath = "/".equals(packagePath) ? "" : packagePath;
366         return packagePath;
367     }

369     /**
370      * Get URLs of resources from a given package.
371      *
372      * @param packagePath package.
373      * @param classLoader resource class loader.
374      * @param recursive   whether the given package path should be scanned recursively by OSGi
375      * @return URLs of the located resources.
376      */
377     @SuppressWarnings("unchecked")
378     public Enumeration<URL> getPackageResources(final String packagePath,
379                                                 final ClassLoader classLoader,
380                                                 final boolean recursive) {
381         final List<URL> result = new LinkedList<URL>();
382 
383         for (final Bundle bundle : bundleContext.getBundles()) {
384             // Look for resources at the given <packagePath> and at WEB-INF/classes/<packagePath> in case a WAR is being examined.
385             for (final String bundlePackagePath : new String[] {packagePath, WEB_INF_CLASSES + packagePath}) {
386                 final Enumeration<URL> enumeration = findEntries(bundle, bundlePackagePath, "*.class", recursive);
387 
388                 if (enumeration != null) {
389                     while (enumeration.hasMoreElements()) {
390                         final URL url = enumeration.nextElement();
391                         final String path = url.getPath();
392 
393                         classToBundleMapping.put(bundleEntryPathToClassName(packagePath, path), bundle);
394                         result.add(url);
395                     }
396                 }
397             }
398 
399             // Now interested only in .jar provided by current bundle.
400             final Enumeration<URL> jars = findEntries(bundle, "/", "*.jar", true);
401             if (jars != null) {
402                 while (jars.hasMoreElements()) {
403                     final URL jar = jars.nextElement();
404                     final InputStream inputStream = classLoader.getResourceAsStream(jar.getPath());
405                     if (inputStream == null) {
406                         LOGGER.config(LocalizationMessages.OSGI_REGISTRY_ERROR_OPENING_RESOURCE_STREAM(jar));
407                         continue;
408                     }
409                     final JarInputStream jarInputStream;
410                     try {
411                         jarInputStream = new JarInputStream(inputStream);
412                     } catch (final IOException ex) {
413                         LOGGER.log(Level.CONFIG, LocalizationMessages.OSGI_REGISTRY_ERROR_PROCESSING_RESOURCE_STREAM(jar), ex);
414                         try {
415                             inputStream.close();
416                         } catch (final IOException e) {
417                             // ignored
418                         }
419                         continue;
420                     }
421 
422                     try {
423                         JarEntry jarEntry;
424                         while ((jarEntry = jarInputStream.getNextJarEntry()) != null) {
425                             final String jarEntryName = jarEntry.getName();
426                             final String jarEntryNameLeadingSlash = jarEntryName.startsWith("/")
427                                     ? jarEntryName : "/" + jarEntryName;
428 
429                             if (jarEntryName.endsWith(".class")
430                                     // Added leading and trailing slashes '/' to package path (e.g. '/com/') helps us to not
431                                     // accidentally match sub-strings of the package path (e.g., if package path 'com' was used
432                                     // for scanning, package 'whatever.foo.telecom' would be matched because of word 'tele[com]').
433                                     // Note that we cannot avoid all corner cases with accidental matches since jar
434                                     // entry name might be almost anything (e.g., if package path 'telecom' was used, package
435                                     // 'whatever.foo.telecom' will be matched and there is no way to avoid it unless user
436                                     // explicitly instructs us to do so somehow (not implemented)
437                                     && jarEntryNameLeadingSlash.contains("/" + normalizedPackagePath(packagePath))) {
438                                 if (!recursive && !isPackageLevelEntry(packagePath, jarEntryName)) {
439                                     continue;
440                                 }
441                                 classToBundleMapping.put(jarEntryName.replace(".class", "").replace('/', '.'), bundle);
442                                 result.add(bundle.getResource(jarEntryName));
443                             }
444                         }
445                     } catch (final Exception ex) {
446                         LOGGER.log(Level.CONFIG, LocalizationMessages.OSGI_REGISTRY_ERROR_PROCESSING_RESOURCE_STREAM(jar), ex);
447                     } finally {
448                         try {
449                             jarInputStream.close();
450                         } catch (final IOException e) {
451                             // ignored
452                         }
453                     }
454                 }
455             }
456         }
457 
458         return Collections.enumeration(result);
459     }

463      * <p>
464      * The context class loader will be utilized if accessible and non-null.
465      * Otherwise the defining class loader of this class will

481     /**
482      * Tries to load resource bundle via OSGi means. No caching involved here,
483      * as localization properties are being cached in Localizer class already.
484      *
485      * @param bundleName name of the resource bundle to load
486      * @return resource bundle instance if found, null otherwise
487      */
488     public ResourceBundle getResourceBundle(final String bundleName) {
489         final int lastDotIndex = bundleName.lastIndexOf('.');
490         final String path = bundleName.substring(0, lastDotIndex).replace('.', '/');
491         final String propertiesName = bundleName.substring(lastDotIndex + 1, bundleName.length()) + ".properties";
492         for (final Bundle bundle : bundleContext.getBundles()) {
493             final Enumeration<URL> entries = findEntries(bundle, path, propertiesName, false);
494             if (entries != null && entries.hasMoreElements()) {
495                 final URL entryUrl = entries.nextElement();
496                 try {
497                     return new PropertyResourceBundle(entryUrl.openStream());
498                 } catch (final IOException ex) {
499                     if (LOGGER.isLoggable(Level.FINE)) {
500                         // does not make sense to localize this
501                         LOGGER.fine("Exception caught when tried to load resource bundle in OSGi");
502                     }
503                     return null;
504                 }
505             }
506         }
507         return null;
508     }

544     private void register(final Bundle bundle) {
545         if (LOGGER.isLoggable(Level.FINEST)) {
546             LOGGER.log(Level.FINEST, "checking bundle {0}", bundle.getBundleId());
547         }
548 
549         Map<String, Callable<List<Class<?>>>> map;
550         lock.writeLock().lock();
551         try {
552             map = factories.get(bundle.getBundleId());
553             if (map == null) {
554                 map = new ConcurrentHashMap<String, Callable<List<Class<?>>>>();
555                 factories.put(bundle.getBundleId(), map);
556             }
557         } finally {
558             lock.writeLock().unlock();
559         }
560 
561         final Enumeration<URL> e = findEntries(bundle, "META-INF/services/", "*", false);
562         if (e != null) {
563             while (e.hasMoreElements()) {
564                 final URL u = e.nextElement();
565                 final String url = u.toString();
566                 if (url.endsWith("/")) {
567                     continue;
568                 }
569                 final String factoryId = url.substring(url.lastIndexOf("/") + 1);
570                 map.put(factoryId, new BundleSpiProvidersLoader(factoryId, u, bundle));
571             }
572         }
573     }

575     private List<Class<?>> locateAllProviders(final String serviceName) {
576         lock.readLock().lock();
577         try {
578             final List<Class<?>> result = new LinkedList<Class<?>>();
579             for (final Map<String, Callable<List<Class<?>>>> value : factories.values()) {
580                 if (value.containsKey(serviceName)) {
581                     try {
582                         result.addAll(value.get(serviceName).call());
583                     } catch (final Exception ex) {
584                         // ignore
585                     }
586                 }
587             }
588             return result;
589         } finally {
590             lock.readLock().unlock();
591         }
592     }

594     private static Class<?> loadClass(final Bundle bundle, final String className) throws ClassNotFoundException {
595         try {
596             return AccessController.doPrivileged(new PrivilegedExceptionAction<Class<?>>() {
597                 @Override
598                 public Class<?> run() throws ClassNotFoundException {
599                     return bundle.loadClass(className);
600                 }
601             });
602         } catch (final PrivilegedActionException ex) {
603             final Exception originalException = ex.getException();
604             if (originalException instanceof ClassNotFoundException) {
605                 throw (ClassNotFoundException) originalException;
606             } else if (originalException instanceof RuntimeException) {
607                 throw (RuntimeException) originalException;
608             } else {
609                 throw new ProcessingException(originalException);
610             }
611         }
612     }
