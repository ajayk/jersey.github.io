41  package org.glassfish.jersey.client.oauth2;
42  
43  import javax.ws.rs.core.Feature;
44  import javax.ws.rs.core.FeatureContext;
45  
46  /**
47   * OAuth2 client filter feature registers the support for performing authenticated requests to the
48   * Service Provider. The feature does not perform Authorization Flow (see {@link OAuth2CodeGrantFlow}
49   * for details how to use Authorization Flow and retrieve Access Token). The feature uses access to initialize
50   * the internal {@link javax.ws.rs.container.ContainerRequestFilter filter}
51   * which will add {@code Authorization} http header containing OAuth 2 authorization information including (based
52   * on {@code bearer} tokens).
53   * <p>
54   * The internal filter can be controlled by properties put into
55   * the {@link javax.ws.rs.client.ClientRequestContext client request}
56   * using {@link javax.ws.rs.client.ClientRequestContext#setProperty(String, Object)} method. The property key
57   * is defined in this class as a static variables
58   * ({@link OAuth2ClientSupport#OAUTH2_PROPERTY_ACCESS_TOKEN} (see its javadoc for usage).
59   * Using the property a specific
60   * access token can be defined for each request.
61   * </p>
62   * Example of using specific access token for one request:
63   * <pre>
64   * final Response response = client.target("foo").request()
65   *           .property(OAUTH2_PROPERTY_ACCESS_TOKEN, "6ab45ab465e46f54d771a").get();
66   * </pre>
67   *
68   * @author Miroslav Fuksa
69   * @since 2.3
70   */
71  class OAuth2ClientFeature implements Feature {
72  
73      private final OAuth2ClientFilter filter;
74  
75      /**
76       * Create a new feature initialized for the access token.
77       *
78       * @param accessToken Access token.
79       */
80      public OAuth2ClientFeature(String accessToken) {
81          this.filter = new OAuth2ClientFilter(accessToken);
82      }
83  
84      /**
85       * Create a new filter feature with no default access token. The token will have to be
86       * specified by {@link OAuth2ClientSupport#OAUTH2_PROPERTY_ACCESS_TOKEN}
87       * for each request otherwise no {@code Authorization}
88       * http header will be added.
89       */
90      public OAuth2ClientFeature() {
91          this.filter = new OAuth2ClientFilter();
92      }
93  
94      /**
95       * Create a new feature with the given {@code filter}.
96       *
97       * @param filter Filter instance.
98       */
99      OAuth2ClientFeature(OAuth2ClientFilter filter) {
100         this.filter = filter;
101     }
102 
103     @Override
104     public boolean configure(FeatureContext context) {
105         context.register(filter);
106         return true;
107     }
108 }
