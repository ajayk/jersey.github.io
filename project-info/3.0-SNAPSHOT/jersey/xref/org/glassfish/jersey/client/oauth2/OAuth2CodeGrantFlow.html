48  /**
49   * The interface that defines OAuth 2 Authorization Code Grant Flow.
50   * <p>
51   * The implementation of this interface is capable of performing of the user
52   * authorization defined in the OAuth2 specification as "Authorization Code Grant Flow" (OAuth 2 spec defines more
53   * Authorization Flows). The result of the authorization
54   * is the {@link TokenResult}. The implementation starts the authorization process by construction of a redirect URI
55   * to which the user should
56   * be redirected (the URI points to authorization consent page hosted by Service Provider). The user
57   * grants an access using this page. Service Provider redirects the user back to the
58   * our server and the authorization process is finished using the same instance of the interface implementation.
59   * </p>
60   * <p>
61   * To perform the authorization follow these steps:
62   * <list>
63   * <li>Get the instance of this interface using {@link OAuth2ClientSupport}.</li>
64   * <li>Call {@link #start()} method. The method returns redirection uri as a String.</li>
65   * <li>Redirect the user to the redirect URI returned from the {@code start} method. If your application deployment
66   * does not allow redirection (for example the app is a console application), then provide the redirection URI
67   * to the user in other ways.</li>
68   * <li>User should authorize your application on the redirect URI.</li>
69   * <li>After authorization the Authorization Server redirects the user back to the URI specified
70   * by {@link OAuth2CodeGrantFlow.Builder#redirectUri(String)} and provide the {@code code} and {@code state} as
71   * a request query parameter. Extract these parameter from the request. If your deployment does not support
72   * redirection (your app is not a web server) then Authorization Server will provide the user with
73   * {@code code} in other ways (for example display on the html page). You need to get
74   * this code from the user. The {@code state} parameter is added to the redirect URI in the {@code start} method and
75   * and the same parameter should be returned from the authorization response as a protection against CSRF attacks.</li>
76   * <li>Use the {@code code} and {@code state} to finish the authorization process by calling the method
77   * {@link #finish(String, String)} supplying the {@code code} and the {@code state} parameter. The method will internally request
78   * the access token from the Authorization Server and return it.</li>
79   * <li>You can use access token from {@code TokenResult} together with {@link ClientIdentifier} to
80   * perform the authenticated requests to the Service Provider. You can also call
81   * methods {@link #getAuthorizedClient()} to get {@link Client client} already configured with support
82   * for authentication from consumer credentials and access token received during authorization process.
83   * </li>
84   * </list>
85   * </p>
86   * <p>
87   * Important note: one instance of the interface can be used only for one authorization process. The methods
88   * must be called exactly in the order specified by the list above. Therefore the instance is also not
89   * thread safe and no concurrent access is expected.
90   * </p>
91   * <p>
92   * Instance must be stored between method calls (between {@code start} and {@code finish})
93   * for one user authorization process as the instance keeps
94   * internal state of the authorization process.
95   * </p>
96   *
97   * @author Miroslav Fuksa
98   * @since 2.3
99   */
100 public interface OAuth2CodeGrantFlow {
101 
102     /**
103      * Start the authorization process and return redirection URI on which the user should give a consent
104      * for our application to access resources.
105      *
106      * @return URI to which user should be redirected.
107      */
108     String start();
109 
110     /**
111      * Finish the authorization process and return the {@link TokenResult}. The method must be called on the
112      * same instance after the {@link #start()} method was called and user granted access to this application.
113      * <p>
114      * The method makes a request to the Authorization Server in order to exchange {@code code} for access token.
115      * </p>
116      * @param code Code received from the user authorization process.
117      * @param state State received from the user authorization response.
118      * @return Token result.
119      */
120 
121     TokenResult finish(String code, String state);
122 
123     /**
124      * Refresh the access token using a refresh token. This method can be called on newly created instance or on
125      * instance on which the authorization flow was already performed.
126      *
127      * @param refreshToken Refresh token.
128      * @return Token result.
129      */
130     TokenResult refreshAccessToken(String refreshToken);
131 
132     /**
133      * Return the client configured for performing authorized requests to the Service Provider. The
134      * authorization process must be successfully finished by instance by calling methods {@link #start()} and
135      * {@link #finish(String, String)}.
136      *
137      * @return Client configured to add correct {@code Authorization} header to requests.
138      */
139     public Client getAuthorizedClient();
140 
141     /**
142      * Return the {@link Feature oauth filter feature} that can be used to configure
143      * {@link Client client} instances to perform authenticated requests to the Service Provider.
144      * <p>
145      * The
146      * authorization process must be successfully finished by instance by calling methods {@link #start()} and
147      * {@link #finish(String, String)}.
148      * </p>
149      *
150      * @return oauth filter feature configured with received {@code AccessToken}.
151      */
152     public Feature getOAuth2Feature();

154     /**
155      * Phase of the Authorization Code Grant Flow.
156      */
157     enum Phase {
158         /**
159          * Authorization phase. The phase when user is redirected to the authorization server to authorize
160          * the application.
161          */
162         AUTHORIZATION {
163             @Override
164             public void property(String key, String value, Map<String, String> authorizationProps,
165                                  Map<String, String> accessTokenProps, Map<String, String> refreshTokenProps) {
166                 nonNullProperty(key, value, authorizationProps);
167             }
168         },
169         /**
170          * Requesting the access token phase.
171          */
172         ACCESS_TOKEN_REQUEST {
173             @Override
174             public void property(String key,
175                                  String value,
176                                  Map<String, String> authorizationProps,
177                                  Map<String, String> accessTokenProps,
178                                  Map<String, String> refreshTokenProps) {
179                 nonNullProperty(key, value, accessTokenProps);
180             }
181         },
182         /**
183          * Refreshing the access token phase.
184          */
185         REFRESH_ACCESS_TOKEN {
186             @Override
187             public void property(String key,
188                                  String value,
189                                  Map<String, String> authorizationProps,
190                                  Map<String, String> accessTokenProps,
191                                  Map<String, String> refreshTokenProps) {
192                 nonNullProperty(key, value, refreshTokenProps);
193             }
194         },
195         /**
196          * All phases.
197          */
198         ALL {
199             @Override
200             public void property(String key,
201                                  String value,
202                                  Map<String, String> authorizationProps,
203                                  Map<String, String> accessTokenProps,
204                                  Map<String, String> refreshTokenProps) {
205                 nonNullProperty(key, value, authorizationProps);
206                 nonNullProperty(key, value, accessTokenProps);
207                 nonNullProperty(key, value, refreshTokenProps);
208             }
209         };
210 
211         /**
212          * Set property defined by {@code key} and {@code value} to the appropriate
213          * property map based on this phase.
214          * @param key Property key.
215          * @param value Property value.
216          * @param authorizationProps Properties used in construction of redirect URI.
217          * @param accessTokenProps Properties (parameters) used in access token request.
218          * @param refreshTokenProps Properties (parameters) used in request for refreshing the access token.
219          */
220         public abstract void property(String key, String value,
221                                       Map<String, String> authorizationProps,
222                                       Map<String, String> accessTokenProps,
223                                       Map<String, String> refreshTokenProps);
224 
225         private static void nonNullProperty(String key, String value, Map<String, String> props) {
226             if (value == null) {
227                 props.remove(key);
228             } else {
229                 props.put(key, value);
230             }
231         }

235     /**
236      * The builder of {@link OAuth2CodeGrantFlow}.
237      *
238      * @param <T> Type of the builder implementation. This parameter is used for convenience to allow
239      *           better implementations of the builders implementing this interface (builder can return
240      *           their own specific type instead of type defined by this interface only).
241      */
242     public interface Builder<T extends Builder> {
243 
244         /**
245          * Set the access token URI on which the access token can be requested. The URI points to the
246          * authorization server and is defined by the Service Provider.
247          *
248          * @param accessTokenUri  Access token URI.
249          * @return Builder instance.
250          */
251         T accessTokenUri(String accessTokenUri);
252 
253         /**
254          * Set the URI to which the user should be redirected to authorize our application. The URI points to the
255          * authorization server and is defined by the Service Provider.
256          *
257          * @param authorizationUri Authorization URI.
258          * @return Builder instance.
259          */
260         T authorizationUri(String authorizationUri);
261 
262         /**
263          * Set the redirect URI to which the user (resource owner) should be redirected after he/she
264          * grants access to our application. In most cases, the URI is under control of this application
265          * and request done on this URI will be used to extract query parameter {@code code} and {@code state}
266          * that will be used in
267          * {@link OAuth2CodeGrantFlow#finish(String, String)} method.
268          * <p>
269          * If URI is not defined by this method, the default value {@code urn:ietf:wg:oauth:2.0:oob} will be used
270          * in the Authorization
271          * Flow which should cause that {@code code} will be passed to application in other way than request
272          * redirection (for example shown to the user using html page).
273          * </p>
274          *
275          * @param redirectUri URI that should receive authorization response from the Service Provider.
276          * @return Builder instance.
277          */
278         T redirectUri(String redirectUri);
279 
280         /**
281          * Set client identifier of the application that should be authorized.
282          *
283          * @param clientIdentifier Client identifier.
284          * @return Builder instance.
285          */
286         T clientIdentifier(ClientIdentifier clientIdentifier);
287 
288         /**
289          * Set a scope to which the application will get authorization grant. Values of this parameter
290          * are defined by the Service Provider and defines usually subset of resource and operations available
291          * in the Service Provider.
292          * <p>
293          * The parameter is optional but ServiceProvider might require it.
294          * </p>
295          *
296          * @param scope Scope string.
297          * @return Builder instance.
298          */
299         T scope(String scope);
300 
301         /**
302          * Set the client that should be used internally by the {@code OAuth1AuthorizationFlow} to make requests to
303          * Authorization Server. If this method is not called, it is up to the implementation to create or get
304          * any private client instance to perform these requests. This method could be used mainly for
305          * performance reasons to avoid creation of new client instances and have control about created client
306          * instances used in the application.
307          *
308          * @param client Client instance.
309          * @return Builder instance.
310          */
311         T client(Client client);
312 
313         /**
314          * Set the refresh token URI on which the access token can be refreshed using a refresh token.
315          * The URI points to the
316          * authorization server and is defined by the Service Provider. If the URI is not defined by this method
317          * it will be the same as URI defined in {@link #accessTokenUri(String)} (which is the default value
318          * defined by the OAuth2 spec).
319          * Some providers do not support
320          * refreshing access tokens at all.
321          *
322          * @param refreshTokenUri  Refresh token URI.
323          * @return Builder instance.
324          */
325         T refreshTokenUri(String refreshTokenUri);
326 
327         /**
328          * Set property (parameter) that will be added to requests or URIs as a query parameters during
329          * the Authorization Flow. Default parameters used during the Authorization Flow can be also
330          * overridden by this method.
331          *
332          * @param phase Phase of the flow in which the properties (parameters) should be used. For example by using
333          *              a {@link Phase#ACCESS_TOKEN_REQUEST}, the parameter will be added only to the http request
334          *              for access token.
335          * @param key Property key.
336          * @param value Property value.
337          * @return Builder instance.
338          */
339         T property(Phase phase, String key, String value);
340 
341         /**
342          * Build the {@code OAuth2CodeGrantFlow} instance.
343          *
344          * @return New instance of {@code OAuth2CodeGrantFlow}.
345          */
346         OAuth2CodeGrantFlow build();
347     }
