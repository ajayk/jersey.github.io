156     private OAuth1AuthorizationFlowImpl(final OAuth1Parameters params, final OAuth1Secrets secrets, final String requestTokenUri,
157                                         final String accessTokenUri, final String authorizationUri, final String callbackUri,
158                                         final Client client, final boolean enableLogging) {
159         this.parameters = params;
160         this.secrets = secrets;
161         this.requestTokenUri = requestTokenUri;
162         this.accessTokenUri = accessTokenUri;
163         this.authorizationUri = authorizationUri;
164 
165         if (client != null) {
166             this.client = client;
167         } else {
168             this.client = ClientBuilder.newBuilder().build();
169         }
170 
171         final Configuration config = this.client.getConfiguration();
172 
173         if (enableLogging && !config.isRegistered(LoggingFeature.class)) {
174             this.client.register(new LoggingFeature(LOGGER, LoggingFeature.Verbosity.PAYLOAD_ANY));
175         }
176         if (!config.isRegistered(OAuth1ClientFeature.class)) {
177             this.client.register(new OAuth1ClientFeature(params, secrets));
178         }
179 
180         if (callbackUri != null) {
181             this.parameters.callback(callbackUri);
182         }
183 
184         if (secrets.getConsumerSecret() == null || parameters.getConsumerKey() == null) {
185             throw new IllegalStateException(LocalizationMessages.ERROR_CONFIGURATION_MISSING_CONSUMER());
186         }
187     }

189     private Invocation.Builder addProperties(final Invocation.Builder invocationBuilder) {
190         return invocationBuilder
191                 .property(OAuth1ClientSupport.OAUTH_PROPERTY_OAUTH_PARAMETERS, parameters)
192                 .property(OAuth1ClientSupport.OAUTH_PROPERTY_OAUTH_SECRETS, secrets);
193     }

195     public String start() {
196         final Response response = addProperties(client.target(requestTokenUri).request())
197                 .post(null);
198         if (response.getStatus() != 200) {
199             throw new RuntimeException(LocalizationMessages.ERROR_REQUEST_REQUEST_TOKEN(response.getStatus()));
200         }
201         final MultivaluedMap<String, String> formParams = response.readEntity(Form.class).asMap();
202         parameters.token(formParams.getFirst(OAuth1Parameters.TOKEN));
203         secrets.tokenSecret(formParams.getFirst(OAuth1Parameters.TOKEN_SECRET));
204 
205         return UriBuilder.fromUri(authorizationUri).queryParam(OAuth1Parameters.TOKEN, parameters.getToken())
206                 .build().toString();
207     }

209     public AccessToken finish() {

213     public AccessToken finish(final String verifier) {
214         parameters.setVerifier(verifier);
215         final Response response = addProperties(client.target(accessTokenUri).request()).post(null);
216         // accessToken request failed
217         if (response.getStatus() >= 400) {
218             throw new RuntimeException(LocalizationMessages.ERROR_REQUEST_ACCESS_TOKEN(response.getStatus()));
219         }
220         final Form form = response.readEntity(Form.class);
221         final String accessToken = form.asMap().getFirst(OAuth1Parameters.TOKEN);
222         final String accessTokenSecret = form.asMap().getFirst(OAuth1Parameters.TOKEN_SECRET);
223 
224         if (accessToken == null) {
225             throw new NotAuthorizedException(LocalizationMessages.ERROR_REQUEST_ACCESS_TOKEN_NULL());
226         }
227 
228         parameters.token(accessToken);
229         secrets.tokenSecret(accessTokenSecret);
230         final AccessToken resultToken = new AccessToken(parameters.getToken(), secrets.getTokenSecret());
231         this.accessToken = resultToken;
232         return resultToken;
233     }

240     @Override
241     public Feature getOAuth1Feature() {
242         if (this.accessToken == null) {
243             throw new IllegalStateException(LocalizationMessages.ERROR_FLOW_NOT_FINISHED());
244         }
245         return oAuth1ClientFilterFeature.get();
