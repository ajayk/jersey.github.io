41  package org.glassfish.jersey.client.oauth1;

43  /**
44   * The main class to build the support for OAuth 1 into the Jersey client.
45   * <p>
46   * The support for OAuth is divided into two parts:
47   * <list>
48   *     <li><b>Authorization Flow:</b> process of acquiring the user
49   *     approval for accessing user's resources on the Service Provider. The authorization process is managed
50   *     by an implementation of {@link OAuth1AuthorizationFlow} interface. The result of the process is
51   *     an {@link AccessToken}.</li>
52   *
53   *     <li><b>Authenticated Requests:</b> requests done by a {@link javax.ws.rs.client.Client client} are
54   *     enhanced by an {@code Authorization} http header that contains OAuth1 authorization information
55   *     based on the {@code AccessToken} received from Authorization flow. This support is provided by
56   *     {@link javax.ws.rs.core.Feature oauth 1 filter feature} that is registered into client configuration.
57   *     </li>
58   * </list>
59   * This class contains static method that allows to build both OAuth1 features (authorization flow and client feature).
60   * </p>
61   *
62   * @author Miroslav Fuksa
63   * @since 2.3
64   */
65  public final class OAuth1ClientSupport {
66

68       * Key of the property that can be attached to the
69       * {@link javax.ws.rs.client.ClientRequestContext client request} using
70       * {@link javax.ws.rs.client.ClientRequestContext#setProperty(String, Object)} and that
71       * defines {@link org.glassfish.jersey.client.oauth1.ConsumerCredentials consumer credentials} that should be used when generating OAuth {@code Authorization}
72       * http header. The property will override the setting of the internal
73       * {@link javax.ws.rs.client.ClientRequestFilter filter} for the current request only.
74       * <p>
75       * The value of the property must be {@link org.glassfish.jersey.client.oauth1.ConsumerCredentials} instance.
76       * </p>
77       */
