40  package org.glassfish.jersey.jettison.internal.entity;
41  
42  import java.io.IOException;
43  import java.io.InputStream;
44  import java.io.OutputStream;
45  import java.io.OutputStreamWriter;
46  import java.lang.annotation.Annotation;
47  import java.lang.reflect.Type;
48  
49  import javax.ws.rs.Consumes;
50  import javax.ws.rs.Produces;
51  import javax.ws.rs.WebApplicationException;
52  import javax.ws.rs.core.MediaType;
53  import javax.ws.rs.core.MultivaluedMap;
54  
55  import org.glassfish.jersey.jettison.internal.LocalizationMessages;
56  
57  import org.codehaus.jettison.json.JSONException;
58  import org.codehaus.jettison.json.JSONObject;
59  
60  /**
61   * Low-level JSON media type message entity provider (reader & writer) for
62   * {@link JSONObject}.
63   *
64   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
65   */
66  public class JettisonObjectProvider extends JettisonLowLevelProvider<JSONObject> {
67  
68      @Produces("application/json")
69      @Consumes("application/json")
70      public static final class App extends JettisonObjectProvider {
71      }
72  
73      @Produces("*/*")
74      @Consumes("*/*")
75      public static final class General extends JettisonObjectProvider {
76  
77          @Override
78          protected boolean isSupported(MediaType m) {
79              return m.getSubtype().endsWith("+json");
80          }
81      }
82  
83      JettisonObjectProvider() {
84          super(JSONObject.class);
85      }
86  
87      @Override
88      public JSONObject readFrom(
89              Class<JSONObject> type,
90              Type genericType,
91              Annotation annotations[],
92              MediaType mediaType,
93              MultivaluedMap<String, String> httpHeaders,
94              InputStream entityStream) throws IOException {
95          try {
96              return new JSONObject(readFromAsString(entityStream, mediaType));
97          } catch (JSONException je) {
98              throw new WebApplicationException(
99                      new Exception(LocalizationMessages.ERROR_PARSING_JSON_OBJECT(), je),
100                     400);
101         }
102     }
103 
104     @Override
105     public void writeTo(
106             JSONObject t,
107             Class<?> type,
108             Type genericType,
109             Annotation annotations[],
110             MediaType mediaType,
111             MultivaluedMap<String, Object> httpHeaders,
112             OutputStream entityStream) throws IOException {
113         try {
114             OutputStreamWriter writer = new OutputStreamWriter(entityStream,
115                     getCharset(mediaType));
116             t.write(writer);
117             writer.flush();
118         } catch (JSONException je) {
119             throw new WebApplicationException(
120                     new Exception(LocalizationMessages.ERROR_WRITING_JSON_OBJECT(), je),
121                     500);
122         }
123     }
124 }
