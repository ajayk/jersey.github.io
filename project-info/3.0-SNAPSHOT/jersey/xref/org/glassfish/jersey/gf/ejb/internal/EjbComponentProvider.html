158     private ApplicationInfo getApplicationInfo(EjbContainerUtil ejbUtil) throws NamingException {
159         ApplicationRegistry appRegistry = ejbUtil.getServices().getService(ApplicationRegistry.class);
160         Applications applications = ejbUtil.getServices().getService(Applications.class);
161         String appNamePrefix = (String) initialContext.lookup("java:app/AppName");
162         Set<String> appNames = appRegistry.getAllApplicationNames();
163         Set<String> disabledApps = new TreeSet<>();
164         for (String appName : appNames) {
165             if (appName.startsWith(appNamePrefix)) {
166                 Application appDesc = applications.getApplication(appName);
167                 if (appDesc != null && !ejbUtil.getDeployment().isAppEnabled(appDesc)) {
168                     // skip disabled version of the app
169                     disabledApps.add(appName);
170                 } else {
171                     return ejbUtil.getDeployment().get(appName);
172                 }
173             }
174         }
175 
176         // grab the latest one, there is no way to make
177         // sure which one the user is actually enabling,
178         // so use the best case, i.e. upgrade
179         Iterator<String> it = disabledApps.iterator();
180         String lastDisabledApp = null;
181         while (it.hasNext()) {
182             lastDisabledApp = it.next();
183         }
184         if (lastDisabledApp != null) {
185             return ejbUtil.getDeployment().get(lastDisabledApp);
186         }
187 
188         throw new NamingException("Application Information Not Found");
189     }

191     private void registerEjbInterceptor() {
192         try {
193             final Object interceptor = new EjbComponentInterceptor(locator);
194             initialContext = getInitialContext();
195             final EjbContainerUtil ejbUtil = EjbContainerUtilImpl.getInstance();
196             final ApplicationInfo appInfo = getApplicationInfo(ejbUtil);
197             final List<String> tempLibNames = new LinkedList<>();
198             for (ModuleInfo moduleInfo : appInfo.getModuleInfos()) {
199                 final String jarName = moduleInfo.getName();
200                 if (jarName.endsWith(".jar") || jarName.endsWith(".war")) {
201                     final String moduleName = jarName.substring(0, jarName.length() - 4);
202                     tempLibNames.add(moduleName);
203                     final Object bundleDescriptor = moduleInfo.getMetaData(EjbBundleDescriptorImpl.class.getName());
204                     if (bundleDescriptor instanceof EjbBundleDescriptorImpl) {
205                         final Collection<EjbDescriptor> ejbs = ((EjbBundleDescriptorImpl) bundleDescriptor).getEjbs();
206 
207                         for (final EjbDescriptor ejb : ejbs) {
208                             final BaseContainer ejbContainer = EjbContainerUtilImpl.getInstance().getContainer(ejb.getUniqueId());
209                             try {
210                                 AccessController.doPrivileged(new PrivilegedExceptionAction() {
211                                     @Override
212                                     public Object run() throws Exception {
213                                         final Method registerInterceptorMethod =
214                                                 BaseContainer.class
215                                                         .getDeclaredMethod("registerSystemInterceptor", java.lang.Object.class);
216                                         registerInterceptorMethod.setAccessible(true);
217 
218                                         registerInterceptorMethod.invoke(ejbContainer, interceptor);
219                                         return null;
220                                     }
221                                 });
222                             } catch (PrivilegedActionException pae) {
223                                 final Throwable cause = pae.getCause();
224                                 LOGGER.log(Level.WARNING,
225                                         LocalizationMessages.EJB_INTERCEPTOR_BINDING_WARNING(ejb.getEjbClassName()), cause);
226                             }
227                         }
228                     }
229                 }
230             }
231             libNames.addAll(tempLibNames);
232             final Object interceptorBinder = initialContext.lookup("java:org.glassfish.ejb.container.interceptor_binding_spi");
233             // Some implementations of InitialContext return null instead of
234             // throwing NamingException if there is no Object associated with
235             // the name
236             if (interceptorBinder == null) {
237                 throw new IllegalStateException(LocalizationMessages.EJB_INTERCEPTOR_BIND_API_NOT_AVAILABLE());
238             }
239 
240             try {
241                 AccessController.doPrivileged(new PrivilegedExceptionAction() {
242                     @Override
243                     public Object run() throws Exception {
244                         Method interceptorBinderMethod = interceptorBinder.getClass()
245                                 .getMethod("registerInterceptor", java.lang.Object.class);
246 
247                         interceptorBinderMethod.invoke(interceptorBinder, interceptor);
248                         EjbComponentProvider.this.ejbInterceptorRegistered = true;
249                         LOGGER.log(Level.CONFIG, LocalizationMessages.EJB_INTERCEPTOR_BOUND());
250                         return null;
251                     }
252                 });
253             } catch (PrivilegedActionException pae) {
254                 throw new IllegalStateException(LocalizationMessages.EJB_INTERCEPTOR_CONFIG_ERROR(), pae.getCause());
255             }
256 
257         } catch (NamingException ex) {
258             throw new IllegalStateException(LocalizationMessages.EJB_INTERCEPTOR_BIND_API_NOT_AVAILABLE(), ex);
259         } catch (LinkageError ex) {
260             throw new IllegalStateException(LocalizationMessages.EJB_INTERCEPTOR_CONFIG_LINKAGE_ERROR(), ex);
261         }
262     }

264     // ComponentProvider
265     @SuppressWarnings("unchecked")
266     @Override
267     public boolean bind(Class<?> component, Set<Class<?>> providerContracts) {
268 
269         if (LOGGER.isLoggable(Level.FINE)) {
270             LOGGER.fine(LocalizationMessages.EJB_CLASS_BEING_CHECKED(component));
271         }
272 
273         if (locator == null) {
274             throw new IllegalStateException(LocalizationMessages.EJB_COMPONENT_PROVIDER_NOT_INITIALIZED_PROPERLY());
275         }
276 
277         if (!isEjbComponent(component)) {
278             return false;
279         }
280 
281         if (!ejbInterceptorRegistered) {
282             registerEjbInterceptor();
283         }
284 
285         DynamicConfiguration dc = Injections.getConfiguration(locator);
286 
287         final ServiceBindingBuilder bindingBuilder = Injections.newFactoryBinder(new EjbFactory(component, initialContext, this));
288 
289         bindingBuilder.to(component);
290         for (Class contract : providerContracts) {
291             bindingBuilder.to(contract);
292         }
293 
294         Injections.addBinding(bindingBuilder, dc);
295 
296         dc.commit();
297 
298         if (LOGGER.isLoggable(Level.CONFIG)) {
299             LOGGER.config(LocalizationMessages.EJB_CLASS_BOUND_WITH_CDI(component));
300         }
301 
302         return true;
303     }

316     private boolean isEjbComponent(Class<?> component) {
317         for (Annotation a : component.getAnnotations()) {
318             if (EjbComponentAnnotations.contains(a.annotationType().getName())) {
319                 return true;
320             }
321         }
322         return false;
323     }

325     @Override
326     public InvocationHandler create(Invocable method) {
327 
328         final Class<?> resourceClass = method.getHandler().getHandlerClass();
329 
330         if (resourceClass == null || !isEjbComponent(resourceClass)) {
331             return null;
332         }
333 
334         final Method handlingMethod = method.getDefinitionMethod();
335 
336         for (Class iFace : remoteAndLocalIfaces(resourceClass)) {
337             try {
338                 final Method iFaceMethod = iFace.getDeclaredMethod(handlingMethod.getName(), handlingMethod.getParameterTypes());
339                 if (iFaceMethod != null) {
340                     return new InvocationHandler() {
341                         @Override
342                         public Object invoke(Object target, Method ignored, Object[] args)
343                                 throws IllegalAccessException, IllegalArgumentException, InvocationTargetException {
344                             return iFaceMethod.invoke(target, args);
345                         }
346                     };
347                 }
348             } catch (NoSuchMethodException | SecurityException ex) {
349                 logLookupException(handlingMethod, resourceClass, iFace, ex);
350             }
351         }
352         return null;
353     }

355     private void logLookupException(final Method method, final Class<?> component, Class<?> iFace, Exception ex) {
356         LOGGER.log(
357                 Level.WARNING,
358                 LocalizationMessages.EJB_INTERFACE_HANDLING_METHOD_LOOKUP_EXCEPTION(method, component, iFace), ex);
359     }

361     private List<Class> remoteAndLocalIfaces(final Class<?> resourceClass) {
362         final List<Class> allLocalOrRemoteIfaces = new LinkedList<>();
363         if (resourceClass.isAnnotationPresent(Remote.class)) {
364             allLocalOrRemoteIfaces.addAll(Arrays.asList(resourceClass.getAnnotation(Remote.class).value()));
365         }
366         if (resourceClass.isAnnotationPresent(Local.class)) {
367             allLocalOrRemoteIfaces.addAll(Arrays.asList(resourceClass.getAnnotation(Local.class).value()));
368         }
369         for (Class<?> i : resourceClass.getInterfaces()) {
370             if (i.isAnnotationPresent(Remote.class) || i.isAnnotationPresent(Local.class)) {
371                 allLocalOrRemoteIfaces.add(i);
372             }
373         }
374         return allLocalOrRemoteIfaces;
375     }

387     private static Object lookup(InitialContext ic, Class<?> c, String name, EjbComponentProvider provider)
388             throws NamingException {
389         try {
390             return lookupSimpleForm(ic, name, provider);
391         } catch (NamingException ex) {
392             LOGGER.log(Level.WARNING, LocalizationMessages.EJB_CLASS_SIMPLE_LOOKUP_FAILED(c.getName()), ex);
393 
394             return lookupFullyQualifiedForm(ic, c, name, provider);
395         }
396     }

398     private static Object lookupSimpleForm(InitialContext ic, String name, EjbComponentProvider provider) throws NamingException {
399         if (provider.libNames.isEmpty()) {
400             String jndiName = "java:module/" + name;
401             return ic.lookup(jndiName);
402         } else {
403             NamingException ne = null;
404             for (String moduleName : provider.libNames) {
405                 String jndiName = "java:app/" + moduleName + "/" + name;
406                 Object result;
407                 try {
408                     result = ic.lookup(jndiName);
409                     if (result != null) {
410                         return result;
411                     }
412                 } catch (NamingException e) {
413                     ne = e;
414                 }
415             }
416             throw (ne != null) ? ne : new NamingException();
417         }
418     }

420     private static Object lookupFullyQualifiedForm(InitialContext ic, Class<?> c, String name, EjbComponentProvider provider)
421             throws NamingException {
422         if (provider.libNames.isEmpty()) {
423             String jndiName = "java:module/" + name + "!" + c.getName();
424             return ic.lookup(jndiName);
425         } else {
426             NamingException ne = null;
427             for (String moduleName : provider.libNames) {
428                 String jndiName = "java:app/" + moduleName + "/" + name + "!" + c.getName();
429                 Object result;
430                 try {
431                     result = ic.lookup(jndiName);
432                     if (result != null) {
433                         return result;
434                     }
435                 } catch (NamingException e) {
436                     ne = e;
437                 }
438             }
439             throw (ne != null) ? ne : new NamingException();
440         }
