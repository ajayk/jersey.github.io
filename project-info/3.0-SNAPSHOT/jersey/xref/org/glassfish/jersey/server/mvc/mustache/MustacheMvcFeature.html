41  package org.glassfish.jersey.server.mvc.mustache;
42  
43  import javax.ws.rs.ConstrainedTo;
44  import javax.ws.rs.RuntimeType;
45  import javax.ws.rs.core.Configuration;
46  import javax.ws.rs.core.Feature;
47  import javax.ws.rs.core.FeatureContext;
48  
49  import org.glassfish.jersey.server.mvc.MvcFeature;
50  
51  /**
52   * {@link Feature} used to add support for {@link MvcFeature MVC} and Mustache templates.
53   * <p/>
54   * Note: This feature also registers {@link MvcFeature}.
55   *
56   * @author Michal Gajdos
57   * @since 2.3
58   */
59  @ConstrainedTo(RuntimeType.SERVER)
60  public class MustacheMvcFeature implements Feature {
61  
62      private static final String SUFFIX = ".mustache";
63  
64      /**
65       * {@link String} property defining the base path to Mustache templates. If set, the value of the property is added in front
66       * of the template name defined in:
67       * <ul>
68       * <li>{@link org.glassfish.jersey.server.mvc.Viewable Viewable}</li>
69       * <li>{@link org.glassfish.jersey.server.mvc.Template Template}, or</li>
70       * <li>{@link org.glassfish.jersey.server.mvc.ErrorTemplate ErrorTemplate}</li>
71       * </ul>
72       * <p/>
73       * Value can be absolute providing a full path to a system directory with templates or relative to current
74       * {@link javax.servlet.ServletContext servlet context}.
75       * <p/>
76       * There is no default value.
77       * <p/>
78       * The name of the configuration property is <tt>{@value}</tt>.
79       */
80      public static final String TEMPLATE_BASE_PATH = MvcFeature.TEMPLATE_BASE_PATH + SUFFIX;
81  
82      /**
83       * If {@code true} then enable caching of Mustache templates to avoid multiple compilation.
84       * <p/>
85       * The default value is {@code false}.
86       * <p/>
87       * The name of the configuration property is <tt>{@value}</tt>.
88       *
89       * @since 2.5
90       */
91      public static final String CACHE_TEMPLATES = MvcFeature.CACHE_TEMPLATES + SUFFIX;
92  
93      /**
94       * Property used to pass user-configured {@link com.github.mustachejava.MustacheFactory factory} able to create
95       * {@link com.github.mustachejava.Mustache Mustache templates}.
96       * <p/>
97       * The default value is not set.
98       * <p/>
99       * The name of the configuration property is <tt>{@value}</tt>.
100      *
101      * @since 2.5
102      */
103     public static final String TEMPLATE_OBJECT_FACTORY = MvcFeature.TEMPLATE_OBJECT_FACTORY + SUFFIX;
104 
105     /**
106      * Property defines output encoding produced by {@link org.glassfish.jersey.server.mvc.spi.TemplateProcessor}.
107      * The value must be a valid encoding defined that can be passed
108      * to the {@link java.nio.charset.Charset#forName(String)} method.
109      *
110      * <p/>
111      * The default value is {@code UTF-8}.
112      * <p/>
113      * The name of the configuration property is <tt>{@value}</tt>.
114      * <p/>
115      *
116      * @since 2.7
117      */
118     public static final String ENCODING = MvcFeature.ENCODING + SUFFIX;
119 
120     @Override
121     public boolean configure(final FeatureContext context) {
122         final Configuration config = context.getConfiguration();
123 
124         if (!config.isRegistered(MustacheTemplateProcessor.class)) {
125             // Template Processor.
126             context.register(MustacheTemplateProcessor.class);
127 
128             // MvcFeature.
129             if (!config.isRegistered(MvcFeature.class)) {
130                 context.register(MvcFeature.class);
131             }
132 
133             return true;
134         }
135         return false;
136     }
137 }
