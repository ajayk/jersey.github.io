40  package org.glassfish.jersey.server.spi;
41  
42  import java.util.Set;
43  
44  import org.glassfish.hk2.api.ServiceLocator;
45  
46  /**
47   * Component provider interface to allow custom management of 3rd party
48   * components life-cycle and dependency injection.
49   * <p />
50   * An implementation (a component-provider) identifies itself by placing a provider-configuration
51   * file (if not already present), {@code org.glassfish.jersey.server.spi.ComponentProvider}
52   * in the resource directory <tt>META-INF/services</tt>, and adding the fully
53   * qualified service-provider-class of the implementation in the file.
54   *
55   * Jersey will not even try to inject component provider instances with Jersey artifacts.
56   * The SPI providers should be designed so that no dependency injection is needed at the bind time phase.
57  
58   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
59   */
60  public interface ComponentProvider {
61  
62  
63      /**
64       * Initializes the component provider with a reference to a HK2 service locator
65       * instance, which will get used in the application to manage individual components.
66       * Providers should keep a reference to the HK2 service locator for later use.
67       * This method will be invoked prior to any bind method calls.
68       * The service locator parameter will not be fully initialized at the time of invocation
69       * and should be used as a reference only.
70       *
71       * @param locator an HK2 service locator.
72       */
73      public void initialize(final ServiceLocator locator);
74  
75      /**
76       * Jersey will invoke this method before binding of each component class internally
77       * during initialization of it's HK2 service locator.
78       *
79       * If the component provider wants to bind the component class
80       * itself, it must do so and return true. In that case, Jersey will not
81       * bind the component and rely on the component provider in this regard.
82       *
83       * @param component a component (resource/provider) class.
84       * @param providerContracts provider contracts implemented by given component.
85       * @return true if the component class has been bound by the provider, false otherwise
86       */
87      public boolean bind(final Class<?> component, Set<Class<?>> providerContracts);
88  
89      /**
90       * Jersey will invoke this method after all component classes have been bound.
91       *
92       * If the component provider wants to do some actions after it has seen all component classes
93       * registered with the application, this is the right place for the corresponding code.
94       */
95      public void done();
96  }
