75  /**
76   * Utility class for constructing resource model from JAX-RS annotated POJO.
77   *
78   * @author Jakub Podlesak (jakub.podlesak at oracle.com)

80  final class IntrospectionModeller {

82      private static final Logger LOGGER = Logger.getLogger(IntrospectionModeller.class.getName());

84      // introspected annotated JAX-RS resource class
85      private final Class<?> handlerClass;
86      // validation flag
87      private final boolean disableValidation;

89      /**
90       * Create a new introspection modeller for a given JAX-RS resource class.
91       *
92       * @param handlerClass      JAX-RS resource (handler) class.
93       * @param disableValidation if set to {@code true}, then any model validation checks will be disabled.
94       */
95      public IntrospectionModeller(Class<?> handlerClass, boolean disableValidation) {
96          this.handlerClass = handlerClass;
97          this.disableValidation = disableValidation;
98      }

100     /**
101      * Create a new resource model builder for the introspected class.
102      * <p>
103      * The model returned is filled with the introspected data.
104      * </p>
105      *
106      * @return new resource model builder for the introspected class.
107      */
108     public Resource.Builder createResourceBuilder() {
109         return Errors.processWithException(new Producer<Resource.Builder>() {
110             @Override
111             public Resource.Builder call() {
112                 return doCreateResourceBuilder();
113             }
114         });
115     }

117     private Resource.Builder doCreateResourceBuilder() {
118         if (!disableValidation) {
119             checkForNonPublicMethodIssues();
120         }
121 
122         final Class<?> annotatedResourceClass = ModelHelper.getAnnotatedResourceClass(handlerClass);
123         final Path rPathAnnotation = annotatedResourceClass.getAnnotation(Path.class);
124 
125         final boolean keepEncodedParams =
126                 (null != annotatedResourceClass.getAnnotation(Encoded.class));
127 
128         final List<MediaType> defaultConsumedTypes =
129                 extractMediaTypes(annotatedResourceClass.getAnnotation(Consumes.class));
130         final List<MediaType> defaultProducedTypes =
131                 extractMediaTypes(annotatedResourceClass.getAnnotation(Produces.class));
132         final Collection<Class<? extends Annotation>> defaultNameBindings =
133                 ReflectionHelper.getAnnotationTypes(annotatedResourceClass, NameBinding.class);
134 
135         final MethodList methodList = new MethodList(handlerClass);
136 
137         final List<Parameter> resourceClassParameters = new LinkedList<>();
138         checkResourceClassSetters(methodList, keepEncodedParams, resourceClassParameters);
139         checkResourceClassFields(keepEncodedParams, InvocableValidator.isSingleton(handlerClass), resourceClassParameters);
140 
141         Resource.Builder resourceBuilder;
142 
143         if (null != rPathAnnotation) {
144             resourceBuilder = Resource.builder(rPathAnnotation.value());
145         } else {
146             resourceBuilder = Resource.builder();
147         }
148 
149         boolean extended = false;
150         if (handlerClass.isAnnotationPresent(ExtendedResource.class)) {
151             resourceBuilder.extended(true);
152             extended = true;
153         }
154 
155         resourceBuilder.name(handlerClass.getName());
156 
157         addResourceMethods(resourceBuilder, methodList, resourceClassParameters, keepEncodedParams,
158                 defaultConsumedTypes, defaultProducedTypes, defaultNameBindings, extended);
159         addSubResourceMethods(resourceBuilder, methodList, resourceClassParameters, keepEncodedParams,
160                 defaultConsumedTypes, defaultProducedTypes, defaultNameBindings, extended);
161         addSubResourceLocators(resourceBuilder, methodList, resourceClassParameters, keepEncodedParams, extended);
162 
163         if (LOGGER.isLoggable(Level.FINEST)) {
164             LOGGER.finest(LocalizationMessages.NEW_AR_CREATED_BY_INTROSPECTION_MODELER(
165                     resourceBuilder.toString()));
166         }
167 
168         return resourceBuilder;
169     }

171     private void checkForNonPublicMethodIssues() {
172         final MethodList allDeclaredMethods = new MethodList(getAllDeclaredMethods(handlerClass));
173 
174         // non-public resource methods
175         for (AnnotatedMethod m : allDeclaredMethods.withMetaAnnotation(HttpMethod.class)
176                 .withoutAnnotation(Path.class).isNotPublic()) {
177             Errors.warning(handlerClass, LocalizationMessages.NON_PUB_RES_METHOD(m.getMethod().toGenericString()));
178         }
179         // non-public sub-resource methods
180         for (AnnotatedMethod m : allDeclaredMethods.withMetaAnnotation(HttpMethod.class)
181                 .withAnnotation(Path.class).isNotPublic()) {
182             Errors.warning(handlerClass, LocalizationMessages.NON_PUB_SUB_RES_METHOD(m.getMethod().toGenericString()));
183         }
184         // non-public sub-resource locators
185         for (AnnotatedMethod m : allDeclaredMethods.withoutMetaAnnotation(HttpMethod.class)
186                 .withAnnotation(Path.class).isNotPublic()) {
187             Errors.warning(handlerClass, LocalizationMessages.NON_PUB_SUB_RES_LOC(m.getMethod().toGenericString()));
188         }
189     }

191     private void checkResourceClassSetters(final MethodList methodList,
192                                            final boolean encodedFlag,
193                                            Collection<Parameter> injectableParameters) {
194 
195         for (AnnotatedMethod method : methodList.withoutMetaAnnotation(HttpMethod.class)
196                 .withoutAnnotation(Path.class)
197                 .hasNumParams(1)
198                 .hasReturnType(void.class)
199                 .nameStartsWith("set")) {
200             Parameter p = Parameter.create(
201                     handlerClass,
202                     method.getMethod().getDeclaringClass(),
203                     encodedFlag || method.isAnnotationPresent(Encoded.class),
204                     method.getParameterTypes()[0],
205                     method.getGenericParameterTypes()[0],
206                     method.getAnnotations());
207             if (null != p) {
208                 ResourceMethodValidator.validateParameter(p, method.getMethod(), method.getMethod().toGenericString(), "1",
209                         InvocableValidator.isSingleton(handlerClass));
210 
211                 // we do not inject entity parameters into class instance fields and properties.
212                 if (p.getSource() != Parameter.Source.ENTITY) {
213                     injectableParameters.add(p);
214                 }
215             }
216         }
217     }

219     private void checkResourceClassFields(final boolean encodedFlag,
220                                           boolean isInSingleton,
221                                           Collection<Parameter> injectableParameters) {
222 
223         for (Field field : AccessController.doPrivileged(ReflectionHelper.getDeclaredFieldsPA(handlerClass))) {
224             if (field.getDeclaredAnnotations().length > 0) {
225                 Parameter p = Parameter.create(
226                         handlerClass,
227                         field.getDeclaringClass(),
228                         encodedFlag || field.isAnnotationPresent(Encoded.class),
229                         field.getType(),
230                         field.getGenericType(),
231                         field.getAnnotations());
232                 if (null != p) {
233                     ResourceMethodValidator.validateParameter(p, field, field.toGenericString(), field.getName(),
234                             isInSingleton);
235                     // we do not inject entity and unknown parameters into class instance fields and properties.
236                     if (p.getSource() != Parameter.Source.ENTITY) {
237                         injectableParameters.add(p);
238                     }
239                 }
240             }
241         }
242     }

244     private List<Method> getAllDeclaredMethods(final Class<?> clazz) {
245         final List<Method> result = new LinkedList<>();
246 
247         AccessController.doPrivileged(new PrivilegedAction<Object>() {
248 
249             @Override
250             public Object run() {
251                 Class current = clazz;
252                 while (current != Object.class && current != null) {
253                     result.addAll(Arrays.asList(current.getDeclaredMethods()));
254                     current = current.getSuperclass();
255                 }
256                 return null;
257             }
258         });
259 
260         return result;
261     }

263     private static List<MediaType> resolveConsumedTypes(final AnnotatedMethod am, final List<MediaType> defaultConsumedTypes) {
264         // Override default types if the annotation is present on the method
265         if (am.isAnnotationPresent(Consumes.class)) {
266             return extractMediaTypes(am.getAnnotation(Consumes.class));
267         }
268 
269         return defaultConsumedTypes;
270     }
271 
272     private static List<MediaType> resolveProducedTypes(final AnnotatedMethod am, final List<MediaType> defaultProducedTypes) {
273         // Override default types if the annotation is present on the method
274         if (am.isAnnotationPresent(Produces.class)) {
275             return extractMediaTypes(am.getAnnotation(Produces.class));
276         }
277 
278         return defaultProducedTypes;
279     }

281     private static List<MediaType> extractMediaTypes(final Consumes annotation) {
282         return (annotation != null) ? extractMediaTypes(annotation.value()) : Collections.<MediaType>emptyList();
283     }
284 
285     private static List<MediaType> extractMediaTypes(final Produces annotation) {
286         return (annotation != null) ? extractMediaTypes(annotation.value()) : Collections.<MediaType>emptyList();
287     }
288 
289     private static List<MediaType> extractMediaTypes(final String[] values) {
290         if (values.length == 0) {
291             return Collections.emptyList();
292         }
293 
294         final List<MediaType> types = new ArrayList<>(values.length);
295         for (final String mtEntry : values) {
296             for (final String mt : Tokenizer.tokenize(mtEntry, ",")) {
297                 types.add(MediaType.valueOf(mt));
298             }
299         }
300 
301         return types;
302     }

304     private static void introspectAsyncFeatures(AnnotatedMethod am, ResourceMethod.Builder resourceMethodBuilder) {
305         if (am.isAnnotationPresent(ManagedAsync.class)) {
306             resourceMethodBuilder.managedAsync();
307         }
308 
309         for (Annotation[] annotations : am.getParameterAnnotations()) {
310             for (Annotation annotation : annotations) {
311                 if (annotation.annotationType() == Suspended.class) {
312                     resourceMethodBuilder.suspended(AsyncResponse.NO_TIMEOUT, TimeUnit.MILLISECONDS);
313                     return;
314                 }
315             }
316         }
317     }

319     private void addResourceMethods(
320             final Resource.Builder resourceBuilder,
321             final MethodList methodList,
322             final List<Parameter> resourceClassParameters, // parameters derived from fields and setters on the resource class
323             final boolean encodedParameters,
324             final List<MediaType> defaultConsumedTypes,
325             final List<MediaType> defaultProducedTypes,
326             final Collection<Class<? extends Annotation>> defaultNameBindings,
327             final boolean extended
328     ) {
329         for (AnnotatedMethod am : methodList.withMetaAnnotation(HttpMethod.class).withoutAnnotation(Path.class)) {
330             ResourceMethod.Builder methodBuilder =
331                     resourceBuilder.addMethod(am.getMetaMethodAnnotations(HttpMethod.class).get(0).value())
332                             .consumes(resolveConsumedTypes(am, defaultConsumedTypes))
333                             .produces(resolveProducedTypes(am, defaultProducedTypes))
334                             .encodedParameters(encodedParameters || am.isAnnotationPresent(Encoded.class))
335                             .nameBindings(defaultNameBindings)
336                             .nameBindings(am.getAnnotations())
337                             .handledBy(handlerClass, am.getMethod())
338                             .handlingMethod(am.getDeclaredMethod())
339                             .handlerParameters(resourceClassParameters)
340                             .extended(extended || am.isAnnotationPresent(ExtendedResource.class));
341 
342             introspectAsyncFeatures(am, methodBuilder);
343         }
344     }

346     private void addSubResourceMethods(
347             final Resource.Builder resourceBuilder,
348             final MethodList methodList,
349             final List<Parameter> resourceClassParameters, // parameters derived from fields and setters on the resource class
350             final boolean encodedParameters,
351             final List<MediaType> defaultConsumedTypes,
352             final List<MediaType> defaultProducedTypes,
353             final Collection<Class<? extends Annotation>> defaultNameBindings,
354             final boolean extended
355     ) {
356 
357         for (AnnotatedMethod am : methodList.withMetaAnnotation(HttpMethod.class).withAnnotation(Path.class)) {
358             Resource.Builder childResourceBuilder = resourceBuilder.addChildResource(am.getAnnotation(Path.class).value());
359 
360             ResourceMethod.Builder methodBuilder =
361                     childResourceBuilder.addMethod(am.getMetaMethodAnnotations(HttpMethod.class).get(0).value())
362                             .consumes(resolveConsumedTypes(am, defaultConsumedTypes))
363                             .produces(resolveProducedTypes(am, defaultProducedTypes))
364                             .encodedParameters(encodedParameters || am.isAnnotationPresent(Encoded.class))
365                             .nameBindings(defaultNameBindings)
366                             .nameBindings(am.getAnnotations())
367                             .handledBy(handlerClass, am.getMethod())
368                             .handlingMethod(am.getDeclaredMethod())
369                             .handlerParameters(resourceClassParameters)
370                             .extended(extended || am.isAnnotationPresent(ExtendedResource.class));
371 
372             introspectAsyncFeatures(am, methodBuilder);
373         }
374     }

376     private void addSubResourceLocators(
377             final Resource.Builder resourceBuilder,
378             final MethodList methodList,
379             final List<Parameter> resourceClassParameters, // parameters derived from fields and setters on the resource class
380             final boolean encodedParameters,
381             final boolean extended) {
382 
383         for (AnnotatedMethod am : methodList.withoutMetaAnnotation(HttpMethod.class).withAnnotation(Path.class)) {
384             final String path = am.getAnnotation(Path.class).value();
385             Resource.Builder builder = resourceBuilder;
386             if (path != null && !path.isEmpty() && !"/".equals(path)) {
387                 builder = resourceBuilder.addChildResource(path);
388             }
389 
390             builder.addMethod()
391                     .encodedParameters(encodedParameters || am.isAnnotationPresent(Encoded.class))
392                     .handledBy(handlerClass, am.getMethod())
393                     .handlingMethod(am.getDeclaredMethod())
394                     .handlerParameters(resourceClassParameters)
395                     .extended(extended || am.isAnnotationPresent(ExtendedResource.class));
396         }
