41  package org.glassfish.jersey.server.oauth1;

43  import java.util.Map;
44  
45  import javax.annotation.Priority;
46  import javax.ws.rs.core.Configuration;
47  import javax.ws.rs.core.Feature;
48  import javax.ws.rs.core.FeatureContext;
49  
50  import org.glassfish.jersey.oauth1.signature.OAuth1SignatureFeature;
51  import org.glassfish.jersey.server.model.ModelProcessor;
52  import org.glassfish.jersey.server.model.Resource;
53  import org.glassfish.jersey.server.model.ResourceModel;
54  import org.glassfish.jersey.server.oauth1.internal.AccessTokenResource;
55  import org.glassfish.jersey.server.oauth1.internal.RequestTokenResource;

57  /**
58   * The feature enables support for OAuth 1.0a on the server.
59   * <p/>
60   * The OAuth 1 server support requires implementation of {@link OAuth1Provider};
61   * which will be used to retrieve Request Tokens, Access tokens, etc. The implementation should be configured
62   * in this feature or registered as a standard provider.
63   * <p/>
64   * Feature can be created and configured by various constructors. Beside that, the feature behaviour
65   * can be overwritten by configuration properties {@link OAuth1ServerProperties#ENABLE_TOKEN_RESOURCES},
66   * {@link OAuth1ServerProperties#REQUEST_TOKEN_URI} and {@link OAuth1ServerProperties#ACCESS_TOKEN_URI}.
67   *
68   * @author Miroslav Fuksa
69   */
70  public class OAuth1ServerFeature implements Feature {
71  
72      private final OAuth1Provider oAuth1Provider;
73      private final String requestTokenUri;
74      private final String accessTokenUri;
75  
76      /**
77       * Create a new feature configured with {@link OAuth1Provider OAuth provider} and request and access token
78       * URIs. The feature also exposes Request and Access Token Resources.
79       * These resources are part of the Authorization process and
80       * grant Request and Access tokens. Resources will be available on
81       * URIs defined by parameters {@code requestTokenUri} and {@code accessTokenUri}.
82       *
83       * @param oAuth1Provider Instance of the {@code OAuth1Provider} that will handle authorization. If the value is
84       *                       {@code null}, then the provider must be registered explicitly outside of this feature
85       *                       as a standard provider.
86       * @param requestTokenUri URI (relative to application context path) of Request Token Resource that will be exposed.
87       * @param accessTokenUri URI (relative to application context path) of Request Token Resource that will be exposed.
88       */
89      public OAuth1ServerFeature(OAuth1Provider oAuth1Provider,
90                                 String requestTokenUri,
91                                 String accessTokenUri) {
92          this.oAuth1Provider = oAuth1Provider;
93          this.requestTokenUri = requestTokenUri;
94          this.accessTokenUri = accessTokenUri;
95      }
96  
97      /**
98       * Create a new feature configured with {@link OAuth1Provider OAuth provider}. The feature will not
99       * expose Request and Access Token Resources. The OAuth 1 support will not be responsible for handling
100      * these authorization request types.
101      *
102      * @param oAuth1Provider Instance of the {@code OAuth1Provider} that will handle authorization.
103      */
104     public OAuth1ServerFeature(OAuth1Provider oAuth1Provider) {
105         this(oAuth1Provider, null, null);
106     }
107 
108     /**
109      * Create a new feature. The feature will not register any {@link OAuth1Provider OAuth provider}
110      * and it will not expose Request and Access Token Resources. {@code OAuth1Provider} must be registered
111      * explicitly as a standard provider. As Token Resources are not exposed, the OAuth 1 support will
112      * not be responsible for handling Token Requests.
113      */
114     public OAuth1ServerFeature() {
115         this(null);
116     }
117 
118     @Override
119     public boolean configure(FeatureContext context) {
120         if (oAuth1Provider != null) {
121             context.register(oAuth1Provider);
122         }
123 
124         context.register(OAuth1ServerFilter.class);
125 
126         if (!context.getConfiguration().isRegistered(OAuth1SignatureFeature.class)) {
127             context.register(OAuth1SignatureFeature.class);
128         }
129 
130         final Map<String, Object> properties = context.getConfiguration().getProperties();
131         final Boolean propertyResourceEnabled = OAuth1ServerProperties.getValue(properties,
132                 OAuth1ServerProperties.ENABLE_TOKEN_RESOURCES, null, Boolean.class);
133 
134         boolean registerResources = propertyResourceEnabled != null
135                 ? propertyResourceEnabled : requestTokenUri != null & accessTokenUri != null;
136 
137         if (registerResources) {
138             String requestUri = OAuth1ServerProperties.getValue(properties, OAuth1ServerProperties.REQUEST_TOKEN_URI,
139                     null, String.class);
140             if (requestUri == null) {
141                 requestUri = requestTokenUri == null ? "requestToken" : requestTokenUri;
142             }
143 
144             String accessUri = OAuth1ServerProperties.getValue(properties, OAuth1ServerProperties.ACCESS_TOKEN_URI,
145                     null, String.class);
146             if (accessUri == null) {
147                 accessUri = accessTokenUri == null ? "accessToken" : accessTokenUri;
148             }
149 
150             final Resource requestResource = Resource.builder(RequestTokenResource.class).path(requestUri).build();
151             final Resource accessResource = Resource.builder(AccessTokenResource.class).path(accessUri).build();
152 
153             context.register(new OAuthModelProcessor(requestResource, accessResource));
154         }
155         return true;
156     }
157 
158 
159     @Priority(100)
160     private static class OAuthModelProcessor implements ModelProcessor {
161         private final Resource[] resources;
162 
163         private OAuthModelProcessor(Resource... resources) {
164             this.resources = resources;
165         }
166 
167         @Override
168         public ResourceModel processResourceModel(ResourceModel resourceModel, Configuration configuration) {
169             final ResourceModel.Builder builder = new ResourceModel.Builder(resourceModel, false);
170             for (Resource resource : resources) {
171                 builder.addResource(resource);
172             }
173 
174             return builder.build();
175         }
176 
177         @Override
178         public ResourceModel processSubResource(ResourceModel subResourceModel, Configuration configuration) {
179             return subResourceModel;
180         }
181     }
