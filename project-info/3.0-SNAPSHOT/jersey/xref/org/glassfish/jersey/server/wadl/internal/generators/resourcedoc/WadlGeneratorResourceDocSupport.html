80  /**
81   * A {@link org.glassfish.jersey.server.wadl.WadlGenerator} implementation that enhances the generated wadl by
82   * information read from a resourcedoc (containing javadoc information about resource
83   * classes).
84   * <p>
85   * The resourcedoc information can either be provided via a {@link File} ({@link #setResourceDocFile(File)}) reference or
86   * via an {@link InputStream} ({@link #setResourceDocStream(InputStream)}).
87   * </p>
88   * <p>
89   * The {@link File} should be used when using the maven-wadl-plugin for generating wadl offline,
90   * the {@link InputStream} should be used when the extended wadl is generated by jersey at runtime, e.g.
91   * using the {@link org.glassfish.jersey.server.wadl.config.WadlGeneratorConfig} for configuration.
92   * </p>
93   *
94   * @author Martin Grotzke (martin.grotzke at freiheit.com)
95   * @author Miroslav Fuksa
96   */
97  public class WadlGeneratorResourceDocSupport implements WadlGenerator {

119     /**
120      * Set the <code>resourceDocFile</code> to the given file. Invoking this method is only allowed, as long as
121      * the <code>resourceDocStream</code> is not set, otherwise an {@link IllegalStateException} will be thrown.
122      * @param resourceDocFile the resourcedoc file to set.
123      */
124     public void setResourceDocFile(final File resourceDocFile) {
125         if (resourceDocStream != null) {
126             throw new IllegalStateException("The resourceDocStream property is already set,"
127                     + " therefore you cannot set the resourceDocFile property. Only one of both can be set at a time.");
128         }
129         this.resourceDocFile = resourceDocFile;
130     }

132     /**
133      * Set the <code>resourceDocStream</code> to the given file. Invoking this method is only allowed, as long as
134      * the <code>resourceDocFile</code> is not set, otherwise an {@link IllegalStateException} will be thrown.
135      * <p>
136      * The resourcedoc stream must be closed by the client providing the stream.
137      * </p>
138      * @param resourceDocStream the resourcedoc stream to set.
139      */
140     public void setResourceDocStream(final InputStream resourceDocStream) {
141         if (this.resourceDocStream != null) {
142             throw new IllegalStateException("The resourceDocFile property is already set,"
143                     + " therefore you cannot set the resourceDocStream property. Only one of both can be set at a time.");
144         }
145         this.resourceDocStream = resourceDocStream;
146     }

148     public void init() throws Exception {
149         if (resourceDocFile == null && resourceDocStream == null) {
150             throw new IllegalStateException("Neither the resourceDocFile nor the resourceDocStream"
151                     + " is set, one of both is required.");
152         }
153         delegate.init();
154 
155         try (final InputStream inputStream = resourceDocFile != null ? new FileInputStream(resourceDocFile) : resourceDocStream) {
156             final ResourceDocType resourceDocType =
157                     WadlUtils.unmarshall(inputStream, saxFactoryProvider.get(), ResourceDocType.class);
158             resourceDoc = new ResourceDocAccessor(resourceDocType);
159         } finally {
160             resourceDocFile = null;
161         }
162     }

181     /**
182      * @param r Jersey resource component for which the WADL reource is to be created.
183      * @param path path where the resource is exposed.
184      * @return the enhanced {@link com.sun.research.ws.wadl.Resource}.
185      * @see org.glassfish.jersey.server.wadl.WadlGenerator#createResource(org.glassfish.jersey.server.model.Resource, String)
186      */
187     public Resource createResource(final org.glassfish.jersey.server.model.Resource r, final String path) {
188         final Resource result = delegate.createResource(r, path);
189         for (final Class<?> resourceClass : r.getHandlerClasses()) {
190             final ClassDocType classDoc = resourceDoc.getClassDoc(resourceClass);
191             if (classDoc != null && !isEmpty(classDoc.getCommentText())) {
192                 final Doc doc = new Doc();
193                 doc.getContent().add(classDoc.getCommentText());
194                 result.getDoc().add(doc);
195             }
196         }
197         return result;
198     }

201      * @param resource Jersey resource component.
202      * @param resourceMethod resource method.
203      * @return the enhanced {@link com.sun.research.ws.wadl.Method}.

205      * org.glassfish.jersey.server.model.ResourceMethod)
206      */
207     public Method createMethod(final org.glassfish.jersey.server.model.Resource resource,
208                                final ResourceMethod resourceMethod) {
209         final Method result = delegate.createMethod(resource, resourceMethod);
210         final java.lang.reflect.Method method = resourceMethod.getInvocable().getDefinitionMethod();
211         final MethodDocType methodDoc = resourceDoc.getMethodDoc(method.getDeclaringClass(), method);
212         if (methodDoc != null && !isEmpty(methodDoc.getCommentText())) {
213             final Doc doc = new Doc();
214             doc.getContent().add(methodDoc.getCommentText());
215             // doc.getOtherAttributes().put( new QName( "xmlns" ), "http://www.w3.org/1999/xhtml" );
216             result.getDoc().add(doc);
217         }
218 
219         return result;
220     }

222     /**
223      * @param r Jersey resource component.
224      * @param m resource method.
225      * @param mediaType media type.
226      * @return the enhanced {@link com.sun.research.ws.wadl.Representation}.
227      * @see org.glassfish.jersey.server.wadl.WadlGenerator#createRequestRepresentation(org.glassfish.jersey.server.model.Resource,
228      * org.glassfish.jersey.server.model.ResourceMethod, javax.ws.rs.core.MediaType)
229      */
230     public Representation createRequestRepresentation(final org.glassfish.jersey.server.model.Resource r,
231                                                       final org.glassfish.jersey.server.model.ResourceMethod m,
232                                                       final MediaType mediaType) {
233         final Representation result = delegate.createRequestRepresentation(r, m, mediaType);
234         final RepresentationDocType requestRepresentation = resourceDoc.getRequestRepresentation(m.getInvocable()
235                         .getDefinitionMethod().getDeclaringClass(),
236                 m.getInvocable().getDefinitionMethod(), result.getMediaType()
237         );
238         if (requestRepresentation != null) {
239             result.setElement(requestRepresentation.getElement());
240             addDocForExample(result.getDoc(), requestRepresentation.getExample());
241         }
242         return result;
243     }

257     /**
258      * @param r Jersey resource component.
259      * @param m resource method.
260      * @return the enhanced {@link com.sun.research.ws.wadl.Response}.
261      * @see org.glassfish.jersey.server.wadl.WadlGenerator#createResponses(org.glassfish.jersey.server.model.Resource,
262      * org.glassfish.jersey.server.model.ResourceMethod)
263      */
264     public List<Response> createResponses(final org.glassfish.jersey.server.model.Resource r,
265                                           final org.glassfish.jersey.server.model.ResourceMethod m) {
266         final ResponseDocType responseDoc = resourceDoc.getResponse(m.getInvocable().getDefinitionMethod().getDeclaringClass(),
267                 m.getInvocable().getDefinitionMethod());
268         List<Response> responses = new ArrayList<Response>();
269         if (responseDoc != null && responseDoc.hasRepresentations()) {
270             for (final RepresentationDocType representationDoc : responseDoc.getRepresentations()) {
271                 final Response response = new Response();
272 
273                 final Representation wadlRepresentation = new Representation();
274                 wadlRepresentation.setElement(representationDoc.getElement());
275                 wadlRepresentation.setMediaType(representationDoc.getMediaType());
276                 addDocForExample(wadlRepresentation.getDoc(), representationDoc.getExample());
277                 addDoc(wadlRepresentation.getDoc(), representationDoc.getDoc());
278 
279                 response.getStatus().add(representationDoc.getStatus());
280                 response.getRepresentation().add(wadlRepresentation);
281 
282                 responses.add(response);
283             }
284 
285             if (!responseDoc.getWadlParams().isEmpty()) {
286                 for (final WadlParamType wadlParamType : responseDoc.getWadlParams()) {
287                     final Param param = new Param();
288                     param.setName(wadlParamType.getName());
289                     param.setStyle(ParamStyle.fromValue(wadlParamType.getStyle()));
290                     param.setType(wadlParamType.getType());
291                     addDoc(param.getDoc(), wadlParamType.getDoc());
292                     for (final Response response : responses) {
293                         response.getParam().add(param);
294                     }
295                 }
296             }
297 
298             if (!isEmpty(responseDoc.getReturnDoc())) {
299                 for (final Response response : responses) {
300                     addDoc(response.getDoc(), responseDoc.getReturnDoc());
301                 }
302             }
303 
304         } else {
305             responses = delegate.createResponses(r, m);
306         }
307 
308         return responses;
309     }

311     private void addDocForExample(final List<Doc> docs, final String example) {
312         if (!isEmpty(example)) {
313             final Doc doc = new Doc();
314 
315             final Elements pElement = Elements.el("p")
316                     .add(Elements.val("h6", "Example"))
317                     .add(Elements.el("pre").add(Elements.val("code", example)));
318 
319             doc.getContent().add(pElement);
320             docs.add(doc);
321         }
322     }
323 
324     private void addDoc(final List<Doc> docs, final String text) {
325         if (!isEmpty(text)) {
326             final Doc doc = new Doc();
327             doc.getContent().add(text);
328             docs.add(doc);
329         }
330     }

349                 result.getDoc().add(doc);
350             }
351         }

353     }
