165         /**
166          * Build routing stage.
167          *
168          * @return routing stage for the runtime resource model.
169          */
170         public ChainableStage<RequestProcessingContext> buildStage() {
171             // No L10N - internally used class
172             if (locator == null) {
173                 throw new NullPointerException("HK2 locator is not set.");
174             }
175             if (resourceContext == null) {
176                 throw new NullPointerException("Resource context is not set.");
177             }
178             if (config == null) {
179                 throw new NullPointerException("Runtime configuration is not set.");
180             }
181             if (entityProviders == null) {
182                 throw new NullPointerException("Entity providers are not set.");
183             }
184             if (processingProviders == null) {
185                 throw new NullPointerException("Processing providers are not set.");
186             }
187 
188             final RuntimeModelBuilder runtimeModelBuilder = new RuntimeModelBuilder(
189                     locator,
190                     resourceContext,
191                     config,
192                     entityProviders,
193                     processingProviders,
194                     locator.getService(ResourceMethodInvoker.Builder.class));
195 
196             return new RoutingStage(runtimeModelBuilder.buildModel(resourceModel, false));
197         }
