54  /**
55   * An abstract extension of {@link WebApplicationException} for the class of
56   * parameter-based exceptions.
57   * <p>
58   * Exceptions that are instances of this class will be thrown if the runtime
59   * encounters an error obtaining a parameter value, from a request, for a
60   * Java type that is annotated with a parameter-based annotation, such as
61   * {@link QueryParam}. For more details see
62   * <a href="http://jsr311.java.net/nonav/releases/1.0/spec/index.html">section 3.2</a>
63   * of the JAX-RS specification.
64   * <p>
65   * An {@link ExceptionMapper} may be configured to map this class or a sub-class
66   * of to customize responses for parameter-based errors.
67   * <p>
68   * Unless otherwise stated all such exceptions of this type will contain a
69   * response with a 400 (Client error) status code.
70   *
71   * @author Paul Sandoz
72   * @author Marek Potociar (marek.potociar at oracle.com)
73   */
74  public abstract class ParamException extends WebApplicationException {

78      /**
79       * An abstract parameter exception for the class of URI-parameter-based
80       * exceptions.
81       * <p>
82       * All such exceptions of this type will contain a response with a 404
83       * (Not Found) status code.
84       */
85      public abstract static class UriParamException extends ParamException {
86  
87          private static final long serialVersionUID = 44233528459885541L;
88  
89          protected UriParamException(Throwable cause,
90                                      Class<? extends Annotation> parameterType, String name, String defaultStringValue) {
91              super(cause, Response.Status.NOT_FOUND, parameterType, name, defaultStringValue);
92          }
93      }
