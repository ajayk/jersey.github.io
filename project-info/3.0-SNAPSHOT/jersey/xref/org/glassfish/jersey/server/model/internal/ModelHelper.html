41  package org.glassfish.jersey.server.model.internal;
42  
43  import javax.ws.rs.Path;
44  
45  /**
46   * Common model helper methods.
47   *
48   * @author Michal Gajdos
49   * @author Constantino Cronemberger (ccronemberger at yahoo.com.br)
50   */
51  public final class ModelHelper {
52  
53      /**
54       * Get the class in the provided resource class ancestor hierarchy that
55       * is actually annotated with the {@link javax.ws.rs.Path &#64;Path} annotation.
56       *
57       * @param resourceClass resource class.
58       * @return resource class or it's ancestor that is annotated with the {@link javax.ws.rs.Path &#64;Path}
59       *         annotation.
60       */
61      public static Class<?> getAnnotatedResourceClass(final Class<?> resourceClass) {
62  
63          Class<?> foundInterface = null;
64  
65          // traverse the class hierarchy to find the annotation
66          // According to specification, annotation in the super-classes must take precedence over annotation in the
67          // implemented interfaces
68          Class<?> cls = resourceClass;
69          do {
70              if (cls.isAnnotationPresent(Path.class)) {
71                  return cls;
72              }
73  
74              // if no annotation found on the class currently traversed, check for annotation in the interfaces on this
75              // level - if not already previously found
76              if (foundInterface == null) {
77                  for (final Class<?> i : cls.getInterfaces()) {
78                      if (i.isAnnotationPresent(Path.class)) {
79                          // store the interface reference in case no annotation will be found in the super-classes
80                          foundInterface = i;
81                          break;
82                      }
83                  }
84              }
85          } while ((cls = cls.getSuperclass()) != null);
86  
87          if (foundInterface != null) {
88              return foundInterface;
89          }
90  
91          return resourceClass;
92      }
93  
94      /**
95       * Prevent instantiation.
96       */
97      private ModelHelper() {
98          throw new AssertionError("Instantiation not allowed.");
99      }
100 }
