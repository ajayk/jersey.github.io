58  /**
59   * A class providing access to information stored in a {@link ResourceDocType}.<br>
60   * Created on: Jun 16, 2008<br>
61   *
62   * @author Martin Grotzke (martin.grotzke at freiheit.com)
63   */
64  public class ResourceDocAccessor {
65  
66      private static final Logger LOGGER = Logger.getLogger(ResourceDocAccessor.class.getName());
67  
68      private ResourceDocType _resourceDoc;
69  
70      public ResourceDocAccessor(ResourceDocType resourceDoc) {
71          _resourceDoc = resourceDoc;
72      }
73  
74      public ClassDocType getClassDoc(Class<?> resourceClass) {
75          if (resourceClass == null) {
76              return null;
77          }
78  
79          for (ClassDocType classDocType : _resourceDoc.getDocs()) {
80              if (resourceClass.getName().equals(classDocType.getClassName())) {
81                  return classDocType;
82              }
83          }
84          return null;
85      }
86  
87      public MethodDocType getMethodDoc(Class<?> resourceClass, Method method) {
88          if (resourceClass == null || method == null) {
89              return null;
90          }
91  
92          final ClassDocType classDoc = getClassDoc(resourceClass);
93          if (classDoc == null) {
94              return null;
95          }
96  
97          MethodDocType candidate = null;
98          int candidateCount = 0;
99  
100         final String methodName = method.getName();
101         final String methodSignature = computeSignature(method);
102         for (MethodDocType methodDocType : classDoc.getMethodDocs()) {
103             if (methodName.equals(methodDocType.getMethodName())) {
104                 candidateCount++;
105                 if (candidate == null) {
106                     candidate = methodDocType;
107                 }
108                 final String docMethodSignature = methodDocType.getMethodSignature();
109                 if (docMethodSignature != null && docMethodSignature.equals(methodSignature)) {
110                     return methodDocType;
111                 }
112             }
113         }
114 
115         if (candidate != null && candidateCount > 1 && LOGGER.isLoggable(Level.CONFIG)) {
116             LOGGER.config(LocalizationMessages.WADL_RESOURCEDOC_AMBIGUOUS_METHOD_ENTRIES(
117                     resourceClass.getName(),
118                     methodName,
119                     methodSignature,
120                     candidateCount));
121         }
122 
123         return candidate;
124     }
125 
126     private String computeSignature(final Method method) {
127         final String methodAsString = method.toGenericString();
128         return methodAsString.substring(methodAsString.indexOf('('), methodAsString.lastIndexOf(')') + 1);
129     }
130 
131     /**
132      * @param resourceClass
133      * @param method
134      * @param p
135      * @return param doc type
136      */
137     public ParamDocType getParamDoc(Class<?> resourceClass, Method method,
138                                     Parameter p) {
139         final MethodDocType methodDoc = getMethodDoc(resourceClass, method);
140         if (methodDoc != null) {
141             for (ParamDocType paramDocType : methodDoc.getParamDocs()) {
142                 for (AnnotationDocType annotationDocType : paramDocType.getAnnotationDocs()) {
143                     final Class<? extends Annotation> annotationType = p.getSourceAnnotation().annotationType();
144                     if (annotationType != null) {
145                         final String sourceName = getSourceName(annotationDocType);
146                         if (sourceName != null && sourceName.equals(p.getSourceName())) {
147                             return paramDocType;
148                         }
149                     }
150                 }
151             }
152         }
153         return null;
154     }
155 
156     public RepresentationDocType getRequestRepresentation(Class<?> resourceClass, Method method, String mediaType) {
157         if (mediaType == null) {
158             return null;
159         }
160         final MethodDocType methodDoc = getMethodDoc(resourceClass, method);
161         return methodDoc != null
162                 && methodDoc.getRequestDoc() != null
163                 && methodDoc.getRequestDoc().getRepresentationDoc() != null
164                 // && mediaType.equals( methodDoc.getRequestDoc().getRepresentationDoc().getMediaType() )
165                 ? methodDoc.getRequestDoc().getRepresentationDoc() : null;
166     }
167 
168     public ResponseDocType getResponse(Class<?> resourceClass, Method method) {
169         final MethodDocType methodDoc = getMethodDoc(resourceClass, method);
170         return methodDoc != null && methodDoc.getResponseDoc() != null
171                 ? methodDoc.getResponseDoc() : null;
172     }
173 
174     private String getSourceName(AnnotationDocType annotationDocType) {
175         if (annotationDocType.hasAttributeDocs()) {
176             for (NamedValueType namedValueType : annotationDocType.getAttributeDocs()) {
177                 /* the value of the "value"-attribute is the param.sourceName...
178                  */
179                 if ("value".equals(namedValueType.getName())) {
180                     return namedValueType.getValue();
181                 }
182             }
183         }
184         return null;
185     }
