65  /**
66   * Abstract resource method dispatcher that provides skeleton implementation of
67   * dispatching requests to a particular {@link Method Java method} using supplied
68   * {@link InvocationHandler Java method invocation handler}.
69   *
70   * @author Marek Potociar (marek.potociar at oracle.com)
71   */
72  abstract class AbstractJavaResourceMethodDispatcher implements ResourceMethodDispatcher {
73  
74      private final Method method;
75      private final InvocationHandler methodHandler;
76      private final Invocable resourceMethod;
77      private final ConfiguredValidator validator;
78  
79      /**
80       * Initialize common java resource method dispatcher structures.
81       *
82       * @param resourceMethod invocable resource class Java method.
83       * @param methodHandler  method invocation handler.
84       * @param validator      input/output parameter validator.
85       */
86      AbstractJavaResourceMethodDispatcher(final Invocable resourceMethod,
87                                           final InvocationHandler methodHandler,
88                                           final ConfiguredValidator validator) {
89          this.method = resourceMethod.getDefinitionMethod();
90          this.methodHandler = methodHandler;
91          this.resourceMethod = resourceMethod;
92          this.validator = validator;
93      }
94  
95      @Override
96      public final Response dispatch(Object resource, ContainerRequest request) throws ProcessingException {
97          Response response = null;
98          try {
99              response = doDispatch(resource, request);
100         } finally {
101             TracingLogger.getInstance(request).log(ServerTraceEvent.DISPATCH_RESPONSE, response);
102         }
103         return response;
104     }
105 
106     /**
107      * Dispatching functionality to be implemented by a concrete dispatcher
108      * implementation sub-class.
109      *
110      * @param resource resource class instance.
111      * @param request  request to be dispatched.
112      * @return response for the dispatched request.
113      * @throws ProcessingException in case of a processing error.
114      * @see ResourceMethodDispatcher#dispatch(Object, org.glassfish.jersey.server.ContainerRequest)
115      */
116     protected abstract Response doDispatch(Object resource, ContainerRequest request) throws ProcessingException;
117 
118     /**
119      * Use the underlying invocation handler to invoke the underlying Java method
120      * with the supplied input method argument values on a given resource instance.
121      *
122      * @param containerRequest container request.
123      * @param resource         resource class instance.
124      * @param args             input argument values for the invoked Java method.
125      * @return invocation result.
126      * @throws ProcessingException (possibly {@link MappableException mappable})
127      *                             container exception in case the invocation failed.
128      */
129     final Object invoke(final ContainerRequest containerRequest, final Object resource, final Object... args)
130             throws ProcessingException {
131         try {
132             // Validate resource class & method input parameters.
133             if (validator != null) {
134                 validator.validateResourceAndInputParams(resource, resourceMethod, args);
135             }
136 
137             final PrivilegedAction invokeMethodAction = new PrivilegedAction() {
138                 @Override
139                 public Object run() {
140                     final TracingLogger tracingLogger = TracingLogger.getInstance(containerRequest);
141                     final long timestamp = tracingLogger.timestamp(ServerTraceEvent.METHOD_INVOKE);
142                     try {
143 
144                         return methodHandler.invoke(resource, method, args);
145 
146                     } catch (IllegalAccessException | IllegalArgumentException | UndeclaredThrowableException ex) {
147                         throw new ProcessingException(LocalizationMessages.ERROR_RESOURCE_JAVA_METHOD_INVOCATION(), ex);
148                     } catch (InvocationTargetException ex) {
149                         throw mapTargetToRuntimeEx(ex.getCause());
150                     } catch (Throwable t) {
151                         throw new ProcessingException(t);
152                     } finally {
153                         tracingLogger.logDuration(ServerTraceEvent.METHOD_INVOKE, timestamp, resource, method);
154                     }
155                 }
156             };
157 
158             final SecurityContext securityContext = containerRequest.getSecurityContext();
159 
160             final Object invocationResult = (securityContext instanceof SubjectSecurityContext)
161                     ? ((SubjectSecurityContext) securityContext).doAsSubject(invokeMethodAction) : invokeMethodAction.run();
162 
163             // Validate response entity.
164             if (validator != null) {
165                 validator.validateResult(resource, resourceMethod, invocationResult);
166             }
167 
168             return invocationResult;
169         } catch (ValidationException ex) { // handle validation exceptions -> potentially mappable
170             throw new MappableException(ex);
171         }
172     }
173 
174     private static RuntimeException mapTargetToRuntimeEx(Throwable throwable) {
175         if (throwable instanceof WebApplicationException) {
176             return (WebApplicationException) throwable;
177         }
178         // handle all exceptions as potentially mappable (incl. ProcessingException)
179         return new MappableException(throwable);
180     }
181 
182     @Override
183     public String toString() {
184         return method.toString();
185     }
