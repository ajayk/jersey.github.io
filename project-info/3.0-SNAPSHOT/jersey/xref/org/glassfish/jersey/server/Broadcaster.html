41  package org.glassfish.jersey.server;
42  
43  import java.io.IOException;
44  import java.util.Iterator;
45  import java.util.concurrent.ConcurrentLinkedQueue;
46  import java.util.concurrent.CopyOnWriteArrayList;
47  import java.util.logging.Level;
48  import java.util.logging.Logger;
49  
50  import org.glassfish.jersey.server.internal.LocalizationMessages;
51  
52  /**
53   * Used for broadcasting response chunks to multiple {@link ChunkedOutput} instances.
54   *
55   * @param <T> broadcast type.
56   * @author Pavel Bucek (pavel.bucek at oracle.com)
57   * @author Martin Matula
58   */
59  public class Broadcaster<T> implements BroadcasterListener<T> {
60      // We do not expect large amounts of broadcaster listeners additions/removals, but large amounts of traversals.
61      private final CopyOnWriteArrayList<BroadcasterListener<T>> listeners =
62              new CopyOnWriteArrayList<BroadcasterListener<T>>();
63  
64      private final ConcurrentLinkedQueue<ChunkedOutput<T>> chunkedOutputs =
65              new ConcurrentLinkedQueue<ChunkedOutput<T>>();
66  
67      /**
68       * Creates a new instance.
69       * If this constructor is called by a subclass, it assumes the the reason for the subclass to exist is to implement
70       * {@link #onClose(ChunkedOutput)} and {@link #onException(ChunkedOutput, Exception)} methods, so it adds
71       * the newly created instance as the listener. To avoid this, subclasses may call {@link #Broadcaster(Class)}
72       * passing their class as an argument.
73       */
74      public Broadcaster() {
75          this(Broadcaster.class);
76      }
77  
78      /**
79       * Can be used by subclasses to override the default functionality of adding self to the set of
80       * {@link BroadcasterListener listeners}. If creating a direct instance of a subclass passed in the parameter,
81       * the broadcaster will not register itself as a listener.
82       *
83       * @param subclass subclass of Broadcaster that should not be registered as a listener - if creating a direct instance
84       *                 of this subclass, this constructor will not register the new instance as a listener.
85       * @see #Broadcaster()
86       */
87      protected Broadcaster(final Class<? extends Broadcaster> subclass) {
88          if (subclass != getClass()) {
89              listeners.add(this);
90          }
91      }
92  
93      /**
94       * Register {@link ChunkedOutput} to this {@code Broadcaster} instance.
95       *
96       * @param chunkedOutput {@link ChunkedOutput} to register.
97       * @return {@code true} if the instance was successfully registered, {@code false} otherwise.
98       */
99      public <OUT extends ChunkedOutput<T>> boolean add(final OUT chunkedOutput) {
100         return chunkedOutputs.offer(chunkedOutput);
101     }
102 
103     /**
104      * Un-register {@link ChunkedOutput} from this {@code Broadcaster} instance.
105      *
106      * This method does not close the {@link ChunkedOutput} being unregistered.
107      *
108      * @param chunkedOutput {@link ChunkedOutput} instance to un-register from this broadcaster.
109      * @return {@code true} if the instance was unregistered, {@code false} otherwise.
110      */
111     public <OUT extends ChunkedOutput<T>> boolean remove(final OUT chunkedOutput) {
112         return chunkedOutputs.remove(chunkedOutput);
113     }
114 
115     /**
116      * Register {@link BroadcasterListener} for {@code Broadcaster} events listening.
117      * <p>
118      * This operation is potentially slow, especially if large number of listeners get registered in the broadcaster.
119      * The {@code Broadcaster} implementation is optimized to efficiently handle small amounts of
120      * concurrent listener registrations and removals and large amounts of registered listener notifications.
121      * </p>
122      *
123      * @param listener listener to be registered.
124      * @return {@code true} if registered, {@code false} otherwise.
125      */
126     public boolean add(final BroadcasterListener<T> listener) {
127         return listeners.add(listener);
128     }
129 
130     /**
131      * Un-register {@link BroadcasterListener}.
132      * <p>
133      * This operation is potentially slow, especially if large number of listeners get registered in the broadcaster.
134      * The {@code Broadcaster} implementation is optimized to efficiently handle small amounts of
135      * concurrent listener registrations and removals and large amounts of registered listener notifications.
136      * </p>
137      *
138      * @param listener listener to be unregistered.
139      * @return {@code true} if unregistered, {@code false} otherwise.
140      */
141     public boolean remove(final BroadcasterListener<T> listener) {
142         return listeners.remove(listener);
143     }
144 
145     /**
146      * Broadcast a chunk to all registered {@link ChunkedOutput} instances.
147      *
148      * @param chunk chunk to be sent.
149      */
150     public void broadcast(final T chunk) {
151         forEachOutput(new Task<ChunkedOutput<T>>() {
152             @Override
153             public void run(final ChunkedOutput<T> cr) throws IOException {
154                 cr.write(chunk);
155             }
156         });
157     }
158 
159     /**
160      * Close all registered {@link ChunkedOutput} instances.
161      */
162     public void closeAll() {
163         forEachOutput(new Task<ChunkedOutput<T>>() {
164             @Override
165             public void run(final ChunkedOutput<T> cr) throws IOException {
166                 cr.close();
167             }
168         });
169     }
170 
171     /**
172      * {@inheritDoc}
173      *
174      * Can be implemented by subclasses to handle the event of exception thrown from a particular {@link ChunkedOutput}
175      * instance when trying to write to it or close it.
176      *
177      * @param chunkedOutput instance that threw exception.
178      * @param exception     exception that was thrown.
179      */
180     @Override
181     public void onException(final ChunkedOutput<T> chunkedOutput, final Exception exception) {
182     }
183 
184     /**
185      * {@inheritDoc}
186      *
187      * Can be implemented by subclasses to handle the event of {@link ChunkedOutput} being closed.
188      *
189      * @param chunkedOutput instance that was closed.
190      */
191     @Override
192     public void onClose(final ChunkedOutput<T> chunkedOutput) {
193     }
194 
195     private static interface Task<T> {
196         void run(T parameter) throws IOException;
197     }
198 
199     private void forEachOutput(final Task<ChunkedOutput<T>> t) {
200         for (Iterator<ChunkedOutput<T>> iterator = chunkedOutputs.iterator(); iterator.hasNext(); ) {
201             ChunkedOutput<T> chunkedOutput = iterator.next();
202             if (!chunkedOutput.isClosed()) {
203                 try {
204                     t.run(chunkedOutput);
205                 } catch (Exception e) {
206                     fireOnException(chunkedOutput, e);
207                 }
208             }
209             if (chunkedOutput.isClosed()) {
210                 iterator.remove();
211                 fireOnClose(chunkedOutput);
212             }
213         }
214     }
215 
216     private void forEachListener(final Task<BroadcasterListener<T>> t) {
217         for (BroadcasterListener<T> listener : listeners) {
218             try {
219                 t.run(listener);
220             } catch (Exception e) {
221                 // log, but don't break
222                 Logger.getLogger(Broadcaster.class.getName()).log(Level.WARNING,
223                         LocalizationMessages.BROADCASTER_LISTENER_EXCEPTION(e.getClass().getSimpleName()), e);
224             }
225         }
226     }
227 
228     private void fireOnException(final ChunkedOutput<T> chunkedOutput, final Exception exception) {
229         forEachListener(new Task<BroadcasterListener<T>>() {
230             @Override
231             public void run(BroadcasterListener<T> parameter) throws IOException {
232                 parameter.onException(chunkedOutput, exception);
233             }
234         });
235     }
236 
237     private void fireOnClose(final ChunkedOutput<T> chunkedOutput) {
238         forEachListener(new Task<BroadcasterListener<T>>() {
239             @Override
240             public void run(BroadcasterListener<T> parameter) throws IOException {
241                 parameter.onClose(chunkedOutput);
242             }
243         });
244     }
