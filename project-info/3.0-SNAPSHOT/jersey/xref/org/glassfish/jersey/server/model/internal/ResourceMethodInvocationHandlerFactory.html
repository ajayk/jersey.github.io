40  package org.glassfish.jersey.server.model.internal;

42  import java.lang.reflect.InvocationHandler;
43  import java.lang.reflect.InvocationTargetException;
44  import java.lang.reflect.Method;
45  import java.util.Set;
46  import java.util.logging.Level;
47  import java.util.logging.Logger;
48  
49  import javax.inject.Inject;
50  import javax.inject.Singleton;
51  
52  import org.glassfish.jersey.internal.inject.Providers;
53  import org.glassfish.jersey.server.internal.LocalizationMessages;
54  import org.glassfish.jersey.server.model.Invocable;
55  import org.glassfish.jersey.server.spi.internal.ResourceMethodInvocationHandlerProvider;
56  
57  import org.glassfish.hk2.api.ServiceLocator;

59  /**
60   * An injectable {@link ResourceMethodInvocationHandlerProvider resource method
61   * invocation handler provider} factory.
62   * <p />
63   * When invoked, the factory iterates over the registered custom {@link ResourceMethodInvocationHandlerProvider
64   * resource method invocation handler providers} invoking their
65   * {@link ResourceMethodInvocationHandlerProvider#create(org.glassfish.jersey.server.model.Invocable) createPatternFor(...)}
66   * methods and returns the first non-null {@link InvocationHandler
67   * invocation handler} instance retrieved from the providers. If no custom providers
68   * are available, or if none of the providers returns a non-null invocation handler,
69   * in such case a default invocation handler provided by the factory is returned.
70   *
71   * @author Marek Potociar (marek.potociar at oracle.com)
72   */
73  @Singleton
74  public final class ResourceMethodInvocationHandlerFactory implements ResourceMethodInvocationHandlerProvider {
75  
76      private static final InvocationHandler DEFAULT_HANDLER = new InvocationHandler() {
77  
78          @Override
79          public Object invoke(Object target, Method method, Object[] args)
80                  throws IllegalAccessException, IllegalArgumentException, InvocationTargetException {
81              return method.invoke(target, args);
82          }
83      };
84      private static final Logger LOGGER = Logger.getLogger(ResourceMethodInvocationHandlerFactory.class.getName());
85      private final Set<ResourceMethodInvocationHandlerProvider> providers;
86  
87      @Inject
88      ResourceMethodInvocationHandlerFactory(ServiceLocator locator) {
89          providers = Providers.getProviders(locator, ResourceMethodInvocationHandlerProvider.class);
90      }
91  
92      // ResourceMethodInvocationHandlerProvider
93      @Override
94      public InvocationHandler create(Invocable resourceMethod) {
95          for (ResourceMethodInvocationHandlerProvider provider : providers) {
96              try {
97                  InvocationHandler handler = provider.create(resourceMethod);
98                  if (handler != null) {
99                      return handler;
100                 }
101             } catch (Exception e) {
102                 LOGGER.log(Level.SEVERE, LocalizationMessages.ERROR_PROCESSING_METHOD(
103                         resourceMethod,
104                         provider.getClass().getName()), e);
105             }
106         }
107 
108         return DEFAULT_HANDLER;
109     }
