177      * delegate is not available.
178      *
179      * @param name the message header.
180      * @return the message header value. If the message header is not present then
181      * {@code null} is returned. If the message header is present but has no

183      * more than once then the values of joined together and separated by a ','

190     /**
191      * Get a single typed header value.
192      *
193      * @param <T>         header value type.
194      * @param name        header name.
195      * @param valueType   header value class.
196      * @param converter   from string conversion function. Is expected to throw {@link ProcessingException}
197      *                    if conversion fails.
198      * @param convertNull if {@code true} this method calls the provided converter even for {@code null}. Otherwise this
199      *                    method returns the {@code null} without calling the converter.
200      * @return value of the header, or (possibly converted) {@code null} if not present.
201      */
202     private <T> T singleHeader(String name, Class<T> valueType, Function<String, T> converter, boolean convertNull) {
203         final List<Object> values = headers.get(name);
204 
205         if (values == null || values.isEmpty()) {
206             return convertNull ? converter.apply(null) : null;
207         }
208         if (values.size() > 1) {
209             throw new HeaderValueException(LocalizationMessages.TOO_MANY_HEADER_VALUES(name, values.toString()),
210                     HeaderValueException.Context.OUTBOUND);
211         }
212 
213         Object value = values.get(0);
214         if (value == null) {
215             return convertNull ? converter.apply(null) : null;
216         }
217 
218         if (valueType.isInstance(value)) {
219             return valueType.cast(value);
220         } else {
221             try {
222                 return converter.apply(HeaderUtils.asString(value, null));
223             } catch (ProcessingException ex) {
224                 throw exception(name, value, ex);
225             }
226         }
227     }
228 
229     private static HeaderValueException exception(final String headerName, Object headerValue, Exception e) {
230         return new HeaderValueException(LocalizationMessages.UNABLE_TO_PARSE_HEADER_VALUE(headerName, headerValue), e,
231                 HeaderValueException.Context.OUTBOUND);
232     }

294     /**
295      * Get a list of media types that are acceptable for the message.
296      *
297      * @return a read-only list of requested message media types sorted according
298      * to their q-value, with highest preference first.
299      */
300     @SuppressWarnings("unchecked")
301     public List<MediaType> getAcceptableMediaTypes() {
302         final List<Object> values = headers.get(HttpHeaders.ACCEPT);
303 
304         if (values == null || values.isEmpty()) {
305             return WILDCARD_ACCEPTABLE_TYPE_SINGLETON_LIST;
306         }
307         final List<MediaType> result = new ArrayList<MediaType>(values.size());
308         final RuntimeDelegate rd = RuntimeDelegate.getInstance();
309         boolean conversionApplied = false;
310         for (final Object value : values) {
311             try {
312                 if (value instanceof MediaType) {
313                     final AcceptableMediaType _value = AcceptableMediaType.valueOf((MediaType) value);
314                     conversionApplied = _value != value; // true if value was not an instance of AcceptableMediaType already
315                     result.add(_value);
316                 } else {
317                     conversionApplied = true;
318                     result.addAll(HttpHeaderReader.readAcceptMediaType(HeaderUtils.asString(value, rd)));
319                 }
320             } catch (java.text.ParseException e) {
321                 throw exception(HttpHeaders.ACCEPT, value, e);
322             }
323         }
324 
325         if (conversionApplied) {
326             // cache converted
327             headers.put(HttpHeaders.ACCEPT, Lists.transform(result, new Function<MediaType, Object>() {
328                 @Override
329                 public Object apply(MediaType input) {
330                     return input;
331                 }
332             }));
333         }
334 
335         return Collections.unmodifiableList(result);
336     }

338     /**
339      * Get a list of languages that are acceptable for the message.
340      *
341      * @return a read-only list of acceptable languages sorted according
342      * to their q-value, with highest preference first.
343      */
344     public List<Locale> getAcceptableLanguages() {
345         final List<Object> values = headers.get(HttpHeaders.ACCEPT_LANGUAGE);
346 
347         if (values == null || values.isEmpty()) {
348             return Collections.singletonList(new AcceptableLanguageTag("*", null).getAsLocale());
349         }
350 
351         final List<Locale> result = new ArrayList<Locale>(values.size());
352         final RuntimeDelegate rd = RuntimeDelegate.getInstance();
353         boolean conversionApplied = false;
354         for (final Object value : values) {
355             if (value instanceof Locale) {
356                 result.add((Locale) value);
357             } else {
358                 conversionApplied = true;
359                 try {
360                     result.addAll(Lists.transform(HttpHeaderReader.readAcceptLanguage(HeaderUtils.asString(value, rd)),
361                             new Function<AcceptableLanguageTag, Locale>() {
362 
363                                 @Override
364                                 public Locale apply(AcceptableLanguageTag input) {
365                                     return input.getAsLocale();
366                                 }
367                             }));
368                 } catch (java.text.ParseException e) {
369                     throw exception(HttpHeaders.ACCEPT_LANGUAGE, value, e);
370                 }
371             }
372         }
373 
374         if (conversionApplied) {
375             // cache converted
376             headers.put(HttpHeaders.ACCEPT_LANGUAGE, Lists.transform(result, new Function<Locale, Object>() {
377                 @Override
378                 public Object apply(Locale input) {
379                     return input;
380                 }
381             }));
382         }
383 
384         return Collections.unmodifiableList(result);
385     }

387     /**
388      * Get any cookies that accompanied the message.
389      *
390      * @return a read-only map of cookie name (String) to {@link javax.ws.rs.core.Cookie}.
391      */
392     public Map<String, Cookie> getRequestCookies() {
393         final List<Object> cookies = headers.get(HttpHeaders.COOKIE);
394         if (cookies == null || cookies.isEmpty()) {
395             return Collections.emptyMap();
396         }
397 
398         Map<String, Cookie> result = new HashMap<String, Cookie>();
399         for (String cookie : HeaderUtils.asStringList(cookies, RuntimeDelegate.getInstance())) {
400             if (cookie != null) {
401                 result.putAll(HttpHeaderReader.readCookies(cookie));
402             }
403         }
404         return result;
405     }

409      *
410      * @return the allowed HTTP methods, all methods will returned as upper case

445      * Get any new cookies set on the message message.
446      *
447      * @return a read-only map of cookie name (String) to a {@link javax.ws.rs.core.NewCookie new cookie}.
448      */
449     public Map<String, NewCookie> getResponseCookies() {
450         List<Object> cookies = headers.get(HttpHeaders.SET_COOKIE);
451         if (cookies == null || cookies.isEmpty()) {
452             return Collections.emptyMap();
453         }
454 
455         Map<String, NewCookie> result = new HashMap<String, NewCookie>();
456         for (String cookie : HeaderUtils.asStringList(cookies, RuntimeDelegate.getInstance())) {
457             if (cookie != null) {
458                 NewCookie newCookie = HttpHeaderReader.readNewCookie(cookie);
459                 result.put(newCookie.getName(), newCookie);
460             }
461         }
462         return result;
463     }

519     /**
520      * Get the links attached to the message as header.
521      *
522      * @return links, may return empty {@link java.util.Set} if no links are present. Never
523      * returns {@code null}.
524      */
525     public Set<Link> getLinks() {
526         List<Object> values = headers.get(HttpHeaders.LINK);
527         if (values == null || values.isEmpty()) {
528             return Collections.emptySet();
529         }
530 
531         final Set<Link> result = new HashSet<Link>(values.size());
532         final RuntimeDelegate rd = RuntimeDelegate.getInstance();
533         boolean conversionApplied = false;
534         for (final Object value : values) {
535             if (value instanceof Link) {
536                 result.add((Link) value);
537             } else {
538                 conversionApplied = true;
539                 try {
540                     result.add(Link.valueOf(HeaderUtils.asString(value, rd)));
541                 } catch (IllegalArgumentException e) {
542                     throw exception(HttpHeaders.LINK, value, e);
543                 }
544             }
545         }
546 
547         if (conversionApplied) {
548             // cache converted
549             headers.put(HttpHeaders.LINK, new ArrayList<Object>(Collections2.transform(result, new Function<Link, Object>() {
550                 @Override
551                 public Object apply(Link input) {
552                     return input;
553                 }
554             })));
555         }
556 
557         return Collections.unmodifiableSet(result);
558     }

560     /**
561      * Check if link for relation exists.
562      *
563      * @param relation link relation.
564      * @return {@code true} if the for the relation link exists, {@code false}
565      * otherwise.
566      */
567     public boolean hasLink(String relation) {
568         for (Link link : getLinks()) {
569             List<String> relations = LinkProvider.getLinkRelations(link.getRel());
570             if (relations != null && relations.contains(relation)) {
571                 return true;
572             }
573         }
574         return false;
575     }

577     /**
578      * Get the link for the relation.
579      *
580      * @param relation link relation.
581      * @return the link for the relation, otherwise {@code null} if not present.
582      */
583     public Link getLink(String relation) {
584         for (Link link : getLinks()) {
585             List<String> relations = LinkProvider.getLinkRelations(link.getRel());
586             if (relations != null && relations.contains(relation)) {
587                 return link;
588             }
589         }
590         return null;
591     }

831     /**
832      * Closes the context. Flushes and closes the entity stream.
833      */
834     public void close() {
835         if (hasEntity()) {
836             try {
837                 final OutputStream es = getEntityStream();
838                 es.flush();
839                 es.close();
840             } catch (IOException e) {
841                 // Happens when the client closed connection before receiving the full response.
842                 // This is OK and not interesting in vast majority of the cases
843                 // hence the log level set to FINE to make sure it does not flood the log unnecessarily
844                 // (especially for clients disconnecting from SSE listening, which is very common).
845                 Logger.getLogger(OutboundMessageContext.class.getName()).log(Level.FINE, e.getMessage(), e);
846             } finally {
847                 // In case some of the output stream wrapper does not delegate close() call we
848                 // close the root stream manually to make sure it commits the data.
849                 if (!committingOutputStream.isClosed()) {
850                     try {
851                         committingOutputStream.close();
852                     } catch (IOException e) {
853                         // Just log the exception
854                         Logger.getLogger(OutboundMessageContext.class.getName()).log(Level.FINE, e.getMessage(), e);
855                     }
856                 }
857             }
858         }
