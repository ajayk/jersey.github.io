40  package org.glassfish.jersey.message.internal;
41  
42  import java.io.IOException;
43  import java.io.OutputStream;
44  import java.lang.annotation.Annotation;
45  import java.lang.reflect.Type;
46  
47  import javax.ws.rs.Produces;
48  import javax.ws.rs.core.MediaType;
49  import javax.ws.rs.core.MultivaluedMap;
50  import javax.ws.rs.core.StreamingOutput;
51  import javax.ws.rs.ext.MessageBodyWriter;
52  
53  import javax.inject.Singleton;
54  
55  /**
56   * Message body writer that supports {@link StreamingOutput streaming output} marshalling.
57   *
58   * @author Paul Sandoz
59   */
60  @Produces({"application/octet-stream", "*/*"})
61  @Singleton
62  public final class StreamingOutputProvider implements MessageBodyWriter<StreamingOutput> {
63  
64      @Override
65      public boolean isWriteable(Class<?> t, Type gt, Annotation[] as, MediaType mediaType) {
66          return StreamingOutput.class.isAssignableFrom(t);
67      }
68  
69      @Override
70      public long getSize(StreamingOutput o, Class<?> type, Type genericType, Annotation[] annotations, MediaType mediaType) {
71          return -1;
72      }
73  
74      @Override
75      public void writeTo(StreamingOutput o, Class<?> t, Type gt, Annotation[] as,
76              MediaType mediaType, MultivaluedMap<String, Object> httpHeaders,
77              OutputStream entity) throws IOException {
78          o.write(entity);
79      }
80  }
