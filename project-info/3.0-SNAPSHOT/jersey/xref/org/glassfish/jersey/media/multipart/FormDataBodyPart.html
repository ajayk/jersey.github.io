51  /**
52   * Subclass of {@link BodyPart} with specialized support for media type
53   * {@code multipart/form-data}.  See
54   * <a href="http://www.ietf.org/rfc/rfc2388.txt">RFC 2388</a>
55   * for the formal definition of this media type.
56   * <p/>
57   * For a server side application wishing to process an incoming
58   * {@code multipart/form-data} message, the following features
59   * are provided:
60   * <ul>
61   * <li>Property accessor to retrieve the control name.</li>
62   * <li>Property accessor to retrieve the field value for a simple
63   * String field.</li>
64   * <li>Convenience accessor to retrieve the field value after conversion
65   * through an appropriate {@code MessageBodyReader}.</li>
66   * </ul>
67   * <p/>
68   * For a client side application wishing to construct an outgoing
69   * {@code multipart/form-data} message, the following features
70   * are provided:
71   * <ul>
72   * <li>Convenience constructors for named fields with either
73   * simple string values, or arbitrary entities and media types.</li>
74   * <li>Property accessor to set the control name.</li>
75   * <li>Property accessor to set the field value for a simple
76   * String field.</li>
77   * <li>Convenience accessor to set the media type and value of a
78   * "file" field.</li>
79   * </ul>
80   *
81   * @author Craig McClanahan
82   * @author Imran M Yousuf (imran at smartitengineering.com)
83   * @author Paul Sandoz
84   * @author Michal Gajdos
85   */

206     /**
207      * Overrides the behaviour on {@link BodyPart} to ensure that
208      * only instances of {@link FormDataContentDisposition} can be obtained.
209      *
210      * @return the content disposition.
211      * @throws IllegalArgumentException if the content disposition header cannot be parsed.
212      */
213     @Override
214     public ContentDisposition getContentDisposition() {
215         if (contentDisposition == null) {
216             String scd = getHeaders().getFirst("Content-Disposition");
217             if (scd != null) {
218                 try {
219                     contentDisposition = new FormDataContentDisposition(scd, fileNameFix);
220                 } catch (ParseException ex) {
221                     throw new IllegalArgumentException(LocalizationMessages.ERROR_PARSING_CONTENT_DISPOSITION(scd), ex);
222                 }
223             }
224         }
225         return contentDisposition;
226     }
227 
228     /**
229      * Overrides the behaviour on {@link BodyPart} to ensure that
230      * only instances of {@link FormDataContentDisposition} can be set.
231      *
232      * @param contentDisposition the content disposition which must be an instance of {@link FormDataContentDisposition}.
233      * @throws IllegalArgumentException if the content disposition is not an instance of {@link FormDataContentDisposition}.
234      */
235     @Override
236     public void setContentDisposition(ContentDisposition contentDisposition) {
237         if (contentDisposition instanceof FormDataContentDisposition) {
238             super.setContentDisposition(contentDisposition);
239         } else {
240             throw new IllegalArgumentException();
241         }
242     }

249     public String getName() {

256     }

258     /**
259      * Sets the control name.
260      *
261      * @param name the control name.
262      */
263     public void setName(String name) {
264         if (name == null) {
265             throw new IllegalArgumentException(LocalizationMessages.CONTROL_NAME_CANNOT_BE_NULL());
266         }
267 
268         if (getFormDataContentDisposition() == null) {
269             FormDataContentDisposition contentDisposition;
270             contentDisposition = FormDataContentDisposition.name(name).build();
271             super.setContentDisposition(contentDisposition);
272         } else {
273             FormDataContentDisposition formDataContentDisposition = FormDataContentDisposition.name(name)
274                     .fileName(contentDisposition.getFileName())
275                     .creationDate(contentDisposition.getCreationDate())
276                     .modificationDate(contentDisposition.getModificationDate())
277                     .readDate(contentDisposition.getReadDate())
278                     .size(contentDisposition.getSize()).build();
279             super.setContentDisposition(formDataContentDisposition);
280         }
281     }

283     /**
284      * Gets the field value for this body part. This should be called only on body parts representing simple field values.
285      *
286      * @return the simple field value.
287      * @throws ProcessingException if an IO error arises during reading the value.
288      * @throws IllegalStateException if called on a body part with a media type other than {@code text/plain}
289      */
290     public String getValue() {
291         if (!MediaTypes.typeEqual(MediaType.TEXT_PLAIN_TYPE, getMediaType())) {
292             throw new IllegalStateException(LocalizationMessages.MEDIA_TYPE_NOT_TEXT_PLAIN());
293         }
294 
295         if (getEntity() instanceof BodyPartEntity) {
296             return getValueAs(String.class);
297         } else {
298             return (String) getEntity();
299         }
300     }

302     /**
303      * Gets the field value after appropriate conversion to the requested type. This is useful only when the containing {@link
304      * FormDataMultiPart} instance has been received, which causes the {@code providers} property to have been set.
305      *
306      * @param <T> the type of the field value.
307      * @param clazz Desired class into which the field value should be converted.
308      * @return the field value.
309      * @throws ProcessingException if an IO error arises during reading an entity.
310      * @throws IllegalArgumentException if no {@code MessageBodyReader} can be found to perform the requested conversion.
311      * @throws IllegalStateException if this method is called when the {@code providers} property has not been set or when
312      * the entity instance is not the unconverted content of the body part entity.
313      */
314     public <T> T getValueAs(Class<T> clazz) {
315         return getEntityAs(clazz);
316     }
317 
318     /**
319      * Sets the field value for this body part. This should be called
320      * only on body parts representing simple field values.
321      *
322      * @param value the field value.
323      * @throws IllegalStateException if called on a body part with a media type other than {@code text/plain}.
324      */
325     public void setValue(String value) {
326         if (!MediaType.TEXT_PLAIN_TYPE.equals(getMediaType())) {
327             throw new IllegalStateException(LocalizationMessages.MEDIA_TYPE_NOT_TEXT_PLAIN());
328         }
329         setEntity(value);
330     }

332     /**
333      * Sets the field media type and value for this body part.
334      *

336      * @param value the field value as a Java object.
337      */
338     public void setValue(MediaType mediaType, Object value) {
339         setMediaType(mediaType);
340         setEntity(value);
341     }

343     /**
344      * @return {@code true} if this body part represents a simple, string-based, field value, otherwise {@code false}.

348     }
