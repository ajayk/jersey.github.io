64  /**
65   * Client-side single inbound Server-Sent Event reader.
66   *
67   * @author Marek Potociar (marek.potociar at oracle.com)
68   */
69  @ConstrainedTo(RuntimeType.CLIENT)
70  class InboundEventReader implements MessageBodyReader<InboundEvent> {
71  
72      private static final Logger LOGGER = Logger.getLogger(InboundEventReader.class.getName());
73      private static final byte[] EOL_DATA = new byte[] {'\n'};
74  
75      @Inject
76      private Provider<MessageBodyWorkers> messageBodyWorkers;
77  
78      private enum State {
79          SKIPPING_PREPENDED_EMPTY_EVENTS,
80          NEW_LINE,
81          COMMENT,
82          FIELD,
83      }
84  
85      @Override
86      public boolean isReadable(final Class<?> type, final Type genericType, final Annotation[] annotations,
87                                final MediaType mediaType) {
88          return InboundEvent.class.equals(type) && SseFeature.SERVER_SENT_EVENTS_TYPE.isCompatible(mediaType);
89      }
90  
91      @Override
92      public InboundEvent readFrom(final Class<InboundEvent> type,
93                                   final Type genericType,
94                                   final Annotation[] annotations,
95                                   final MediaType mediaType,
96                                   final MultivaluedMap<String, String> headers,
97                                   final InputStream entityStream) throws IOException, WebApplicationException {
98          /**
99           * SSE Event parsing based on:
100          *
101          * http://dev.w3.org/html5/eventsource/
102          * last editors draft from 13 March 2012
103          */
104         final ByteArrayOutputStream tokenData = new ByteArrayOutputStream();
105         final String charsetName = MessageUtils.getCharset(mediaType).name();
106         final InboundEvent.Builder eventBuilder =
107                 new InboundEvent.Builder(messageBodyWorkers.get(), annotations, mediaType, headers);
108 
109         int b = -1;
110         State currentState = State.SKIPPING_PREPENDED_EMPTY_EVENTS;
111         loop:
112         do {
113             switch (currentState) {
114                 /* There is a problem with the SSE event parsing, because Jersey uses ChunkedInput to separate events.
115                    The problem is that ChunkedInput uses fixed character string as a separator, which is \r\n\r\n when it
116                    parses SSE.
117                    The problem is that SSE events are separated only by \r\n and \r\n also works as an end of a field inside
118                    the event, so the fixed separator \r\n\r\n  only works if the server does not send empty events.
119                    For example:
120 
121                     event: e1\r\n
122                     data: d1\r\n
123                     \r\n
124                     \r\n
125                     event: e2\r\n
126                     data: d2\r\n
127                     \r\n
128 
129                     is a stream of <e1> <empty event> <e2>
130                     Unfortunately the ChunkedInput parser will parse it only into 2 events <e1> and <e2> and <e2> will have
131                     \r\n (an empty event) prepended at the beginning. This is not fixable on the ChunkedInput parser level,
132                     which is not SSE aware, so this InboundEventReader must be aware of this and skip any prepended empty events.
133                     Also as a result Jersey will not deliver empty events to the user. */
134                 case SKIPPING_PREPENDED_EMPTY_EVENTS:
135                 case NEW_LINE:
136                     if (b == '\r') {
137                         // read next byte in case of CRLF delimiter
138                         b = entityStream.read();
139                         b = b == '\n' ? entityStream.read() : b;
140                     } else {
141                         b = entityStream.read();
142                     }
143 
144                     if (b == '\n' || b == '\r' || b == -1) {
145                         if (currentState == State.SKIPPING_PREPENDED_EMPTY_EVENTS) {
146                             break;
147                         }
148 
149                         break loop;
150                     }
151 
152                     if (b == ':') {
153                         currentState = State.COMMENT;
154                     } else {
155                         tokenData.write(b);
156                         currentState = State.FIELD;
157                     }
158                     break;
159                 case COMMENT:
160                     // skipping comment data
161                     b = readLineUntil(entityStream, '\n', tokenData);
162                     final String commentLine = tokenData.toString(charsetName);
163                     tokenData.reset();
164                     eventBuilder.commentLine(commentLine.trim());
165                     currentState = State.NEW_LINE;
166                     break;
167                 case FIELD:
168                     // read field name
169                     b = readLineUntil(entityStream, ':', tokenData);
170                     final String fieldName = tokenData.toString(charsetName);
171                     tokenData.reset();
172 
173                     if (b == ':') {
174                         do {
175                             b = entityStream.read();
176                         } while (b == ' ');
177 
178                         if (b != '\n' && b != '\r' && b != -1) {
179                             tokenData.write(b);
180                             b = readLineUntil(entityStream, '\n', tokenData);
181                         }
182                     }
183 
184                     processField(eventBuilder, fieldName, mediaType, tokenData.toByteArray());
185                     tokenData.reset();
186 
187                     currentState = State.NEW_LINE;
188                     break;
189             }
190         } while (b != -1);
191 
192         return eventBuilder.build();
193     }
194 
195     /**
196      * Read input stream until a delimiter or {@code EOL ('\n')} or {@code EOF} is reached
197      * and write the read data to the supplied output stream if not {@code null}, or discard
198      * the data if the output stream is {@code null}.
199      *
200      * @param in        input stream to be read.
201      * @param delimiter delimiter to break the read (apart from {@code EOL ('\n', '\r')} or {@code EOF}).
202      * @param out       output stream to write the read data to. May be {@code null}, in which case the
203      *                  read data are silently discarded.
204      * @return value of the last byte read.
205      * @throws IOException in case the reading or writing of the data failed.
206      */
207     private int readLineUntil(final InputStream in, final int delimiter, final OutputStream out) throws IOException {
208         int b;
209         while ((b = in.read()) != -1) {
210             if (b == delimiter || b == '\n' || b == '\r') {
211                 break;
212             } else if (out != null) {
213                 out.write(b);
214             }
215         }
216 
217         return b;
218     }
219 
220     private void processField(final InboundEvent.Builder inboundEventBuilder, final String name,
221                               final MediaType mediaType, final byte[] value) {
222         final String valueString = new String(value, MessageUtils.getCharset(mediaType));
223         if ("event".equals(name)) {
224             inboundEventBuilder.name(valueString);
225         } else if ("data".equals(name)) {
226             inboundEventBuilder.write(value);
227             inboundEventBuilder.write(EOL_DATA);
228         } else if ("id".equals(name)) {
229             inboundEventBuilder.id(valueString);
230         } else if ("retry".equals(name)) {
231             try {
232                 inboundEventBuilder.reconnectDelay(Long.parseLong(valueString));
233             } catch (final NumberFormatException ex) {
234                 LOGGER.log(Level.FINE, LocalizationMessages.IN_EVENT_RETRY_PARSE_ERROR(valueString), ex);
235             }
236         } else {
237             LOGGER.fine(LocalizationMessages.IN_EVENT_FIELD_NOT_RECOGNIZED(name, valueString));
238         }
239     }
