78      private static final Logger LOGGER = Logger.getLogger(SecureSaxParserFactory.class.getName());
79      private static final EntityResolver EMPTY_ENTITY_RESOLVER = new EntityResolver() {
80          @Override
81          public InputSource resolveEntity(String publicId, String systemId) {
82              return new InputSource(new ByteArrayInputStream(new byte[0]));
83          }
84      };

86      private static final String EXTERNAL_GENERAL_ENTITIES_FEATURE = "http://xml.org/sax/features/external-general-entities";
87      private static final String EXTERNAL_PARAMETER_ENTITIES_FEATURE = "http://xml.org/sax/features/external-parameter-entities";
88  
89      private final SAXParserFactory spf;
90  
91      /**
92       * Create new secure SAX parser factory wrapper.
93       *
94       * @param spf SAX parser factory.
95       */
96      public SecureSaxParserFactory(SAXParserFactory spf) {
97          this.spf = spf;
98  
99          if (SaxHelper.isXdkParserFactory(spf)) {
100             LOGGER.log(Level.WARNING, LocalizationMessages.SAX_XDK_NO_SECURITY_FEATURES());
101         } else {
102             try {
103                 spf.setFeature(EXTERNAL_GENERAL_ENTITIES_FEATURE, Boolean.FALSE);
104             } catch (Exception ex) {
105                 LOGGER.log(
106                         Level.CONFIG,
107                         LocalizationMessages.SAX_CANNOT_DISABLE_GENERAL_ENTITY_PROCESSING_FEATURE(spf.getClass()),
108                         ex);
109             }
110 
111             try {
112                 spf.setFeature(EXTERNAL_PARAMETER_ENTITIES_FEATURE, Boolean.FALSE);
113             } catch (Exception ex) {
114                 LOGGER.log(
115                         Level.CONFIG,
116                         LocalizationMessages.SAX_CANNOT_DISABLE_PARAMETER_ENTITY_PROCESSING_FEATURE(spf.getClass()),
117                         ex);
118             }
119 
120             try {
121                 spf.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, Boolean.TRUE);
122             } catch (Exception ex) {
123                 LOGGER.log(
124                         Level.CONFIG,
125                         LocalizationMessages.SAX_CANNOT_ENABLE_SECURE_PROCESSING_FEATURE(spf.getClass()),
126                         ex);
127             }
128         }
129     }
