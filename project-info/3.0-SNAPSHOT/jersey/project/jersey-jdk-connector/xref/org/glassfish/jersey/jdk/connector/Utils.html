51      /**
52       * Concatenates two buffers into one. If buffer given as first argument has enough space for putting
53       * the other one, it will be done and the original buffer will be returned. Otherwise new buffer will
54       * be created.
55       *
56       * @param buffer  first buffer.
57       * @param buffer1 second buffer.
58       * @return concatenation.
59       */
60      static ByteBuffer appendBuffers(ByteBuffer buffer, ByteBuffer buffer1, int incomingBufferSize, int BUFFER_STEP_SIZE) {
61  
62          final int limit = buffer.limit();
63          final int capacity = buffer.capacity();
64          final int remaining = buffer.remaining();
65          final int len = buffer1.remaining();
66  
67          // buffer1 will be appended to buffer
68          if (len < (capacity - limit)) {
69  
70              buffer.mark();
71              buffer.position(limit);
72              buffer.limit(capacity);
73              buffer.put(buffer1);
74              buffer.limit(limit + len);
75              buffer.reset();
76              return buffer;
77              // Remaining data is moved to left. Then new data is appended
78          } else if (remaining + len < capacity) {
79              buffer.compact();
80              buffer.put(buffer1);
81              buffer.flip();
82              return buffer;
83              // create new buffer
84          } else {
85              int newSize = remaining + len;
86              if (newSize > incomingBufferSize) {
87                  throw new IllegalArgumentException("Buffer overflow");
88              } else {
89                  final int roundedSize =
90                          (newSize % BUFFER_STEP_SIZE) > 0 ? ((newSize / BUFFER_STEP_SIZE) + 1) * BUFFER_STEP_SIZE : newSize;
91                  final ByteBuffer result = ByteBuffer.allocate(roundedSize > incomingBufferSize ? newSize : roundedSize);
92                  result.put(buffer);
93                  result.put(buffer1);
94                  result.flip();
95                  return result;
96              }
97          }
98      }

100     static ByteBuffer split(ByteBuffer buffer, int position) {
101         int bytesLength = position - buffer.position();
102         byte[] bytes = new byte[bytesLength];
103         buffer.get(bytes);
104         return ByteBuffer.wrap(bytes);
105     }

107     static int getPort(URI uri) {
108         if (uri.getPort() != -1) {
109             return uri.getPort();
110         }
