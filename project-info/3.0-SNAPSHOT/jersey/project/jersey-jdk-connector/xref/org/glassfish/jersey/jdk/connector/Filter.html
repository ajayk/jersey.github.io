41  package org.glassfish.jersey.jdk.connector;

43  import java.net.SocketAddress;

45  /**
46   * A filter can add functionality to JDK client transport. Filters are composed together to
47   * create JDK client transport.
48   *
49   * @author Petr Janouch (petr.janouch at oracle.com)
50   * @author Pavel Bucek (pavel.bucek at oracle.com)
51   */
52  class Filter<UP_IN, UP_OUT, DOWN_OUT, DOWN_IN> {
53  
54      protected volatile Filter<?, ?, UP_IN, UP_OUT> upstreamFilter = null;
55      protected final Filter<DOWN_OUT, DOWN_IN, ?, ?> downstreamFilter;
56  
57      /**
58       * Constructor.
59       *
60       * @param downstreamFilter downstream filter. Accessible directly as {@link #downstreamFilter} protected field.
61       */
62      Filter(Filter<DOWN_OUT, DOWN_IN, ?, ?> downstreamFilter) {
63          this.downstreamFilter = downstreamFilter;
64      }
65  
66      /**
67       * Perform write operation for this filter and invokes write method on the next filter in the filter chain.
68       *
69       * @param data              on which write operation is performed.
70       * @param completionHandler will be invoked when the write operation is completed or has failed.
71       */
72      void write(UP_IN data, CompletionHandler<UP_IN> completionHandler) {
73      }
74  
75      /**
76       * Close the filter, invokes close operation on the next filter in the filter chain.
77       * <p/>
78       * The filter is expected to clean up any allocated resources and pass the invocation to downstream filter.
79       */
80      void close() {
81          if (downstreamFilter != null) {
82              downstreamFilter.close();
83          }
84      }
85  
86      /**
87       * Signal to turn on SSL, it is passed on in the filter chain until a filter responsible for SSL is reached.
88       */
89      void startSsl() {
90          if (downstreamFilter != null) {
91              downstreamFilter.startSsl();
92          }
93      }
94  
95      /**
96       * Initiate connect.
97       * <p/>
98       * If the {@link Filter} needs to do something during this phase, it must implement {@link
99       * #handleConnect(java.net.SocketAddress, Filter)}
100      * method.
101      *
102      * @param address        an address where to connect (server or proxy).
103      * @param upstreamFilter a filter positioned upstream.
104      */
105     void connect(SocketAddress address, Filter<?, ?, UP_IN, UP_OUT> upstreamFilter) {
106         this.upstreamFilter = upstreamFilter;
107 
108         handleConnect(address, upstreamFilter);
109 
110         if (downstreamFilter != null) {
111             downstreamFilter.connect(address, this);
112         }
113     }
114 
115     /**
116      * An event listener that is called when a connection is set up.
117      * This event travels up in the filter chain.
118      * <p/>
119      * If the {@link Filter} needs to process this event, it must implement {@link #processConnect()} method.
120      */
121     void onConnect() {
122         processConnect();
123 
124         if (upstreamFilter != null) {
125             upstreamFilter.onConnect();
126         }
127     }
128 
129     /**
130      * An event listener that is called when some data is read.
131      * <p/>
132      * If the {@link Filter} needs to process this event, it must implement {@link #onRead(Object)} ()} method.
133      * If the method returns {@code true}, the processing will continue with upstream filters; if the method invocation
134      * returns {@code false}, the processing won't continue.
135      *
136      * @param data that has been read.
137      */
138     @SuppressWarnings("unchecked")
139     final void onRead(DOWN_IN data) {
140         if (processRead(data)) {
141             if (upstreamFilter != null) {
142                 UP_OUT _data;
143                 try {
144                     _data = (UP_OUT) data;
145                 } catch (Exception e) {
146                     throw new IllegalStateException("Cannot pass message of different type from filter input to filter output");
147                 }
148                 upstreamFilter.onRead(_data);
149             }
150         }
151     }
152 
153     /**
154      * An event listener that is called when the connection is closed by the peer.
155      * <p/>
156      * If the {@link Filter} needs to process this event, it must implement {@link #processConnectionClosed()} method.
157      */
158     final void onConnectionClosed() {
159         processConnectionClosed();
160 
161         if (upstreamFilter != null) {
162             upstreamFilter.onConnectionClosed();
163         }
164     }
165 
166     /**
167      * An event listener that is called, when SSL completes its handshake.
168      * <p/>
169      * If the {@link Filter} needs to process this event, it must implement {@link #processSslHandshakeCompleted()} method.
170      */
171     final void onSslHandshakeCompleted() {
172         processSslHandshakeCompleted();
173 
174         if (upstreamFilter != null) {
175             upstreamFilter.onSslHandshakeCompleted();
176         }
177     }
178 
179     /**
180      * An event listener that is called when an error has occurred.
181      * <p/>
182      * Errors travel in direction from downstream filter to upstream filter.
183      * <p/>
184      * If the {@link Filter} needs to process this event, it must implement {@link #processError(Throwable)} method.
185      *
186      * @param t an error that has occurred.
187      */
188     final void onError(Throwable t) {
189         processError(t);
190 
191         if (upstreamFilter != null) {
192             upstreamFilter.onError(t);
193         }
194     }

204     }

211     void processConnect() {
212     }

215      * Process {@link #onRead(Object)}.

217      * @param data read data.
218      * @return {@code true} if the data should be sent to processing to upper filter in the chain, {@code false} otherwise.
219      * @see #onRead(Object).
220      */
221     boolean processRead(DOWN_IN data) {
222         return true;
223     }

226      * Process {@link #onConnectionClosed()}.

247     void processError(Throwable t) {
