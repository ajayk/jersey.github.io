214             }

218     /**
219      * CDI producer for CDI bean constructor String parameters, that should be injected by JAX-RS.

221     @ApplicationScoped
222     public static class JaxRsParamProducer {
223 
224         @Qualifier
225         @Retention(RUNTIME)
226         @Target({METHOD, FIELD, PARAMETER, TYPE})
227         public static @interface JaxRsParamQualifier {
228         }
229 
230         private static final JaxRsParamQualifier JaxRsParamQUALIFIER = new JaxRsParamQualifier() {
231 
232             @Override
233             public Class<? extends Annotation> annotationType() {
234                 return JaxRsParamQualifier.class;
235             }
236         };
237 
238         static final Set<Class<? extends Annotation>> JAX_RS_STRING_PARAM_ANNOTATIONS =
239                 new HashSet<Class<? extends Annotation>>() {{
240                     add(javax.ws.rs.PathParam.class);
241                     add(javax.ws.rs.QueryParam.class);
242                     add(javax.ws.rs.CookieParam.class);
243                     add(javax.ws.rs.HeaderParam.class);
244                     add(javax.ws.rs.MatrixParam.class);
245                     add(javax.ws.rs.FormParam.class);
246                 }};
247 
248         /**
249          * Internal cache to store CDI {@link InjectionPoint} to Jersey {@link Parameter} mapping.
250          */
251         final Cache<InjectionPoint, Parameter> parameterCache = new Cache<>(new Computable<InjectionPoint, Parameter>() {
252 
253             @Override
254             public Parameter compute(final InjectionPoint injectionPoint) {
255                 final Annotated annotated = injectionPoint.getAnnotated();
256                 final Class<?> clazz = injectionPoint.getMember().getDeclaringClass();
257 
258                 if (annotated instanceof AnnotatedParameter) {
259 
260                     final AnnotatedParameter annotatedParameter = (AnnotatedParameter) annotated;
261                     final AnnotatedCallable callable = annotatedParameter.getDeclaringCallable();
262 
263                     if (callable instanceof AnnotatedConstructor) {
264 
265                         final AnnotatedConstructor ac = (AnnotatedConstructor) callable;
266                         final int position = annotatedParameter.getPosition();
267                         final List<Parameter> parameters = Parameter.create(clazz, clazz, ac.getJavaMember(), false);
268 
269                         return parameters.get(position);
270                     }
271                 }
272 
273                 return null;
274             }
275         });
276 
277         /**
278          * Provide a value for given injection point. If the injection point does not refer
279          * to a CDI bean constructor parameter, or the value could not be found, the method will return null.
280          *
281          * @param injectionPoint actual injection point.
282          * @param beanManager    current application bean manager.
283          * @return concrete JAX-RS parameter value for given injection point.
284          */
285         @javax.enterprise.inject.Produces
286         @JaxRsParamQualifier
287         public String getParameterValue(final InjectionPoint injectionPoint, final BeanManager beanManager) {
288             final Parameter parameter = parameterCache.compute(injectionPoint);
289 
290             if (parameter != null) {
291                 final ServiceLocator locator = beanManager.getExtension(CdiComponentProvider.class).getEffectiveLocator();
292                 final Set<ValueFactoryProvider> providers = Providers.getProviders(locator, ValueFactoryProvider.class);
293 
294                 for (final ValueFactoryProvider vfp : providers) {
295                     final Factory<?> valueFactory = vfp.getValueFactory(parameter);
296                     if (valueFactory != null) {
297                         return (String) valueFactory.provide();
298                     }
299                 }
300             }
301 
302             return null;
303         }
304     }

306     @Override
307     public boolean bind(final Class<?> clazz, final Set<Class<?>> providerContracts) {
308         if (LOGGER.isLoggable(Level.FINE)) {
309             LOGGER.fine(LocalizationMessages.CDI_CLASS_BEING_CHECKED(clazz));
310         }
311 
312         if (beanManager == null) {
313             return false;
314         }
315 
316         if (isJerseyOrDependencyType(clazz)) {
317             return false;
318         }
319 
320         final boolean isCdiManaged = isCdiComponent(clazz);
321         final boolean isManagedBean = isManagedBean(clazz);
322         final boolean isJaxRsComponent = isJaxRsComponentType(clazz);
323 
324         if (!isCdiManaged && !isManagedBean && !isJaxRsComponent) {
325             return false;
326         }
327 
328         final boolean isJaxRsResource = jaxRsResourceCache.compute(clazz);
329 
330         final DynamicConfiguration dc = Injections.getConfiguration(locator);
331 
332         final Class<? extends Annotation> beanScopeAnnotation = CdiUtil.getBeanScope(clazz, beanManager);
333         final boolean isRequestScoped = beanScopeAnnotation == RequestScoped.class
334                                         || (beanScopeAnnotation == Dependent.class && isJaxRsResource);
335 
336         Factory beanFactory = isRequestScoped
337                 ? new RequestScopedCdiBeanHk2Factory(clazz, locator, beanManager, isCdiManaged)
338                 : new GenericCdiBeanHk2Factory(clazz, locator, beanManager, isCdiManaged);
339 
340         final ServiceBindingBuilder bindingBuilder =
341                     Injections.newFactoryBinder(beanFactory);
342 
343         bindingBuilder.to(clazz);
344         for (final Class contract : providerContracts) {
345             bindingBuilder.to(contract);
346         }
347 
348         Injections.addBinding(bindingBuilder, dc);
349 
350         dc.commit();
351 
352         if (isRequestScoped) {
353             requestScopedComponents.add(clazz);
354         }
355 
356         if (LOGGER.isLoggable(Level.CONFIG)) {
357             LOGGER.config(LocalizationMessages.CDI_CLASS_BOUND_WITH_CDI(clazz));
358         }
359 
360         return true;
361     }
362 
363     @Override
364     public void done() {
365         if (requestScopedComponents.size() > 0) {
366             final DynamicConfiguration dc = Injections.getConfiguration(locator);
367             Injections.addBinding(Injections.newBinder(new ForeignRequestScopeBridge() {
368                 @Override
369                 public Set<Class<?>> getRequestScopedComponents() {
370                     return requestScopedComponents;
371                 }
372             }).to(ForeignRequestScopeBridge.class), dc);
373             dc.commit();
374             if (LOGGER.isLoggable(Level.CONFIG)) {
375                 LOGGER.config(LocalizationMessages.CDI_REQUEST_SCOPED_COMPONENTS_RECOGNIZED(
376                         listElements(new StringBuilder().append("\n"), requestScopedComponents).toString()));
377             }
378         }
379     }

390     private static AnnotatedConstructor<?> enrichedConstructor(final AnnotatedConstructor<?> ctor) {
391         return new AnnotatedConstructor(){
392 
393             @Override
394             public Constructor getJavaMember() {
395                 return ctor.getJavaMember();
396             }
397 
398             @Override
399             public List<AnnotatedParameter> getParameters() {
400                 final List<AnnotatedParameter> parameters = new ArrayList<>(ctor.getParameters().size());
401 
402                 for (final AnnotatedParameter<?> ap : ctor.getParameters()) {
403                     parameters.add(new AnnotatedParameter() {
404 
405                         @Override
406                         public int getPosition() {
407                             return ap.getPosition();
408                         }
409 
410                         @Override
411                         public AnnotatedCallable getDeclaringCallable() {
412                             return ap.getDeclaringCallable();
413                         }
414 
415                         @Override
416                         public Type getBaseType() {
417                             return ap.getBaseType();
418                         }
419 
420                         @Override
421                         public Set<Type> getTypeClosure() {
422                             return ap.getTypeClosure();
423                         }
424 
425                         @Override
426                         public <T extends Annotation> T getAnnotation(final Class<T> annotationType) {
427                             if (annotationType == JaxRsParamProducer.JaxRsParamQualifier.class) {
428                                 return hasAnnotation(ap, JaxRsParamProducer.JAX_RS_STRING_PARAM_ANNOTATIONS)
429                                         ? (T) JaxRsParamProducer.JaxRsParamQUALIFIER : null;
430                             } else {
431                                 return ap.getAnnotation(annotationType);
432                             }
433                         }
434 
435                         @Override
436                         public Set<Annotation> getAnnotations() {
437                             final Set<Annotation> result = new HashSet<>();
438                             for (final Annotation a : ap.getAnnotations()) {
439                                 result.add(a);
440                                 final Class<? extends Annotation> annotationType = a.annotationType();
441                                 if (JaxRsParamProducer.JAX_RS_STRING_PARAM_ANNOTATIONS.contains(annotationType)) {
442                                     result.add(JaxRsParamProducer.JaxRsParamQUALIFIER);
443                                 }
444                             }
445                             return result;
446                         }
447 
448                         @Override
449                         public boolean isAnnotationPresent(final Class<? extends Annotation> annotationType) {
450                             return (annotationType == JaxRsParamProducer.JaxRsParamQualifier.class
451                                             && hasAnnotation(ap, JaxRsParamProducer.JAX_RS_STRING_PARAM_ANNOTATIONS))
452                                     || ap.isAnnotationPresent(annotationType);
453                         }
454                     });
455                 }
456                 return parameters;
457             }
458 
459             @Override
460             public boolean isStatic() {
461                 return ctor.isStatic();
462             }
463 
464             @Override
465             public AnnotatedType getDeclaringType() {
466                 return ctor.getDeclaringType();
467             }
468 
469             @Override
470             public Type getBaseType() {
471                 return ctor.getBaseType();
472             }
473 
474             @Override
475             public Set<Type> getTypeClosure() {
476                 return ctor.getTypeClosure();
477             }
478 
479             @Override
480             public <T extends Annotation> T getAnnotation(final Class<T> annotationType) {
481                 return ctor.getAnnotation(annotationType);
482             }
483 
484             @Override
485             public Set<Annotation> getAnnotations() {
486                 return ctor.getAnnotations();
487             }
488 
489             @Override
490             public boolean isAnnotationPresent(final Class<? extends Annotation> annotationType) {
491                 return ctor.isAnnotationPresent(annotationType);
492             }
493         };
494     }

496     @SuppressWarnings("unused")
497     private void processAnnotatedType(@Observes
498     // We can not apply the following constraint
499     // if we want to fully support {@link org.glassfish.jersey.ext.cdi1x.spi.Hk2CustomBoundTypesProvider}.
500     // Covered by tests/integration/cdi-with-jersey-injection-custom-cfg-webapp test application:
501 //                                      @WithAnnotations({
502 //                                              Context.class,
503 //                                              ApplicationPath.class,
504 //                                              HeaderParam.class,
505 //                                              QueryParam.class,
506 //                                              FormParam.class,
507 //                                              MatrixParam.class,
508 //                                              BeanParam.class,
509 //                                              PathParam.class})
510                                       final ProcessAnnotatedType processAnnotatedType) {
511         final AnnotatedType<?> annotatedType = processAnnotatedType.getAnnotatedType();
512 
513         // if one of the JAX-RS annotations is present in the currently seen class, add it to the "whitelist"
514         if (containsJaxRsConstructorInjection(annotatedType)
515                 || containsJaxRsFieldInjection(annotatedType)
516                 || containsJaxRsMethodInjection(annotatedType)) {
517             jaxrsInjectableTypes.add(annotatedType.getBaseType());
518         }
519 
520         if (customHk2TypesProvider != null) {
521             final Type baseType = annotatedType.getBaseType();
522             if (customHk2TypesProvider.getHk2Types().contains(baseType)) {
523                 processAnnotatedType.veto();
524                 jerseyVetoedTypes.add(baseType);
525             }
526         }
527 
528         if (containsJaxRsParameterizedCtor(annotatedType)) {
529             processAnnotatedType.setAnnotatedType(new AnnotatedType() {
530 
531                 @Override
532                 public Class getJavaClass() {
533                     return annotatedType.getJavaClass();
534                 }
535 
536                 @Override
537                 public Set<AnnotatedConstructor> getConstructors() {
538                     final Set<AnnotatedConstructor> result = new HashSet<>();
539                     for (final AnnotatedConstructor c : annotatedType.getConstructors()) {
540                         result.add(enrichedConstructor(c));
541                     }
542                     return result;
543                 }
544 
545                 @Override
546                 public Set getMethods() {
547                     return annotatedType.getMethods();
548                 }
549 
550                 @Override
551                 public Set getFields() {
552                     return annotatedType.getFields();
553                 }
554 
555                 @Override
556                 public Type getBaseType() {
557                     return annotatedType.getBaseType();
558                 }
559 
560                 @Override
561                 public Set<Type> getTypeClosure() {
562                     return annotatedType.getTypeClosure();
563                 }
564 
565                 @Override
566                 public <T extends Annotation> T getAnnotation(final Class<T> annotationType) {
567                     return annotatedType.getAnnotation(annotationType);
568                 }
569 
570                 @Override
571                 public Set<Annotation> getAnnotations() {
572                     return annotatedType.getAnnotations();
573                 }
574 
575                 @Override
576                 public boolean isAnnotationPresent(final Class<? extends Annotation> annotationType) {
577                     return annotatedType.isAnnotationPresent(annotationType);
578                 }
579             });
580         }
581     }

583     private boolean containsJaxRsParameterizedCtor(final AnnotatedType annotatedType) {
584         return containAnnotatedParameters(annotatedType.getConstructors(), JaxRsParamProducer.JAX_RS_STRING_PARAM_ANNOTATIONS);
585     }

587     private boolean containsJaxRsConstructorInjection(final AnnotatedType annotatedType) {
588         return containAnnotatedParameters(annotatedType.getConstructors(), JAX_RS_INJECT_ANNOTATIONS);
589     }
590 
591     private boolean containsJaxRsMethodInjection(final AnnotatedType annotatedType) {
592         return containAnnotatedParameters(annotatedType.getMethods(), JAX_RS_INJECT_ANNOTATIONS);
593     }
594 
595     private boolean containsJaxRsFieldInjection(final AnnotatedType annotatedType) {
596         return containAnnotation(annotatedType.getFields(), JAX_RS_INJECT_ANNOTATIONS);
597     }

599     private boolean containAnnotatedParameters(final Collection<AnnotatedCallable> annotatedCallables,
600                                                final Set<Class<? extends Annotation>> annotationSet) {
601         for (final AnnotatedCallable c : annotatedCallables) {
602             if (containAnnotation(c.getParameters(), annotationSet)) {
603                 return true;
604             }
605         }
606         return false;
607     }
608 
609     private boolean containAnnotation(final Collection<Annotated> elements,
610                                       final Set<Class<? extends Annotation>> annotationSet) {
611         for (final Annotated element : elements) {
612             if (hasAnnotation(element, annotationSet)) {
613                 return true;
614             }
615         }
616         return false;
617     }
618 
619     private static boolean hasAnnotation(final Annotated element, final Set<Class<? extends Annotation>> annotations) {
620         for (final Class<? extends Annotation> a : annotations) {
621             if (element.isAnnotationPresent(a)) {
622                 return true;
623             }
624         }
625         return false;
626     }

641     @SuppressWarnings("unused")
642     private void processInjectionTarget(@Observes final ProcessInjectionTarget event) {
643         final InjectionTarget it = event.getInjectionTarget();
644         final Class<?> componentClass = event.getAnnotatedType().getJavaClass();
645 
646         final Set<InjectionPoint> cdiInjectionPoints = filterHk2InjectionPointsOut(it.getInjectionPoints());
647 
648         for (final InjectionPoint injectionPoint : cdiInjectionPoints) {
649             final Member member = injectionPoint.getMember();
650             if (member instanceof Field) {
651                 addInjecteeToSkip(componentClass, fieldsToSkip, (Field) member);
652             } else if (member instanceof Method) {
653                 addInjecteeToSkip(componentClass, methodsToSkip, (Method) member);
654             }
655         }
656

658         if (isJerseyOrDependencyType(componentClass)) {

663                     // Tell CDI to ignore Jersey (or it's dependencies) classes when injecting.
664                     // CDI will not treat these classes as CDI beans (as they are not).
665                     return Collections.emptySet();

687     private Set<InjectionPoint> filterHk2InjectionPointsOut(final Set<InjectionPoint> originalInjectionPoints) {
688         final Set<InjectionPoint> filteredInjectionPoints = new HashSet<>();
689         for (final InjectionPoint ip : originalInjectionPoints) {
690             final Type injectedType = ip.getType();
691             if (customHk2TypesProvider != null && customHk2TypesProvider.getHk2Types().contains(injectedType)) {
692                 //remember the type, we would need to mock it's CDI binding at runtime
693                 hk2ProvidedTypes.add(injectedType);
694             } else {
695                 if (injectedType instanceof Class<?>) {
696                     final Class<?> injectedClass = (Class<?>) injectedType;
697                     if (isJerseyOrDependencyType(injectedClass)) {
698                         //remember the type, we would need to mock it's CDI binding at runtime
699                         hk2ProvidedTypes.add(injectedType);
700                     } else {
701                         filteredInjectionPoints.add(ip);
702                     }
703                 } else { // it is not a class, maybe provider type?:
704                     if (isInjectionProvider(injectedType)
705                             && (isProviderOfJerseyType((ParameterizedType) injectedType))) {
706                         //remember the type, we would need to mock it's CDI binding at runtime
707                         hk2ProvidedTypes.add(((ParameterizedType) injectedType).getActualTypeArguments()[0]);
708                     } else {
709                         filteredInjectionPoints.add(ip);
710                     }
711                 }
712             }
713         }
714         return filteredInjectionPoints;
715     }

825     private static boolean isJerseyOrDependencyType(final Class<?> clazz) {
826         if (clazz.isPrimitive() || clazz.isSynthetic()) {
827             return false;
828         }
829 
830         final Package pkg = clazz.getPackage();
831         if (pkg == null) { // Class.getPackage() could return null
832             LOGGER.warning(String.format("Class %s has null package", clazz));
833             return false;
834         }
835 
836         final String pkgName = pkg.getName();
837         return !clazz.isAnnotationPresent(JerseyVetoed.class)
838                 && (pkgName.contains("org.glassfish.hk2")
839                             || pkgName.contains("jersey.repackaged")
840                             || pkgName.contains("org.jvnet.hk2")
841                             || (pkgName.startsWith("org.glassfish.jersey")
842                                         && !pkgName.startsWith("org.glassfish.jersey.examples")
843                                         && !pkgName.startsWith("org.glassfish.jersey.tests"))
844                             || (pkgName.startsWith("com.sun.jersey")
845                                         && !pkgName.startsWith("com.sun.jersey.examples")
846                                         && !pkgName.startsWith("com.sun.jersey.tests")));
847     }

849     private void bindHk2ClassAnalyzer() {
850         final ClassAnalyzer defaultClassAnalyzer =
851                 locator.getService(ClassAnalyzer.class, ClassAnalyzer.DEFAULT_IMPLEMENTATION_NAME);
852 
853         final int skippedElements = methodsToSkip.size() + fieldsToSkip.size();
854 
855         final ClassAnalyzer customizedClassAnalyzer = skippedElements > 0
856                 ? new InjecteeSkippingAnalyzer(defaultClassAnalyzer, methodsToSkip, fieldsToSkip)
857                 : defaultClassAnalyzer;
858 
859         final DynamicConfiguration dc = Injections.getConfiguration(locator);
860 
861         final ScopedBindingBuilder bindingBuilder =
862                 Injections.newBinder(customizedClassAnalyzer);
863 
864         bindingBuilder.analyzeWith(ClassAnalyzer.DEFAULT_IMPLEMENTATION_NAME)
865                 .to(ClassAnalyzer.class)
866                 .named(CDI_CLASS_ANALYZER);
867 
868         Injections.addBinding(bindingBuilder, dc);
869 
870         dc.commit();
871     }
