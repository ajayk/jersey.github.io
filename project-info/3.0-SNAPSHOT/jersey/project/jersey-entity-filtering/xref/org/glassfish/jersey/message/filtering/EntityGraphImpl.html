56  /**
57   * Default implementation of {@link EntityGraph}.
58   *
59   * @author Michal Gajdos
60   */
61  final class EntityGraphImpl implements EntityGraph {
62  
63      private final Class<?> entityClass;
64  
65      private final Set<String> globalScopes;
66      private final Set<String> localScopes;
67  
68      // <FilteringScope, FieldName>
69      private final HashMultimap<String, String> fields;
70      // <FilteringScope, FieldName, Class>
71      private final Table<String, String, Class<?>> subgraphs;
72  
73      /**
74       * Create an entity graph for given class.
75       *
76       * @param entityClass entity class the graph should be created for.
77       */
78      public EntityGraphImpl(final Class<?> entityClass) {
79          this.entityClass = entityClass;
80  
81          this.fields = HashMultimap.create();
82          this.subgraphs = HashBasedTable.create();
83  
84          this.globalScopes = Sets.newHashSet();
85          this.localScopes = Sets.newHashSet();
86      }
87  
88      @Override
89      public EntityGraphImpl addField(final String fieldName) {
90          return addField(fieldName, globalScopes);
91      }
92  
93      @Override
94      public EntityGraphImpl addField(final String fieldName, final String... filteringScopes) {
95          return addField(fieldName, Sets.newHashSet(filteringScopes));
96      }
97  
98      @Override
99      public EntityGraphImpl addField(final String fieldName, final Set<String> filteringScopes) {
100         for (final String filteringScope : filteringScopes) {
101             createFilteringScope(filteringScope);
102             fields.get(filteringScope).add(fieldName);
103         }
104 
105         return this;
106     }
107 
108     @Override
109     public EntityGraphImpl addFilteringScopes(final Set<String> filteringScopes) {
110         this.globalScopes.addAll(filteringScopes);
111         return this;
112     }
113 
114     @Override
115     public EntityGraphImpl addSubgraph(final String fieldName, final Class<?> fieldClass) {
116         return addSubgraph(fieldName, fieldClass, globalScopes);
117     }
118 
119     @Override
120     public EntityGraphImpl addSubgraph(final String fieldName, final Class<?> fieldClass, final String... filteringScopes) {
121         return addSubgraph(fieldName, fieldClass, Sets.newHashSet(filteringScopes));
122     }
123 
124     @Override
125     public EntityGraphImpl addSubgraph(final String fieldName, final Class<?> fieldClass, final Set<String> filteringScopes) {
126         for (final String filteringScope : filteringScopes) {
127             createFilteringScope(filteringScope);
128             subgraphs.put(filteringScope, fieldName, fieldClass);
129         }
130 
131         return this;
132     }
133 
134     @Override
135     public Class<?> getEntityClass() {
136         return entityClass;
137     }
138 
139     @Override
140     public Set<String> getFields(final String filteringScope) {
141         return fields.containsKey(filteringScope)
142                 ? Collections.unmodifiableSet(fields.get(filteringScope)) : Collections.<String>emptySet();
143     }
144 
145     @Override
146     public Set<String> getFields(final String... filteringScopes) {
147         return filteringScopes.length == 0 ? Collections.<String>emptySet()
148                 : (filteringScopes.length == 1 ? getFields(filteringScopes[0]) : getFields(Sets.newHashSet(filteringScopes)));
149     }
150 
151     @Override
152     public Set<String> getFields(final Set<String> filteringScopes) {
153         final Set<String> matched = Sets.newHashSet();
154 
155         for (final String filteringContext : filteringScopes) {
156             matched.addAll(fields.get(filteringContext));
157         }
158 
159         return matched;
160     }
161 
162     @Override
163     public Set<String> getFilteringScopes() {
164         return Collections.unmodifiableSet(Sets.union(globalScopes, localScopes));
165     }
166 
167     @Override
168     public Set<String> getClassFilteringScopes() {
169         return Collections.unmodifiableSet(globalScopes);
170     }
171 
172     @Override
173     public Map<String, Class<?>> getSubgraphs(final String filteringScope) {
174         return subgraphs.containsRow(filteringScope)
175                 ? Collections.unmodifiableMap(subgraphs.row(filteringScope)) : Collections.<String, Class<?>>emptyMap();
176     }
177 
178     @Override
179     public Map<String, Class<?>> getSubgraphs(final String... filteringScopes) {
180         return filteringScopes.length == 0
181                 ? Collections.<String, Class<?>>emptyMap()
182                 : (filteringScopes.length == 1
183                            ? getSubgraphs(filteringScopes[0])
184                            : getSubgraphs(Sets.newHashSet(filteringScopes)));
185     }
186 
187     @Override
188     public Map<String, Class<?>> getSubgraphs(final Set<String> filteringScopes) {
189         final Map<String, Class<?>> matched = Maps.newHashMap();
190 
191         for (final String filteringContext : filteringScopes) {
192             matched.putAll(subgraphs.row(filteringContext));
193         }
194 
195         return matched;
196     }
197 
198     @Override
199     public boolean presentInScopes(final String name) {
200         return fields.containsValue(name) || subgraphs.containsColumn(name);
201     }
202 
203     @Override
204     public boolean presentInScope(final String field, final String filteringScope) {
205         return fields.containsEntry(filteringScope, field) || subgraphs.contains(filteringScope, field);
206     }
207 
208     @Override
209     public EntityGraphImpl remove(final String fieldName) {
210         for (final String scope : getFilteringScopes()) {
211             if (fields.containsEntry(scope, fieldName)) {
212                 fields.remove(scope, fieldName);
213             }
214             if (subgraphs.containsColumn(fieldName)) {
215                 subgraphs.remove(scope, fieldName);
216             }
217         }
218         return this;
219     }
220 
221     /**
222      * Create a new entity-filtering scope based on the {@link ScopeProvider#DEFAULT_SCOPE default one}.
223      *
224      * @param filteringScope entity-filtering scope to be created.
225      */
226     private void createFilteringScope(final String filteringScope) {
227         // Do not create a scope if it already exists.
228         if (!getFilteringScopes().contains(filteringScope)) {
229             // Copy contents of default scope into the new one.
230             if (localScopes.contains(ScopeProvider.DEFAULT_SCOPE)) {
231                 fields.putAll(filteringScope, fields.get(ScopeProvider.DEFAULT_SCOPE));
232 
233                 final Map<String, Class<?>> row = subgraphs.row(ScopeProvider.DEFAULT_SCOPE);
234                 for (final Map.Entry<String, Class<?>> entry : row.entrySet()) {
235                     subgraphs.put(filteringScope, entry.getKey(), entry.getValue());
236                 }
237             }
238             localScopes.add(filteringScope);
239         }
240     }

242     @Override
243     public boolean equals(final Object o) {
244         if (this == o) {
245             return true;
246         }
247         if (o == null || getClass() != o.getClass()) {
248             return false;
249         }
250 
251         final EntityGraphImpl that = (EntityGraphImpl) o;
252 
253         return entityClass.equals(that.entityClass)
254                 && fields.equals(that.fields)
255                 && globalScopes.equals(that.globalScopes)
256                 && localScopes.equals(that.localScopes)
257                 && subgraphs.equals(that.subgraphs);
258     }
259 
260     @Override
261     public int hashCode() {
262         int result = entityClass.hashCode();
263         result = 53 * result + globalScopes.hashCode();
264         result = 53 * result + localScopes.hashCode();
265         result = 53 * result + fields.hashCode();
266         result = 53 * result + subgraphs.hashCode();
267         return result;
268     }
