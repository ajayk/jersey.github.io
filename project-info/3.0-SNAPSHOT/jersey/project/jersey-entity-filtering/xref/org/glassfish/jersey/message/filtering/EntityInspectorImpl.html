69  /**
70   * Class responsible for inspecting entity classes. This class invokes all available {@link EntityProcessor entity processors} in
71   * different {@link EntityProcessorContext contexts}.
72   *
73   * @author Michal Gajdos
74   */
75  @Singleton
76  final class EntityInspectorImpl implements EntityInspector {
77  
78      private final List<EntityProcessor> entityProcessors;
79  
80      @Inject
81      private EntityGraphProvider graphProvider;
82  
83      /**
84       * Constructor for HK2 expecting {@link ServiceLocator} to be injected.
85       *
86       * @param locator service locator to be injected.
87       */
88      @Inject
89      public EntityInspectorImpl(final ServiceLocator locator) {
90          this.entityProcessors = Lists.newArrayList(Providers.getAllProviders(locator, EntityProcessor.class,
91                  new RankedComparator<EntityProcessor>()));
92      }
93  
94      @Override
95      public void inspect(final Class<?> entityClass, final boolean forWriter) {
96          if (!graphProvider.containsEntityGraph(entityClass, forWriter)) {
97              final EntityGraph graph = graphProvider.getOrCreateEntityGraph(entityClass, forWriter);
98              final Set<Class<?>> inspect = Sets.newHashSet();
99  
100             // Class.
101             if (!inspectEntityClass(entityClass, graph, forWriter)) {
102                 // Properties.
103                 final Map<String, Method> unmatchedAccessors = inspectEntityProperties(entityClass, graph, inspect, forWriter);
104 
105                 // Setters/Getters without fields.
106                 inspectStandaloneAccessors(unmatchedAccessors, graph, forWriter);
107 
108                 // Inspect new classes.
109                 for (final Class<?> clazz : inspect) {
110                     inspect(clazz, forWriter);
111                 }
112             }
113         }
114     }
115 
116     /**
117      * Invoke available {@link EntityProcessor}s on given entity class.
118      *
119      * @param entityClass entity class to be examined.
120      * @param graph entity graph to be modified by examination.
121      * @param forWriter flag determining whether the class should be examined for reader or writer.
122      * @return {@code true} if the inspecting should be roll-backed, {@code false} otherwise.
123      */
124     private boolean inspectEntityClass(final Class<?> entityClass, final EntityGraph graph, final boolean forWriter) {
125         final EntityProcessorContextImpl context = new EntityProcessorContextImpl(
126                 forWriter ? EntityProcessorContext.Type.CLASS_WRITER : EntityProcessorContext.Type.CLASS_READER,
127                 entityClass, graph);
128 
129         for (final EntityProcessor processor : entityProcessors) {
130             final EntityProcessor.Result result = processor.process(context);
131 
132             if (EntityProcessor.Result.ROLLBACK == result) {
133                 graphProvider.getOrCreateEmptyEntityGraph(entityClass, false);
134                 return true;
135             }
136         }
137         return false;
138     }

140     /**
141      * Invoke available {@link EntityProcessor}s on fields of given entity class. Method returns a map ({@code fieldName},
142      * {@code method}) of unprocessed property accessors (getters/setters) and fills {@code inspect} set with entity classes
143      * that should be further processed.
144      *
145      * @param entityClass entity class to obtain properties to be examined.
146      * @param graph entity graph to be modified by examination.
147      * @param inspect non-null set of classes to-be-examined.
148      * @param forWriter flag determining whether the class should be examined for reader or writer.
149      * @return map of unprocessed property accessors.
150      */
151     private Map<String, Method> inspectEntityProperties(final Class<?> entityClass, final EntityGraph graph,
152                                                         final Set<Class<?>> inspect, final boolean forWriter) {
153         final Field[] fields = AccessController.doPrivileged(ReflectionHelper.getAllFieldsPA(entityClass));
154         final Map<String, Method> methods = FilteringHelper.getPropertyMethods(entityClass, forWriter);
155 
156         for (final Field field : fields) {
157             // Ignore static fields.
158             if (Modifier.isStatic(field.getModifiers())) {
159                 continue;
160             }
161 
162             final String name = field.getName();
163             final Class<?> clazz = FilteringHelper.getEntityClass(field.getGenericType());
164             final Method method = methods.remove(name);
165 
166             final EntityProcessorContextImpl context = new EntityProcessorContextImpl(
167                     forWriter ? EntityProcessorContext.Type.PROPERTY_WRITER : EntityProcessorContext.Type.PROPERTY_READER,
168                     field, method, graph);
169 
170             boolean rollback = false;
171             for (final EntityProcessor processor : entityProcessors) {
172                 final EntityProcessor.Result result = processor.process(context);
173 
174                 if (EntityProcessor.Result.ROLLBACK == result) {
175                     rollback = true;
176                     graph.remove(name);
177                     break;
178                 }
179             }
180 
181             if (!rollback && FilteringHelper.filterableEntityClass(clazz)) {
182                 inspect.add(clazz);
183             }
184         }
185 
186         return methods;
187     }
188 
189     /**
190      * Invoke available {@link EntityProcessor}s on accessors (getter/setter) that has no match in classes' fields.
191      *
192      * @param unprocessedAccessors map of unprocessed accessors.
193      * @param graph entity graph to be modified by examination.
194      * @param forWriter flag determining whether the class should be examined for reader or writer.
195      */
196     private void inspectStandaloneAccessors(final Map<String, Method> unprocessedAccessors, final EntityGraph graph,
197                                             final boolean forWriter) {
198         for (final Map.Entry<String, Method> entry : unprocessedAccessors.entrySet()) {
199             final EntityProcessorContextImpl context = new EntityProcessorContextImpl(
200                     forWriter ? EntityProcessorContext.Type.METHOD_WRITER : EntityProcessorContext.Type.METHOD_READER,
201                     entry.getValue(), graph);
202 
203             for (final EntityProcessor processor : entityProcessors) {
204                 final EntityProcessor.Result result = processor.process(context);
205 
206                 if (EntityProcessor.Result.ROLLBACK == result) {
207                     graph.remove(entry.getKey());
208                     break;
209                 }
210             }
211         }
212     }
