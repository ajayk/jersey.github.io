77  /**
78   * Entity Data Filtering provider based on Jackson JSON provider.
79   *
80   * @author Michal Gajdos
81   */
82  @Singleton
83  public final class FilteringJacksonJaxbJsonProvider extends JacksonJaxbJsonProvider {
84  
85      @Inject
86      private Provider<ObjectProvider<FilterProvider>> provider;
87  
88      @Override
89      protected JsonEndpointConfig _configForWriting(final ObjectMapper mapper, final Annotation[] annotations,
90                                                     final Class<?> defaultView) {
91          final AnnotationIntrospector customIntrospector = mapper.getSerializationConfig().getAnnotationIntrospector();
92          // Set the custom (user) introspector to be the primary one.
93          final ObjectMapper filteringMapper = mapper.setAnnotationIntrospector(AnnotationIntrospector.pair(customIntrospector,
94                  new JacksonAnnotationIntrospector() {
95                      @Override
96                      public Object findFilterId(final Annotated a) {
97                          final Object filterId = super.findFilterId(a);
98  
99                          if (filterId != null) {
100                             return filterId;
101                         }
102 
103                         if (a instanceof AnnotatedMethod) {
104                             final Method method = ((AnnotatedMethod) a).getAnnotated();
105 
106                             // Interested only in getters - trying to obtain "field" name from them.
107                             if (ReflectionHelper.isGetter(method)) {
108                                 return ReflectionHelper.getPropertyName(method);
109                             }
110                         }
111                         if (a instanceof AnnotatedField || a instanceof AnnotatedClass) {
112                             return a.getName();
113                         }
114 
115                         return null;
116                     }
117                 }));
118 
119         return super._configForWriting(filteringMapper, annotations, defaultView);
120     }
121 
122     @Override
123     public void writeTo(final Object value,
124                         final Class<?> type,
125                         final Type genericType,
126                         final Annotation[] annotations,
127                         final MediaType mediaType,
128                         final MultivaluedMap<String, Object> httpHeaders,
129                         final OutputStream entityStream) throws IOException {
130         final FilterProvider filterProvider = provider.get().getFilteringObject(genericType, true, annotations);
131         if (filterProvider != null) {
132             ObjectWriterInjector.set(new FilteringObjectWriterModifier(filterProvider, ObjectWriterInjector.getAndClear()));
133         }
134 
135         super.writeTo(value, type, genericType, annotations, mediaType, httpHeaders, entityStream);
136     }
137 
138     private static final class FilteringObjectWriterModifier extends ObjectWriterModifier {
139 
140         private final ObjectWriterModifier original;
141         private final FilterProvider filterProvider;
142 
143         private FilteringObjectWriterModifier(final FilterProvider filterProvider, final ObjectWriterModifier original) {
144             this.original = original;
145             this.filterProvider = filterProvider;
146         }
147 
148         @Override
149         public ObjectWriter modify(final EndpointConfigBase<?> endpoint,
150                                    final MultivaluedMap<String, Object> responseHeaders,
151                                    final Object valueToWrite,
152                                    final ObjectWriter w,
153                                    final JsonGenerator g) throws IOException {
154             final ObjectWriter writer = original == null ? w : original.modify(endpoint, responseHeaders, valueToWrite, w, g);
155             final FilterProvider customFilterProvider = writer.getConfig().getFilterProvider();
156 
157             // Try the custom (user) filter provider first.
158             return customFilterProvider == null
159                     ? writer.with(filterProvider)
160                     : writer.with(new FilterProvider() {
161                         @Override
162                         public BeanPropertyFilter findFilter(final Object filterId) {
163                             return customFilterProvider.findFilter(filterId);
164                         }
165 
166                         @Override
167                         public PropertyFilter findPropertyFilter(final Object filterId, final Object valueToFilter) {
168                             final PropertyFilter filter = customFilterProvider.findPropertyFilter(filterId, valueToFilter);
169                             if (filter != null) {
170                                 return filter;
171                             }
172 
173                             return filterProvider.findPropertyFilter(filterId, valueToFilter);
174                         }
175                     });
176         }
177     }
