40  package org.glassfish.jersey.servlet.async;
41  
42  import java.util.concurrent.atomic.AtomicBoolean;
43  import java.util.concurrent.atomic.AtomicReference;
44  import java.util.logging.Level;
45  import java.util.logging.Logger;
46  
47  import javax.servlet.AsyncContext;
48  import javax.servlet.http.HttpServletRequest;
49  import javax.servlet.http.HttpServletResponse;
50  
51  import org.glassfish.jersey.servlet.init.internal.LocalizationMessages;
52  import org.glassfish.jersey.servlet.spi.AsyncContextDelegate;
53  import org.glassfish.jersey.servlet.spi.AsyncContextDelegateProvider;
54  
55  /**
56   * Servlet 3.x container response writer async extension and related extension factory implementation.
57   *
58   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
59   * @author Marek Potociar (marek.potociar at oracle.com)
60   */
61  public class AsyncContextDelegateProviderImpl implements AsyncContextDelegateProvider {
62  
63      private static final Logger LOGGER = Logger.getLogger(AsyncContextDelegateProviderImpl.class.getName());
64  
65      @Override
66      public final AsyncContextDelegate createDelegate(final HttpServletRequest request, final HttpServletResponse response) {
67          return new ExtensionImpl(request, response);
68      }
69  
70      private static final class ExtensionImpl implements AsyncContextDelegate {
71  
72          private static final int NEVER_TIMEOUT_VALUE = -1;
73  
74          private final HttpServletRequest request;
75          private final HttpServletResponse response;
76          private final AtomicReference<AsyncContext> asyncContextRef;
77          private final AtomicBoolean completed;
78  
79          /**
80           * Create a Servlet 3.x {@link AsyncContextDelegate} with given {@code request} and {@code response}.
81           *
82           * @param request  request to create {@link AsyncContext} for.
83           * @param response response to create {@link AsyncContext} for.
84           */
85          private ExtensionImpl(final HttpServletRequest request, final HttpServletResponse response) {
86              this.request = request;
87              this.response = response;
88              this.asyncContextRef = new AtomicReference<>();
89              this.completed = new AtomicBoolean(false);
90          }
91  
92          @Override
93          public void suspend() throws IllegalStateException {
94              // Suspend only if not completed and not suspended before.
95              if (!completed.get() && asyncContextRef.get() == null) {
96                  asyncContextRef.set(getAsyncContext());
97              }
98          }
99  
100         private AsyncContext getAsyncContext() {
101             final AsyncContext asyncContext;
102             if (request.isAsyncStarted()) {
103                 asyncContext = request.getAsyncContext();
104                 try {
105                     asyncContext.setTimeout(NEVER_TIMEOUT_VALUE);
106                 } catch (IllegalStateException ex) {
107                     // Let's hope the time out is set properly, otherwise JAX-RS AsyncResponse time-out support
108                     // may not work as expected... At least we can log this at fine level...
109                     LOGGER.log(Level.FINE, LocalizationMessages.SERVLET_ASYNC_CONTEXT_ALREADY_STARTED(), ex);
110                 }
111             } else {
112                 asyncContext = request.startAsync(request, response);
113                 // Tell underlying asyncContext to never time out.
114                 asyncContext.setTimeout(NEVER_TIMEOUT_VALUE);
115             }
116             return asyncContext;
117         }
118 
119         @Override
120         public void complete() {
121             completed.set(true);
122 
123             final AsyncContext asyncContext = asyncContextRef.getAndSet(null);
124             if (asyncContext != null) {
125                 asyncContext.complete();
126             }
127         }
128     }
129 }
