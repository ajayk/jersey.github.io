225     private AuthCodeGrantImpl(final String authorizationUri, final String accessTokenUri, final String redirectUri,
226                               final String refreshTokenUri,

229                               final Map<String, String> accessTokenProperties,

232         this.authorizationUri = authorizationUri;

235         this.accessTokenProperties = accessTokenProperties;
236         this.refreshTokenProperties = refreshTokenProperties;

238         if (refreshTokenUri != null) {

245         this.client = configureClient(client);

247         initDefaultProperties(redirectUri, scope);

250     private Client configureClient(Client client) {
251         if (client == null) {
252             client = ClientBuilder.newClient();
253         }
254 
255         final Configuration config = client.getConfiguration();
256         if (!config.isRegistered(AuthCodeGrantImpl.DefaultTokenMessageBodyReader.class)) {
257             client.register(AuthCodeGrantImpl.DefaultTokenMessageBodyReader.class);
258         }
259         if (!config.isRegistered(JacksonFeature.class)) {
260             client.register(JacksonFeature.class);
261         }
262 
263         return client;
264     }

266     private void setDefaultProperty(final String key, final String value, final Map<String, String>... properties) {
267         if (value == null) {
268             return;
269         }

271             if (props.get(key) == null) {

279     private void initDefaultProperties(final String redirectUri, final String scope) {
280         setDefaultProperty(OAuth2Parameters.RESPONSE_TYPE, "code", authorizationProperties);
281         setDefaultProperty(OAuth2Parameters.CLIENT_ID, clientIdentifier.getClientId(), authorizationProperties,
282                 accessTokenProperties, refreshTokenProperties);
283         setDefaultProperty(OAuth2Parameters.REDIRECT_URI, redirectUri == null
284                 ? OAuth2Parameters.REDIRECT_URI_UNDEFINED : redirectUri, authorizationProperties, accessTokenProperties);
285         setDefaultProperty(OAuth2Parameters.STATE, UUID.randomUUID().toString(), authorizationProperties);
286         setDefaultProperty(OAuth2Parameters.SCOPE, scope, authorizationProperties);
287 
288         setDefaultProperty(OAuth2Parameters.CLIENT_SECRET, clientIdentifier.getClientSecret(), accessTokenProperties,
289                 refreshTokenProperties);
290         setDefaultProperty(OAuth2Parameters.GrantType.key, OAuth2Parameters.GrantType.AUTHORIZATION_CODE.name().toLowerCase(),
291                 accessTokenProperties);
292 
293         setDefaultProperty(OAuth2Parameters.GrantType.key, OAuth2Parameters.GrantType.REFRESH_TOKEN.name().toLowerCase(),
294                 refreshTokenProperties);
295     }

310     @Override
311     public String start() {
312         final UriBuilder uriBuilder = UriBuilder.fromUri(authorizationUri);
313         for (final Map.Entry<String, String> entry : authorizationProperties.entrySet()) {
314             uriBuilder.queryParam(entry.getKey(), entry.getValue());
315         }
316         return uriBuilder.build().toString();
317     }
318 
319     @Override
320     public TokenResult finish(final String authorizationCode, final String state) {
321         if (!this.authorizationProperties.get(OAuth2Parameters.STATE).equals(state)) {
322             throw new IllegalArgumentException(LocalizationMessages.ERROR_FLOW_WRONG_STATE());
323         }
324 
325         accessTokenProperties.put(OAuth2Parameters.CODE, authorizationCode);
326         final Form form = new Form();
327         for (final Map.Entry<String, String> entry : accessTokenProperties.entrySet()) {
328             form.param(entry.getKey(), entry.getValue());
329         }
330 
331         final Response response = client.target(accessTokenUri)
332                 .request(MediaType.APPLICATION_JSON_TYPE)
333                 .post(Entity.entity(form, MediaType.APPLICATION_FORM_URLENCODED_TYPE));
334 
335         if (response.getStatus() != 200) {
336             throw new ProcessingException(LocalizationMessages.ERROR_FLOW_REQUEST_ACCESS_TOKEN(response.getStatus()));
337         }
338         this.tokenResult = response.readEntity(TokenResult.class);
339         return tokenResult;
340     }
341 
342     @Override
343     public TokenResult refreshAccessToken(final String refreshToken) {
344         refreshTokenProperties.put(OAuth2Parameters.REFRESH_TOKEN, refreshToken);
345         final Form form = new Form();
346         for (final Map.Entry<String, String> entry : refreshTokenProperties.entrySet()) {
347             form.param(entry.getKey(), entry.getValue());
348         }
349 
350         final Response response = client.target(refreshTokenUri)
351                 .request(MediaType.APPLICATION_JSON_TYPE)
352                 .post(Entity.entity(form, MediaType.APPLICATION_FORM_URLENCODED_TYPE));
353 
354         if (response.getStatus() != 200) {
355             throw new ProcessingException(LocalizationMessages.ERROR_FLOW_REQUEST_REFRESH_TOKEN(response.getStatus()));
356         }
357 
358         this.tokenResult = response.readEntity(TokenResult.class);
359         return tokenResult;
360     }

375     static class DefaultTokenMessageBodyReader implements MessageBodyReader<TokenResult> {
376 
377         // Provider here prevents circular dependency error from HK2 (workers inject providers and this provider inject workers)
378         @Inject
379         private Provider<MessageBodyWorkers> workers;
380 
381         @Inject
382         private Provider<PropertiesDelegate> propertiesDelegateProvider;
383 
384         private static Iterable<ReaderInterceptor> EMPTY_INTERCEPTORS = new ArrayList<>();
385 
386         @Override
387         public boolean isReadable(final Class<?> type,
388                                   final Type genericType,
389                                   final Annotation[] annotations,
390                                   final MediaType mediaType) {
391             return type.equals(TokenResult.class);
392         }
393 
394         @Override
395         public TokenResult readFrom(final Class<TokenResult> type, final Type genericType, final Annotation[] annotations,
396                                     final MediaType mediaType, final MultivaluedMap<String, String> httpHeaders,
397                                     final InputStream entityStream) throws IOException, WebApplicationException {
398 
399             final GenericType<Map<String, Object>> mapType = new GenericType<Map<String, Object>>() {
400             };
401 
402             final Map<String, Object> map = (Map<String, Object>) workers.get().readFrom(mapType.getRawType(),
403                     mapType.getType(), annotations,
404                     mediaType, httpHeaders,
405                     propertiesDelegateProvider.get(),
406                     entityStream, EMPTY_INTERCEPTORS, false);
407 
408             return new TokenResult(map);
409         }
410     }
