41  package org.glassfish.jersey.media.sse;
42  
43  import org.glassfish.jersey.server.Broadcaster;
44  
45  /**
46   * Used for broadcasting SSE to multiple {@link EventOutput} instances.
47   *
48   * @author Pavel Bucek (pavel.bucek at oracle.com)
49   * @author Martin Matula
50   */
51  public class SseBroadcaster extends Broadcaster<OutboundEvent> {
52  
53      /**
54       * Creates a new instance.
55       * If this constructor is called by a subclass, it assumes the the reason for the subclass to exist is to implement
56       * {@link #onClose(org.glassfish.jersey.server.ChunkedOutput)} and {@link #onException(org.glassfish.jersey.server.ChunkedOutput, Exception)} methods, so it adds
57       * the newly created instance as the listener. To avoid this, subclasses may call {@link #SseBroadcaster(Class)}
58       * passing their class as an argument.
59       */
60      public SseBroadcaster() {
61          this(SseBroadcaster.class);
62      }
63  
64      /**
65       * Can be used by subclasses to override the default functionality of adding self to the set of
66       * {@link org.glassfish.jersey.server.BroadcasterListener listeners}.
67       * If creating a direct instance of a subclass passed in the parameter,
68       * the broadcaster will not register itself as a listener.
69       *
70       * @param subclass subclass of SseBroadcaster that should not be registered as a listener - if creating a direct instance
71       *                 of this subclass, this constructor will not register the new instance as a listener.
72       * @see #SseBroadcaster()
73       */
74      protected SseBroadcaster(final Class<? extends SseBroadcaster> subclass) {
75          super(subclass);
76      }
77  }
