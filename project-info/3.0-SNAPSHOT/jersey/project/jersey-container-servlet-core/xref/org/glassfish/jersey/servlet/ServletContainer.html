83  /**
84   * A {@link javax.servlet.Servlet} or {@link Filter} for deploying root resource classes.
85   * <p />
86   * The following sections make reference to initialization parameters. Unless
87   * otherwise specified the initialization parameters apply to both server
88   * and filter initialization parameters.
89   * <p />
90   * The servlet or filter may be configured to have an initialization
91   * parameter {@value ServletProperties#JAXRS_APPLICATION_CLASS}
92   * (see {@link org.glassfish.jersey.servlet.ServletProperties#JAXRS_APPLICATION_CLASS}) and whose value is a
93   * fully qualified name of a class that implements {@link javax.ws.rs.core.Application}.
94   * The class is instantiated as a singleton component
95   * managed by the runtime, and injection may be performed (the artifacts that
96   * may be injected are limited to injectable providers registered when
97   * the servlet or filter is configured).
98   * <p />
99   * If the initialization parameter {@value ServletProperties#JAXRS_APPLICATION_CLASS}
100  * is not present and a initialization parameter {@value org.glassfish.jersey.server.ServerProperties#PROVIDER_PACKAGES}
101  * is present (see {@link ServerProperties#PROVIDER_PACKAGES}) a new instance of
102  * {@link ResourceConfig} with this configuration is created. The initialization parameter
103  * {@value org.glassfish.jersey.server.ServerProperties#PROVIDER_PACKAGES} MUST be set to provide one or
104  * more package names. Each package name MUST be separated by ';'.
105  * <p />
106  * If none of the above resource configuration related initialization parameters
107  * are present a new instance of {@link ResourceConfig} with {@link WebAppResourcesScanner}
108  * is created. The initialization parameter {@value org.glassfish.jersey.server.ServerProperties#PROVIDER_CLASSPATH}
109  * is present (see {@link ServerProperties#PROVIDER_CLASSPATH}) MAY be
110  * set to provide one or more resource paths. Each path MUST be separated by ';'.
111  * If the initialization parameter is not present then the following resource
112  * paths are utilized: {@code "/WEB-INF/lib"} and {@code "/WEB-INF/classes"}.
113  * <p />
114  * All initialization parameters are added as properties of the created
115  * {@link ResourceConfig}.
116  * <p />
117  * A new {@link org.glassfish.jersey.server.ApplicationHandler} instance will be created and configured such
118  * that the following classes may be injected onto a root resource, provider
119  * and {@link javax.ws.rs.core.Application} classes using {@link javax.ws.rs.core.Context
120  * &#64;Context} annotation:
121  * {@link HttpServletRequest}, {@link HttpServletResponse},
122  * {@link ServletContext}, {@link javax.servlet.ServletConfig} and {@link WebConfig}.
123  * If this class is used as a Servlet then the {@link javax.servlet.ServletConfig} class may
124  * be injected. If this class is used as a servlet filter then the {@link FilterConfig}
125  * class may be injected. {@link WebConfig} may be injected to abstract
126  * servlet or filter deployment.
127  * <p />
128  * Persistence units that may be injected must be configured in web.xml
129  * in the normal way plus an additional servlet parameter to enable the
130  * Jersey servlet to locate them in JNDI. E.g. with the following
131  * persistence unit configuration:
132  * <pre>{@code
133  * <persistence-unit-ref>
134  *     <persistence-unit-ref-name>persistence/widget</persistence-unit-ref-name>
135  *     <persistence-unit-name>WidgetPU</persistence-unit-name>
136  * </persistence-unit-ref>
137  * }</pre>
138  * the Jersey servlet requires an additional servlet parameter as
139  * follows:
140  * <pre>{@code
141  * <init-param>
142  *     <param-name>unit:WidgetPU</param-name>
143  *     <param-value>persistence/widget</param-value>
144  * </init-param>
145  * }</pre>
146  * Given the above, Jersey will inject the {@link javax.persistence.EntityManagerFactory EntityManagerFactory} found
147  * at {@code java:comp/env/persistence/widget} in JNDI when encountering a
148  * field or parameter annotated with {@code @PersistenceUnit(unitName="WidgetPU")}.
149  *
150  * @author Paul Sandoz
151  * @author Pavel Bucek (pavel.bucek at oracle.com)
152  * @author Michal Gajdos
153  * @author Libor Kramolis (libor.kramolis at oracle.com)
154  */

197     /**
198      * Dispatches client requests to the protected
199      * {@code service} method. There's no need to
200      * override this method.
201      *
202      * @param req the {@link HttpServletRequest} object that
203      *            contains the request the client made of
204      *            the servlet
205      * @param res the {@link HttpServletResponse} object that
206      *            contains the response the servlet returns
207      *            to the client
208      * @throws IOException      if an input or output error occurs
209      *                          while the servlet is handling the
210      *                          HTTP request
211      * @throws ServletException if the HTTP request cannot
212      *                          be handled
213      * @see javax.servlet.Servlet#service
214      */
215     @Override
216     public void service(final ServletRequest req, final ServletResponse res)
217             throws ServletException, IOException {
218         final HttpServletRequest request;
219         final HttpServletResponse response;
220 
221         if (!(req instanceof HttpServletRequest && res instanceof HttpServletResponse)) {
222             throw new ServletException("non-HTTP request or response");
223         }
224 
225         request = (HttpServletRequest) req;
226         response = (HttpServletResponse) res;
227 
228         service(request, response);
229     }

232      * Receives standard HTTP requests from the public {@code service} method and dispatches
233      * them to the {@code do}<i>XXX</i> methods defined in
234      * this class. This method is an HTTP-specific version of the
235      * {@link javax.servlet.Servlet#service} method. There's no
236      * need to override this method.
237      *
238      * @param request  the {@link HttpServletRequest} object that
239      *                 contains the request the client made of
240      *                 the servlet
241      * @param response the {@link HttpServletResponse} object that
242      *                 contains the response the servlet returns
243      *                 to the client
244      * @throws IOException      if an input or output error occurs
245      *                          while the servlet is handling the
246      *                          HTTP request
247      * @throws ServletException if the HTTP request
248      *                          cannot be handled
249      * @see javax.servlet.Servlet#service
250      */
251     @Override
252     protected void service(final HttpServletRequest request, final HttpServletResponse response)
253             throws ServletException, IOException {
254         /**
255          * There is an annoying edge case where the service method is
256          * invoked for the case when the URI is equal to the deployment URL
257          * minus the '/', for example http://locahost:8080/HelloWorldWebApp
258          */
259         final String servletPath = request.getServletPath();
260         final StringBuffer requestUrl = request.getRequestURL();
261         final String requestURI = request.getRequestURI();
262 
263         //        final String pathInfo = request.getPathInfo();
264         //        final boolean checkPathInfo = pathInfo == null || pathInfo.isEmpty() || pathInfo.equals("/");
265         //        if (checkPathInfo && !request.getRequestURI().endsWith("/")) {
266         // Only do this if the last segment of the servlet path does not contain '.'
267         // This handles the case when the extension mapping is used with the servlet
268         // see issue 506
269         // This solution does not require parsing the deployment descriptor,
270         // however still leaves it broken for the very rare case if a standard path
271         // servlet mapping would include dot in the last segment (e.g. /.webresources/*)
272         // and somebody would want to hit the root resource without the trailing slash
273         //            final int i = servletPath.lastIndexOf('/');
274         //            if (servletPath.substring(i + 1).indexOf('.') < 0) {
275         // TODO (+ handle request URL with invalid characters - see the creation of absoluteUriBuilder below)
276         //                if (webComponent.getResourceConfig().getFeature(ResourceConfig.FEATURE_REDIRECT)) {
277         //                    URI l = UriBuilder.fromUri(request.getRequestURL().toString()).
278         //                            path("/").
279         //                            replaceQuery(request.getQueryString()).build();
280         //
281         //                    response.setStatus(307);
282         //                    response.setHeader("Location", l.toASCIIString());
283         //                    return;
284         //                } else {
285         //                pathInfo = "/";
286         //                requestURL.append("/");
287         //                requestURI += "/";
288         //                }
289         //            }
290         //        }
291 
292         /**
293          * The HttpServletRequest.getRequestURL() contains the complete URI
294          * minus the query and fragment components.
295          */
296         final UriBuilder absoluteUriBuilder;
297         try {
298             absoluteUriBuilder = UriBuilder.fromUri(requestUrl.toString());
299         } catch (final IllegalArgumentException iae) {
300             setResponseForInvalidUri(response, iae);
301             return;
302         }
303 
304         /**
305          * The HttpServletRequest.getPathInfo() and
306          * HttpServletRequest.getServletPath() are in decoded form.
307          *
308          * On some servlet implementations the getPathInfo() removed
309          * contiguous '/' characters. This is problematic if URIs
310          * are embedded, for example as the last path segment.
311          * We need to work around this and not use getPathInfo
312          * for the decodedPath.
313          */
314         final String decodedBasePath = request.getContextPath() + servletPath + "/";
315 
316         final String encodedBasePath = UriComponent.encode(decodedBasePath,
317                 UriComponent.Type.PATH);
318 
319         if (!decodedBasePath.equals(encodedBasePath)) {
320             throw new ProcessingException("The servlet context path and/or the "
321                     + "servlet path contain characters that are percent encoded");
322         }
323 
324         final URI baseUri;
325         final URI requestUri;
326         try {
327             baseUri = absoluteUriBuilder.replacePath(encodedBasePath).build();
328             String queryParameters = ContainerUtils.encodeUnsafeCharacters(request.getQueryString());
329             if (queryParameters == null) {
330                 queryParameters = "";
331             }
332 
333             requestUri = absoluteUriBuilder.replacePath(requestURI)
334                     .replaceQuery(queryParameters)
335                     .build();
336         } catch (final UriBuilderException | IllegalArgumentException ex) {
337             setResponseForInvalidUri(response, ex);
338             return;
339         }
340 
341         service(baseUri, requestUri, request, response);
342     }

344     private void setResponseForInvalidUri(final HttpServletResponse response, final Throwable throwable) throws IOException {
345         LOGGER.log(Level.FINER, "Error while processing request.", throwable);
346 
347         final Response.Status badRequest = Response.Status.BAD_REQUEST;
348         if (webComponent.configSetStatusOverSendError) {
349             response.reset();
350             //noinspection deprecation
351             response.setStatus(badRequest.getStatusCode(), badRequest.getReasonPhrase());
352         } else {
353             response.sendError(badRequest.getStatusCode(), badRequest.getReasonPhrase());
354         }
355     }

411     // Filter
412     @Override
413     public void init(final FilterConfig filterConfig) throws ServletException {
414         this.filterConfig = filterConfig;
415         init(new WebFilterConfig(filterConfig));
416 
417         final String regex = (String) getConfiguration().getProperty(ServletProperties.FILTER_STATIC_CONTENT_REGEX);
418         if (regex != null && !regex.isEmpty()) {
419             try {
420                 staticContentPattern = Pattern.compile(regex);
421             } catch (final PatternSyntaxException ex) {
422                 throw new ContainerException(LocalizationMessages.INIT_PARAM_REGEX_SYNTAX_INVALID(
423                         regex, ServletProperties.FILTER_STATIC_CONTENT_REGEX), ex);
424             }
425         }
426 
427         this.filterContextPath = filterConfig.getInitParameter(ServletProperties.FILTER_CONTEXT_PATH);
428         if (filterContextPath != null) {
429             if (filterContextPath.isEmpty()) {
430                 filterContextPath = null;
431             } else {
432                 if (!filterContextPath.startsWith("/")) {
433                     filterContextPath = '/' + filterContextPath;
434                 }
435                 if (filterContextPath.endsWith("/")) {
436                     filterContextPath = filterContextPath.substring(0, filterContextPath.length() - 1);
437                 }
438             }
439         }
440 
441         // get the url-pattern defined (e.g.) in the filter-mapping section of web.xml
442         final FilterUrlMappingsProvider filterUrlMappingsProvider = getFilterUrlMappingsProvider();
443         if (filterUrlMappingsProvider != null) {
444             filterUrlMappings = filterUrlMappingsProvider.getFilterUrlMappings(filterConfig);
445         }
446 
447         // we need either the url-pattern from the filter mapping (in case of Servlet 3) or specific init-param to
448         // determine the baseUri and request relative URI. If we do not have either one, the app will most likely
449         // not work (won't be accessible)
450         if (filterUrlMappings == null && filterContextPath == null) {
451             LOGGER.warning(LocalizationMessages.FILTER_CONTEXT_PATH_MISSING());
452         }
453     }

482     /**
483      * Dispatches client requests to the
484      * {@link #service(URI, URI, HttpServletRequest, HttpServletResponse)} method.
485      * <p />
486      * If the servlet path matches the regular expression declared by the
487      * property {@link ServletProperties#FILTER_STATIC_CONTENT_REGEX} then the
488      * request is forwarded to the next filter in the filter chain so that the
489      * underlying servlet engine can process the request otherwise Jersey
490      * will process the request.
491      *
492      * @param request  the {@link HttpServletRequest} object that
493      *                 contains the request the client made to
494      *                 the servlet.
495      * @param response the {@link HttpServletResponse} object that
496      *                 contains the response the servlet returns
497      *                 to the client.
498      * @param chain    the chain of filters from which the next filter can be invoked.
499      * @throws java.io.IOException            in case of an I/O error.
500      * @throws javax.servlet.ServletException in case of an error while executing the
501      *                                        filter chain.
502      */
503     public void doFilter(final HttpServletRequest request, final HttpServletResponse response, final FilterChain chain)
504             throws IOException, ServletException {
505         if (request.getAttribute("javax.servlet.include.request_uri") != null) {
506             final String includeRequestURI = (String) request.getAttribute("javax.servlet.include.request_uri");
507 
508             if (!includeRequestURI.equals(request.getRequestURI())) {
509                 doFilter(request, response, chain,
510                         includeRequestURI,
511                         (String) request.getAttribute("javax.servlet.include.servlet_path"),
512                         (String) request.getAttribute("javax.servlet.include.query_string"));
513                 return;
514             }
515         }
516 
517         /**
518          * JERSEY-880 - WAS interprets HttpServletRequest#getServletPath() and HttpServletRequest#getPathInfo()
519          * differently when accessing a static resource.
520          */
521         final String servletPath = request.getServletPath()
522                 + (request.getPathInfo() == null ? "" : request.getPathInfo());
523 
524         doFilter(request, response, chain,
525                 request.getRequestURI(),
526                 servletPath,
527                 request.getQueryString());
528     }
529 
530     private void doFilter(final HttpServletRequest request, final HttpServletResponse response, final FilterChain chain,
531                           final String requestURI, final String servletPath, final String queryString)
532             throws IOException, ServletException {
533         // if we match the static content regular expression lets delegate to
534         // the filter chain to use the default container servlets & handlers
535         final Pattern p = getStaticContentPattern();
536         if (p != null && p.matcher(servletPath).matches()) {
537             chain.doFilter(request, response);
538             return;
539         }
540 
541         if (filterContextPath != null) {
542             if (!servletPath.startsWith(filterContextPath)) {
543                 throw new ContainerException(LocalizationMessages.SERVLET_PATH_MISMATCH(servletPath, filterContextPath));
544                 //TODO:
545                 //            } else if (servletPath.length() == filterContextPath.length()) {
546                 //                // Path does not end in a slash, may need to redirect
547                 //                if (webComponent.getResourceConfig().getFeature(ResourceConfig.FEATURE_REDIRECT)) {
548                 //                    URI l = UriBuilder.fromUri(request.getRequestURL().toString()).
549                 //                            path("/").
550                 //                            replaceQuery(queryString).build();
551                 //
552                 //                    response.setStatus(307);
553                 //                    response.setHeader("Location", l.toASCIIString());
554                 //                    return;
555                 //                } else {
556                 //                    requestURI += "/";
557                 //                }
558             }
559         }
560 
561         final URI baseUri;
562         final URI requestUri;
563         try {
564             final UriBuilder absoluteUriBuilder = UriBuilder.fromUri(request.getRequestURL().toString());
565 
566             // depending on circumstances, use the correct path to replace in the absolute request URI
567             final String pickedUrlMapping = pickUrlMapping(request.getRequestURL().toString(), filterUrlMappings);
568             final String replacingPath = pickedUrlMapping != null
569                     ? pickedUrlMapping
570                     : (filterContextPath != null ? filterContextPath : "");
571 
572             baseUri = absoluteUriBuilder.replacePath(request.getContextPath()).path(replacingPath).path("/").build();
573 
574 
575             requestUri = absoluteUriBuilder.replacePath(requestURI)
576                     .replaceQuery(ContainerUtils.encodeUnsafeCharacters(queryString))
577                     .build();
578         } catch (final IllegalArgumentException iae) {
579             setResponseForInvalidUri(response, iae);
580             return;
581         }
582 
583         final ResponseWriter responseWriter = serviceImpl(baseUri, requestUri, request, response);
584 
585         // If forwarding is configured and response is a 404 with no entity
586         // body then call the next filter in the chain
587 
588         if (webComponent.forwardOn404 && !response.isCommitted()) {
589             boolean hasEntity = false;
590             Response.StatusType status = null;
591             if (responseWriter.responseContextResolved()) {
592                 final ContainerResponse responseContext = responseWriter.getResponseContext();
593                 hasEntity = responseContext.hasEntity();
594                 status = responseContext.getStatusInfo();
595             }
596             if (!hasEntity && status == Response.Status.NOT_FOUND) {
597                 // lets clear the response to OK before we forward to the next in the chain
598                 // as OK is the default set by servlet containers before filters/servlets do any work
599                 // so lets hide our footsteps and pretend we were never in the chain at all and let the
600                 // next filter or servlet return the 404 if they can't find anything to return
601                 //
602                 // We could add an optional flag to disable this step if anyone can ever find a case where
603                 // this causes a problem, though I suspect any problems will really be with downstream
604                 // servlets not correctly setting an error status if they cannot find something to return
605                 response.setStatus(HttpServletResponse.SC_OK);
606                 chain.doFilter(request, response);
607             }
608         }
609     }

611     /**
612      * Picks the most suitable url mapping (in case more than one is defined) based on the request URI.
613      *
614      * @param requestUri String representation of the request URI
615      * @param filterUrlMappings set of configured filter url-patterns
616      * @return the most suitable context path, or {@code null} if empty
617      */
618     private String pickUrlMapping(final String requestUri, final List<String> filterUrlMappings) {
619         if (filterUrlMappings == null || filterUrlMappings.isEmpty()) {
620             return null;
621         }
622 
623         if (filterUrlMappings.size() == 1) {
624             return filterUrlMappings.get(0);
625         }
626 
627         for (final String pattern : filterUrlMappings) {
628             if (requestUri.contains(pattern)) {
629                 return pattern;
630             }
631         }
632 
633         return null;
634     }
