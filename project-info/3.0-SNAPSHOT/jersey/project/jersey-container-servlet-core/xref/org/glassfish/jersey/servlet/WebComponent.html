206         private final Map<String, Object> applicationProperties;
207 
208         /**
209          * Create binder for {@link WebComponent} passing a map of properties to determine whether certain features are allowed
210          * or
211          * not.
212          *
213          * @param applicationProperties map of properties to determine whether certain features are allowed or not.
214          */
215         private WebComponentBinder(final Map<String, Object> applicationProperties) {
216             this.applicationProperties = applicationProperties;
217         }
218 
219         @Override
220         protected void configure() {
221 
222             if (!requestResponseBindingExternalized) {
223 
224                 // request
225                 bindFactory(HttpServletRequestReferencingFactory.class).to(HttpServletRequest.class)
226                         .proxy(true).proxyForSameScope(false).in(RequestScoped.class);
227 
228                 bindFactory(ReferencingFactory.<HttpServletRequest>referenceFactory())
229                         .to(new TypeLiteral<Ref<HttpServletRequest>>() {}).in(RequestScoped.class);
230 
231                 // response
232                 bindFactory(HttpServletResponseReferencingFactory.class).to(HttpServletResponse.class)
233                         .proxy(true).proxyForSameScope(false).in(RequestScoped.class);
234                 bindFactory(ReferencingFactory.<HttpServletResponse>referenceFactory())
235                         .to(new TypeLiteral<Ref<HttpServletResponse>>() {}).in(RequestScoped.class);
236             }
237 
238             bindFactory(new Factory<ServletContext>() {
239                 @Override
240                 public ServletContext provide() {
241                     return webConfig.getServletContext();
242                 }
243 
244                 @Override
245                 public void dispose(final ServletContext instance) {
246                     //not used
247                 }
248             }).to(ServletContext.class).in(Singleton.class);
249 
250             final ServletConfig servletConfig = webConfig.getServletConfig();
251             if (webConfig.getConfigType() == WebConfig.ConfigType.ServletConfig) {
252                 bindFactory(new Factory<ServletConfig>() {
253                     @Override
254                     public ServletConfig provide() {
255                         return servletConfig;
256                     }
257 
258                     @Override
259                     public void dispose(final ServletConfig instance) {
260                         //not used
261                     }
262                 }).to(ServletConfig.class).in(Singleton.class);
263 
264                 // @PersistenceUnit
265                 final Enumeration initParams = servletConfig.getInitParameterNames();
266                 while (initParams.hasMoreElements()) {
267                     final String initParamName = (String) initParams.nextElement();
268 
269                     if (initParamName.startsWith(PersistenceUnitBinder.PERSISTENCE_UNIT_PREFIX)) {
270                         install(new PersistenceUnitBinder());
271                         break;
272                     }
273                 }
274             } else {
275                 bindFactory(new Factory<FilterConfig>() {
276                     @Override
277                     public FilterConfig provide() {
278                         return webConfig.getFilterConfig();
279                     }
280 
281                     @Override
282                     public void dispose(final FilterConfig instance) {
283                         //not used
284                     }
285                 }).to(FilterConfig.class).in(Singleton.class);
286             }
287 
288             bindFactory(new Factory<WebConfig>() {
289                 @Override
290                 public WebConfig provide() {
291                     return webConfig;
292                 }
293 
294                 @Override
295                 public void dispose(final WebConfig instance) {
296                     //not used
297                 }
298             }).to(WebConfig.class).in(Singleton.class);
299 
300             install(new ServiceFinderBinder<>(AsyncContextDelegateProvider.class, applicationProperties, RuntimeType.SERVER));
301             install(new ServiceFinderBinder<>(FilterUrlMappingsProvider.class, applicationProperties, RuntimeType.SERVER));
302         }

343     /**
344      * Create and initialize new web component instance.
345      *
346      * @param webConfig      we component configuration.
347      * @param resourceConfig Jersey application configuration.
348      * @throws ServletException in case the Jersey application cannot be created from the supplied
349      *                          resource configuration.
350      */
351     public WebComponent(final WebConfig webConfig, ResourceConfig resourceConfig) throws ServletException {
352 
353         this.webConfig = webConfig;
354 
355         if (resourceConfig == null) {
356             resourceConfig = createResourceConfig(webConfig);
357         }
358 
359 
360         final ServletContainerProvider[] allServletContainerProviders =
361                 ServletContainerProviderFactory.getAllServletContainerProviders();
362 
363         // SPI/extension hook to configure ResourceConfig
364         configure(resourceConfig, allServletContainerProviders);
365 
366         boolean rrbExternalized = false;
367         RequestScopedInitializerProvider rsiProvider = null;
368 
369         for (final ServletContainerProvider servletContainerProvider : allServletContainerProviders) {
370             if (servletContainerProvider instanceof ExtendedServletContainerProvider) {
371                 final ExtendedServletContainerProvider extendedProvider =
372                         (ExtendedServletContainerProvider) servletContainerProvider;
373 
374                 if (extendedProvider.bindsServletRequestResponse()) {
375                     rrbExternalized = true;
376                 }
377                 if (rsiProvider == null) { // try to take the first non-null provider
378                     rsiProvider = extendedProvider.getRequestScopedInitializerProvider();
379                 }
380             }
381         }
382 
383         requestScopedInitializer = rsiProvider != null ? rsiProvider : DEFAULT_REQUEST_SCOPE_INITIALIZER_PROVIDER;
384         requestResponseBindingExternalized = rrbExternalized;
385 
386         final AbstractBinder webComponentBinder = new WebComponentBinder(resourceConfig.getProperties());
387         resourceConfig.register(webComponentBinder);
388 
389         final ServiceLocator locator = (ServiceLocator) webConfig.getServletContext()
390                 .getAttribute(ServletProperties.SERVICE_LOCATOR);
391 
392         this.appHandler = new ApplicationHandler(resourceConfig, webComponentBinder, locator);
393 
394         this.asyncExtensionDelegate = getAsyncExtensionDelegate();
395         this.forwardOn404 = webConfig.getConfigType() == WebConfig.ConfigType.FilterConfig
396                 && resourceConfig.isProperty(ServletProperties.FILTER_FORWARD_ON_404);
397         this.queryParamsAsFormParams = !resourceConfig.isProperty(ServletProperties.QUERY_PARAMS_AS_FORM_PARAMS_DISABLED);
398         this.configSetStatusOverSendError = ServerProperties.getValue(resourceConfig.getProperties(),
399                 ServerProperties.RESPONSE_SET_STATUS_OVER_SEND_ERROR, false, Boolean.class);
400         this.backgroundTaskScheduler = appHandler.getServiceLocator()
401                 .getService(ScheduledExecutorService.class, BackgroundSchedulerLiteral.INSTANCE);
402     }

417      * @throws java.io.IOException            if an input or output error occurs

422     public Value<Integer> service(
423             final URI baseUri,
424             final URI requestUri,
425             final HttpServletRequest servletRequest,
426             final HttpServletResponse servletResponse) throws ServletException, IOException {
427         final ResponseWriter responseWriter = serviceImpl(baseUri, requestUri, servletRequest, servletResponse);
428         return Values.lazy(new Value<Integer>() {
429             @Override
430             public Integer get() {
431                 return responseWriter.responseContextResolved() ? responseWriter.getResponseStatus() : -1;
432             }
433         });
434     }

436     /**
437      * Dispatch client requests to a resource class.
438      *
439      * @param baseUri         the base URI of the request.
440      * @param requestUri      the URI of the request.
441      * @param servletRequest  the {@link javax.servlet.http.HttpServletRequest} object that
442      *                        contains the request the client made to
443      *                        the Web component.
444      * @param servletResponse the {@link javax.servlet.http.HttpServletResponse} object that
445      *                        contains the response the Web component returns
446      *                        to the client.
447      * @return returns {@link ResponseWriter}, Servlet's {@link org.glassfish.jersey.server.spi.ContainerResponseWriter}
448      *         implementation, into which processed request response was written to.
449      * @throws java.io.IOException            if an input or output error occurs
450      *                                        while the Web component is handling the
451      *                                        HTTP request.
452      * @throws javax.servlet.ServletException if the HTTP request cannot be handled.
453      */
454      /* package */ ResponseWriter serviceImpl(
455             final URI baseUri,
456             final URI requestUri,
457             final HttpServletRequest servletRequest,
458             final HttpServletResponse servletResponse) throws ServletException, IOException {
459 
460         final ResponseWriter responseWriter = new ResponseWriter(
461                 forwardOn404,
462                 configSetStatusOverSendError,
463                 servletResponse,
464                 asyncExtensionDelegate.createDelegate(servletRequest, servletResponse),
465                 backgroundTaskScheduler);
466 
467         try {
468             final ContainerRequest requestContext = new ContainerRequest(baseUri, requestUri, servletRequest.getMethod(),
469                     getSecurityContext(servletRequest), new ServletPropertiesDelegate(servletRequest));
470 
471             initContainerRequest(requestContext, servletRequest, servletResponse, responseWriter);
472 
473             appHandler.handle(requestContext);
474         } catch (final HeaderValueException hve) {
475             if (LOGGER.isLoggable(Level.FINE)) {
476                 LOGGER.log(Level.FINE, LocalizationMessages.HEADER_VALUE_READ_FAILED(), hve);
477             }
478 
479             final Response.Status status = Response.Status.BAD_REQUEST;
480 
481             if (configSetStatusOverSendError) {
482                 servletResponse.reset();
483                 //noinspection deprecation
484                 servletResponse.setStatus(status.getStatusCode(), status.getReasonPhrase());
485             } else {
486                 servletResponse.sendError(status.getStatusCode(), status.getReasonPhrase());
487             }
488         } catch (final Exception e) {
489             throw new ServletException(e);
490         }
491         return responseWriter;
492     }

494     /**
495      * Initialize {@code ContainerRequest} instance to used used to handle {@code servletRequest}.
496      */
497     private void initContainerRequest(
498             final ContainerRequest requestContext,
499             final HttpServletRequest servletRequest,
500             final HttpServletResponse servletResponse,
501             final ResponseWriter responseWriter) throws IOException {
502 
503         requestContext.setEntityStream(servletRequest.getInputStream());
504         requestContext.setRequestScopedInitializer(requestScopedInitializer.get(new RequestContextProvider() {
505             @Override
506             public HttpServletRequest getHttpServletRequest() {
507                 return servletRequest;
508             }
509             @Override
510             public HttpServletResponse getHttpServletResponse() {
511                 return servletResponse;
512             }
513         }));
514         requestContext.setWriter(responseWriter);
515 
516         addRequestHeaders(servletRequest, requestContext);
517         // Check if any servlet filters have consumed a request entity
518         // of the media type application/x-www-form-urlencoded
519         // This can happen if a filter calls request.getParameter(...)
520         filterFormParameters(servletRequest, requestContext);
521     }

554     /**
555      * Create a {@link ResourceConfig} instance from given {@link WebConfig}.
556      *
557      * @param config web config to create resource config from.
558      * @return resource config instance.
559      * @throws ServletException if an error has occurred.
560      */
561     private static ResourceConfig createResourceConfig(final WebConfig config) throws ServletException {
562         final ServletContext servletContext = config.getServletContext();
563 
564         // check if ResourceConfig has already been created, if so use it
565         ResourceConfig resourceConfig = Utils.retrieve(config.getServletContext(), config.getName());
566         if (resourceConfig != null) {
567             return resourceConfig;
568         }
569 
570         final Map<String, Object> initParams = getInitParams(config);
571         final Map<String, Object> contextParams = Utils.getContextParams(servletContext);
572 
573         // check if the JAX-RS application config class property is present
574         final String jaxrsApplicationClassName = config.getInitParameter(ServletProperties.JAXRS_APPLICATION_CLASS);
575 
576         if (jaxrsApplicationClassName == null) {
577             // If no resource config class property is present, create default config
578             resourceConfig = new ResourceConfig().addProperties(initParams).addProperties(contextParams);
579 
580             final String webApp = config.getInitParameter(ServletProperties.PROVIDER_WEB_APP);
581             if (webApp != null && !"false".equals(webApp)) {
582                 resourceConfig.registerFinder(new WebAppResourcesScanner(servletContext));
583             }
584             return resourceConfig;
585         }
586 
587         try {
588             final Class<? extends javax.ws.rs.core.Application> jaxrsApplicationClass = AccessController.doPrivileged(
589                     ReflectionHelper.<javax.ws.rs.core.Application>classForNameWithExceptionPEA(jaxrsApplicationClassName)
590             );
591 
592             if (javax.ws.rs.core.Application.class.isAssignableFrom(jaxrsApplicationClass)) {
593                 return ResourceConfig.forApplicationClass(jaxrsApplicationClass)
594                         .addProperties(initParams).addProperties(contextParams);
595             } else {
596                 throw new ServletException(LocalizationMessages.RESOURCE_CONFIG_PARENT_CLASS_INVALID(
597                         jaxrsApplicationClassName, javax.ws.rs.core.Application.class));
598             }
599         } catch (final PrivilegedActionException e) {
600             throw new ServletException(
601                     LocalizationMessages.RESOURCE_CONFIG_UNABLE_TO_LOAD(jaxrsApplicationClassName), e.getCause());
602         } catch (final ClassNotFoundException e) {
603             throw new ServletException(LocalizationMessages.RESOURCE_CONFIG_UNABLE_TO_LOAD(jaxrsApplicationClassName), e);
604         }
605     }

621     /**
622      * Copy request headers present in {@code request} into {@code requestContext} ignoring {@code null} values.
623      *
624      * @param request        http servlet request to copy headers from.
625      * @param requestContext container request to copy headers to.
626      */
627     @SuppressWarnings("unchecked")
628     private void addRequestHeaders(final HttpServletRequest request, final ContainerRequest requestContext) {
629         final Enumeration<String> names = request.getHeaderNames();
630         while (names.hasMoreElements()) {
631             final String name = names.nextElement();
632 
633             final Enumeration<String> values = request.getHeaders(name);
634             while (values.hasMoreElements()) {
635                 final String value = values.nextElement();
636                 if (value != null) { // filter out null values
637                     requestContext.header(name, value);
638                 }
639             }
640         }
641     }

660      * Extract parameters contained in {@link HttpServletRequest servlet request} and put them into
661      * {@link ContainerRequest container request} under
662      * {@value org.glassfish.jersey.server.internal.InternalServerProperties#FORM_DECODED_PROPERTY} property (as {@link Form}
663      * instance).
664      *
665      * @param servletRequest   http servlet request to extract params from.
666      * @param containerRequest container request to put {@link Form} property to.
667      */
668     private void filterFormParameters(final HttpServletRequest servletRequest, final ContainerRequest containerRequest) {
669         if (MediaTypes.typeEqual(MediaType.APPLICATION_FORM_URLENCODED_TYPE, containerRequest.getMediaType())
670                 && !containerRequest.hasEntity()) {
671             final Form form = new Form();
672             final Enumeration parameterNames = servletRequest.getParameterNames();
673 
674             final String queryString = servletRequest.getQueryString();
675             final List<String> queryParams = queryString != null ? getDecodedQueryParamList(queryString)
676                     : Collections.<String>emptyList();
677 
678             final boolean keepQueryParams = queryParamsAsFormParams || queryParams.isEmpty();
679             final MultivaluedMap<String, String> formMap = form.asMap();
680 
681             while (parameterNames.hasMoreElements()) {
682                 final String name = (String) parameterNames.nextElement();
683                 final List<String> values = Arrays.asList(servletRequest.getParameterValues(name));
684 
685                 formMap.put(name, keepQueryParams ? values : filterQueryParams(name, values, queryParams));
686             }
687 
688             if (!formMap.isEmpty()) {
689                 containerRequest.setProperty(InternalServerProperties.FORM_DECODED_PROPERTY, form);
690 
691                 if (LOGGER.isLoggable(Level.WARNING)) {
692                     LOGGER.log(Level.WARNING, LocalizationMessages.FORM_PARAM_CONSUMED(containerRequest.getRequestUri()));
693                 }
694             }
695         }
696     }

698     private List<String> getDecodedQueryParamList(final String queryString) {
699         final List<String> params = new ArrayList<>();
700         for (final String param : queryString.split("&")) {
701             params.add(UriComponent.decode(param, UriComponent.Type.QUERY_PARAM));
702         }
703         return params;
704     }

706     /**
707      * From given list of values remove values that represents values of query params of the same name as the processed form
708      * parameter.
709      *
710      * @param name   name of form/query parameter.
711      * @param values values of form/query parameter.
712      * @param params collection of unprocessed query parameters.
713      * @return list of form param values for given name without values of query param of the same name.
714      */
715     private List<String> filterQueryParams(final String name, final List<String> values, final Collection<String> params) {
716         return new ArrayList<>(Collections2.filter(values, new Predicate<String>() {
717             @Override
718             public boolean apply(final String input) {
719                 return !params.remove(name + "=" + input)
720                         && !params.remove(name + "[]=" + input);
721             }
722         }));
723     }
