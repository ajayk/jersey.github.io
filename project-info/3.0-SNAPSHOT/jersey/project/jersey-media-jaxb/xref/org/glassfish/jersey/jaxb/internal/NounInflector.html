48  /**
49   * <p>API for performing inflections (pluralization, singularization, and so on)
50   * on various strings.  These inflections will be useful in code generators that
51   * convert things like database table names into Java class names.</p>
52   *
53   * <p>The {@code getInstance()} method returns a singleton instance of
54   * this class with a default set of rules, which can then be customized.
55   * Rules added during customization will take precedence over the standard ones.
56   * Use the {@code addIrregular()}, {@code addPlural()}, {@code addSingular()},
57   * and {@code addUncountable()} methods to add additional rules ot the default
58   * ones.</p>
59   *
60   * <p><strong>IMPLEMENTATION NOTE</strong> - The default implementation is
61   * intended to be functionally compatible with the {@code Inflector::inflections}
62   * class in Ruby on Rails.  The {@code gsub()} method on Ruby strings
63   * matches regular expressions anywhere in the input.  However, nearly all of
64   * the actual patterns used in this component use {@code $} at the end to
65   * match the end of the input string (so that only the last word in a multiple
66   * word phrase will be singularized or pluralized).  Therefore, the Java versions
67   * of the regular expressions have been modified to capture all text before the
68   * interesting characters at the end, and emit them as part of the result, so
69   * that the entire string can be matched against a pattern once.</p>
70   *
71   * @author Florian Rosenberg
72   */
73  final class NounInflector {

200     /**
201      * <p>Convert word strings consisting of lower case letters and
202      * underscore characters between words into <code>embeddedCamelCase</code>
203      * or <code>EmbeddedCamelCase</code>, depending on the <code>lower</code>
204      * flag.  Embedded underscores will be removed.  Embedded '/'
205      * characters will be replaced by '.', making this method useful
206      * in converting path-like names into fully qualified classnames.</p>
207      *
208      * <p><strong>IMPLEMENTATION DIFFERENCE</strong> - The Rails version of this
209      * method also converts '/' characters to '::' because that reflects
210      * the normal syntax for fully qualified names in Ruby.</p>
211      *
212      * <table border="1" width="100%">
213      *   <tr>
214      *     <th>Input</th>
215      *     <th>Output</th>
216      *   </tr>
217      *   <tr>
218      *     <td>"foo_bar", false</td>
219      *     <td>"FooBar"</td>
220      *   </tr>
221      *   <tr>
222      *     <td>"foo_bar", true</td>
223      *     <td>"fooBar"</td>
224      *   </tr>
225      *   <tr>
226      *     <td>"foo_bar/baz", false</td>
227      *     <td>"FooBar.Baz"</td>
228      *   </tr>
229      *   <tr>
230      *     <td>"foo_bar/baz", true</td>
231      *     <td>"fooBar.Baz"</td>
232      *   </tr>
233      * </table>
234      *
235      * @param word Word to be converted
236      * @param flag Flag indicating that the initial character should
237      *  be lower cased instead of upper cased
238      */
239     public String camelize(final String word, final boolean flag) {
240         if (word.length() == 0) {
241             return word;
242         }
243 
244         final StringBuilder sb = new StringBuilder(word.length());
245         if (flag) {
246             sb.append(Character.toLowerCase(word.charAt(0)));
247         } else {
248             sb.append(Character.toUpperCase(word.charAt(0)));
249         }
250         boolean capitalize = false;
251         for (int i = 1; i < word.length(); i++) {
252             final char ch = word.charAt(i);
253             if (capitalize) {
254                 sb.append(Character.toUpperCase(ch));
255                 capitalize = false;
256             } else if (ch == '_') {
257                 capitalize = true;
258             } else if (ch == '/') {
259                 capitalize = true;
260                 sb.append('.');
261             } else {
262                 sb.append(ch);
263             }
264         }
265         return sb.toString();
266 
267     }

405     /**
406      * <p>Capitalize the first word in a lower cased and underscored string,
407      * turn underscores into spaces, and string any trailing "_id".  Like
408      * <code>titleize()</code>, this is meant for creating pretty output,
409      * and is not intended for code generation.</p>
410      *
411      * <table border="1" width="100%">
412      *   <tr>
413      *     <th>Input</th>
414      *     <th>Output</th>
415      *   </tr>
416      *   <tr>

422      *     <td>"Author"</td>
423      *   </tr>
424      * </table>
425      *
426      * @param words Word string to be converted
427      */
428     public String humanize(String words) {
429 
430         if (words.endsWith("_id")) {
431             words = words.substring(0, words.length() - 3);
432         }
433         final StringBuilder sb = new StringBuilder(words.length());
434         sb.append(Character.toUpperCase(words.charAt(0)));
435         for (int i = 1; i < words.length(); i++) {
436             final char ch = words.charAt(i);
437             if (ch == '_') {
438                 sb.append(' ');
439             } else {
440                 sb.append(ch);
441             }
442         }
443         return sb.toString();
444 
445     }

448      * <p>Turn a number into a corresponding ordinal string used to

451      * <table border="1" width="100%">

454      *     <th>Output</th>
455      *   </tr>

458      *     <td>"1st"</td>

460      *   <tr>
461      *     <td>2</td>
462      *     <td>"2nd"</td>
463      *   </tr>
464      *   <tr>
465      *     <td>3</td>
466      *     <td>"3rd"</td>
467      *   </tr>
468      *   <tr>
469      *     <td>4</td>
470      *     <td>"rth"</td>
471      *   </tr>
472      *   <tr>
473      *     <td>1002</td>
474      *     <td>"1002nd"</td>
475      *   </tr>
476      *   <tr>
477      *     <td>2012</td>
478      *     <td>"2012th"</td>
479      *   </tr>
480      * </table>
481      *
482      * @param number Number to be converted
483      */
484     public String ordinalize(final int number) {
485 
486         final int modulo = number % 100;
487         if ((modulo >= 11) && (modulo <= 13)) {
488             return "" + number + "th";
489         }
490         switch (number % 10) {
491             case 1:
492                 return "" + number + "st";
493             case 2:
494                 return "" + number + "nd";
495             case 3:
496                 return "" + number + "rd";
497             default:
498                 return "" + number + "th";
499         }
500 
501     }

503     /**
504      * <p>Return a addPlural version of the specified (addSingular) word.</p>
505      *
506      *
507      * @param word Singular word to be converted
508      */
509     public String pluralize(final String word) {
510 
511         // Scan uncountables and leave alone
512         for (final String uncountable : uncountables) {
513             if (uncountable.equals(word)) {
514                 return word;
515             }
516         }
517 
518         // Scan our patterns for a match and return the correct replacement
519         for (final Replacer plural : plurals) {
520             final String replacement = plural.replacement(word);
521             if (replacement != null) {
522                 return replacement;
523             }
524         }
525 
526         // Return the original string unchanged
527         return word;
528 
529     }

587     /**
588      * <p>Capitalize all the words, and replace some characters in the string
589      * to create a nicer looking title.  This is meant for creating pretty
590      * output, and is not intended for code generation.</p>
591      *
592      * <table border="1" width="100%">
593      *   <tr>
594      *     <th>Input</th>
595      *     <th>Output</th>
596      *   </tr>
597      *   <tr>
598      *     <td>"the honeymooners"</td>
599      *     <td>"The Honeymooners"</td>
600      *   </tr>
601      *   <tr>
602      *     <td>"x-men: the last stand"</td>
603      *     <td>"X Men: The Last Stand"</td>
604      *   </tr>
605      * </table>
606      *
607      * @param words Word string to be converted
608      */
609     public String titleize(final String words) {
610 
611         final StringBuilder sb = new StringBuilder(words.length());
612         boolean capitalize = true; // To get the first character right
613         for (int i = 0; i < words.length(); i++) {
614             final char ch = words.charAt(i);
615             if (Character.isWhitespace(ch)) {
616                 sb.append(' ');
617                 capitalize = true;
618             } else if (ch == '-') {
619                 sb.append(' ');
620                 capitalize = true;
621             } else if (capitalize) {
622                 sb.append(Character.toUpperCase(ch));
623                 capitalize = false;
624             } else {
625                 sb.append(ch);
626             }
627         }
628         return sb.toString();
629 
630     }

632     public String decapitalize(final String word) {
633         // do nothing if null or empty
634         if ((word == null) || (word.length() < 1)) {
635             return word;
636         }
637         // or if already decapitalized
638         final char first = word.charAt(0);
639         if (Character.isLowerCase(first)) {
640             return word;
641         }
642         // otherwise turn the first character to lower case and attach the rest
643         final StringBuilder sb = new StringBuilder(word.length());
644         sb.append(Character.toLowerCase(first));
645         sb.append(word.substring(1));
646         return sb.toString();
647     }

684             final char ch = word.charAt(i);

696             } else {

698             }
699         }
