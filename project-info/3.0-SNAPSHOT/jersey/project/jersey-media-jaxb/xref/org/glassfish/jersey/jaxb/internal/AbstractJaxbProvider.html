101     /**
102      * Inheritance constructor.
103      *
104      * @param providers         JAX-RS providers.
105      * @param resolverMediaType JAXB component context resolver media type to be used.
106      */
107     public AbstractJaxbProvider(final Providers providers, final MediaType resolverMediaType) {
108         this.jaxrsProviders = providers;
109 
110         fixedResolverMediaType = resolverMediaType != null;
111         if (fixedResolverMediaType) {
112             this.mtContext = Values.lazy(new Value<ContextResolver<JAXBContext>>() {
113 
114                 @Override
115                 public ContextResolver<JAXBContext> get() {
116                     return providers.getContextResolver(JAXBContext.class, resolverMediaType);
117                 }
118             });
119             this.mtUnmarshaller = Values.lazy(new Value<ContextResolver<Unmarshaller>>() {
120 
121                 @Override
122                 public ContextResolver<Unmarshaller> get() {
123                     return providers.getContextResolver(Unmarshaller.class, resolverMediaType);
124                 }
125             });
126             this.mtMarshaller = Values.lazy(new Value<ContextResolver<Marshaller>>() {
127 
128                 @Override
129                 public ContextResolver<Marshaller> get() {
130                     return providers.getContextResolver(Marshaller.class, resolverMediaType);
131                 }
132             });
133         } else {
134             this.mtContext = null;
135             this.mtUnmarshaller = null;
136             this.mtMarshaller = null;
137         }
138     }

173     /**
174      * Get the JAXB unmarshaller for the given class and media type.
175      * <p>
176      * In case this provider instance has been {@link #AbstractJaxbProvider(Providers, MediaType)
177      * created with a fixed resolver media type}, the supplied media type argument will be ignored.
178      * </p>
179      *
180      * @param type      Java type to be unmarshalled.
181      * @param mediaType entity media type.
182      * @return JAXB unmarshaller for the requested Java type, media type combination.
183      * @throws JAXBException in case retrieving the unmarshaller fails with a JAXB exception.
184      */
185     protected final Unmarshaller getUnmarshaller(Class type, MediaType mediaType) throws JAXBException {
186         if (fixedResolverMediaType) {
187             return getUnmarshaller(type);
188         }
189 
190         final ContextResolver<Unmarshaller> unmarshallerResolver =
191                 jaxrsProviders.getContextResolver(Unmarshaller.class, mediaType);
192         if (unmarshallerResolver != null) {
193             Unmarshaller u = unmarshallerResolver.getContext(type);
194             if (u != null) {
195                 return u;
196             }
197         }
198 
199         final JAXBContext ctx = getJAXBContext(type, mediaType);
200         return (ctx == null) ? null : ctx.createUnmarshaller();
201     }

216     /**
217      * Get the JAXB marshaller for the given class and media type.
218      * <p>
219      * In case this provider instance has been {@link #AbstractJaxbProvider(Providers, MediaType)
220      * created with a fixed resolver media type}, the supplied media type argument will be ignored.
221      * </p>
222      *
223      * @param type      Java type to be marshalled.
224      * @param mediaType entity media type.
225      * @return JAXB marshaller for the requested Java type, media type combination.
226      * @throws JAXBException in case retrieving the marshaller fails with a JAXB exception.
227      */
228     protected final Marshaller getMarshaller(Class type, MediaType mediaType) throws JAXBException {
229         if (fixedResolverMediaType) {
230             return getMarshaller(type);
231         }
232 
233         final ContextResolver<Marshaller> mcr = jaxrsProviders.getContextResolver(Marshaller.class, mediaType);
234         if (mcr != null) {
235             Marshaller m = mcr.getContext(type);
236             if (m != null) {
237                 return m;
238             }
239         }
240 
241         final JAXBContext ctx = getJAXBContext(type, mediaType);
242         if (ctx == null) {
243             return null;
244         }
245 
246         Marshaller m = ctx.createMarshaller();
247         if (formattedOutput.get()) {
248             m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, formattedOutput.get());
249         }
250         return m;
251 
252     }

254     private Marshaller getMarshaller(Class type) throws JAXBException {
255         final ContextResolver<Marshaller> resolver = mtMarshaller.get();
256         if (resolver != null) {
257             Marshaller u = resolver.getContext(type);
258             if (u != null) {
259                 return u;
260             }
261         }

263         final JAXBContext ctx = getJAXBContext(type);
264         if (ctx == null) {

266         }
267

275     private JAXBContext getJAXBContext(Class type, MediaType mt) throws JAXBException {
276         final ContextResolver<JAXBContext> cr = jaxrsProviders.getContextResolver(JAXBContext.class, mt);
277         if (cr != null) {
278             JAXBContext c = cr.getContext(type);
279             if (c != null) {
280                 return c;
281             }
282         }
283 
284         return getStoredJaxbContext(type);
285     }

287     private JAXBContext getJAXBContext(Class type) throws JAXBException {
288         final ContextResolver<JAXBContext> resolver = mtContext.get();
289         if (resolver != null) {
290             JAXBContext c = resolver.getContext(type);
291             if (c != null) {
292                 return c;
293             }
294         }
295 
296         return getStoredJaxbContext(type);
297     }

299     /**
300      * Retrieve cached JAXB context capable of handling the given Java type.
301      *
302      * @param type Java type .
303      * @return JAXB context associated with the Java type.
304      * @throws JAXBException in case the JAXB context retrieval fails.
305      */
306     protected JAXBContext getStoredJaxbContext(Class type) throws JAXBException {
307         synchronized (jaxbContexts) {
308             final WeakReference<JAXBContext> ref = jaxbContexts.get(type);
309             JAXBContext c = (ref != null) ? ref.get() : null;
310             if (c == null) {
311                 c = JAXBContext.newInstance(type);
312                 jaxbContexts.put(type, new WeakReference<JAXBContext>(c));
313             }
314             return c;
315         }
316     }

318     /**
319      * Create new {@link javax.xml.transform.sax.SAXSource} for a given entity input stream.
320      *
321      * @param spf          SAX parser factory to be used to create the SAX source.
322      * @param entityStream entity input stream.
323      * @return new {@link javax.xml.transform.sax.SAXSource} representing the entity input stream.
324      * @throws JAXBException in case SAX source creation fails.
325      */
326     protected static SAXSource getSAXSource(SAXParserFactory spf, InputStream entityStream) throws JAXBException {
327         try {
328             return new SAXSource(spf.newSAXParser().getXMLReader(), new InputSource(entityStream));
329         } catch (Exception ex) {
330             throw new JAXBException("Error creating SAXSource", ex);
331         }
332     }

342     /**
343      * Set the custom XML header on a JAXB marshaller if specified via {@link org.glassfish.jersey.message.XmlHeader} annotation,
344      * present in the supplied array of annotations.
345      *
346      * @param marshaller  JAXB marshaller.
347      * @param annotations array of annotations that MAY contain a {@code XmlHeader} annotation instance.
348      */
349     protected void setHeader(Marshaller marshaller, Annotation[] annotations) {
350         for (Annotation a : annotations) {
351             if (a instanceof XmlHeader) {
352                 try {
353                     // standalone jaxb ri
354                     marshaller.setProperty("com.sun.xml.bind.xmlHeaders", ((XmlHeader) a).value());
355                 } catch (PropertyException e) {
356                     try {
357                         // jaxb ri from jdk
358                         marshaller.setProperty("com.sun.xml.internal.bind.xmlHeaders", ((XmlHeader) a).value());
359                     } catch (PropertyException ex) {
360                         // other jaxb implementation
361                         Logger.getLogger(AbstractJaxbProvider.class.getName()).log(
362                                 Level.WARNING, "@XmlHeader annotation is not supported with this JAXB implementation."
363                                         + " Please use JAXB RI if you need this feature.");
364                     }
365                 }
366                 break;
367             }
368         }
