60  /**
61   * A common interface for invocable resource components. This includes resource
62   * methods, sub-resource methods and sub-resource locators bound to a concrete
63   * handler class and a Java method (either directly or indirectly) declared &
64   * implemented by the handler class.
65   * <p/>
66   * Invocable component information is used at runtime by a Java method dispatcher
67   * when processing requests.
68   * <p>
69   * Class defines two kinds of {@link Method java methods}: {@link #getDefinitionMethod() definition method} and
70   * {@link #getHandlingMethod() handling method}. Definition method is the java {@code Method} that is defined
71   * by the user to be
72   * executed. This can be java {@code Method} of the class but also method of the interface. If it is the
73   * method of the interface (method handler class is the {@code class} but method itself is from
74   * the {@code interface}) then the definition method is the method from the inherited {@code class}. In other words, the
75   * handling method is the concrete method but definition method can be its parent abstract definition. However, in most
76   * cases these methods are the same.
77   * </p>
78   *
79   *
80   * @author Marek Potociar (marek.potociar at oracle.com)
81   * @see ResourceMethod
82   * @see org.glassfish.jersey.server.spi.internal.ResourceMethodDispatcher
83   */
84  public final class Invocable implements Parameterized, ResourceModelComponent {

87       * Method instance representing the {@link Inflector#apply(Object)} method.
88       */
89      static final Method APPLY_INFLECTOR_METHOD = initApplyMethod();

91      private static Method initApplyMethod() {
92          try {
93              return Inflector.class.getMethod("apply", Object.class);
94          } catch (NoSuchMethodException e) {
95              IncompatibleClassChangeError error = new IncompatibleClassChangeError("Inflector.apply(Object) method not found");
96              error.initCause(e);
97              throw error;
98          }
99      }

163     /**
164      * Create a new resource method invocable model.
165      *
166      * @param handler           resource method handler.
167      * @param definitionMethod  method that is defined to be executed on the {@code handler}.
168      * @param handlingMethod    specific and concrete method to be actually executed as a resource method. If {@code null}
169      *                          then the {@code definitionMethod} will be used.
170      * @param encodedParameters {@code true} if the automatic parameter decoding
171      *                          should be disabled, false otherwise.
172      * @param routingResponseType response type that will be used during the routing for purpose
173      *                            of selection of the resource method to be executed. If this parameter is
174      *                            non-{@code null} then it will override the return type of the
175      *                            {@link #getHandlingMethod() the Java handling method}) for purposes of
176      *                            resource method selection. This might be useful in cases when resource
177      *                            method returns a type {@code A} but thanks to registered providers
178      *                            (eg. {@link javax.ws.rs.ext.WriterInterceptor}) it will be always converted
179      *                            to type {@code B}. Then the method selecting algorithm would check presence of
180      *                            {@link javax.ws.rs.ext.MessageBodyWriter} for type {@code A} (which will
181      *                            never be actually needed) and might result in choosing undesired method.
182      *                            If the parameter is {@code null} then the default response type will be used.
183      *
184      * @return Invocable.
185      */
186     public static Invocable create(MethodHandler handler, Method definitionMethod, Method handlingMethod,
187                                    boolean encodedParameters, Type routingResponseType) {
188         return new Invocable(handler, definitionMethod, handlingMethod, encodedParameters, routingResponseType);

201     private Invocable(MethodHandler handler, Method definitionMethod, Method handlingMethod, boolean encodedParameters,
202                       Type routingResponseType) {
203         this.handler = handler;
204         this.definitionMethod = definitionMethod;
205         this.handlingMethod = handlingMethod == null ? ReflectionHelper
206                 .findOverridingMethodOnClass(handler.getHandlerClass(), definitionMethod) : handlingMethod;
207 
208         final Class<?> handlerClass = handler.getHandlerClass();
209         final Class<?> definitionClass = definitionMethod.getDeclaringClass();
210         final ClassTypePair handlingCtPair = ReflectionHelper.resolveGenericType(
211                 handlerClass,
212                 this.handlingMethod.getDeclaringClass(),
213                 this.handlingMethod.getReturnType(),
214                 this.handlingMethod.getGenericReturnType());
215 
216         // here we need to find types also for definition method. Definition method is in most
217         // cases used for parent methods (for example for interface method of resource class). But here we
218         // consider also situation when resource is a proxy (for example proxy of EJB) and definition
219         // method is the original method and handling method is method on proxy. So, we try to find generic
220         // type in the original class using definition method.
221         final ClassTypePair definitionCtPair = ReflectionHelper.resolveGenericType(
222                 definitionClass,
223                 this.definitionMethod.getDeclaringClass(),
224                 this.definitionMethod.getReturnType(),
225                 this.definitionMethod.getGenericReturnType());
226         this.rawResponseType = handlingCtPair.rawClass();
227         final boolean handlerReturnTypeIsParameterized = handlingCtPair.type() instanceof ParameterizedType;
228         final boolean definitionReturnTypeIsParameterized = definitionCtPair.type() instanceof ParameterizedType;
229         this.responseType =
230                 (handlingCtPair.rawClass() == definitionCtPair.rawClass()
231                         && definitionReturnTypeIsParameterized && !handlerReturnTypeIsParameterized)
232                         ? definitionCtPair.type() : handlingCtPair.type();
233         if (routingResponseType == null) {
234             this.routingResponseType = responseType;
235             this.rawRoutingResponseType = rawResponseType;
236         } else {
237             GenericType routingResponseGenericType = new GenericType(routingResponseType);
238             this.routingResponseType = routingResponseGenericType.getType();
239             this.rawRoutingResponseType = routingResponseGenericType.getRawType();
240         }
241 
242         this.parameters = Collections.unmodifiableList(Parameter.create(
243                 handlerClass, definitionMethod.getDeclaringClass(), definitionMethod, encodedParameters));
244     }

299     /**
300      * Check if the invocable represents an {@link Inflector#apply(Object) inflector
301      * processing method}.
302      *
303      * @return {@code true}, if this invocable represents an inflector invocation,
304      *         {@code false} otherwise.
305      */
306     public boolean isInflector() {
307         // Method.equals(...) does not perform the identity check (in Java SE 6)
308         return APPLY_INFLECTOR_METHOD == definitionMethod || APPLY_INFLECTOR_METHOD.equals(definitionMethod);
