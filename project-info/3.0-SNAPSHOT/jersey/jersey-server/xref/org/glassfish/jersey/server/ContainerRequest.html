154     /**
155      * Create new Jersey container request context.
156      *
157      * @param baseUri            base application URI.
158      * @param requestUri         request URI.
159      * @param httpMethod         request HTTP method name.
160      * @param securityContext    security context of the current request. Must not be {@code null}.
161      *                           The {@link SecurityContext#getUserPrincipal()} must return
162      *                           {@code null} if the current request has not been authenticated
163      *                           by the container.
164      * @param propertiesDelegate custom {@link PropertiesDelegate properties delegate}
165      *                           to be used by the context.

167     public ContainerRequest(
168             final URI baseUri,
169             final URI requestUri,
170             final String httpMethod,
171             final SecurityContext securityContext,
172             final PropertiesDelegate propertiesDelegate) {
173         super(true);
174 
175         this.baseUri = baseUri == null ? DEFAULT_BASE_URI : baseUri.normalize();
176         this.requestUri = requestUri;
177         this.httpMethod = httpMethod;
178         this.securityContext = securityContext;
179         this.propertiesDelegate = new TracingAwarePropertiesDelegate(propertiesDelegate);
180         this.uriRoutingContext = new UriRoutingContext(this);
181     }

439     /**
440      * Get the path of the current request relative to the application root (base)
441      * URI as a string.
442      *
443      * @param decode controls whether sequences of escaped octets are decoded
444      *               ({@code true}) or not ({@code false}).
445      * @return relative request path.
446      */
447     public String getPath(final boolean decode) {
448         if (decode) {
449             if (decodedRelativePath != null) {
450                 return decodedRelativePath;
451             }
452 
453             return decodedRelativePath = UriComponent.decode(encodedRelativePath(), UriComponent.Type.PATH);
454         } else {
455             return encodedRelativePath();
456         }
457     }

459     private String encodedRelativePath() {
460         if (encodedRelativePath != null) {
461             return encodedRelativePath;
462         }
463 
464         final String requestUriRawPath = requestUri.getRawPath();
465 
466         if (baseUri == null) {
467             return encodedRelativePath = requestUriRawPath;
468         }
469 
470         final int baseUriRawPathLength = baseUri.getRawPath().length();
471         return encodedRelativePath = baseUriRawPathLength < requestUriRawPath.length()
472                 ? requestUriRawPath.substring(baseUriRawPathLength) : "";
473     }

600     @Override
601     public Response.ResponseBuilder evaluatePreconditions(final EntityTag eTag) {
602         if (eTag == null) {
603             throw new IllegalArgumentException(METHOD_PARAMETER_CANNOT_BE_NULL_ETAG);
604         }
605 
606         final Response.ResponseBuilder r = evaluateIfMatch(eTag);
607         if (r != null) {
608             return r;
609         }
610         return evaluateIfNoneMatch(eTag);
611     }

613     @Override
614     public Response.ResponseBuilder evaluatePreconditions(final Date lastModified) {
615         if (lastModified == null) {
616             throw new IllegalArgumentException(METHOD_PARAMETER_CANNOT_BE_NULL_LAST_MODIFIED);
617         }
618 
619         final long lastModifiedTime = lastModified.getTime();
620         final Response.ResponseBuilder r = evaluateIfUnmodifiedSince(lastModifiedTime);
621         if (r != null) {
622             return r;
623         }
624         return evaluateIfModifiedSince(lastModifiedTime);
625     }
626 
627     @Override
628     public Response.ResponseBuilder evaluatePreconditions(final Date lastModified, final EntityTag eTag) {
629         if (lastModified == null) {
630             throw new IllegalArgumentException(METHOD_PARAMETER_CANNOT_BE_NULL_LAST_MODIFIED);
631         }
632         if (eTag == null) {
633             throw new IllegalArgumentException(METHOD_PARAMETER_CANNOT_BE_NULL_ETAG);
634         }
635 
636         Response.ResponseBuilder r = evaluateIfMatch(eTag);
637         if (r != null) {
638             return r;
639         }
640 
641         final long lastModifiedTime = lastModified.getTime();
642         r = evaluateIfUnmodifiedSince(lastModifiedTime);
643         if (r != null) {
644             return r;
645         }
646 
647         final boolean isGetOrHead = "GET".equals(getMethod()) || "HEAD".equals(getMethod());
648         final Set<MatchingEntityTag> matchingTags = getIfNoneMatch();
649         if (matchingTags != null) {
650             r = evaluateIfNoneMatch(eTag, matchingTags, isGetOrHead);
651             // If the If-None-Match header is present and there is no
652             // match then the If-Modified-Since header must be ignored
653             if (r == null) {
654                 return null;
655             }
656 
657             // Otherwise if the If-None-Match header is present and there
658             // is a match then the If-Modified-Since header must be checked
659             // for consistency
660         }
661 
662         final String ifModifiedSinceHeader = getHeaderString(HttpHeaders.IF_MODIFIED_SINCE);
663         if (ifModifiedSinceHeader != null && !ifModifiedSinceHeader.isEmpty() && isGetOrHead) {
664             r = evaluateIfModifiedSince(lastModifiedTime, ifModifiedSinceHeader);
665             if (r != null) {
666                 r.tag(eTag);
667             }
668         }
669 
670         return r;
671     }

673     @Override
674     public Response.ResponseBuilder evaluatePreconditions() {
675         final Set<MatchingEntityTag> matchingTags = getIfMatch();
676         if (matchingTags == null) {
677             return null;
678         }
679 
680         // Since the resource does not exist the method must not be
681         // perform and 412 Precondition Failed is returned
682         return Response.status(Response.Status.PRECONDITION_FAILED);
683     }

685     // Private methods
686     private Response.ResponseBuilder evaluateIfMatch(final EntityTag eTag) {
687         final Set<? extends EntityTag> matchingTags = getIfMatch();
688         if (matchingTags == null) {
689             return null;
690         }
691 
692         // The strong comparison function must be used to compare the entity
693         // tags. Thus if the entity tag of the entity is weak then matching
694         // of entity tags in the If-Match header should fail.
695         if (eTag.isWeak()) {
696             return Response.status(Response.Status.PRECONDITION_FAILED);
697         }
698 
699         if (matchingTags != MatchingEntityTag.ANY_MATCH && !matchingTags.contains(eTag)) {
700             // 412 Precondition Failed
701             return Response.status(Response.Status.PRECONDITION_FAILED);
702         }
703 
704         return null;
705     }

707     private Response.ResponseBuilder evaluateIfNoneMatch(final EntityTag eTag) {
708         final Set<MatchingEntityTag> matchingTags = getIfNoneMatch();
709         if (matchingTags == null) {
710             return null;
711         }
712 
713         final String httpMethod = getMethod();
714         return evaluateIfNoneMatch(eTag, matchingTags, "GET".equals(httpMethod) || "HEAD".equals(httpMethod));

717     private Response.ResponseBuilder evaluateIfNoneMatch(final EntityTag eTag, final Set<? extends EntityTag> matchingTags,
718                                                          final boolean isGetOrHead) {
719         if (isGetOrHead) {
720             if (matchingTags == MatchingEntityTag.ANY_MATCH) {
721                 // 304 Not modified
722                 return Response.notModified(eTag);
723             }
724 
725             // The weak comparison function may be used to compare entity tags
726             if (matchingTags.contains(eTag) || matchingTags.contains(new EntityTag(eTag.getValue(), !eTag.isWeak()))) {
727                 // 304 Not modified
728                 return Response.notModified(eTag);
729             }
730         } else {
731             // The strong comparison function must be used to compare the entity
732             // tags. Thus if the entity tag of the entity is weak then matching
733             // of entity tags in the If-None-Match header should fail if the
734             // HTTP method is not GET or not HEAD.
735             if (eTag.isWeak()) {
736                 return null;
737             }
738 
739             if (matchingTags == MatchingEntityTag.ANY_MATCH || matchingTags.contains(eTag)) {
740                 // 412 Precondition Failed
741                 return Response.status(Response.Status.PRECONDITION_FAILED);
742             }
743         }
744 
745         return null;
746     }

748     private Response.ResponseBuilder evaluateIfUnmodifiedSince(final long lastModified) {
749         final String ifUnmodifiedSinceHeader = getHeaderString(HttpHeaders.IF_UNMODIFIED_SINCE);
750         if (ifUnmodifiedSinceHeader != null && !ifUnmodifiedSinceHeader.isEmpty()) {

756                 }
757             } catch (final ParseException ex) {

759             }
760         }

766         final String ifModifiedSinceHeader = getHeaderString(HttpHeaders.IF_MODIFIED_SINCE);

768             return null;

770
