138     @Override
139     public void pushLeftHandPath() {
140         final String rightHandPath = getFinalMatchingGroup();
141         final int rhpLength = (rightHandPath != null) ? rightHandPath.length() : 0;
142 
143         final String encodedRequestPath = getPath(false);
144 
145         final int length = encodedRequestPath.length() - rhpLength;
146         if (length <= 0) {
147             paths.addFirst("");
148         } else {
149             paths.addFirst(encodedRequestPath.substring(0, length));
150         }
151     }

153     @Override
154     public void pushTemplates(final UriTemplate resourceTemplate, final UriTemplate methodTemplate) {
155         final Iterator<MatchResult> matchResultIterator = matchResults.iterator();
156         templates.push(resourceTemplate);
157         if (methodTemplate != null) {
158             templates.push(methodTemplate);
159             // fast-forward the match result iterator to second element in the stack
160             matchResultIterator.next();
161         }
162 
163         pushMatchedTemplateValues(resourceTemplate, matchResultIterator.next());
164         if (methodTemplate != null) {
165             // use the match result from the top of the stack
166             pushMatchedTemplateValues(methodTemplate, matchResults.peek());
167         }
168     }
169 
170     private void pushMatchedTemplateValues(final UriTemplate template, final MatchResult matchResult) {
171         int i = 1;
172         for (final String templateVariable : template.getTemplateVariables()) {
173             final String value = matchResult.group(i++);
174             encodedTemplateValues.addFirst(templateVariable, value);
175             if (decodedTemplateValues != null) {
176                 decodedTemplateValues.addFirst(
177                         UriComponent.decode(templateVariable, UriComponent.Type.PATH_SEGMENT),
178                         UriComponent.decode(value, UriComponent.Type.PATH));
179             }
180         }
181     }

183     @Override
184     public String getFinalMatchingGroup() {
185         final MatchResult mr = matchResults.peek();
186         if (mr == null) {
187             return null;
188         }
189 
190         final String finalGroup = mr.group(mr.groupCount());
191         // We have found a match but the right hand path pattern did not match anything
192         // so just returning an empty string as a final matching group result.
193         // Otherwise a non-empty patterns would fail to match the right-hand-path properly.
194         // See also PatternWithGroups.match(CharSequence) implementation.
195         return finalGroup == null ? "" : finalGroup;
196     }

307     @Override
308     public MultivaluedMap<String, String> getPathParameters(final boolean decode) {
309         if (decode) {
310             if (decodedTemplateValuesView != null) {
311                 return decodedTemplateValuesView;
312             } else if (decodedTemplateValues == null) {
313                 decodedTemplateValues = new MultivaluedHashMap<>();
314                 for (final Map.Entry<String, List<String>> e : encodedTemplateValues.entrySet()) {
315                     decodedTemplateValues.put(
316                             UriComponent.decode(e.getKey(), UriComponent.Type.PATH_SEGMENT),
317                             // we need to keep the ability to add new entries
318                             new LinkedList<>(Lists.transform(e.getValue(), new Function<String, String>() {
319 
320                                 @Override
321                                 public String apply(final String input) {
322                                     return UriComponent.decode(input, UriComponent.Type.PATH);
323                                 }
324                             })));
325                 }
326             }
327             decodedTemplateValuesView = new ImmutableMultivaluedMap<>(decodedTemplateValues);
328 
329             return decodedTemplateValuesView;
330         } else {
331             return encodedTemplateValuesView;
332         }
333     }

417     @Override
418     public List<PathSegment> getPathSegments(final String name, final boolean decode) {
419         final int[] bounds = getPathParameterBounds(name);
420         if (bounds != null) {
421             final String path = matchResults.getLast().group();
422             // Work out how many path segments are up to the start
423             // and end position of the matching path parameter value
424             // This assumes that the path always starts with a '/'
425             int segmentsStart = 0;
426             for (int x = 0; x < bounds[0]; x++) {
427                 if (path.charAt(x) == '/') {
428                     segmentsStart++;
429                 }
430             }
431             int segmentsEnd = segmentsStart;
432             for (int x = bounds[0]; x < bounds[1]; x++) {
433                 if (path.charAt(x) == '/') {
434                     segmentsEnd++;
435                 }
436             }
437 
438             return getPathSegments(decode).subList(segmentsStart - 1, segmentsEnd);
439         } else {
440             return Collections.emptyList();
441         }
442     }
443 
444     private int[] getPathParameterBounds(final String name) {
445         final Iterator<UriTemplate> templatesIterator = templates.iterator();
446         final Iterator<MatchResult> matchResultsIterator = matchResults.iterator();
447         while (templatesIterator.hasNext()) {
448             MatchResult mr = matchResultsIterator.next();
449             // Find the index of path parameter
450             final int pIndex = getLastPathParameterIndex(name, templatesIterator.next());
451             if (pIndex != -1) {
452                 int pathLength = mr.group().length();
453                 int segmentIndex = mr.end(pIndex + 1);
454                 final int groupLength = segmentIndex - mr.start(pIndex + 1);
455 
456                 // Find the absolute position of the end of the
457                 // capturing group in the request path
458                 while (matchResultsIterator.hasNext()) {
459                     mr = matchResultsIterator.next();
460                     segmentIndex += mr.group().length() - pathLength;
461                     pathLength = mr.group().length();
462                 }
463                 return new int[] {segmentIndex - groupLength, segmentIndex};
464             }
465         }
466         return null;
467     }
468 
469     private int getLastPathParameterIndex(final String name, final UriTemplate t) {
470         int i = 0;
471         int pIndex = -1;
472         for (final String parameterName : t.getTemplateVariables()) {
473             if (parameterName.equals(name)) {
474                 pIndex = i;
475             }
476             i++;
477         }
478         return pIndex;
479     }
