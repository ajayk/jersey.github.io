69  /**
70   * Validator checking resource methods and sub resource locators. The validator mainly checks the parameters of resource
71   * methods and sub resource locators.
72   *
73   * @author Miroslav Fuksa
74   */
75  class ResourceMethodValidator extends AbstractResourceModelVisitor {
76  
77      private final ServiceLocator locator;
78  
79      /**
80       * Create new resource method validator.
81       *
82       * @param locator HK2 service locator.
83       */
84      public ResourceMethodValidator(ServiceLocator locator) {
85          this.locator = locator;
86      }
87  
88      @Override
89      public void visitResourceMethod(final ResourceMethod method) {
90          switch (method.getType()) {
91              case RESOURCE_METHOD:
92                  visitJaxrsResourceMethod(method);
93                  break;
94              case SUB_RESOURCE_LOCATOR:
95                  visitSubResourceLocator(method);
96                  break;
97          }
98  
99      }
100 
101     private void visitJaxrsResourceMethod(ResourceMethod method) {
102         checkMethod(method);
103     }
104 
105     private void checkMethod(ResourceMethod method) {
106         checkValueProviders(method);
107         final Invocable invocable = method.getInvocable();
108 
109         checkParameters(method);
110 
111         if ("GET".equals(method.getHttpMethod())) {
112             // ensure GET returns non-void value if not suspendable
113             if (void.class == invocable.getHandlingMethod().getReturnType() && !method.isSuspendDeclared()) {
114                 Errors.hint(method, LocalizationMessages.GET_RETURNS_VOID(invocable.getHandlingMethod()));
115             }
116 
117             // ensure GET does not consume an entity parameter, if not inflector-based
118             if (invocable.requiresEntity() && !invocable.isInflector()) {
119                 Errors.warning(method, LocalizationMessages.GET_CONSUMES_ENTITY(invocable.getHandlingMethod()));
120             }
121             // ensure GET does not consume any @FormParam annotated parameter
122             for (Parameter p : invocable.getParameters()) {
123                 if (p.isAnnotationPresent(FormParam.class)) {
124                     Errors.fatal(method, LocalizationMessages.GET_CONSUMES_FORM_PARAM(invocable.getHandlingMethod()));
125                     break;
126                 }
127             }
128         }
129 
130         // ensure there is not multiple HTTP method designators specified on the method
131         List<String> httpMethodAnnotations = new LinkedList<>();
132         for (Annotation a : invocable.getHandlingMethod().getDeclaredAnnotations()) {
133             if (null != a.annotationType().getAnnotation(HttpMethod.class)) {
134                 httpMethodAnnotations.add(a.toString());
135             }
136         }
137 
138         if (httpMethodAnnotations.size() > 1) {
139             Errors.fatal(method, LocalizationMessages.MULTIPLE_HTTP_METHOD_DESIGNATORS(invocable.getHandlingMethod(),
140                     httpMethodAnnotations.toString()));
141         }
142 
143         final Type responseType = invocable.getResponseType();
144         if (!isConcreteType(responseType)) {
145             Errors.warning(invocable.getHandlingMethod(),
146                     LocalizationMessages.TYPE_OF_METHOD_NOT_RESOLVABLE_TO_CONCRETE_TYPE(responseType,
147                             invocable.getHandlingMethod().toGenericString()));
148         }
149 
150         final Path pathAnnotation = invocable.getHandlingMethod().getAnnotation(Path.class);
151         if (pathAnnotation != null) {
152             final String path = pathAnnotation.value();
153             if (path == null || path.isEmpty() || "/".equals(path)) {
154 
155                 Errors.warning(invocable.getHandlingMethod(),
156                         LocalizationMessages.METHOD_EMPTY_PATH_ANNOTATION(
157                                 invocable.getHandlingMethod().getName(), invocable.getHandler().getHandlerClass().getName()));
158 
159             }
160         }
161 
162     }
163 
164     private void checkValueProviders(ResourceMethod method) {
165         final List<? extends Factory<?>> valueProviders = method.getInvocable().getValueProviders(locator);
166         if (valueProviders.contains(null)) {
167             int index = valueProviders.indexOf(null);
168             Errors.fatal(method, LocalizationMessages.ERROR_PARAMETER_MISSING_VALUE_PROVIDER(index, method.getInvocable()
169                     .getHandlingMethod()));
170         }
171     }
172 
173     private void visitSubResourceLocator(ResourceMethod locator) {
174         checkParameters(locator);
175         checkValueProviders(locator);
176 
177         final Invocable invocable = locator.getInvocable();
178         if (void.class == invocable.getRawResponseType()) {
179             Errors.fatal(locator, LocalizationMessages.SUBRES_LOC_RETURNS_VOID(invocable.getHandlingMethod()));
180         }
181     }
182 
183     private void checkParameters(ResourceMethod method) {
184         final Invocable invocable = method.getInvocable();
185         final Method handlingMethod = invocable.getHandlingMethod();
186         int paramCount = 0;
187         int nonAnnotatedParameters = 0;
188 
189         for (Parameter p : invocable.getParameters()) {
190             validateParameter(p, handlingMethod, handlingMethod.toGenericString(), Integer.toString(++paramCount), false);
191             if (method.getType() == ResourceMethod.JaxrsType.SUB_RESOURCE_LOCATOR
192                     && Parameter.Source.ENTITY == p.getSource()) {
193                 Errors.fatal(method, LocalizationMessages.SUBRES_LOC_HAS_ENTITY_PARAM(invocable.getHandlingMethod()));
194             } else if (p.getAnnotations().length == 0) {
195                 nonAnnotatedParameters++;
196                 if (nonAnnotatedParameters > 1) {
197                     Errors.fatal(method, LocalizationMessages.AMBIGUOUS_NON_ANNOTATED_PARAMETER(invocable.getHandlingMethod(),
198                             invocable.getHandlingMethod().getDeclaringClass()));
199                 }
200             }
201         }
202     }

204     private static final Set<Class> PARAM_ANNOTATION_SET = createParamAnnotationSet();
205 
206     private static Set<Class> createParamAnnotationSet() {
207         Set<Class> set = new HashSet<>(6);
208         set.add(HeaderParam.class);
209         set.add(CookieParam.class);
210         set.add(MatrixParam.class);
211         set.add(QueryParam.class);
212         set.add(PathParam.class);
213         set.add(BeanParam.class);
214         return Collections.unmodifiableSet(set);
215     }

217     /**
218      * Validate a single parameter instance.
219      *
220      * @param parameter             parameter to be validated.
221      * @param source                parameter source; used for issue reporting.
222      * @param reportedSourceName    source name; used for issue reporting.
223      * @param reportedParameterName parameter name; used for issue reporting.
224      * @param injectionsForbidden   {@code true} if parameters cannot be injected by
225      *                              parameter annotations, eg. {@link HeaderParam @HeaderParam}.
226      */
227     static void validateParameter(final Parameter parameter,
228                                   final Object source,
229                                   final String reportedSourceName,
230                                   final String reportedParameterName,
231                                   final boolean injectionsForbidden) {
232         Errors.processWithException(new Runnable() {
233             @Override
234             public void run() {
235                 int counter = 0;
236                 final Annotation[] annotations = parameter.getAnnotations();
237                 for (Annotation a : annotations) {
238                     if (PARAM_ANNOTATION_SET.contains(a.annotationType())) {
239                         if (injectionsForbidden) {
240                             Errors.fatal(source, LocalizationMessages.SINGLETON_INJECTS_PARAMETER(reportedSourceName,
241                                     reportedParameterName));
242                             break;
243                         }
244                         counter++;
245                         if (counter > 1) {
246                             Errors.warning(source, LocalizationMessages.AMBIGUOUS_PARAMETER(reportedSourceName,
247                                     reportedParameterName));
248                             break;
249                         }
250                     }
251                 }
252 
253                 final Type paramType = parameter.getType();
254                 if (!isConcreteType(paramType)) {
255                     Errors.warning(source, LocalizationMessages.PARAMETER_UNRESOLVABLE(reportedParameterName, paramType,
256                             reportedSourceName));
257                 }
258             }
259         });
260     }

262     @SuppressWarnings("ChainOfInstanceofChecks")
263     private static boolean isConcreteType(Type t) {
264         if (t instanceof ParameterizedType) {
265             return isConcreteParameterizedType((ParameterizedType) t);
266         } else if (!(t instanceof Class)) {
267             // GenericArrayType, WildcardType, TypeVariable
268             return false;
269         }
270 
271         return true;
272     }
273 
274     private static boolean isConcreteParameterizedType(ParameterizedType pt) {
275         boolean isConcrete = true;
276         for (Type t : pt.getActualTypeArguments()) {
277             isConcrete &= isConcreteType(t);
278         }
279 
280         return isConcrete;
281     }
