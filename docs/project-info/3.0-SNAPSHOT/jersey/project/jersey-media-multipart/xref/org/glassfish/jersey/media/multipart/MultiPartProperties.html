41  package org.glassfish.jersey.media.multipart;
42  
43  import java.io.IOException;
44  import java.io.InputStream;
45  import java.util.Properties;
46  
47  import javax.ws.rs.ext.ContextResolver;
48  
49  import org.glassfish.jersey.internal.util.PropertiesClass;
50  
51  /**
52   * Injectable JavaBean containing the configuration parameters for
53   * {@code jersey-multipart} as used in this particular application.
54   *
55   * @author Craig McClanahan
56   * @author Paul Sandoz
57   * @author Michal Gajdos
58   */
59  @PropertiesClass
60  public class MultiPartProperties {
61  
62      /**
63       * Default threshold size for buffer.
64       */
65      public static final int DEFAULT_BUFFER_THRESHOLD = 4096;
66  
67      /**
68       * Name of a properties resource that (if found in the classpath
69       * for this application) will be used to configure the settings returned
70       * by our getter methods.
71       */
72      public static final String MULTI_PART_CONFIG_RESOURCE = "jersey-multipart-config.properties";
73  
74      /**
75       * Name of the resource property for the threshold size (in bytes) above which a body part entity will be
76       * buffered to disk instead of being held in memory.
77       *
78       * The default value is {@value #DEFAULT_BUFFER_THRESHOLD}.
79       */
80      public static final String BUFFER_THRESHOLD = "jersey.config.multipart.bufferThreshold";
81  
82      /**
83       * The {@link #BUFFER_THRESHOLD} property value to keep a body part entity in memory only.
84       */
85      public static final int BUFFER_THRESHOLD_MEMORY_ONLY = -1;
86  
87      /**
88       * Name of the resource property for the directory to store temporary files containing body parts of multipart message that
89       * extends allowed memory threshold..
90       *
91       * The default value is not set (will be taken from {@code java.io.tmpdir} system property).
92       */
93      public static final String TEMP_DIRECTORY = "jersey.config.multipart.tempDir";
94  
95      /**
96       * The threshold size (in bytes) above which a body part entity will be
97       * buffered to disk instead of being held in memory.
98       */
99      private int bufferThreshold = DEFAULT_BUFFER_THRESHOLD;
100 
101     /**
102      * Directory to store temporary files containing body parts of multipart message that extends allowed memory threshold.
103      */
104     private String tempDir = null;
105 
106     /**
107      * Load and customize (if necessary) the configuration values for the
108      * {@code jersey-multipart} injection binder.
109      *
110      * @throws IllegalArgumentException if the configuration resource
111      *                                  exists, but there are problems reading it
112      */
113     public MultiPartProperties() {
114         configure();
115     }
116 
117     /**
118      * Get the size (in bytes) of the entity of an incoming
119      * {@link BodyPart} before it will be buffered to disk.  If not
120      * customized, the default value is 4096.
121      *
122      * @return return threshold size for starting to buffer the incoming entity
123      *         to disk.
124      */
125     public int getBufferThreshold() {
126         return bufferThreshold;
127     }
128 
129     /**
130      * Get the directory to store temporary files containing body parts of multipart message that extends allowed memory
131      * threshold.
132      *
133      * @return path to the temporary directory.
134      * @since 2.4.1
135      */
136     public String getTempDir() {
137         return tempDir;
138     }
139 
140     /**
141      * Set the size (in bytes) of the entity of an incoming {@link BodyPart} before it will be buffered to disk.
142      *
143      * @param threshold size of body part.
144      * @return {@code MultiPartProperties} instance.
145      * @since 2.4.1
146      */
147     public MultiPartProperties bufferThreshold(final int threshold) {
148         this.bufferThreshold = threshold < BUFFER_THRESHOLD_MEMORY_ONLY ? BUFFER_THRESHOLD_MEMORY_ONLY : threshold;
149         return this;
150     }
151 
152     /**
153      * Set the path to the directory to store temporary files containing body parts of multipart message that extends allowed
154      * memory threshold.
155      *
156      * @param path path to the temporary directory.
157      * @return {@code MultiPartProperties} instance.
158      * @since 2.4.1
159      */
160     public MultiPartProperties tempDir(final String path) {
161         this.tempDir = path;
162         return this;
163     }
164 
165     /**
166      * Configure the values returned by this instance's getters based on
167      * the contents of a properties resource, if it exists on the classpath
168      * for this application.
169      *
170      * @throws IllegalArgumentException if the configuration resource
171      *                                  exists, but there are problems reading it
172      */
173     private void configure() {
174         // Identify the class loader we will use
175         ClassLoader loader = Thread.currentThread().getContextClassLoader();
176         if (loader == null) {
177             loader = this.getClass().getClassLoader();
178         }
179 
180         // Attempt to find our properties resource
181         InputStream stream = null;
182         try {
183             stream = loader.getResourceAsStream(MULTI_PART_CONFIG_RESOURCE);
184             if (stream == null) {
185                 return;
186             }
187             final Properties props = new Properties();
188             props.load(stream);
189 
190             if (props.containsKey(BUFFER_THRESHOLD)) {
191                 this.bufferThreshold = Integer.parseInt(props.getProperty(BUFFER_THRESHOLD));
192             }
193             if (props.containsKey(TEMP_DIRECTORY)) {
194                 this.tempDir = props.getProperty(TEMP_DIRECTORY);
195             }
196         } catch (final IOException e) {
197             throw new IllegalArgumentException(e);
198         } finally {
199             if (stream != null) {
200                 try {
201                     stream.close();
202                 } catch (final IOException e) {
203                     // Pass through
204                 }
205             }
206         }
207     }
208 
209     /**
210      * Create a {@link ContextResolver context resolver} for a current state of this {@code MultiPartProperties}.
211      *
212      * @return context resolver for this config.
213      * @since 2.4.1
214      */
215     public ContextResolver<MultiPartProperties> resolver() {
216         return new ContextResolver<MultiPartProperties>() {
217 
218             @Override
219             public MultiPartProperties getContext(final Class<?> type) {
220                 return MultiPartProperties.this;
221             }
222         };
223     }
