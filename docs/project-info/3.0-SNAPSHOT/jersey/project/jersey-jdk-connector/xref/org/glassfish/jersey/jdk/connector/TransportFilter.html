67  /**
68   * Writes and reads data to and from a socket. Only one {@link #write(java.nio.ByteBuffer,
69   * org.glassfish.jersey.jdk.connector.CompletionHandler)}
70   * method call can be processed at a time. Only one {@link #_read(java.nio.ByteBuffer)} operation is supported at a time,
71   * another one is started only after the previous one has completed. Blocking in {@link #onRead(Object)}
72   * or {@link #onConnect()} method will result in data not being read from a socket until these methods have completed.
73   *
74   * @author Petr Janouch (petr.janouch at oracle.com)
75   */
76  class TransportFilter extends Filter<ByteBuffer, ByteBuffer, Void, ByteBuffer> {

78      private static final Logger LOGGER = Logger.getLogger(TransportFilter.class.getName());
79      private static final AtomicInteger openedConnections = new AtomicInteger(0);
80      private static final ScheduledExecutorService connectionCloseScheduler = Executors.newSingleThreadScheduledExecutor(r -> {
81          Thread thread = new Thread(r);
82          thread.setName("jdk-connector-container-idle-timeout");
83          thread.setDaemon(true);
84          return thread;
85      });

88      private static volatile ScheduledFuture<?> closeWaitTask;

91       * {@link ThreadPoolConfig} current {@link #channelGroup} has been created with.

97      private static volatile Integer currentContainerIdleTimeout;

105     /**
106      * Constructor.
107      * <p/>
108      * If the channel group is not active (all connections have been closed and the shutdown timeout is running) and a new
109      * transport is created with tread pool configuration different from the one of the current thread pool, the current
110      * thread pool will be shut down and a new one created with the new configuration.
111      *
112      * @param inputBufferSize      size of buffer to be allocated for reading data from a socket.
113      * @param threadPoolConfig     thread pool configuration used for creating thread pool.
114      * @param containerIdleTimeout idle time after which the shared thread pool will be destroyed.
115      */
116     TransportFilter(int inputBufferSize, ThreadPoolConfig threadPoolConfig, int containerIdleTimeout) {
117         super(null);
118         this.inputBufferSize = inputBufferSize;
119         this.threadPoolConfig = threadPoolConfig;
120         this.containerIdleTimeout = containerIdleTimeout;
121     }

123     @Override
124     void write(ByteBuffer data,
125                final org.glassfish.jersey.jdk.connector.CompletionHandler<ByteBuffer> completionHandler) {
126         socketChannel.isOpen();
127         socketChannel.write(data, data, new CompletionHandler<Integer, ByteBuffer>() {
128 
129             @Override
130             public void completed(Integer result, ByteBuffer buffer) {
131                 if (buffer.hasRemaining()) {
132                     write(buffer, completionHandler);
133                     return;
134                 }
135                 completionHandler.completed(buffer);
136             }
137 
138             @Override
139             public void failed(Throwable exc, ByteBuffer buffer) {
140                 completionHandler.failed(exc);
141             }
142         });
143     }

145     @Override
146     void close() {
147         if (socketChannel == null || !socketChannel.isOpen()) {
148             return;
149         }
150         try {
151             socketChannel.close();
152         } catch (IOException e) {
153             LOGGER.log(Level.INFO, LocalizationMessages.TRANSPORT_CONNECTION_NOT_CLOSED(), e);
154         }
155         synchronized (TransportFilter.class) {
156             openedConnections.decrementAndGet();
157             if (openedConnections.get() == 0 && channelGroup != null) {
158                 scheduleClose();
159             }
160         }
161     }

163     @Override
164     void startSsl() {
165         onSslHandshakeCompleted();
166     }

168     @Override
169     public void handleConnect(SocketAddress serverAddress, Filter upstreamFilter) {
170         this.upstreamFilter = upstreamFilter;
171 
172         try {
173             synchronized (TransportFilter.class) {
174                 updateThreadPoolConfig();
175                 initializeChannelGroup();
176                 socketChannel = AsynchronousSocketChannel.open(channelGroup);
177                 openedConnections.incrementAndGet();
178             }
179         } catch (IOException e) {
180             onError(e);
181             return;
182         }
183 
184         socketChannel.connect(serverAddress, null, new CompletionHandler<Void, Void>() {
185 
186             @Override
187             public void completed(Void result, Void nothing) {
188                 final ByteBuffer inputBuffer = ByteBuffer.allocate(inputBufferSize);
189                 onConnect();
190                 _read(inputBuffer);
191             }
192 
193             @Override
194             public void failed(Throwable exc, Void nothing) {
195                 onError(exc);
196 
197                 try {
198                     socketChannel.close();
199                 } catch (IOException e) {
200                     LOGGER.log(Level.FINE, LocalizationMessages.TRANSPORT_CONNECTION_NOT_CLOSED(), exc.getMessage());
201                 }
202             }
203         });
204     }

206     private void updateThreadPoolConfig() {
207 
208         // the channel group is active, no change in configuration
209         if (openedConnections.get() != 0) {
210             return;
211         }
212 
213         // check if the new configuration is different from the one of the current container
214         if (!threadPoolConfig.equals(currentThreadPoolConfig) || containerIdleTimeout != currentContainerIdleTimeout) {
215 
216             currentThreadPoolConfig = threadPoolConfig;
217             currentContainerIdleTimeout = containerIdleTimeout;
218 
219             if (channelGroup == null) {
220                 // the channel group has not been initialized (this is a first client) - no need to shut it down
221                 return;
222             }
223 
224             closeWaitTask.cancel(true);
225             closeWaitTask = null;
226             channelGroup.shutdown();
227             channelGroup = null;
228         }
229     }
230 
231     private void initializeChannelGroup() throws IOException {
232         if (closeWaitTask != null) {
233             closeWaitTask.cancel(true);
234             closeWaitTask = null;
235         }
236 
237         if (channelGroup == null) {
238             ThreadFactory threadFactory = threadPoolConfig.getThreadFactory();
239             if (threadFactory == null) {
240                 threadFactory = new TransportThreadFactory(threadPoolConfig);
241             }
242 
243             ExecutorService executor;
244             if (threadPoolConfig.getQueue() != null) {
245                 executor = new QueuingExecutor(threadPoolConfig.getCorePoolSize(), threadPoolConfig.getMaxPoolSize(),
246                         threadPoolConfig.getKeepAliveTime(TimeUnit.MILLISECONDS), TimeUnit.MILLISECONDS,
247                         threadPoolConfig.getQueue(), false, threadFactory);
248             } else {
249                 int taskQueueLimit = threadPoolConfig.getQueueLimit();
250                 if (taskQueueLimit == -1) {
251                     taskQueueLimit = Integer.MAX_VALUE;
252                 }
253 
254                 executor = new QueuingExecutor(threadPoolConfig.getCorePoolSize(), threadPoolConfig.getMaxPoolSize(),
255                         threadPoolConfig.getKeepAliveTime(TimeUnit.MILLISECONDS), TimeUnit.MILLISECONDS,
256                         new LinkedBlockingDeque<>(taskQueueLimit), true, threadFactory);
257             }
258 
259             // Thread pool is owned by the channel group and will be shut down when channel group is shut down
260             channelGroup = AsynchronousChannelGroup.withCachedThreadPool(executor, threadPoolConfig.getCorePoolSize());
261         }
262     }

264     private void _read(final ByteBuffer inputBuffer) {
265         /**
266          * It must be checked that the channel has not been closed by {@link #close()} method.
267          */
268         if (!socketChannel.isOpen()) {
269             return;
270         }
271 
272         socketChannel.read(inputBuffer, null, new CompletionHandler<Integer, Void>() {
273             @Override
274             public void completed(Integer bytesRead, Void result) {
275                 // connection closed by the server
276                 if (bytesRead == -1) {
277                     // close will set TransportFilter.this.upstreamFilter to null
278                     Filter upstreamFilter = TransportFilter.this.upstreamFilter;
279                     close();
280                     upstreamFilter.onConnectionClosed();
281                     return;
282                 }
283 
284                 inputBuffer.flip();
285                 onRead(inputBuffer);
286                 inputBuffer.compact();
287                 _read(inputBuffer);
288             }
289 
290             @Override
291             public void failed(Throwable exc, Void result) {
292                 /**
293                  * Reading from the channel will fail if it is closing. In such cases {@link java.nio.channels
294                  * .AsynchronousCloseException}
295                  * is thrown. This should not be logged and no action undertaken.
296                  */
297                 if (exc instanceof AsynchronousCloseException) {
298                     return;
299                 }
300 
301                 onError(exc);
302             }
303         });
304     }
305 
306     private void scheduleClose() {
307         closeWaitTask = connectionCloseScheduler.schedule(() -> {
308             synchronized (TransportFilter.class) {
309                 if (closeWaitTask == null) {
310                     return;
311                 }
312                 channelGroup.shutdown();
313                 channelGroup = null;
314                 closeWaitTask = null;
315             }
316         }, currentContainerIdleTimeout, TimeUnit.MILLISECONDS);
317     }

319     /**
320      * A default thread factory that gets used if {@link ThreadPoolConfig#getThreadFactory()}
321      * is not specified.
322      */
323     private static class TransportThreadFactory implements ThreadFactory {
324 
325         private static final String THREAD_NAME_BASE = " jdk-connector-";
326         private static final AtomicInteger threadCounter = new AtomicInteger(0);
327 
328         private final ThreadPoolConfig threadPoolConfig;
329 
330         TransportThreadFactory(ThreadPoolConfig threadPoolConfig) {
331             this.threadPoolConfig = threadPoolConfig;
332         }
333 
334         @Override
335         public Thread newThread(Runnable r) {
336             final Thread thread = new Thread(r);
337             thread.setName(THREAD_NAME_BASE + threadCounter.incrementAndGet());
338             thread.setPriority(threadPoolConfig.getPriority());
339             thread.setDaemon(threadPoolConfig.isDaemon());
340 
341             try {
342                 AccessController.doPrivileged(new PrivilegedAction<Void>() {
343                     @Override
344                     public Void run() {
345                         if (threadPoolConfig.getInitialClassLoader() == null) {
346                             thread.setContextClassLoader(this.getClass().getClassLoader());
347                         } else {
348                             thread.setContextClassLoader(threadPoolConfig.getInitialClassLoader());
349                         }
350                         return null;
351                     }
352                 });
353             } catch (Throwable t) {
354                 // just log - client still can work without setting context class loader
355                 LOGGER.log(Level.WARNING, LocalizationMessages.TRANSPORT_SET_CLASS_LOADER_FAILED(), t);
356             }
357 
358             return thread;
359         }
360     }

362     /**
363      * A thread pool executor that prefers creating new worker threads over queueing tasks until the maximum poll size
364      * has been reached, after which it will start queueing tasks.
365      */
366     private static class QueuingExecutor extends ThreadPoolExecutor {
367 
368         private final Queue<Runnable> taskQueue;
369         private final boolean threadSafeQueue;
370 
371         /**
372          * Constructor.
373          *
374          * @param threadSafeQueue indicates if {@link #taskQueue} is thread safe or not.
375          */
376         QueuingExecutor(int corePoolSize,
377                         int maximumPoolSize,
378                         long keepAliveTime,
379                         TimeUnit unit,
380                         Queue<Runnable> taskQueue,
381                         boolean threadSafeQueue,
382                         ThreadFactory threadFactory) {
383             super(corePoolSize, maximumPoolSize, keepAliveTime, unit, new HandOffQueue(taskQueue, threadSafeQueue),
384                     threadFactory);
385             this.taskQueue = taskQueue;
386             this.threadSafeQueue = threadSafeQueue;
387         }
388 
389         /**
390          * Submit a task for execution, if the maximum thread limit has been reached and all the threads are occupied,
391          * enqueue the task. The task is not executed by the current thread, but by a thread from the thread pool.
392          *
393          * @param task to be executed.
394          */
395         @Override
396         public void execute(Runnable task) {
397             try {
398                 super.execute(task);
399             } catch (RejectedExecutionException e) {
400 
401                 /* execution has been rejected either because the executor has been shut down or all worker threads are
402                  * busy - check the former one
403                  */
404                 if (isShutdown()) {
405                     throw new RejectedExecutionException(LocalizationMessages.TRANSPORT_EXECUTOR_CLOSED(), e);
406                 }
407 
408                 /* All threads are occupied, try enqueuing the task.
409                  * Each worker thread checks the queue after it has finished executing its task.
410                  */
411                 if (threadSafeQueue) {
412                     if (!taskQueue.offer(task)) {
413                         throw new RejectedExecutionException(LocalizationMessages.TRANSPORT_EXECUTOR_QUEUE_LIMIT_REACHED(), e);
414                     }
415                 } else {
416                     synchronized (taskQueue) {
417                         if (!taskQueue.offer(task)) {
418                             throw new RejectedExecutionException(LocalizationMessages.TRANSPORT_EXECUTOR_QUEUE_LIMIT_REACHED(),
419                                     e);
420                         }
421                     }
422                 }
423 
424                 /**
425                  * There is a small time interval between a worker thread checks {@link #taskQueue} and it starts to block
426                  * waiting for a new tasks to be submitted (Ideally checking that the {@link #taskQueue} is empty and starting to
427                  * block at the task hand off queue would be atomic). This can be detected by the situation when a thread
428                  * submitting
429                  * a new tasks has been rejected, but not all worker threads are active (However this does not indicate
430                  * exclusively
431                  * the problematic situation).
432                  */
433                 if (getActiveCount() < getMaximumPoolSize()) {
434                     /*
435                      * There is no guarantee that the same tasks that has been enqueued above will be dequeued,
436                      * but trying to execute one arbitrary task by everyone in this situation is enough to clear the queue.
437                      */
438                     Runnable dequeuedTask;
439                     if (threadSafeQueue) {
440                         dequeuedTask = taskQueue.poll();
441                     } else {
442                         synchronized (taskQueue) {
443                             dequeuedTask = taskQueue.poll();
444                         }
445                     }
446 
447                     // check if the task has not been consumed by a worker thread after all
448                     if (dequeuedTask != null) {
449                         execute(dequeuedTask);
450                     }
451                 }
452             }
453         }
454 
455         /**
456          * Synchronous queue that tries to empty {@link #taskQueue} before it blocks waiting for new tasks to be submitted.
457          * It is passed to {@link ThreadPoolExecutor}, where it is used used to hand off tasks from task-submitting
458          * thread to worker threads.
459          */
460         private static class HandOffQueue extends SynchronousQueue<Runnable> {
461 
462             private static final long serialVersionUID = -1607064661828834847L;
463             private final Queue<Runnable> taskQueue;
464             private final boolean threadSafeQueue;
465 
466             private HandOffQueue(Queue<Runnable> taskQueue, boolean threadSafeQueue) {
467                 this.taskQueue = taskQueue;
468                 this.threadSafeQueue = threadSafeQueue;
469             }
470 
471             @Override
472             public Runnable take() throws InterruptedException {
473                 // try to empty the task queue
474                 Runnable task;
475                 if (threadSafeQueue) {
476                     task = taskQueue.poll();
477                 } else {
478                     synchronized (taskQueue) {
479                         task = taskQueue.poll();
480                     }
481                 }
482                 if (task != null) {
483                     return task;
484                 }
485 
486                 // block and wait for a task to be submitted
487                 return super.take();
488             }
489 
490             @Override
491             public Runnable poll(long timeout, TimeUnit unit) throws InterruptedException {
492                 // try to empty the task queue
493                 Runnable task;
494                 if (threadSafeQueue) {
495                     task = taskQueue.poll();
496                 } else {
497                     synchronized (taskQueue) {
498                         task = taskQueue.poll();
499                     }
500                 }
501                 if (task != null) {
502                     return task;
503                 }
504 
505                 // block and wait for a task to be submitted
506                 return super.poll(timeout, unit);
507             }
508         }
509     }
