78  /**
79   * {@link ModelProcessor Model processor} enhancing (sub-)resources with {@value HttpMethod#GET} methods responsible of producing
80   * implicit {@link org.glassfish.jersey.server.mvc.Viewable viewables}.
81   * <p/>
82   * Note: Resource classes has to be annotated with {@link Template} annotation in order to be enhanced by this model processor.
83   *
84   * @author Michal Gajdos
85   * @see Template
86   */
87  class TemplateModelProcessor implements ModelProcessor {

90       * Path parameter representing implicit template name.
91       */
92      private static final String IMPLICIT_VIEW_PATH_PARAMETER = "implicit-view-path-parameter";
93      private static final String IMPLICIT_VIEW_PATH_PARAMETER_TEMPLATE = "{" + IMPLICIT_VIEW_PATH_PARAMETER + "}";

97      private final Provider<ExtendedUriInfo> extendedUriInfoProvider;
98      private final Provider<ConfiguredValidator> validatorProvider;

100     /**
101      * Inflector producing response with {@link org.glassfish.jersey.server.mvc.spi.ResolvedViewable resolved viewable} where
102      * model is the resource class annotated with {@link Template} or 404 as its status code.
103      */
104     private class TemplateInflectorImpl implements TemplateInflector, Inflector<ContainerRequestContext, Response> {
105 
106         private final String templateName;
107 
108         private final Class<?> resourceClass;
109         private final Object resourceInstance;
110 
111         private Class<?> modelClass;
112 
113         /**
114          * Create enhancing template {@link Inflector inflector} method.
115          *
116          * @param templateName     template name for the produced {@link org.glassfish.jersey.server.mvc.Viewable viewable}.
117          * @param resourceClass    model class for the produced {@link org.glassfish.jersey.server.mvc.Viewable viewable}.
118          *                         Should not be {@code null}.
119          * @param resourceInstance model for the produced {@link org.glassfish.jersey.server.mvc.Viewable viewable}. May be
120          *                         {@code null}.
121          */
122         private TemplateInflectorImpl(final String templateName, final Class<?> resourceClass,
123                                       final Object resourceInstance) {
124             this.templateName = templateName;
125 
126             this.resourceClass = resourceClass;
127             this.resourceInstance = resourceInstance;
128         }

130         @Override
131         public Response apply(ContainerRequestContext requestContext) {
132             final List<String> templateNames = getTemplateNames(requestContext);
133             final Object model = getModel(extendedUriInfoProvider.get());
134 
135             // Validate resource class.
136             final ConfiguredValidator validator = validatorProvider.get();
137             if (validator != null) {
138                 validator.validateResourceAndInputParams(model, null, null);
139             }
140 
141             return Response.ok().entity(new ImplicitViewable(templateNames, model, resourceClass)).build();
142         }
143 
144         @Override
145         public Class<?> getModelClass() {
146             return modelClass;
147         }
148 
149         private Object setModelClass(final Object model) {
150             if (modelClass == null) {
151                 modelClass = model.getClass();
152             }
153             return model;
154         }

156         /**
157          * Obtains a model object for a viewable.
158          *
159          * @param extendedUriInfo uri info to obtain last matched resource from.
160          * @return a model object.
161          */
162         private Object getModel(final ExtendedUriInfo extendedUriInfo) {
163             final List<Object> matchedResources = extendedUriInfo.getMatchedResources();
164 
165             if (resourceInstance != null) {
166                 return setModelClass(resourceInstance);
167             } else if (matchedResources.size() > 1) {
168                 return setModelClass(matchedResources.get(1));
169             } else {
170                 return setModelClass(resourceContext.getResource(resourceClass));
171             }
172         }
173 
174         /**
175          * Returns a list of template names to be considered as candidates for resolving
176          * {@link org.glassfish.jersey.server.mvc.Viewable viewable} into
177          * {@link org.glassfish.jersey.server.mvc.spi.ResolvedViewable resolved viewable}.
178          * <p/>
179          * Order of template names to be resolved is as follows:
180          * <ul>
181          * <li>{{@value #IMPLICIT_VIEW_PATH_PARAMETER}} value</li>
182          * <li>{@link org.glassfish.jersey.server.mvc.Template#name()}</li>
183          * <li>last sub-resource locator path</li>
184          * <li>index</li>
185          * </ul>
186          *
187          * @param requestContext request context to obtain {@link #IMPLICIT_VIEW_PATH_PARAMETER} value from.
188          * @return a non-empty list of template names.
189          */
190         private List<String> getTemplateNames(final ContainerRequestContext requestContext) {
191             final List<String> templateNames = Lists.newArrayList();
192 
193             // Template name extracted from path param.
194             final String pathTemplate = requestContext.getUriInfo().getPathParameters().getFirst(IMPLICIT_VIEW_PATH_PARAMETER);
195             if (pathTemplate != null) {
196                 templateNames.add(pathTemplate);
197             }
198 
199             // Annotation.
200             if (this.templateName != null && !"".equals(this.templateName)) {
201                 templateNames.add(this.templateName);
202             }
203 
204             // Sub-resource path.
205             final ExtendedUriInfo uriInfo = extendedUriInfoProvider.get();
206             final List<RuntimeResource> matchedRuntimeResources = uriInfo.getMatchedRuntimeResources();
207             if (matchedRuntimeResources.size() > 1) {
208                 // > 1 to check that we matched sub-resource
209                 final RuntimeResource lastMatchedRuntimeResource = matchedRuntimeResources.get(0);
210                 final Resource lastMatchedResource = lastMatchedRuntimeResource.getResources().get(0);
211 
212                 String path = lastMatchedResource.getPath();
213 
214                 if (path != null && !IMPLICIT_VIEW_PATH_PARAMETER_TEMPLATE.equals(path)) {
215                     path = path.charAt(0) == '/' ? path.substring(1, path.length()) : path;
216                     templateNames.add(path);
217                 }
218             }
219 
220             // Index.
221             if (templateNames.isEmpty()) {
222                 templateNames.add("index");
223             }
224 
225             return templateNames;
226         }

255     /**
256      * Enhance {@link org.glassfish.jersey.server.model.RuntimeResource runtime resources} from the given
257      * {@link org.glassfish.jersey.server.model.ResourceModel resource model} with methods obtained via
258      * {@link #getEnhancingMethods(org.glassfish.jersey.server.model.RuntimeResource)}.
259      *
260      * @param resourceModel    resource model with runtime resources to enhance.
261      * @param subResourceModel determines whether the resource model represents sub-resource.
262      * @return enhanced resource model.
263      */
264     private ResourceModel processModel(final ResourceModel resourceModel, final boolean subResourceModel) {
265         ResourceModel.Builder newModelBuilder = processTemplateAnnotatedInvocables(resourceModel, subResourceModel);
266 
267         for (RuntimeResource resource : resourceModel.getRuntimeResourceModel().getRuntimeResources()) {
268             ModelProcessorUtil.enhanceResource(resource, newModelBuilder, getEnhancingMethods(resource), false);
269         }
270 
271         return newModelBuilder.build();
272     }

274     /**
275      * Process all {@link Invocable invocables} and defines
276      * {@link org.glassfish.jersey.server.model.Invocable#getRoutingResponseType() routing response types}
277      * as {@link Viewable} for all methods annotated with {@link Template}.
278      *
279      * @param resourceModel    resource model to process.
280      * @param subResourceModel determines whether the resource model represents sub-resource.
281      * @return Modified resource model.
282      */
283     private ResourceModel.Builder processTemplateAnnotatedInvocables(ResourceModel resourceModel,
284                                                                      final boolean subResourceModel) {
285         ResourceModel.Builder modelBuilder = new ResourceModel.Builder(subResourceModel);
286         for (Resource resource : resourceModel.getResources()) {
287             Resource newResource = processResource(resource);
288             modelBuilder.addResource(newResource);
289         }
290         return modelBuilder;
291     }
292 
293     private Resource processResource(Resource resource) {
294         Resource.Builder resourceBuilder = Resource.builder(resource.getPath());
295         for (ResourceMethod resourceMethod : resource.getResourceMethods()) {
296             ResourceMethod.Builder builder = resourceBuilder.addMethod(resourceMethod);
297             if (resourceMethod.getInvocable().getHandlingMethod().isAnnotationPresent(Template.class)) {
298                 builder.routingResponseType(Viewable.class);
299             }
300         }
301         if (resource.getResourceLocator() != null) {
302             resourceBuilder.addMethod(resource.getResourceLocator());
303         }
304 
305         for (Resource child : resource.getChildResources()) {
306             resourceBuilder.addChildResource(processResource(child));
307         }
308         return resourceBuilder.build();
309 
310     }

312     /**
313      * Returns a list of enhancing methods for a given {@link org.glassfish.jersey.server.model.RuntimeResource runtime
314      * resource}.
315      *
316      * @param runtimeResource runtime resource to create enhancing methods for.
317      * @return list of enhancing methods.
318      */
319     private List<ModelProcessorUtil.Method> getEnhancingMethods(final RuntimeResource runtimeResource) {
320         final List<ModelProcessorUtil.Method> newMethods = Lists.newArrayList();
321 
322         for (final Resource resource : runtimeResource.getResources()) {
323             // Handler classes.
324             for (final Class<?> handlerClass : resource.getHandlerClasses()) {
325                 createEnhancingMethods(handlerClass, null, newMethods);
326             }
327 
328             // Names - if there are no handler classes / instances.
329             if (resource.getHandlerClasses().isEmpty() && resource.getHandlerInstances().isEmpty()) {
330                 for (String resourceName : resource.getNames()) {
331                     final Class<Object> resourceClass = AccessController
332                             .doPrivileged(ReflectionHelper.classForNamePA(resourceName));
333                     if (resourceClass != null) {
334                         createEnhancingMethods(resourceClass, null, newMethods);
335                     }
336                 }
337             }
338 
339             // Handler instances.
340             Errors.process(new Producer<Void>() {
341                 @Override
342                 public Void call() {
343                     for (final Object handlerInstance : resource.getHandlerInstances()) {
344                         final Class<?> handlerInstanceClass = handlerInstance.getClass();
345 
346                         if (!resource.getHandlerClasses().contains(handlerInstanceClass)) {
347                             createEnhancingMethods(handlerInstanceClass, handlerInstance, newMethods);
348                         } else {
349                             Errors.warning(resource,
350                                     LocalizationMessages.TEMPLATE_HANDLER_ALREADY_ENHANCED(handlerInstanceClass));
351                         }
352                     }
353 
354                     return null;
355                 }
356             });
357         }
358 
359         return newMethods;
360     }

362     /**
363      * Creates enhancing methods for given resource.
364      *
365      * @param resourceClass    resource class for which enhancing methods should be created.
366      * @param resourceInstance resource instance for which enhancing methods should be created. May be {@code null}.
367      * @param newMethods       list to store new methods into.
368      */
369     private void createEnhancingMethods(final Class<?> resourceClass, final Object resourceInstance,
370                                         final List<ModelProcessorUtil.Method> newMethods) {
371         final Template template = resourceClass.getAnnotation(Template.class);
372 
373         if (template != null) {
374             final Class<?> annotatedResourceClass = ModelHelper.getAnnotatedResourceClass(resourceClass);
375 
376             final List<MediaType> produces = MediaTypes
377                     .createQualitySourceMediaTypes(annotatedResourceClass.getAnnotation(Produces.class));
378             final List<MediaType> consumes = MediaTypes.createFrom(annotatedResourceClass.getAnnotation(Consumes.class));
379 
380             final TemplateInflectorImpl inflector = new TemplateInflectorImpl(template.name(),
381                     resourceClass, resourceInstance);
382 
383             newMethods.add(new ModelProcessorUtil.Method(HttpMethod.GET, consumes, produces, inflector));
384             newMethods.add(new ModelProcessorUtil.Method(IMPLICIT_VIEW_PATH_PARAMETER_TEMPLATE, HttpMethod.GET,
385                     consumes, produces, inflector));
386         }
