68  /**
69   * An abstract implementation of {@link RuntimeDelegate} that
70   * provides support common to the client and server.
71   *
72   * @author Paul Sandoz
73   */
74  public abstract class AbstractRuntimeDelegate extends RuntimeDelegate {
75  
76      private final Set<HeaderDelegateProvider> hps;
77      private final Map<Class<?>, HeaderDelegate<?>> map;
78  
79      /**
80       * Initialization constructor. The service locator will be shut down.
81       *
82       * @param serviceLocator HK2 service locator.
83       */
84      protected AbstractRuntimeDelegate(final ServiceLocator serviceLocator) {
85          try {
86              hps = Providers.getProviders(serviceLocator, HeaderDelegateProvider.class);
87  
88              /**
89               * Construct a map for quick look up of known header classes
90               */
91              map = new WeakHashMap<Class<?>, HeaderDelegate<?>>();
92              map.put(EntityTag.class, _createHeaderDelegate(EntityTag.class));
93              map.put(MediaType.class, _createHeaderDelegate(MediaType.class));
94              map.put(CacheControl.class, _createHeaderDelegate(CacheControl.class));
95              map.put(NewCookie.class, _createHeaderDelegate(NewCookie.class));
96              map.put(Cookie.class, _createHeaderDelegate(Cookie.class));
97              map.put(URI.class, _createHeaderDelegate(URI.class));
98              map.put(Date.class, _createHeaderDelegate(Date.class));
99              map.put(String.class, _createHeaderDelegate(String.class));
100         } finally {
101             serviceLocator.shutdown();
102         }
103     }
104 
105     @Override
106     public javax.ws.rs.core.Variant.VariantListBuilder createVariantListBuilder() {
107         return new VariantListBuilder();
108     }
109 
110     @Override
111     public ResponseBuilder createResponseBuilder() {
112         return new OutboundJaxrsResponse.Builder(new OutboundMessageContext());
113     }
114 
115     @Override
116     public UriBuilder createUriBuilder() {
117         return new JerseyUriBuilder();
118     }
119 
120     @Override
121     public Link.Builder createLinkBuilder() {
122         return new JerseyLink.Builder();
123     }
124 
125     @Override
126     public <T> HeaderDelegate<T> createHeaderDelegate(final Class<T> type) {
127         if (type == null) {
128             throw new IllegalArgumentException("type parameter cannot be null");
129         }
130 
131         @SuppressWarnings("unchecked") final HeaderDelegate<T> delegate = (HeaderDelegate<T>) map.get(type);
132         if (delegate != null) {
133             return delegate;
134         }
135 
136         return _createHeaderDelegate(type);
137     }
138 
139     @SuppressWarnings("unchecked")
140     private <T> HeaderDelegate<T> _createHeaderDelegate(final Class<T> type) {
141         for (final HeaderDelegateProvider hp : hps) {
142             if (hp.supports(type)) {
143                 return hp;
144             }
145         }
146 
147         return null;
148     }
