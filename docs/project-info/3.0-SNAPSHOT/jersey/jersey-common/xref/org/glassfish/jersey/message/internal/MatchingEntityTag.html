40  package org.glassfish.jersey.message.internal;
41  
42  import java.text.ParseException;
43  import java.util.Collections;
44  import java.util.Set;
45  
46  import javax.ws.rs.core.EntityTag;
47  
48  import org.glassfish.jersey.internal.LocalizationMessages;
49  
50  /**
51   * A matching entity tag.
52   * <p>
53   * Note that this type and it's super type cannot be used to create request
54   * header values for {@code If-Match} and {@code If-None-Match}
55   * of the form {@code If-Match: *} or {@code If-None-Match: *} as
56   * {@code *} is not a valid entity tag.
57   *
58   * @author Paul Sandoz
59   * @author Marek Potociar (marek.potociar at oracle.com)
60   */
61  public class MatchingEntityTag extends EntityTag {
62  
63      /**
64       * An empty set that corresponds to {@code If-Match: *} or {@code If-None-Match: *}.
65       */
66      public static final Set<MatchingEntityTag> ANY_MATCH = Collections.emptySet();
67  
68      /**
69       * Create new strongly validating entity tag.
70       *
71       * @param value ETag header value.
72       */
73      public MatchingEntityTag(String value) {
74          super(value, false);
75      }
76  
77      /**
78       * Create new matching entity tag.
79       *
80       * @param value ETag header value.
81       * @param weak should be set to false, if strong validation is required,
82       *            otherwise should be set to true.
83       */
84      public MatchingEntityTag(String value, boolean weak) {
85          super(value, weak);
86      }
87  
88      /**
89       * Create new matching entity tag out of provided header reader.
90       *
91       * @param reader HTTP header content reader.
92       * @return a new matching entity tag.
93       * @throws ParseException in case the header could not be parsed.
94       */
95      public static MatchingEntityTag valueOf(HttpHeaderReader reader) throws ParseException {
96          final CharSequence tagString = reader.getRemainder();
97  
98          HttpHeaderReader.Event e = reader.next(false);
99          if (e == HttpHeaderReader.Event.QuotedString) {
100             return new MatchingEntityTag(reader.getEventValue().toString());
101         } else if (e == HttpHeaderReader.Event.Token) {
102             CharSequence ev = reader.getEventValue();
103             if (ev != null && ev.length() == 1 && 'W' == ev.charAt(0)) {
104                 reader.nextSeparator('/');
105                 return new MatchingEntityTag(reader.nextQuotedString().toString(), true);
106             }
107         }
108 
109         throw new ParseException(LocalizationMessages.ERROR_PARSING_ENTITY_TAG(tagString), reader.getIndex());
110     }
