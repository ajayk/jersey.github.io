215     /**
216      * Real implementation of tracing logger.
217      */
218     private static final class TracingLoggerImpl extends TracingLogger {
219 
220         private final Logger logger;
221         private final Level threshold;
222         private final TracingInfo tracingInfo;
223 
224         public TracingLoggerImpl(final Level threshold, String loggerNameSuffix) {
225             this.threshold = threshold;
226 
227             this.tracingInfo = new TracingInfo();
228 
229             loggerNameSuffix = loggerNameSuffix != null ? loggerNameSuffix : DEFAULT_LOGGER_NAME_SUFFIX;
230             this.logger = Logger.getLogger(TRACING_LOGGER_NAME_PREFIX + "." + loggerNameSuffix);
231         }
232 
233         @Override
234         public boolean isLogEnabled(final Event event) {
235             return isEnabled(event.level());
236         }
237 
238         @Override
239         public void log(final Event event, final Object... args) {
240             logDuration(event, -1, args);
241         }
242 
243         @Override
244         public void logDuration(final Event event, final long fromTimestamp, final Object... args) {
245             if (isEnabled(event.level())) {
246                 final long toTimestamp;
247                 if (fromTimestamp == -1) {
248                     toTimestamp = -1;
249                 } else {
250                     toTimestamp = System.nanoTime();
251                 }
252                 long duration = 0;
253                 if ((fromTimestamp != -1) && (toTimestamp != -1)) {
254                     duration = toTimestamp - fromTimestamp;
255                 }
256                 logImpl(event, duration, args);
257             }
258         }
259 
260         @Override
261         public long timestamp(final Event event) {
262             if (isEnabled(event.level())) {
263                 return System.nanoTime();
264             }
265             return -1;
266         }
267 
268         @Override
269         public void flush(final MultivaluedMap<String, Object> headers) {
270             final String[] messages = tracingInfo.getMessages();
271             for (int i = 0; i < messages.length; i++) {
272                 headers.putSingle(String.format(TracingLogger.HEADER_RESPONSE_FORMAT, i), messages[i]);
273             }
274         }
275 
276         /**
277          * Log message for specified event type.
278          *
279          * The event contains name, category, level and also message format.
280          * Message format will be formatted by parameter {@code messageArgs} is used to format event.
281          * If there is no message format then each message arg is separated by space.
282          * Final message also contains event name (JDK Log) or category (HTTP header) and time stamp.
283          *
284          * @param event       Event type of log
285          * @param duration    Time duration of logged event. Can be {@code 0}.
286          * @param messageArgs message arguments
287          */
288         private void logImpl(final Event event, final long duration, final Object... messageArgs) {
289             if (isEnabled(event.level())) {
290                 final String[] messageArgsStr = new String[messageArgs.length];
291                 for (int i = 0; i < messageArgs.length; i++) {
292                     messageArgsStr[i] = formatInstance(messageArgs[i]);
293                 }
294                 final TracingInfo.Message message = new TracingInfo.Message(event, duration, messageArgsStr);
295                 tracingInfo.addMessage(message);
296 
297                 final java.util.logging.Level loggingLevel;
298                 switch (event.level()) {
299                     case SUMMARY:
300                         loggingLevel = java.util.logging.Level.FINE;
301                         break;
302                     case TRACE:
303                         loggingLevel = java.util.logging.Level.FINER;
304                         break;
305                     case VERBOSE:
306                         loggingLevel = java.util.logging.Level.FINEST;
307                         break;
308                     default:
309                         loggingLevel = java.util.logging.Level.OFF;
310                 }
311                 if (logger.isLoggable(loggingLevel)) {
312                     logger.log(loggingLevel,
313                             event.name() + ' ' + message.toString() + " [" + TracingInfo.formatDuration(duration) + " ms]");
314                 }
315             }
316         }
317 
318         private boolean isEnabled(final Level level) {
319             return threshold.ordinal() >= level.ordinal();
320         }
321 
322         /**
323          * Format info of instance.
324          *
325          * It shows its class name, identity hash code, and following info if available: priority value, response detail.
326          *
327          * @param instance instance to be formatted
328          * @return Formatted info of instance.
329          */
330         private static String formatInstance(final Object instance) {
331             final StringBuilder textSB = new StringBuilder();
332             if (instance == null) {
333                 textSB.append("null");
334             } else if ((instance instanceof Number) || (instance instanceof String) || (instance instanceof Method)) {
335                 textSB.append(instance.toString());
336             } else if (instance instanceof Response.StatusType) {
337                 textSB.append(formatStatusInfo((Response.StatusType) instance));
338             } else {
339                 textSB.append('[');
340                 formatInstance(instance, textSB);
341                 if (instance.getClass().isAnnotationPresent(Priority.class)) {
342                     textSB.append(" #").append(instance.getClass().getAnnotation(Priority.class).value());
343                 }
344                 if (instance instanceof WebApplicationException) {
345                     formatResponse(((WebApplicationException) instance).getResponse(), textSB);
346                 } else if (instance instanceof Response) {
347                     formatResponse(((Response) instance), textSB);
348                 }
349                 textSB.append(']');
350             }
351             return textSB.toString();
352         }
353 
354         /**
355          * Basic format of instance - just class name and identity hash code.
356          *
357          * @param instance instance to be formatted
358          * @param textSB   Formatted info will be appended to {@code StringBuilder}
359          */
360         private static void formatInstance(final Object instance, final StringBuilder textSB) {
361             textSB.append(instance.getClass().getName()).append(" @")
362                     .append(Integer.toHexString(System.identityHashCode(instance)));
363         }
364 
365         /**
366          * Format of response - status code, status family, reason phrase and info about entity.
367          *
368          * @param response response to be formatted
369          * @param textSB   Formatted info will be appended to {@code StringBuilder}
370          */
371         private static void formatResponse(final Response response, final StringBuilder textSB) {
372             textSB.append(" <").append(formatStatusInfo(response.getStatusInfo())).append('|');
373             if (response.hasEntity()) {
374                 formatInstance(response.getEntity(), textSB);
375             } else {
376                 textSB.append("-no-entity-");
377             }
378             textSB.append('>');
379         }
380 
381         private static String formatStatusInfo(final Response.StatusType statusInfo) {
382             return String.valueOf(statusInfo.getStatusCode()) + '/' + statusInfo.getFamily() + '|' + statusInfo.getReasonPhrase();
383         }

404     /**
405      * Type of event.
406      */
407     public static interface Event {
408 
409         /**
410          * Name of event, should be unique.
411          * Is logged by JDK logger.
412          *
413          * @return event name.
414          */
415         public String name();
416 
417         /**
418          * Category of event, more events share same category.
419          * Is used to format response HTTP header.
420          *

422          */
423         public String category();
424 
425         /**
426          * Level of event.
427          * Is used to check if the event is logged according to application/request settings.
