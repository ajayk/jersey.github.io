175     private static InputStream getInputStream(final HttpURLConnection uc) throws IOException {
176         return new InputStream() {
177             private final UnsafeValue<InputStream, IOException> in = Values.lazy(new UnsafeValue<InputStream, IOException>() {
178                 @Override
179                 public InputStream get() throws IOException {
180                     if (uc.getResponseCode() < Response.Status.BAD_REQUEST.getStatusCode()) {
181                         return uc.getInputStream();
182                     } else {
183                         InputStream ein = uc.getErrorStream();
184                         return (ein != null) ? ein : new ByteArrayInputStream(new byte[0]);
185                     }
186                 }
187             });
188 
189             private volatile boolean closed = false;
190 
191             /**
192              * The motivation for this method is to straighten up a behaviour of {@link sun.net.www.http.KeepAliveStream} which
193              * is used here as a backing {@link InputStream}. The problem is that its access methods (e.g., {@link
194              * sun.net.www.http.KeepAliveStream#read()}) do not throw {@link IOException} if the stream is closed. This behaviour
195              * contradicts with {@link InputStream} contract.
196              * <p/>
197              * This is a part of fix of JERSEY-2878
198              * <p/>
199              * Note that {@link java.io.FilterInputStream} also changes the contract of
200              * {@link java.io.FilterInputStream#read(byte[], int, int)} as it doesn't state that closed stream causes an {@link
201              * IOException} which might be questionable. Nevertheless, our contract is {@link InputStream} and as such, the
202              * stream we're offering must comply with it.
203              *
204              * @throws IOException
205              */
206             private void throwIOExceptionIfClosed() throws IOException {
207                 if (closed) {
208                     throw new IOException("Stream closed");
209                 }
210             }
211 
212             @Override
213             public int read() throws IOException {
214                 int result = in.get().read();
215                 throwIOExceptionIfClosed();
216                 return result;
217             }
218 
219             @Override
220             public int read(byte[] b) throws IOException {
221                 int result = in.get().read(b);
222                 throwIOExceptionIfClosed();
223                 return result;
224             }
225 
226             @Override
227             public int read(byte[] b, int off, int len) throws IOException {
228                 int result = in.get().read(b, off, len);
229                 throwIOExceptionIfClosed();
230                 return result;
231             }
232 
233             @Override
234             public long skip(long n) throws IOException {
235                 long result = in.get().skip(n);
236                 throwIOExceptionIfClosed();
237                 return result;
238             }
239 
240             @Override
241             public int available() throws IOException {
242                 int result = in.get().available();
243                 throwIOExceptionIfClosed();
244                 return result;
245             }
246 
247             @Override
248             public void close() throws IOException {
249                 try {
250                     in.get().close();
251                 } finally {
252                     closed = true;
253                 }
254             }
255 
256             @Override
257             public void mark(int readLimit) {
258                 try {
259                     in.get().mark(readLimit);
260                 } catch (IOException e) {
261                     throw new IllegalStateException("Unable to retrieve the underlying input stream.", e);
262                 }
263             }
264 
265             @Override
266             public void reset() throws IOException {
267                 in.get().reset();
268                 throwIOExceptionIfClosed();
269             }
270 
271             @Override
272             public boolean markSupported() {
273                 try {
274                     return in.get().markSupported();
275                 } catch (IOException e) {
276                     throw new IllegalStateException("Unable to retrieve the underlying input stream.", e);
277                 }
278             }
279         };
280     }

312     /**
313      * Secure connection if necessary.
314      * <p/>
315      * Provided implementation sets {@link HostnameVerifier} and {@link SSLSocketFactory} to give connection, if that
316      * is an instance of {@link HttpsURLConnection}.
317      *
318      * @param client client associated with this client runtime.
319      * @param uc     http connection to be secured.
320      */
321     protected void secureConnection(final JerseyClient client, final HttpURLConnection uc) {
322         if (uc instanceof HttpsURLConnection) {
323             HttpsURLConnection suc = (HttpsURLConnection) uc;
324 
325             final HostnameVerifier verifier = client.getHostnameVerifier();
326             if (verifier != null) {
327                 suc.setHostnameVerifier(verifier);
328             }
329 
330             if (HttpsURLConnection.getDefaultSSLSocketFactory() == suc.getSSLSocketFactory()) {
331                 // indicates that the custom socket factory was not set
332                 suc.setSSLSocketFactory(sslSocketFactory.get());
333             }
334         }
335     }

337     private ClientResponse _apply(final ClientRequest request) throws IOException {
338         final HttpURLConnection uc;
339 
340         uc = this.connectionFactory.getConnection(request.getUri().toURL());
341         uc.setDoInput(true);
342 
343         final String httpMethod = request.getMethod();
344         if (request.resolveProperty(HttpUrlConnectorProvider.SET_METHOD_WORKAROUND, setMethodWorkaround)) {
345             setRequestMethodViaJreBugWorkaround(uc, httpMethod);
346         } else {
347             uc.setRequestMethod(httpMethod);
348         }
349 
350         uc.setInstanceFollowRedirects(request.resolveProperty(ClientProperties.FOLLOW_REDIRECTS, true));
351 
352         uc.setConnectTimeout(request.resolveProperty(ClientProperties.CONNECT_TIMEOUT, uc.getConnectTimeout()));
353 
354         uc.setReadTimeout(request.resolveProperty(ClientProperties.READ_TIMEOUT, uc.getReadTimeout()));
355 
356         secureConnection(request.getClient(), uc);
357 
358         final Object entity = request.getEntity();
359         if (entity != null) {
360             RequestEntityProcessing entityProcessing = request.resolveProperty(
361                     ClientProperties.REQUEST_ENTITY_PROCESSING, RequestEntityProcessing.class);
362 
363             if (entityProcessing == null || entityProcessing != RequestEntityProcessing.BUFFERED) {
364                 final int length = request.getLength();
365                 if (fixLengthStreaming && length > 0) {
366                     uc.setFixedLengthStreamingMode(length);
367                 } else if (entityProcessing == RequestEntityProcessing.CHUNKED) {
368                     uc.setChunkedStreamingMode(chunkSize);
369                 }
370             }
371             uc.setDoOutput(true);
372 
373             if ("GET".equalsIgnoreCase(httpMethod)) {
374                 final Logger logger = Logger.getLogger(HttpUrlConnector.class.getName());
375                 if (logger.isLoggable(Level.INFO)) {
376                     logger.log(Level.INFO, LocalizationMessages.HTTPURLCONNECTION_REPLACES_GET_WITH_ENTITY());
377                 }
378             }
379 
380             request.setStreamProvider(new OutboundMessageContext.StreamProvider() {
381 
382                 @Override
383                 public OutputStream getOutputStream(int contentLength) throws IOException {
384                     setOutboundHeaders(request.getStringHeaders(), uc);
385                     return uc.getOutputStream();
386                 }
387             });
388             request.writeEntity();
389 
390         } else {
391             setOutboundHeaders(request.getStringHeaders(), uc);
392         }
393 
394         final int code = uc.getResponseCode();
395         final String reasonPhrase = uc.getResponseMessage();
396         final Response.StatusType status =
397                 reasonPhrase == null ? Statuses.from(code) : Statuses.from(code, reasonPhrase);
398         final URI resolvedRequestUri;
399         try {
400             resolvedRequestUri = uc.getURL().toURI();
401         } catch (URISyntaxException e) {
402             throw new ProcessingException(e);
403         }
404 
405         ClientResponse responseContext = new ClientResponse(status, request, resolvedRequestUri);
406         responseContext.headers(Maps.filterKeys(uc.getHeaderFields(), Predicates.notNull()));
407         responseContext.setEntityStream(getInputStream(uc));
408 
409         return responseContext;
410     }

412     private void setOutboundHeaders(MultivaluedMap<String, String> headers, HttpURLConnection uc) {
413         boolean restrictedSent = false;
414         for (Map.Entry<String, List<String>> header : headers.entrySet()) {
415             String headerName = header.getKey();
416             String headerValue;
417 
418             List<String> headerValues = header.getValue();
419             if (headerValues.size() == 1) {
420                 headerValue = headerValues.get(0);
421                 uc.setRequestProperty(headerName, headerValue);
422             } else {
423                 StringBuilder b = new StringBuilder();
424                 boolean add = false;
425                 for (Object value : headerValues) {
426                     if (add) {
427                         b.append(',');
428                     }
429                     add = true;
430                     b.append(value);
431                 }
432                 headerValue = b.toString();
433                 uc.setRequestProperty(headerName, headerValue);
434             }
435             // if (at least one) restricted header was added and the allowRestrictedHeaders
436             if (!isRestrictedHeaderPropertySet && !restrictedSent) {
437                 if (isHeaderRestricted(headerName, headerValue)) {
438                     restrictedSent = true;
439                 }
440             }
441         }
442         if (restrictedSent) {
443             LOGGER.warning(LocalizationMessages.RESTRICTED_HEADER_POSSIBLY_IGNORED(ALLOW_RESTRICTED_HEADERS_SYSTEM_PROPERTY));
444         }
445     }

447     private boolean isHeaderRestricted(String name, String value) {
448         name = name.toLowerCase();
449         return name.startsWith("sec-")
450                 || restrictedHeaderSet.contains(name)
451                 && !("connection".equalsIgnoreCase(name) && "close".equalsIgnoreCase(value));
452     }

454     /**
455      * Workaround for a bug in {@code HttpURLConnection.setRequestMethod(String)}
456      * The implementation of Sun/Oracle is throwing a {@code ProtocolException}
457      * when the method is not in the list of the HTTP/1.1 default methods.
458      * This means that to use e.g. {@code PROPFIND} and others, we must apply this workaround.
459      * <p/>
460      * See issue http://java.net/jira/browse/JERSEY-639
461      */
462     private static void setRequestMethodViaJreBugWorkaround(final HttpURLConnection httpURLConnection,
463                                                             final String method) {
464         try {
465             httpURLConnection.setRequestMethod(method); // Check whether we are running on a buggy JRE
466         } catch (final ProtocolException pe) {
467             try {
468                 AccessController
469                         .doPrivileged(new PrivilegedExceptionAction<Object>() {
470                             @Override
471                             public Object run() throws NoSuchFieldException,
472                                     IllegalAccessException {
473                                 try {
474                                     httpURLConnection.setRequestMethod(method);
475                                     // Check whether we are running on a buggy
476                                     // JRE
477                                 } catch (final ProtocolException pe) {
478                                     Class<?> connectionClass = httpURLConnection
479                                             .getClass();
480                                     try {
481                                         final Field delegateField = connectionClass.getDeclaredField("delegate");
482                                         delegateField.setAccessible(true);
483 
484                                         HttpURLConnection delegateConnection =
485                                                 (HttpURLConnection) delegateField.get(httpURLConnection);
486                                         setRequestMethodViaJreBugWorkaround(delegateConnection, method);
487                                     } catch (NoSuchFieldException e) {
488                                         // Ignore for now, keep going
489                                     } catch (IllegalArgumentException e) {
490                                         throw new RuntimeException(e);
491                                     } catch (IllegalAccessException e) {
492                                         throw new RuntimeException(e);
493                                     }
494                                     try {
495                                         Field methodField;
496                                         while (connectionClass != null) {
497                                             try {
498                                                 methodField = connectionClass
499                                                         .getDeclaredField("method");
500                                             } catch (NoSuchFieldException e) {
501                                                 connectionClass = connectionClass
502                                                         .getSuperclass();
503                                                 continue;
504                                             }
505                                             methodField.setAccessible(true);
506                                             methodField.set(httpURLConnection, method);
507                                             break;
508                                         }
509                                     } catch (final Exception e) {
510                                         throw new RuntimeException(e);
511                                     }
512                                 }
513                                 return null;
514                             }
515                         });
516             } catch (final PrivilegedActionException e) {
517                 final Throwable cause = e.getCause();
518                 if (cause instanceof RuntimeException) {
519                     throw (RuntimeException) cause;
520                 } else {
521                     throw new RuntimeException(cause);
522                 }
523             }
524         }
525     }
