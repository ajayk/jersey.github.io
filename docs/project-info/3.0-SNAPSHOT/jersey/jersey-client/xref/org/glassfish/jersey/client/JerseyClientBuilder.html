56  /**
57   * Jersey provider of {@link javax.ws.rs.client.ClientBuilder JAX-RS client builder}.
58   *
59   * @author Marek Potociar (marek.potociar at oracle.com)
60   */
61  public class JerseyClientBuilder extends ClientBuilder {
62  
63      private final ClientConfig config;
64      private HostnameVerifier hostnameVerifier;
65      private SslConfigurator sslConfigurator;
66      private SSLContext sslContext;
67  
68      /**
69       * Create a new custom-configured {@link JerseyClient} instance.
70       *
71       * @return new configured Jersey client instance.
72       * @since 2.5
73       */
74      public static JerseyClient createClient() {
75          return new JerseyClientBuilder().build();
76      }
77  
78      /**
79       * Create a new custom-configured {@link JerseyClient} instance.
80       *
81       * @param configuration data used to provide initial configuration for the new
82       *                      Jersey client instance.
83       * @return new configured Jersey client instance.
84       * @since 2.5
85       */
86      public static JerseyClient createClient(Configuration configuration) {
87          return new JerseyClientBuilder().withConfig(configuration).build();
88      }
89  
90      /**
91       * Create new Jersey client builder instance.
92       */
93      public JerseyClientBuilder() {
94          this.config = new ClientConfig();
95      }
96  
97      @Override
98      public JerseyClientBuilder sslContext(SSLContext sslContext) {
99          if (sslContext == null) {
100             throw new NullPointerException(LocalizationMessages.NULL_SSL_CONTEXT());
101         }
102         this.sslContext = sslContext;
103         sslConfigurator = null;
104         return this;
105     }
106 
107     @Override
108     public JerseyClientBuilder keyStore(KeyStore keyStore, char[] password) {
109         if (keyStore == null) {
110             throw new NullPointerException(LocalizationMessages.NULL_KEYSTORE());
111         }
112         if (password == null) {
113             throw new NullPointerException(LocalizationMessages.NULL_KEYSTORE_PASWORD());
114         }
115         if (sslConfigurator == null) {
116             sslConfigurator = SslConfigurator.newInstance();
117         }
118         sslConfigurator.keyStore(keyStore);
119         sslConfigurator.keyPassword(password);
120         sslContext = null;
121         return this;
122     }
123 
124     @Override
125     public JerseyClientBuilder trustStore(KeyStore trustStore) {
126         if (trustStore == null) {
127             throw new NullPointerException(LocalizationMessages.NULL_TRUSTSTORE());
128         }
129         if (sslConfigurator == null) {
130             sslConfigurator = SslConfigurator.newInstance();
131         }
132         sslConfigurator.trustStore(trustStore);
133         sslContext = null;
134         return this;
135     }
136 
137     @Override
138     public JerseyClientBuilder hostnameVerifier(HostnameVerifier hostnameVerifier) {
139         this.hostnameVerifier = hostnameVerifier;
140         return this;
141     }
142 
143     @Override
144     public JerseyClient build() {
145         if (sslContext != null) {
146             return new JerseyClient(config, sslContext, hostnameVerifier);
147         } else if (sslConfigurator != null) {
148             final SslConfigurator sslConfiguratorCopy = sslConfigurator.copy();
149             return new JerseyClient(
150                     config,
151                     Values.lazy(new UnsafeValue<SSLContext, IllegalStateException>() {
152                         @Override
153                         public SSLContext get() {
154                             return sslConfiguratorCopy.createSSLContext();
155                         }
156                     }),
157                     hostnameVerifier);
158         } else {
159             return new JerseyClient(config, (UnsafeValue<SSLContext, IllegalStateException>) null, hostnameVerifier);
160         }
161     }
162 
163     @Override
164     public ClientConfig getConfiguration() {

166     }
