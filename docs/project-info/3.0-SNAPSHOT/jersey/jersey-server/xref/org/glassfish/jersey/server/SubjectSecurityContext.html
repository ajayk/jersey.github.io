40  package org.glassfish.jersey.server;
41  
42  import java.security.PrivilegedAction;
43  
44  import javax.ws.rs.core.SecurityContext;
45  
46  /**
47   * Security context that allows establishing a subject before a resource method
48   * or a sub-resource locator is called. Container or filters should set an
49   * implementation of this interface to the request context using
50   * {@link ContainerRequest#setSecurityContext(javax.ws.rs.core.SecurityContext)}.
51   *
52   * When Jersey detects this kind of context is in the request scope,
53   * it will use {@link #doAsSubject(java.security.PrivilegedAction)} method to
54   * dispatch the request to a resource method (or to call a sub-resource locator).
55   *
56   * @author Martin Matula
57   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
58   * @author Marek Potociar (marek.potociar at oracle.com)
59   */
60  public interface SubjectSecurityContext extends SecurityContext {
61  
62      /**
63       * Jersey wraps calls to resource methods and sub-resource locators in
64       * {@link PrivilegedAction} instance and passes it to this method when
65       * dispatching a request. Implementations should do the needful to establish
66       * a {@link javax.security.auth.Subject} and invoke the {@link PrivilegedAction}
67       * passed as the parameter using
68       * {@link javax.security.auth.Subject#doAs(javax.security.auth.Subject, java.security.PrivilegedAction)}.
69       * <p>
70       * The privileged action passed into the method may, when invoked, fail with either
71       * {@link javax.ws.rs.WebApplicationException} or {@link javax.ws.rs.ProcessingException}.
72       * Both these exceptions must be propagated to the caller without a modification.
73       * </p>
74       *
75       * @param action {@link PrivilegedAction} that represents a resource or sub-resource locator
76       *               method invocation to be executed by this method after establishing a subject.
77       * @return result of the action.
78       * @throws NullPointerException if the {@code PrivilegedAction} is {@code null}.
79       * @throws SecurityException    if the caller does not have permission to invoke the
80       *                              {@code Subject#doAs(Subject, PrivilegedAction)} method.
81       * @throws javax.ws.rs.WebApplicationException
82       *                              propagated exception from the privileged action. May be thrown in case the invocation
83       *                              of resource or sub-resource locator method in the privileged action results in
84       *                              this exception.
85       * @throws javax.ws.rs.ProcessingException
86       *                              propagated exception from the privileged action. May be thrown in case the invocation
87       *                              of resource or sub-resource locator method in the privileged action has failed
88       *                              or resulted in a non-checked exception.
89       */
90      public Object doAsSubject(PrivilegedAction action);
91  }
