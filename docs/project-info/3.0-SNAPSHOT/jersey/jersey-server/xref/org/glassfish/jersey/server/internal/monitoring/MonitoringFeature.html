66  /**
67   * Feature that enables calculating of {@link MonitoringStatistics monitoring statistics} and
68   * optionally also enables exposure of monitoring MBeans.
69   * <p>
70   * Calculation of {@code MonitoringStatistics} is necessary in order to expose monitoring MBeans, so by default
71   * this feature always enables calculation of {@code MonitoringStatistics}. Additionally, the feature can be
72   * configured by setting {@code true} to {@link #setmBeansEnabled(boolean)} in order to enable exposure
73   * of monitoring MBeans. The same can be achieved by configuration of a property
74   * {@link org.glassfish.jersey.server.ServerProperties#MONITORING_STATISTICS_MBEANS_ENABLED} which
75   * overrides the setting defined by the {@link #setmBeansEnabled(boolean)} method.
76   * <p/>
77   * <p>
78   * The MonitoringStatistics can be controlled also by definition of a property
79   * {@link org.glassfish.jersey.server.ServerProperties#MONITORING_STATISTICS_ENABLED} which overrides
80   * the registration of this feature.
81   * </p>
82   * When auto-discovery is enabled then monitoring statistics and exposure of MBeans can be controlled only
83   * by properties above without a need to explicitly register this feature.
84   *
85   * @see org.glassfish.jersey.server.ServerProperties#MONITORING_STATISTICS_ENABLED for more details.
86   * @author Miroslav Fuksa
87   */
88  public final class MonitoringFeature implements Feature {
89  
90      private static final Logger LOGGER = Logger.getLogger(MonitoringFeature.class.getName());
91  
92      private boolean monitoringEnabled = true;
93      private boolean statisticsEnabled = true; // monitoring statistics are enabled only if monitoring is enabled
94      private boolean mBeansEnabled; // monitoring mbeans are enabled only if monitoring statistics is enabled
95  
96      @Override
97      public boolean configure(FeatureContext context) {
98          final Boolean monitoringEnabledProperty = ServerProperties.getValue(context.getConfiguration().getProperties(),
99                  ServerProperties.MONITORING_ENABLED, null, Boolean.class);
100         final Boolean statisticsEnabledProperty = ServerProperties.getValue(context.getConfiguration().getProperties(),
101                 ServerProperties.MONITORING_STATISTICS_ENABLED, null, Boolean.class);
102         final Boolean mbeansEnabledProperty = ServerProperties.getValue(context.getConfiguration().getProperties(),
103                 ServerProperties.MONITORING_STATISTICS_MBEANS_ENABLED, null, Boolean.class);
104 
105         if (monitoringEnabledProperty != null) {
106             monitoringEnabled = monitoringEnabledProperty;
107             statisticsEnabled = monitoringEnabled; // monitoring statistics are enabled by default if monitoring is enabled
108         }
109 
110         if (statisticsEnabledProperty != null) {
111             monitoringEnabled = monitoringEnabled || statisticsEnabledProperty;
112             statisticsEnabled = statisticsEnabledProperty;
113         }
114 
115         if (mbeansEnabledProperty != null) {
116             monitoringEnabled = monitoringEnabled || mbeansEnabledProperty;
117             statisticsEnabled = statisticsEnabled || mbeansEnabledProperty;
118             mBeansEnabled = mbeansEnabledProperty;
119         }
120 
121         if (statisticsEnabledProperty != null && !statisticsEnabledProperty) {
122             if (mbeansEnabledProperty != null && mBeansEnabled) {
123                 LOGGER.log(Level.WARNING,
124                         LocalizationMessages.WARNING_MONITORING_FEATURE_ENABLED(ServerProperties.MONITORING_STATISTICS_ENABLED));
125             } else {
126                 LOGGER.log(Level.WARNING,
127                         LocalizationMessages.WARNING_MONITORING_FEATURE_DISABLED(ServerProperties.MONITORING_STATISTICS_ENABLED));
128             }
129         }
130 
131         if (monitoringEnabled) {
132             context.register(ApplicationInfoListener.class);
133             context.register(new AbstractBinder() {
134                 @Override
135                 protected void configure() {
136                     bindFactory(ReferencingFactory.<ApplicationInfo>referenceFactory()).to(
137                             new TypeLiteral<Ref<ApplicationInfo>>() {
138                             }
139                     ).in(Singleton.class);
140 
141                     bindFactory(ApplicationInfoInjectionFactory.class).to(
142                             ApplicationInfo.class).in(PerLookup.class);
143                 }
144             });
145         }
146 
147         if (statisticsEnabled) {
148             context.register(MonitoringEventListener.class);
149             context.register(new AbstractBinder() {
150                 @Override
151                 protected void configure() {
152                     bindFactory(ReferencingFactory.<MonitoringStatistics>referenceFactory()).to(
153                             new TypeLiteral<Ref<MonitoringStatistics>>() {
154                             }).in(Singleton.class);
155 
156                     bindFactory(StatisticsInjectionFactory.class).to(MonitoringStatistics.class).in(PerLookup.class);
157 
158                     bind(StatisticsListener.class).to(MonitoringStatisticsListener.class).in(Singleton.class);
159                 }
160             });
161         }
162 
163         if (mBeansEnabled) {
164             // instance registration is needed here as MBeanExposer needs to be a singleton so that
165             // one instance handles listening to events of MonitoringStatisticsListener and ContainerLifecycleListener
166             context.register(new MBeanExposer());
167         }
168 
169         return monitoringEnabled;
170     }
171 
172     /**
173      * Set whether the feature should also enable exposure of monitoring statistics MBeans.
174      * The set value can be overwritten by the definition of the property
175      * {@link org.glassfish.jersey.server.ServerProperties#MONITORING_STATISTICS_MBEANS_ENABLED}.
176      *
177      * @param mBeansEnabled {@code true} is monitoring MBeans should be exposed.
178      */
179     public void setmBeansEnabled(boolean mBeansEnabled) {
180         this.mBeansEnabled = mBeansEnabled;
181     }
