61  /**
62   * Container filtering stage responsible for execution of request and response filters
63   * on each request-response message exchange.
64   *
65   * @author Marek Potociar (marek.potociar at oracle.com)
66   * @author Martin Matula
67   */
68  class ContainerFilteringStage extends AbstractChainableStage<RequestProcessingContext> {
69  
70      private final Iterable<RankedProvider<ContainerRequestFilter>> requestFilters;
71      private final Iterable<RankedProvider<ContainerResponseFilter>> responseFilters;
72  
73      /**
74       * Create a new container filtering stage specifying global request and response filters. This stage class
75       * is reused for both pre and post match filtering phases.
76       * <p>
77       * All global response filters are passed in the pre-match stage, since if a pre-match filter aborts,
78       * response filters should still be executed. For the post-match filter stage creation, {@code null} is passed
79       * to the responseFilters parameter.
80       * </p>
81       *
82       * @param requestFilters  list of global (unbound) request filters (either pre or post match - depending on the
83       *                        stage being created).
84       * @param responseFilters list of global response filters (for pre-match stage) or {@code null} (for post-match
85       *                        stage).
86       */
87      ContainerFilteringStage(
88              Iterable<RankedProvider<ContainerRequestFilter>> requestFilters,
89              Iterable<RankedProvider<ContainerResponseFilter>> responseFilters) {
90  
91          this.requestFilters = requestFilters;
92          this.responseFilters = responseFilters;
93      }
94  
95      @Override
96      @SuppressWarnings("unchecked")
97      public Continuation<RequestProcessingContext> apply(RequestProcessingContext context) {
98          Iterable<ContainerRequestFilter> sortedRequestFilters;
99          final boolean postMatching = responseFilters == null;
100 
101         final ContainerRequest request = context.request();
102 
103         final TracingLogger tracingLogger = TracingLogger.getInstance(request);
104         if (postMatching) {
105             // post-matching
106             final ArrayList<Iterable<RankedProvider<ContainerRequestFilter>>> rankedProviders =
107                     new ArrayList<>(2);
108             rankedProviders.add(requestFilters);
109             rankedProviders.add(request.getRequestFilters());
110             sortedRequestFilters = Providers.mergeAndSortRankedProviders(
111                     new RankedComparator<ContainerRequestFilter>(), rankedProviders);
112 
113             context.monitoringEventBuilder().setContainerRequestFilters(sortedRequestFilters);
114             context.triggerEvent(RequestEvent.Type.REQUEST_MATCHED);
115 
116         } else {
117             // pre-matching (response filter stage is pushed in pre-matching phase, so that if pre-matching filter
118             // throws exception, response filters get still invoked)
119             context.push(new ResponseFilterStage(context, responseFilters, tracingLogger));
120             sortedRequestFilters = Providers.sortRankedProviders(new RankedComparator<ContainerRequestFilter>(), requestFilters);
121         }
122 
123         final TracingLogger.Event summaryEvent =
124                 (postMatching ? ServerTraceEvent.REQUEST_FILTER_SUMMARY : ServerTraceEvent.PRE_MATCH_SUMMARY);
125         final long timestamp = tracingLogger.timestamp(summaryEvent);
126         int processedCount = 0;
127         try {
128             final TracingLogger.Event filterEvent = (postMatching ? ServerTraceEvent.REQUEST_FILTER : ServerTraceEvent.PRE_MATCH);
129             for (ContainerRequestFilter filter : sortedRequestFilters) {
130                 final long filterTimestamp = tracingLogger.timestamp(filterEvent);
131                 try {
132                     filter.filter(request);
133                 } catch (Exception exception) {
134                     throw new MappableException(exception);
135                 } finally {
136                     processedCount++;
137                     tracingLogger.logDuration(filterEvent, filterTimestamp, filter);
138                 }
139 
140                 final Response abortResponse = request.getAbortResponse();
141                 if (abortResponse != null) {
142                     // abort accepting & return response
143                     return Continuation.of(context, Stages.asStage(
144                             new Endpoint() {
145                                 @Override
146                                 public ContainerResponse apply(
147                                         final RequestProcessingContext requestContext) {
148                                     return new ContainerResponse(requestContext.request(), abortResponse);
149                                 }
150                             }));
151                 }
152             }
153         } finally {
154             if (postMatching) {
155                 context.triggerEvent(RequestEvent.Type.REQUEST_FILTERED);
156             }
157             tracingLogger.logDuration(summaryEvent, timestamp, processedCount);
158         }
159 
160         return Continuation.of(context, getDefaultNext());
161     }
162 
163     private static class ResponseFilterStage extends AbstractChainableStage<ContainerResponse> {
164         // TODO remove the field - processing context should be made available on the response chain directly.
165         private final RequestProcessingContext processingContext;
166         private final Iterable<RankedProvider<ContainerResponseFilter>> filters;
167         private final TracingLogger tracingLogger;
168 
169         private ResponseFilterStage(final RequestProcessingContext processingContext,
170                                     final Iterable<RankedProvider<ContainerResponseFilter>> filters,
171                                     final TracingLogger tracingLogger) {
172             this.processingContext = processingContext;
173             this.filters = filters;
174             this.tracingLogger = tracingLogger;
175         }
176 
177         @Override
178         @SuppressWarnings("unchecked")
179         public Continuation<ContainerResponse> apply(ContainerResponse responseContext) {
180             final ArrayList<Iterable<RankedProvider<ContainerResponseFilter>>> rankedProviders = new ArrayList<>(2);
181             rankedProviders.add(filters);
182             rankedProviders.add(responseContext.getRequestContext().getResponseFilters());
183             Iterable<ContainerResponseFilter> sortedResponseFilters = Providers.mergeAndSortRankedProviders(
184                     new RankedComparator<ContainerResponseFilter>(RankedComparator.Order.DESCENDING), rankedProviders);
185 
186             final ContainerRequest request = responseContext.getRequestContext();
187             processingContext.monitoringEventBuilder().setContainerResponseFilters(sortedResponseFilters);
188             processingContext.triggerEvent(RequestEvent.Type.RESP_FILTERS_START);
189 
190             final long timestamp = tracingLogger.timestamp(ServerTraceEvent.RESPONSE_FILTER_SUMMARY);
191             int processedCount = 0;
192             try {
193                 for (ContainerResponseFilter filter : sortedResponseFilters) {
194                     final long filterTimestamp = tracingLogger.timestamp(ServerTraceEvent.RESPONSE_FILTER);
195                     try {
196                         filter.filter(request, responseContext);
197                     } catch (Exception ex) {
198                         throw new MappableException(ex);
199                     } finally {
200                         processedCount++;
201                         tracingLogger.logDuration(ServerTraceEvent.RESPONSE_FILTER, filterTimestamp, filter);
202                     }
203                 }
204             } finally {
205                 processingContext.triggerEvent(RequestEvent.Type.RESP_FILTERS_FINISHED);
206                 tracingLogger.logDuration(ServerTraceEvent.RESPONSE_FILTER_SUMMARY, timestamp, processedCount);
207             }
208 
209             return Continuation.of(responseContext, getDefaultNext());
210         }
211     }
