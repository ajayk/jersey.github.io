107     /**
108      * Convert a message header value, represented as a general object, to it's
109      * string representation. If the supplied header value is {@code null},
110      * this method returns {@code null}.
111      * <p>
112      * This method defers to {@link RuntimeDelegate#createHeaderDelegate} to
113      * obtain a {@link HeaderDelegate} to convert the value to a {@code String}.
114      * If a {@link HeaderDelegate} is not found then the {@code toString()}
115      * method on the header object is utilized.
116      *
117      * @param headerValue the header value represented as an object.
118      * @param rd          runtime delegate instance to be used for header delegate
119      *                    retrieval. If {@code null}, a default {@code RuntimeDelegate}
120      *                    instance will be {@link RuntimeDelegate#getInstance() obtained} and
121      *                    used.
122      * @return the string representation of the supplied header value or {@code null}
123      *         if the supplied header value is {@code null}.
124      */
125     @SuppressWarnings("unchecked")
126     public static String asString(final Object headerValue, RuntimeDelegate rd) {
127         if (headerValue == null) {
128             return null;
129         }
130         if (headerValue instanceof String) {
131             return (String) headerValue;
132         }
133         if (rd == null) {
134             rd = RuntimeDelegate.getInstance();
135         }
136 
137         final HeaderDelegate hp = rd.createHeaderDelegate(headerValue.getClass());
138         return (hp != null) ? hp.toString(headerValue) : headerValue.toString();
139     }

141     /**
142      * Returns string view of list of header values. Any modifications to the underlying list are visible to the view,
143      * the view also supports removal of elements. Does not support other modifications.
144      *
145      * @param headerValues header values.
146      * @param rd           RuntimeDelegate instance or {@code null} (in that case {@link RuntimeDelegate#getInstance()}
147      *                     will be called for before element conversion.
148      * @return String view of header values.
149      */
150     public static List<String> asStringList(final List<Object> headerValues, final RuntimeDelegate rd) {
151         if (headerValues == null || headerValues.isEmpty()) {
152             return Collections.emptyList();
153         }
154 
155         final RuntimeDelegate delegate;
156         if (rd == null) {
157             delegate = RuntimeDelegate.getInstance();
158         } else {
159             delegate = rd;
160         }
161 
162         return Lists.transform(headerValues, new Function<Object, String>() {
163             @Override
164             public String apply(final Object input) {
165                 return (input == null) ? "[null]" : HeaderUtils.asString(input, delegate);
166             }
167 
168         });
169     }

171     /**
172      * Returns string view of passed headers. Any modifications to the headers are visible to the view, the view also
173      * supports removal of elements. Does not support other modifications.
174      *
175      * @param headers headers.
176      * @return String view of headers or {@code null} if {code headers} input parameter is {@code null}.
177      */
178     public static MultivaluedMap<String, String> asStringHeaders(final MultivaluedMap<String, Object> headers) {
179         if (headers == null) {
180             return null;
181         }
182 
183         final RuntimeDelegate rd = RuntimeDelegate.getInstance();
184         return new AbstractMultivaluedMap<String, String>(
185                 Maps.transformValues(headers, new Function<List<Object>, List<String>>() {
186                     @Override
187                     public List<String> apply(final List<Object> input) {
188                         return HeaderUtils.asStringList(input, rd);
189                     }
190                 })
191         ) {
192         };
193     }

195     /**
196      * Transforms multi value map of headers to single {@code String} value map.
197      *
198      * Returned map is immutable. Map values are formatted using method {@link #asHeaderString}.
199      *
200      * @param headers headers to be formatted
201      * @return immutable single {@code String} value map or
202      *      {@code null} if {@code headers} input parameter is {@code null}.
203      */
204     public static Map<String, String> asStringHeadersSingleValue(final MultivaluedMap<String, Object> headers) {
205         if (headers == null) {
206             return null;
207         }
208 
209         final RuntimeDelegate rd = RuntimeDelegate.getInstance();
210         final ImmutableMap.Builder<String, String> immutableMapBuilder = new ImmutableMap.Builder<String, String>();
211         for (final Map.Entry<? extends String, ? extends List<Object>> entry : headers.entrySet()) {
212             immutableMapBuilder.put(entry.getKey(), asHeaderString(entry.getValue(), rd));
213         }
214         return immutableMapBuilder.build();
215     }

217     /**
218      * Converts a list of message header values to a single string value (with individual values separated by
219      * {@code ','}).
220      *
221      * Each single header value is converted to String using a
222      * {@link javax.ws.rs.ext.RuntimeDelegate.HeaderDelegate} if one is available
223      * via {@link javax.ws.rs.ext.RuntimeDelegate#createHeaderDelegate(java.lang.Class)}
224      * for the header value class or using its {@code toString()} method if a header
225      * delegate is not available.
226      *
227      * @param values list of individual header values.
228      * @param rd     {@link RuntimeDelegate} instance or {@code null} (in that case {@link RuntimeDelegate#getInstance()}
229      *               will be called for before conversion of elements).
230      * @return single string consisting of all the values passed in as a parameter. If values parameter is {@code null},
231      *         {@code null} is returned. If the list of values is empty, an empty string is returned.
232      */
233     public static String asHeaderString(final List<Object> values, final RuntimeDelegate rd) {
234         if (values == null) {
235             return null;
236         }
237         final Iterator<String> stringValues = asStringList(values, rd).iterator();
238         if (!stringValues.hasNext()) {
239             return "";
240         }
241 
242         final StringBuilder buffer = new StringBuilder(stringValues.next());
243         while (stringValues.hasNext()) {
244             buffer.append(',').append(stringValues.next());
245         }
246 
247         return buffer.toString();
248     }

250     /**
251      * Compares two snapshots of headers from jersey {@code ClientRequest} and logs {@code WARNING} in case of difference.
252      *
253      * Current container implementations does not support header modification in {@link javax.ws.rs.ext.WriterInterceptor}
254      * and {@link javax.ws.rs.ext.MessageBodyWriter}. The method checks there are some newly added headers
255      * (probably by WI or MBW) and logs {@code WARNING} message about it.
256      *
257      * @param headersSnapshot first immutable snapshot of headers
258      * @param currentHeaders  current instance of headers tobe compared to
259      * @param connectorName   name of connector the method is invoked from, used just in logged message
260      * @see <a href="https://java.net/jira/browse/JERSEY-2341">JERSEY-2341</a>
261      */
262     public static void checkHeaderChanges(final Map<String, String> headersSnapshot,
263                                           final MultivaluedMap<String, Object> currentHeaders,
264                                           final String connectorName) {
265         if (HeaderUtils.LOGGER.isLoggable(Level.WARNING)) {
266             final RuntimeDelegate rd = RuntimeDelegate.getInstance();
267             final Set<String> changedHeaderNames = new HashSet<String>();
268             for (final Map.Entry<? extends String, ? extends List<Object>> entry : currentHeaders.entrySet()) {
269                 if (!headersSnapshot.containsKey(entry.getKey())) {
270                     changedHeaderNames.add(entry.getKey());
271                 } else {
272                     final String prevValue = headersSnapshot.get(entry.getKey());
273                     final String newValue = asHeaderString(currentHeaders.get(entry.getKey()), rd);
274                     if (!prevValue.equals(newValue)) {
275                         changedHeaderNames.add(entry.getKey());
276                     }
277                 }
278             }
279             if (!changedHeaderNames.isEmpty()) {
280                 if (HeaderUtils.LOGGER.isLoggable(Level.WARNING)) {
281                     HeaderUtils.LOGGER.warning(LocalizationMessages.SOME_HEADERS_NOT_SENT(connectorName,
282                             changedHeaderNames.toString()));
283                 }
284             }
285         }
286     }
