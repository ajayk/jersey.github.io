63  /**
64   * Implementation of an inbound client-side JAX-RS {@link Response} message.
65   * <p>
66   * This response delegates method calls to the underlying
67   * {@link org.glassfish.jersey.client.ClientResponse client response context} and
68   * ensures that all request-scoped method invocations are run in the proper request scope.
69   * </p>
70   *
71   * @author Marek Potociar (marek.potociar at oracle.com)
72   */
73  class InboundJaxrsResponse extends Response {
74  
75      private final ClientResponse context;
76      private final RequestScope scope;
77      private final RequestScope.Instance scopeInstance;
78  
79      /**
80       * Create new scoped client response.
81       *
82       * @param context jersey client response context.
83       * @param scope   request scope instance.
84       */
85      public InboundJaxrsResponse(final ClientResponse context, final RequestScope scope) {
86          this.context = context;
87          this.scope = scope;
88          if (this.scope != null) {
89              this.scopeInstance = scope.referenceCurrent();
90          } else {
91              this.scopeInstance = null;
92          }
93      }

265     private <T> T runInScopeIfPossible(Producer<T> producer) {
266         if (scope != null && scopeInstance != null) {
267             return scope.runInScope(scopeInstance, producer);
268         } else {
269             return producer.call();
270         }
