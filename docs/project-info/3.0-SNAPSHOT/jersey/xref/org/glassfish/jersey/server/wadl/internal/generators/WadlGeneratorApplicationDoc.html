67  /**
68   * This {@link org.glassfish.jersey.server.wadl.WadlGenerator} adds all doc elements provided by {@link ApplicationDocs#getDocs()}
69   * to the generated wadl-file.
70   * <p>
71   * The {@link ApplicationDocs} content can either be provided via a {@link File} reference
72   * ({@link #setApplicationDocsFile(File)}) or
73   * via an {@link InputStream} ({@link #setApplicationDocsStream(InputStream)}).
74   * </p>
75   * <p>
76   * The {@link File} should be used when using the maven-wadl-plugin for generating wadl offline,
77   * the {@link InputStream} should be used when the extended wadl is generated by jersey at runtime, e.g.
78   * using the {@link org.glassfish.jersey.server.wadl.config.WadlGeneratorConfig} for configuration.
79   * </p>
80   *
81   * Created on: Jun 16, 2008<br>
82   *
83   * @author Martin Grotzke (martin.grotzke at freiheit.com)
84   */
85  public class WadlGeneratorApplicationDoc implements WadlGenerator {

89      private InputStream _applicationDocsStream;
90      private ApplicationDocs _applicationDocs;
91      @Context
92      private Provider<SAXParserFactory> saxFactoryProvider;

96          _delegate = delegate;
97      }

100         return _delegate.getRequiredJaxbContextPath();
101     }

103     public void setApplicationDocsFile(File applicationDocsFile) {
104         if (_applicationDocsStream != null) {
105             throw new IllegalStateException("The applicationDocsStream property is already set,"
106                     + " therefore you cannot set the applicationDocsFile property. Only one of both can be set at a time.");
107         }
108         _applicationDocsFile = applicationDocsFile;
109     }
110 
111     public void setApplicationDocsStream(InputStream applicationDocsStream) {
112         if (_applicationDocsFile != null) {
113             throw new IllegalStateException("The applicationDocsFile property is already set,"
114                     + " therefore you cannot set the applicationDocsStream property. Only one of both can be set at a time.");
115         }
116         _applicationDocsStream = applicationDocsStream;
117     }
118 
119 
120     public void init() throws Exception {
121         if (_applicationDocsFile == null && _applicationDocsStream == null) {
122             throw new IllegalStateException("Neither the applicationDocsFile nor the applicationDocsStream"
123                     + " is set, one of both is required.");
124         }
125         _delegate.init();
126 
127         InputStream inputStream;
128         if (_applicationDocsFile != null) {
129             inputStream = new FileInputStream(_applicationDocsFile);
130         } else {
131             inputStream = _applicationDocsStream;
132         }
133         _applicationDocs = WadlUtils.unmarshall(inputStream, saxFactoryProvider.get(), ApplicationDocs.class);
134     }
135 
136     /**
137      * @return the application
138      * @see org.glassfish.jersey.server.wadl.WadlGenerator#createApplication()
139      */
140     public Application createApplication() {
141         final Application result = _delegate.createApplication();
142         if (_applicationDocs != null && _applicationDocs.getDocs() != null
143                 && !_applicationDocs.getDocs().isEmpty()) {
144             result.getDoc().addAll(_applicationDocs.getDocs());
145         }
146         return result;
147     }
