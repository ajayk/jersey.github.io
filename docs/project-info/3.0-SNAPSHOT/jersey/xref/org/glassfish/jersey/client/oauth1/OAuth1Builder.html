41  package org.glassfish.jersey.client.oauth1;
42  
43  import javax.ws.rs.client.Client;
44  import javax.ws.rs.core.Feature;
45  
46  /**
47   * Builder of OAuth 1 client support. This builder can build {@link OAuth1AuthorizationFlow} using a method
48   * {@link #authorizationFlow(String, String, String)} and {@link Feature filter feature} using a method
49   * {@link #feature()}. Before one of these methods is invoked, properties common for both features
50   * can be defined using methods of this builder. However, for default OAuth configuration there should not be
51   * no need to call these {@code set*} methods of this builder and the builder can be directly used to build
52   * mentioned features.
53   *
54   * @author Miroslav Fuksa
55   * @since 2.3
56   */
57  public interface OAuth1Builder {
58  
59      /**
60       * Set the signature method name. The signature methods implement
61       * {@link org.glassfish.jersey.oauth1.signature.OAuth1SignatureMethod} and the name is retrieved from
62       * {@link org.glassfish.jersey.oauth1.signature.OAuth1SignatureMethod#name()} method. Build-in signature
63       * methods are {@code HMAC-SHA1}, {@code RSA-SHA1} and {@code PLAINTEXT}.
64       * <p>
65       * Default value is {@code HMAC-SHA1}.
66       * </p>
67       *
68       * @param signatureMethod Signature method name.
69       * @return This builder instance.
70       */
71      public OAuth1Builder signatureMethod(String signatureMethod);
72  
73      /**
74       * Set the realm to which the user wants to authenticate. The parameter
75       *              will be sent in Authenticated request and used during Authorization Flow.
76       *
77       * @param realm Realm on the server to which the user authentication is required.
78       * @return This builder instance.
79       */
80      public OAuth1Builder realm(String realm);
81  
82      /**
83       * Set the timestamp. The timestamp if defined will be used in {@code Authorization} header. Usually this
84       * parameter is not defined explicitly by this method and will be automatically filled with the current
85       * time during the request.
86       *
87       * @param timestamp Timestamp value.
88       * @return This builder instance.
89       */
90      public OAuth1Builder timestamp(String timestamp);
91  
92      /**
93       * Set the nonce. Nonce (shortcut of "number used once") is used to uniquely identify the request and
94       * prevent from multiple usage of the same signed request. The nonce if defined will be used
95       * in the {@code Authorization} header if defined. Usually this
96       * parameter is not defined explicitly by this method and will be automatically filled with the randomly
97       * generated UID during the request.
98       *
99       * @param nonce Nonce value.
100      * @return This builder instance.
101      */
102     public OAuth1Builder nonce(String nonce);
103 
104     /**
105      * Set the version of the OAuth protocol. The version, if defined, will be used in the {@code Authorization}
106      * header otherwise default value {@code 1.1} will be used. Usually this parameter does not need to be
107      * overwritten by this method.
108      *
109      * @param version OAuth protocol version parameter.
110      * @return This builder instance.
111      */
112     public OAuth1Builder version(String version);
113 
114 
115     /**
116      * Get the builder of {@link Feature filter feature}.
117      *
118      * @return The builder that can be used to build {@code OAuth1ClientFeature}.
119      */
120     public FilterFeatureBuilder feature();
121 
122 
123     /**
124      * Get the builder of {@link OAuth1AuthorizationFlow}.
125      *
126      * @param requestTokenUri URI of the endpoint on the Authorization Server where Request Token can be obtained.
127      *                        The URI is defined by the Service Provider.
128      * @param accessTokenUri URI of the endpoint on the Authorization Server where Access Token can be obtained.
129      *                        The URI is defined by the Service Provider.
130      * @param authorizationUri URI of the endpoint on the Authorization Server to which the user (resource owner)
131      *                         should be redirected in order to grant access to this application (our consumer).
132      *                         The URI is defined by the Service Provider.
133      * @return The builder that can be used to build {@code OAuth1AuthorizationFlow}.
134      */
135     public FlowBuilder authorizationFlow(String requestTokenUri, String accessTokenUri, String authorizationUri);
136 
137 
138     /**
139      * Builder of the {@link javax.ws.rs.core.Feature}.
140      */
141     public static interface FilterFeatureBuilder {
142         /**
143          * Set the Access Token that will be used for signing the requests. If this method is not called,
144          * no access token will be defined in the resulting filter and it will have to be supplied using
145          * {@link OAuth1ClientSupport#OAUTH_PROPERTY_ACCESS_TOKEN} request property for each request.
146          * The property could be also used to override settings defined by this method.
147          *
148          * @param accessToken Access token.
149          * @return this builder.
150          */
151         public FilterFeatureBuilder accessToken(AccessToken accessToken);
152 
153         /**
154          * Build the {@link Feature oauth 1 filter feature}. The feature can be then registered into the
155          * client configuration.
156          *
157          * @return Client OAuth 1 feature.
158          */
159         public Feature build();
160     }
161 
162     /**
163      * Builder of the {@link OAuth1AuthorizationFlow}.
164      */
165     public static interface FlowBuilder {
166 
167         /**
168          * Set the callback URI to which the user (resource owner) should be redirected after he/she
169          * grants access to this application. In most cases, the URI is under control of this application
170          * and request done on this URI will be used to extract query parameter {@code verifier} that will be used in
171          * {@link OAuth1AuthorizationFlow#finish(String)} method.
172          * <p>
173          * If URI is not defined by this method, the default value {@code oob} will be used in the Authorization
174          * Flow which should cause that {@code verifier} will be passed to application in other way than request
175          * redirection (for example shown to the user using html page).
176          * </p>
177          *
178          * @param callbackUri URI that should receive authorization response from the Service Provider.
179          * @return this builder.
180          */
181         public FlowBuilder callbackUri(String callbackUri);
182 
183         /**
184          * Set the client that should be used internally by the {@code OAuth1AuthorizationFlow} to make requests to
185          * Authorization Server. If this method is not called, it is up to the implementation to create or get
186          * any private client instance to perform these requests. This method could be used mainly for
187          * performance reasons to avoid creation of new client instances and have control about created client
188          * instances used in the application.
189          *
190          * @param client Client instance.
191          * @return this builder.
192          */
193         public FlowBuilder client(Client client);
194 
195         /**
196          * Enable logging (headers and entities) of OAuth requests and responses.
197          *
198          * @return this builder.
199          * @since 2.7
200          */
201         public FlowBuilder enableLogging();
202 
203         /**
204          * Build the {@code OAuth1AuthorizationFlow}.
205          *
206          * @return Authorization flow.
207          */
208         public OAuth1AuthorizationFlow build();
209     }
