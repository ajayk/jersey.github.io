178             if (this == obj) {

182                 return false;
183             }

195             return hash;
196         }

199     /**
200      * Create a new {@code RuntimeConfig} instance.
201      * <p>
202      * The constructor provides a way for defining a {@link ContractProvider contract
203      * provider model} registration strategy. Once a registration model is built
204      * for a newly registered contract, the provided registration strategy filter is
205      * consulted whether the model should be registered or not.
206      * </p>
207      * <p>
208      * Clients can use the method to cancel any contract provider model registration
209      * that does not meet the criteria of a given configuration context, such as a model
210      * that does not have any recognized contracts associated with it.
211      * </p>
212      *
213      * @param type                 configuration runtime type.
214      * @param registrationStrategy function driving the decision (based on the introspected
215      *                             {@link ContractProvider contract provider model}) whether
216      *                             or not should the component class registration continue
217      *                             towards a successful completion.
218      */
219     public CommonConfig(final RuntimeType type, final Predicate<ContractProvider> registrationStrategy) {
220         this.type = type;
221 
222         this.properties = new HashMap<String, Object>();
223         this.immutablePropertiesView = Collections.unmodifiableMap(properties);
224         this.immutablePropertyNames = Collections.unmodifiableCollection(properties.keySet());
225 
226         this.componentBag = ComponentBag.newInstance(registrationStrategy);
227 
228         this.newFeatureRegistrations = new LinkedList<FeatureRegistration>();
229 
230         this.enabledFeatureClasses = Sets.newIdentityHashSet();
231         this.enabledFeatures = Sets.newHashSet();
232 
233         this.disableMetaProviderConfiguration = false;
234     }

236     /**
237      * Copy constructor.
238      *
239      * @param config configurable to copy class properties from.
240      */
241     public CommonConfig(final CommonConfig config) {
242         this.type = config.type;
243 
244         this.properties = new HashMap<String, Object>(config.properties.size());
245         this.immutablePropertiesView = Collections.unmodifiableMap(this.properties);
246         this.immutablePropertyNames = Collections.unmodifiableCollection(this.properties.keySet());
247 
248         this.componentBag = config.componentBag.copy();
249 
250         this.newFeatureRegistrations = Lists.newLinkedList();
251         this.enabledFeatureClasses = Sets.newIdentityHashSet();
252         this.enabledFeatures = Sets.newHashSet();
253 
254         copy(config, false);
255     }

257     /**
258      * Copy config properties, providers from given {@code config} to this instance.
259      *
260      * @param config configurable to copy class properties from.
261      * @param loadComponentBag {@code true} if the component bag from config should be copied as well, {@code false} otherwise.
262      */
263     private void copy(final CommonConfig config, final boolean loadComponentBag) {
264         this.properties.clear();
265         this.properties.putAll(config.properties);
266 
267         this.newFeatureRegistrations.clear();
268         this.newFeatureRegistrations.addAll(config.newFeatureRegistrations);
269 
270         this.enabledFeatureClasses.clear();
271         this.enabledFeatureClasses.addAll(config.enabledFeatureClasses);
272 
273         this.enabledFeatures.clear();
274         this.enabledFeatures.addAll(config.enabledFeatures);
275 
276         this.disableMetaProviderConfiguration = config.disableMetaProviderConfiguration;
277 
278         if (loadComponentBag) {
279             this.componentBag.loadFrom(config.componentBag);
280         }
281     }

510     private void processFeatureRegistration(final Object component, final Class<?> componentClass) {
511         final ContractProvider model = componentBag.getModel(componentClass);
512         if (model.getContracts().contains(Feature.class)) {
513             @SuppressWarnings("unchecked")
514             final FeatureRegistration registration = (component != null)
515                     ? new FeatureRegistration((Feature) component)
516                     : new FeatureRegistration((Class<? extends Feature>) componentClass);
517             newFeatureRegistrations.add(registration);
518         }
519     }

521     /**
522      * Load the internal configuration state from an externally provided configuration state.
523      * <p/>
524      * Calling this method effectively replaces existing configuration state of the instance with the state represented by the
525      * externally provided configuration. If the features, auto-discoverables of given config has been already configured then
526      * this method will make sure to not configure them for the second time.
527      *
528      * @param config external configuration state to replace the configuration of this configurable instance.
529      * @return the updated common configuration instance.
530      */
531     public CommonConfig loadFrom(final Configuration config) {
532         if (config instanceof CommonConfig) {
533             // If loading from CommonConfig then simply copy properties and check whether given config has been initialized.
534             final CommonConfig commonConfig = (CommonConfig) config;
535 
536             copy(commonConfig, true);
537             this.disableMetaProviderConfiguration = !commonConfig.enabledFeatureClasses.isEmpty();
538         } else {
539             setProperties(config.getProperties());
540 
541             this.enabledFeatures.clear();
542             this.enabledFeatureClasses.clear();
543 
544             componentBag.clear();
545             resetRegistrations();
546 
547             for (final Class<?> clazz : config.getClasses()) {
548                 if (Feature.class.isAssignableFrom(clazz) && config.isEnabled((Class<? extends Feature>) clazz)) {
549                     this.disableMetaProviderConfiguration = true;
550                 }
551 
552                 register(clazz, config.getContracts(clazz));
553             }
554 
555             for (final Object instance : config.getInstances()) {
556                 if (instance instanceof Feature && config.isEnabled((Feature) instance)) {
557                     this.disableMetaProviderConfiguration = true;
558                 }
559 
560                 register(instance, config.getContracts(instance.getClass()));
561             }
562         }
563 
564         return this;
565     }

585     /**
586      * Configure {@link AutoDiscoverable auto-discoverables} in the HK2 service locator.
587      *
588      * @param locator locator in which the auto-discoverables should be configured.
589      * @param forcedOnly defines whether all or only forced auto-discoverables should be configured.
590      */
591     public void configureAutoDiscoverableProviders(final ServiceLocator locator, final boolean forcedOnly) {
592         // Check whether meta providers have been initialized for a config this config has been loaded from.
593         if (!disableMetaProviderConfiguration) {
594             final Set<AutoDiscoverable> providers = new TreeSet<AutoDiscoverable>(new Comparator<AutoDiscoverable>() {
595                 @Override
596                 public int compare(final AutoDiscoverable o1, final AutoDiscoverable o2) {
597                     final int p1 = o1.getClass().isAnnotationPresent(Priority.class)
598                             ? o1.getClass().getAnnotation(Priority.class).value() : Priorities.USER;
599                     final int p2 = o2.getClass().isAnnotationPresent(Priority.class)
600                             ? o2.getClass().getAnnotation(Priority.class).value() : Priorities.USER;
601 
602                     return (p1 < p2 || p1 == p2) ? -1 : 1;
603                 }
604             });
605 
606             // Forced (always invoked).
607             final List<ForcedAutoDiscoverable> forcedAutoDiscroverables = new LinkedList<ForcedAutoDiscoverable>();
608             for (Class<ForcedAutoDiscoverable> forcedADType : ServiceFinder.find(ForcedAutoDiscoverable.class, true)
609                     .toClassArray()) {
610                 forcedAutoDiscroverables.add(locator.createAndInitialize(forcedADType));
611             }
612             providers.addAll(forcedAutoDiscroverables);
613 
614             // Regular.
615             if (!forcedOnly) {
616                 providers.addAll(Providers.getProviders(locator, AutoDiscoverable.class));
617             }
618 
619             for (final AutoDiscoverable autoDiscoverable : providers) {
620                 final ConstrainedTo constrainedTo = autoDiscoverable.getClass().getAnnotation(ConstrainedTo.class);
621 
622                 if (constrainedTo == null || type.equals(constrainedTo.value())) {
623                     try {
624                         autoDiscoverable.configure(this);
625                     } catch (final Exception e) {
626                         LOGGER.log(Level.FINE,
627                                 LocalizationMessages.AUTODISCOVERABLE_CONFIGURATION_FAILED(autoDiscoverable.getClass()), e);
628                     }
629                 }
630             }
631         }
632     }

634     /**
635      * Configure HK2 binders in the HK2 service locator and enable JAX-RS features.
636      *
637      * @param locator locator in which the binders and features should be configured.
638      */
639     public void configureMetaProviders(final ServiceLocator locator) {
640         // First, configure existing binders
641         final Set<Binder> configuredBinders = configureBinders(locator, Collections.<Binder>emptySet());
642 
643         // Check whether meta providers have been initialized for a config this config has been loaded from.
644         if (!disableMetaProviderConfiguration) {
645 
646             registerManagedObjectsFinalizer(locator);
647             // Next, configure all features
648             configureFeatures(
649                     locator,
650                     new HashSet<FeatureRegistration>(),
651                     resetRegistrations());
652 
653             // At last, configure any new binders added by features
654             configureBinders(locator, configuredBinders);
655         }
656     }

658     private void registerManagedObjectsFinalizer(ServiceLocator locator) {
659         DynamicConfiguration dc = Injections.getConfiguration(locator);
660         ScopedBindingBuilder<ManagedObjectsFinalizer> binder = Injections.newBinder(ManagedObjectsFinalizer.class)
661                 .to(ManagedObjectsFinalizer.class)
662                 .in(Singleton.class);
663         Injections.addBinding(binder, dc);
664         dc.commit();
665     }

667     private Set<Binder> configureBinders(final ServiceLocator locator, final Set<Binder> configured) {
668         final Set<Binder> allConfigured = Sets.newIdentityHashSet();
669         allConfigured.addAll(configured);
670 
671         final Collection<Binder> binders = getBinders(configured);
672         if (!binders.isEmpty()) {
673             final DynamicConfiguration dc = Injections.getConfiguration(locator);
674 
675             for (final Binder binder : binders) {
676                 binder.bind(dc);
677                 allConfigured.add(binder);
678             }
679             dc.commit();
680         }
681 
682         return allConfigured;
683     }
684 
685     private Collection<Binder> getBinders(final Set<Binder> configured) {
686         return Collections2.filter(
687                 Collections2.transform(componentBag.getInstances(ComponentBag.BINDERS_ONLY), CAST_TO_BINDER),
688                 new Predicate<Binder>() {
689                     @Override
690                     public boolean apply(final Binder binder) {
691                         return !configured.contains(binder);
692                     }
693                 });
694     }

696     private void configureFeatures(final ServiceLocator locator,
697                                    final Set<FeatureRegistration> processed,
698                                    final List<FeatureRegistration> unprocessed) {
699 
700         ManagedObjectsFinalizer managedObjectsFinalizer = locator.getService(ManagedObjectsFinalizer.class);
701 
702         FeatureContextWrapper featureContextWrapper = null;
703         for (final FeatureRegistration registration : unprocessed) {
704             if (processed.contains(registration)) {
705                 LOGGER.config(LocalizationMessages.FEATURE_HAS_ALREADY_BEEN_PROCESSED(registration.getFeatureClass()));
706                 continue;
707             }
708 
709             Feature feature = registration.getFeature();
710             if (feature == null) {
711                 feature = locator.createAndInitialize(registration.getFeatureClass());
712                 managedObjectsFinalizer.registerForPreDestroyCall(feature);
713             } else {
714                 // Disable injection of Feature instances on the client-side. Instances may be registered into multiple
715                 // web-targets which means that injecting anything into these instances is not safe.
716                 if (!RuntimeType.CLIENT.equals(type)) {
717                     locator.inject(feature);
718                 }
719             }
720 
721             if (enabledFeatures.contains(feature)) {
722                 LOGGER.config(LocalizationMessages.FEATURE_HAS_ALREADY_BEEN_PROCESSED(feature));
723                 continue;
724             }
725 
726             if (featureContextWrapper == null) {
727                 // init lazily
728                 featureContextWrapper = new FeatureContextWrapper(this, locator);
729             }
730             final boolean success = feature.configure(featureContextWrapper);
731 
732             if (success) {
733                 processed.add(registration);
734 
735                 configureFeatures(locator, processed, resetRegistrations());
736 
737                 enabledFeatureClasses.add(registration.getFeatureClass());
738                 enabledFeatures.add(feature);
739             }
740         }
741     }
