41  package org.glassfish.jersey.ext.cdi1x.validation.internal;
42  
43  import java.util.Set;
44  
45  import javax.annotation.Priority;
46  import javax.enterprise.inject.spi.Bean;
47  import javax.enterprise.inject.spi.BeanManager;
48  import javax.inject.Inject;
49  import javax.interceptor.AroundConstruct;
50  import javax.interceptor.AroundInvoke;
51  import javax.interceptor.Interceptor;
52  import javax.interceptor.InvocationContext;
53  
54  import org.hibernate.validator.internal.cdi.interceptor.MethodValidated;
55  import org.hibernate.validator.internal.cdi.interceptor.ValidationInterceptor;
56  
57  /**
58   * JAX-RS wrapper for Hibernate CDI bean validation interceptor.
59   * Since Jersey already executes validation on JAX-RS resources,
60   * Jersey registers this wrapper into CDI container so that JAX-RS
61   * components do not get validated twice.
62   *
63   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
64   */
65  @MethodValidated
66  @Interceptor
67  @Priority(Interceptor.Priority.PLATFORM_AFTER + 800)
68  public class CdiInterceptorWrapper {
69  
70      private final ValidationInterceptor interceptor;
71  
72      @Inject
73      public CdiInterceptorWrapper(BeanManager beanManager) {
74          // get the original interceptor from the bean manager directly
75          // to avoid CDI bootstrap issues caused by wrong extension ordering
76          final Set<Bean<?>> interceptorBeans = beanManager.getBeans(ValidationInterceptor.class);
77          final Bean<?> interceptorBean = beanManager.resolve(interceptorBeans);
78          this.interceptor = (ValidationInterceptor) beanManager.getReference(
79                  interceptorBean, ValidationInterceptor.class, beanManager.createCreationalContext(interceptorBean));
80      }
81  
82      @Inject
83      private CdiInterceptorWrapperExtension extension;
84  
85      @AroundInvoke
86      public Object validateMethodInvocation(InvocationContext ctx) throws Exception {
87          final boolean isJaxRsMethod = extension.jaxRsResourceCache.compute(ctx.getMethod().getDeclaringClass());
88          return isJaxRsMethod ? ctx.proceed() : interceptor.validateMethodInvocation(ctx);
89      }
90  
91      @AroundConstruct
92      public void validateConstructorInvocation(InvocationContext ctx) throws Exception {
93          final boolean isJaxRsConstructor = extension.jaxRsResourceCache.compute(ctx.getConstructor().getDeclaringClass());
94          if (!isJaxRsConstructor) {
95              interceptor.validateConstructorInvocation(ctx);
96          }
97      }
98  }
