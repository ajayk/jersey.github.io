68  /**
69   * Server-side implementation of {@link org.glassfish.jersey.message.filtering.spi.ScopeProvider scope provider}. In addition to
70   * {@link CommonScopeProvider base implementation} this class provides entity-filtering scopes by examining matched resource
71   * method and sub-resource locators. This examination comes into play only in case if entity-filtering scopes cannot be found in
72   * entity annotations or application configuration.
73   *
74   * @author Michal Gajdos
75   */
76  @Singleton
77  @Priority(Priorities.ENTITY_CODER + 200)
78  @ConstrainedTo(RuntimeType.SERVER)
79  class ServerScopeProvider extends CommonScopeProvider {
80  
81      @Inject
82      private Provider<ExtendedUriInfo> uriInfoProvider;
83  
84      private final ConcurrentMap<String, Set<String>> uriToContexts;
85  
86      /**
87       * Create new server scope provider with injected {@link Configuration configuration} and
88       * {@link ServiceLocator HK2 service locator}.
89       */
90      @Inject
91      public ServerScopeProvider(final Configuration config, final ServiceLocator serviceLocator) {
92          super(config, serviceLocator);
93          this.uriToContexts = DataStructures.createConcurrentMap();
94      }
95  
96      @Override
97      public Set<String> getFilteringScopes(final Annotation[] entityAnnotations, final boolean defaultIfNotFound) {
98          Set<String> filteringScope = super.getFilteringScopes(entityAnnotations, false);
99  
100         if (filteringScope.isEmpty()) {
101             final ExtendedUriInfo uriInfo = uriInfoProvider.get();
102             final String path = uriInfo.getPath();
103 
104             if (uriToContexts.containsKey(path)) {
105                 return uriToContexts.get(path);
106             }
107 
108             for (final ResourceMethod method : ServerScopeProvider.getMatchedMethods(uriInfo)) {
109                 final Invocable invocable = method.getInvocable();
110 
111                 mergeFilteringScopes(filteringScope,
112                         getFilteringScopes(invocable.getHandlingMethod(), invocable.getHandler().getHandlerClass()));
113 
114                 if (!filteringScope.isEmpty()) {
115                     uriToContexts.putIfAbsent(path, filteringScope);
116                     return filteringScope;
117                 }
118             }
119         }
120 
121         // Use default scope if not in other scope.
122         return returnFilteringScopes(filteringScope, defaultIfNotFound);
123     }
124 
125     /**
126      * Get entity-filtering scopes from examining annotations present on resource method and resource class.
127      *
128      * @param resourceMethod matched resource method to be examined.
129      * @param resourceClass matched resource class to be examined.
130      * @return entity-filtering scopes or an empty set if the scopes cannot be obtained.
131      */
132     protected Set<String> getFilteringScopes(final Method resourceMethod, final Class<?> resourceClass) {
133         // Method annotations first.
134         Set<String> scope = getFilteringScopes(resourceMethod.getAnnotations());
135 
136         // Class annotations second.
137         if (scope.isEmpty()) {
138             scope = getFilteringScopes(resourceClass.getAnnotations());
139         }
140 
141         return scope;
142     }
143 
144     private static List<ResourceMethod> getMatchedMethods(final ExtendedUriInfo uriInfo) {
145         final List<ResourceMethod> matchedResourceLocators = uriInfo.getMatchedResourceLocators();
146         final List<ResourceMethod> methods = Lists.newArrayListWithCapacity(1 + matchedResourceLocators.size());
147 
148         methods.add(uriInfo.getMatchedResourceMethod());
149         methods.addAll(matchedResourceLocators);
150 
151         return methods;
152     }
