47  import org.glassfish.jersey.server.ContainerRequest;

52  /**
53   * An event informing about details of a request processing. The event is created by a Jersey runtime and
54   * handled by {@link RequestEventListener} (javadoc of listener describes how to register the listener for
55   * particular request).
56   * <p/>
57   * The event contains the {@link Type} which distinguishes between types of event. There are various
58   * properties in the event (accessible by getters) and some of them might be relevant only to specific event types.
59   * <p/>
60   * Note that internal state of the event must be modified. Even the event is immutable it exposes objects
61   * which might be mutable and the code of event listener must not change state of these objects.
62   *
63   * @author Miroslav Fuksa
64   */
65  public interface RequestEvent {
66  
67      /**
68       * The type of the event which describes in which request processing phase the event
69       * is triggered.
70       */
71      public static enum Type {
72          /**
73           * The request processing has started. This event type is handled only by
74           * {@link org.glassfish.jersey.server.monitoring.ApplicationEventListener#onRequest(RequestEvent)} and will
75           * never be called for {@link RequestEventListener#onEvent(RequestEvent)}.
76           */
77          START,
78  
79          /**
80           * The matching of the resource and resource method has started.
81           */
82          MATCHING_START,
83          /**
84           * The sub resource locator method is found and it will be called.
85           * The locator method can be retrieved from {@link #getUriInfo()} by method
86           * {@link org.glassfish.jersey.server.ExtendedUriInfo#getMatchedResourceLocators()}.
87           */
88          LOCATOR_MATCHED,
89  
90          /**
91           * The sub resource has been returned from sub resource locator, model was constructed, enhanced by
92           * {@link org.glassfish.jersey.server.model.ModelProcessor model processor}, validated and the matching
93           * is going to be performed on the sub {@link Resource resource}.
94           * The sub resource can be retrieved from {@link #getUriInfo()} by method
95           * {@link org.glassfish.jersey.server.ExtendedUriInfo#getLocatorSubResources()}.
96           */
97          SUBRESOURCE_LOCATED,
98  
99          /**
100          * The matching has been finished and {@link ContainerRequestFilter container request filters}
101          * are going to be executed. The request filters can be retrieved from event by
102          * {@link #getContainerRequestFilters()} method. This method also determines end of the matching
103          * process and therefore the matching results can be retrieved using {@link #getUriInfo()}.
104          */
105         REQUEST_MATCHED,
106 
107         /**
108          * Execution of {@link ContainerRequestFilter container request filters} has been finished.
109          */
110         REQUEST_FILTERED,
111 
112         /**
113          * Resource method is going to be executed. The resource method can be extracted from {@link ExtendedUriInfo}
114          * returned by {@link #getUriInfo()}.
115          */
116         RESOURCE_METHOD_START,
117 
118         /**
119          * Resource method execution has finished. In the case of synchronous processing
120          * the response is not available yet. In the case of asynchronous processing the situation depends on the
121          * method design and it in some cases on race conditions. In asynchronous cases this event can be
122          * triggered even after the response is completely processed. Exactly defined, this event
123          * is triggered when the thread executing the resource method returns from the resource method.
124          */
125         RESOURCE_METHOD_FINISHED,
126 
127         /**
128          * {@link ContainerResponseFilter Container response filters} are going to be executed. In this point
129          * the response is already available and can be retrieved by {@link #getContainerResponse()}. The
130          * response filters can be retrieved by {@link #getContainerResponseFilters()}.
131          * <p/>
132          * This phase is executed in the regular response processing but might also been executed for
133          * processing on response mapped from exceptions by {@link ExceptionMapper exception mappers}.
134          * In this case the {@link #ON_EXCEPTION} event type precedes this event.
135          */
136         RESP_FILTERS_START,
137 
138         /**
139          * Execution of {@link ContainerResponseFilter Container response filters} has finished.
140          * <p/>
141          * This phase is executed in the regular response processing but might also been executed for
142          * processing on response mapped from exceptions by {@link ExceptionMapper exception mappers}.
143          * In this case the {@link #ON_EXCEPTION} event type precedes this event.
144          */
145         RESP_FILTERS_FINISHED,
146 
147         /**
148          * Exception has been thrown during the request/response processing. This situation can
149          * occur in almost all phases of request processing and therefore there is no fixed order of
150          * events in which this event type can be triggered.
151          * <p/>
152          * The origin of exception can be retrieved
153          * by {@link #getExceptionCause()}. This event type can be received even two types in the case
154          * when first exception is thrown during the standard request processing and the second one
155          * is thrown during the processing of the response mapped from the exception.
156          * <p/>
157          * The exception thrown can be retrieved by {@link #getException()}.
158          */
159         ON_EXCEPTION,
160 
161         /**
162          * An {@link ExceptionMapper} is successfully found and it is going to be executed. The
163          * {@code ExceptionMapper} can be retrieved by {@link #getExceptionMapper()}.
164          */
165         EXCEPTION_MAPPER_FOUND,
166 
167         /**
168          * Exception mapping is finished. The result of exception mapping can be checked by
169          * {@link #isResponseSuccessfullyMapped()} which returns true when the exception mapping
170          * was successful. In this case the new response is available in the {@link #getContainerResponse()}.
171          */
172         EXCEPTION_MAPPING_FINISHED,
173 
174         /**
175          * The request and response processing has finished. The result of request processing can be checked
176          * by {@link #isSuccess()} method. This method is called even when request processing fails and ends
177          * up with not handled exceptions.
178          */
179         FINISHED
180     }
181 
182 
183     /**
184      * Describes the origin of the exception.
185      */
186     public static enum ExceptionCause {
187         /**
188          * An exception was thrown during the standard request and response processing and was not thrown
189          * from processing of mapped response.
190          */
191         ORIGINAL,
192         /**
193          * An exception was thrown during the processing of response
194          * mapped from {@link ExceptionMapper exception mappers}.
195          */
196         MAPPED_RESPONSE
197     }
198 
199     /**
200      * Returns the {@link Type type} of this event.
201      *
202      * @return Request event type.
203      */
204     public Type getType();
205 
206     /**
207      * Get the container request. The container request is available for all event types. Returned
208      * request must not be modified by the {@link RequestEventListener request event listener}.
209      *
210      * @return The non-null container request.
211      */
212     public ContainerRequest getContainerRequest();
213 
214     /**
215      * Get the container response. The response is available only for certain {@link Type event types}. The
216      * returned response might vary also on the event type. The getter returns always the latest response being
217      * processed. So, for example for event {@link Type#EXCEPTION_MAPPING_FINISHED} event type the method
218      * returns mapped response and not the original response created from execution of the resource method.
219      *
220      * @return Latest response being processed or {@code null} if no response has been produced yet.
221      */
222     public ContainerResponse getContainerResponse();
223 
224     /**
225      * Get the latest exception, if any, thrown by the request and response processing. When this method
226      * returns not null value, the method {@link #getExceptionCause()} returns the origin of the exception.
227      *
228      * @return Exception thrown or {@code null} if no exception has been thrown.
229      */
230     public Throwable getException();
231 
232     /**
233      * Get the {@link ExtendedUriInfo extended uri info} associated with this request. This method returns
234      * null for {@link Type#START} event. The returned {@code ExtendedUriInfo} can be used to retrieve
235      * information relevant to many event types (especially event types describing the matching process).
236      *
237      * @return Extended uri info or {@code null} if it is not available yet.
238      */
239     public ExtendedUriInfo getUriInfo();
240 
241     /**
242      * Get the {@link ExceptionMapper} that was found and used during the exception mapping phase.
243      *
244      * @return Exception mapper or {@code null} if no exception mapper was found or even needed.
245      */
246     public ExceptionMapper<?> getExceptionMapper();
247 
248 
249     /**
250      * Get {@link ContainerRequestFilter container request filters} used during the request filtering
251      * phase.
252      *
253      * @return Container request filters or {@code null} if no filters were used yet.
254      */
255     public Iterable<ContainerRequestFilter> getContainerRequestFilters();
256 
257     /**
258      * Get {@link ContainerResponseFilter container response filters} used during the response filtering
259      * phase.
260      *
261      * @return Container response filter or {@code null} if no filters were used yet.
262      */
263     public Iterable<ContainerResponseFilter> getContainerResponseFilters();
264 
265     /**
266      * Return {@code true} if the request and response has been successfully processed. Response is successfully
267      * processed when the response code is smaller than 400 and response was successfully written. If the exception
268      * occurred but was mapped into a response with successful response code and written, this method returns
269      * {@code true}.
270      *
271      * @return True if the response was successfully processed.
272      */
273     public boolean isSuccess();
274 
275     /**
276      * Returns {@code true} if the response was successfully mapped from an exception
277      * by {@link ExceptionMapper exception mappers}. When exception mapping phase failed or when
278      * no exception was thrown at all the, the method returns false. This method is convenient when
279      * handling the {@link Type#EXCEPTION_MAPPING_FINISHED} event type.
280      *
281      * @return True if the exception occurred and it was successfully mapped into a response.
282      */
283     public boolean isResponseSuccessfullyMapped();
284 
285     /**
286      * Get the {@link ExceptionCause exception cause}. This method is relevant only in cases when
287      * {@link #getException()} returns non-null value (for example when handling {@link Type#ON_EXCEPTION})
288      * event type.
289      *
290      * @return Exception cause of the latest exception or {@code null} if no exception has occurred.
291      */
292     public ExceptionCause getExceptionCause();
293 
294     /**
295      * Returns {@code true} if the response has been successfully written. {@code true} is returned
296      * even for cases when the written response contains error response code.
297      *
298      * @return {@code true} if the response was successfully written;{@code false} when the response
299      *         has not been written yet or when writing of response failed.
300      */
301     public boolean isResponseWritten();
302 }
