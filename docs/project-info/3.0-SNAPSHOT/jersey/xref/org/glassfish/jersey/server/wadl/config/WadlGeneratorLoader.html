64  /**
65   * Loads {@link WadlGenerator}s from a provided list of {@link WadlGeneratorDescription}s.<br/>
66   * The properties of the {@link WadlGeneratorDescription}s can refer to {@link WadlGenerator} properties
67   * of these types:
68   * <ul>
69   *
70   * <li>exact match: if the WadlGenerator property is of type <code>org.example.Foo</code> and the
71   * property value provided by the {@link WadlGeneratorDescription} is of type <code>org.example.Foo</code></li>
72   *
73   * <li>java.io.InputStream: The {@link InputStream} can e.g. represent a file. The stream is loaded from the
74   * property value (provided by the {@link WadlGeneratorDescription}) via
75   * {@link ClassLoader#getResourceAsStream(String)} or via OSGi API means if OSGi runtime is detected.
76   * The stream will be closed after {@link WadlGenerator#init()} was called.
77   * </li>
78   *
79   * <li>Types that provide a constructor for the provided type (mostly java.lang.String)</li>
80   *
81   * <li><strong>Deprecated, will be removed in future versions from the {@link WadlGeneratorLoader}:</strong><br/>
82   * java.lang.File: The property value can contain the prefix <em>classpath:</em> to denote, that the
83   * path to the file is relative to the classpath. In this case, the property value is stripped by
84   * the prefix <em>classpath:</em> and the java.lang.File is created via
85   * <pre><code>new File( generator.getClass().getResource( strippedFilename ).toURI() )</code></pre></li>
86   *
87   * </ul>
88   *
89   * @author Martin Grotzke (martin.grotzke at freiheit.com)
90   */
91  class WadlGeneratorLoader {

95      static WadlGenerator loadWadlGenerators(
96              List<WadlGenerator> wadlGenerators) throws Exception {
97          WadlGenerator wadlGenerator = new WadlGeneratorJAXBGrammarGenerator();
98          if (wadlGenerators != null && !wadlGenerators.isEmpty()) {
99              for (WadlGenerator generator : wadlGenerators) {
100                 generator.setWadlGeneratorDelegate(wadlGenerator);
101                 wadlGenerator = generator;
102             }
103         }
104         wadlGenerator.init();
105         return wadlGenerator;
106     }

108     static WadlGenerator loadWadlGeneratorDescriptions(ServiceLocator serviceLocator,
109                                                        WadlGeneratorDescription... wadlGeneratorDescriptions) throws Exception {

115     static WadlGenerator loadWadlGeneratorDescriptions(ServiceLocator serviceLocator,
116                                                        List<WadlGeneratorDescription> wadlGeneratorDescriptions)
117             throws Exception {
118         WadlGenerator wadlGenerator = new WadlGeneratorJAXBGrammarGenerator();
119 
120         final CallbackList callbacks = new CallbackList();
121         try {
122             if (wadlGeneratorDescriptions != null && !wadlGeneratorDescriptions.isEmpty()) {
123                 for (WadlGeneratorDescription wadlGeneratorDescription : wadlGeneratorDescriptions) {
124                     final WadlGeneratorControl control = loadWadlGenerator(serviceLocator, wadlGeneratorDescription,
125                             wadlGenerator);
126                     wadlGenerator = control.wadlGenerator;
127                     callbacks.add(control.callback);
128                 }
129             }
130             wadlGenerator.init();
131         } finally {
132             callbacks.callback();
133         }
134 
135         return wadlGenerator;
136 
137     }

139     private static WadlGeneratorControl loadWadlGenerator(ServiceLocator serviceLocator,
140                                                           WadlGeneratorDescription wadlGeneratorDescription,
141                                                           WadlGenerator wadlGeneratorDelegate) throws Exception {
142         LOGGER.info("Loading wadlGenerator " + wadlGeneratorDescription.getGeneratorClass().getName());
143         final WadlGenerator generator = Injections.getOrCreate(serviceLocator, wadlGeneratorDescription.getGeneratorClass());
144         generator.setWadlGeneratorDelegate(wadlGeneratorDelegate);
145         CallbackList callbacks = null;
146         if (wadlGeneratorDescription.getProperties() != null
147                 && !wadlGeneratorDescription.getProperties().isEmpty()) {
148             callbacks = new CallbackList();
149             final Properties wadlGeneratorProperties = wadlGeneratorDescription.getProperties();
150             Class<?> osgiConfiguratorClass = wadlGeneratorDescription.getConfiguratorClass();
151             for (Entry<Object, Object> entry : wadlGeneratorProperties.entrySet()) {
152                 final Callback callback = setProperty(generator, entry.getKey().toString(), entry.getValue(),
153                         osgiConfiguratorClass);
154                 callbacks.add(callback);
155             }
156         }
157 
158         return new WadlGeneratorControl(generator, callbacks);
159     }

161     /**
162      * Set the object (generator) property with the given name to the specified value.
163      * @param generator the object, on which the property shall be set
164      * @param propertyName the name of the property, that shall be set
165      * @param propertyValue the value to populate the property with
166      * @return a {@link Callback} object that must be called later, or null if no callback is required.
167      * @throws Exception if s.th. goes wrong
168      */
169     private static Callback setProperty(final Object generator,
170                                         final String propertyName,
171                                         final Object propertyValue,
172                                         final Class<?> osgiConfigClass) throws Exception {
173         Callback result = null;
174 
175         final String methodName = "set" + propertyName.substring(0, 1).toUpperCase() + propertyName.substring(1);
176         final Method method = getMethodByName(methodName, generator.getClass());
177         if (method.getParameterTypes().length != 1) {
178             throw new RuntimeException(
179                     "Method " + methodName + " is no setter, it does not expect exactly one parameter, but "
180                             + method.getParameterTypes().length);
181         }
182         final Class<?> paramClazz = method.getParameterTypes()[0];
183         if (paramClazz.isAssignableFrom(propertyValue.getClass())) {
184             method.invoke(generator, propertyValue);
185         } else if (File.class.equals(paramClazz) && propertyValue instanceof String) {
186 
187             /* This is now deprecated and can be removed in future versions.
188              * It's being replaced by the InputStream support, which must be used in
189              * a JEE environment instead of files.
190              */
191 
192             LOGGER.warning("Configuring the " + method.getDeclaringClass().getSimpleName()
193                     + " with the file based property " + propertyName + " is deprecated and will be removed"
194                     + " in future versions of jersey! You should use the InputStream based property instead.");
195 
196             final String filename = propertyValue.toString();
197             if (filename.startsWith("classpath:")) {
198                 final String strippedFilename = filename.substring("classpath:".length());
199                 final URL resource = generator.getClass().getResource(strippedFilename);
200                 if (resource == null) {
201                     throw new RuntimeException("The file '" + strippedFilename + "' does not exist in the classpath."
202                             + " It's loaded by the generator class, so if you use a relative filename it's relative to"
203                             + " the generator class, otherwise you might want to load it via an absolute classpath reference like"
204                             + " classpath:/somefile.xml");
205                 }
206                 final File file = new File(resource.toURI());
207                 method.invoke(generator, file);
208             } else {
209                 method.invoke(generator, new File(filename));
210             }
211         } else if (InputStream.class.equals(paramClazz) && propertyValue instanceof String) {
212             final String resource = propertyValue.toString();
213             ClassLoader loader = Thread.currentThread().getContextClassLoader();
214             if (loader == null) {
215                 loader = WadlGeneratorLoader.class.getClassLoader();
216             }
217             final InputStream is = ReflectionHelper.getResourceAsStream(loader, osgiConfigClass, resource);
218             if (is == null) {
219                 String message = "The resource '" + resource + "' does not exist.";
220                 throw new RuntimeException(message);
221             }
222             result = new Callback() {
223 
224                 public void callback() {
225                     try {
226                         is.close();
227                     } catch (IOException e) {
228                         LOGGER.log(Level.WARNING, "Could not close InputStream from resource " + resource, e);
229                     }
230                 }
231             };
232             /* if the method invocation fails we need to close the input stream
233              * by ourselves...
234              */
235             try {
236                 method.invoke(generator, is);
237             } catch (Exception e) {
238                 is.close();
239                 throw e;
240             }
241         } else {
242             /* does the param class provide a constructor for string?
243              */
244             final Constructor<?> paramTypeConstructor = paramClazz.getConstructor(propertyValue.getClass());
245             if (paramTypeConstructor != null) {
246                 final Object typedPropertyValue = paramTypeConstructor.newInstance(propertyValue);
247                 method.invoke(generator, typedPropertyValue);
248             } else {
249                 throw new RuntimeException("The property '" + propertyName + "' could not be set"
250                         + " because the expected parameter is neither of type " + propertyValue.getClass()
251                         + " nor of any type that provides a constructor expecting a " + propertyValue.getClass() + "."
252                         + " The expected parameter is of type " + paramClazz.getName());
253             }
254         }
255 
256         return result;
257     }

259     private static Method getMethodByName(final String methodName, final Class<?> clazz) {
260         for (Method method : clazz.getMethods()) {
261             if (method.getName().equals(methodName)) {
262                 return method;
263             }
264         }
265         throw new RuntimeException("Method '" + methodName + "' not found for class " + clazz.getName());
266     }

268     private static class WadlGeneratorControl {
269 
270         WadlGenerator wadlGenerator;
271         Callback callback;

273         /**
274          * The constructor.
275          * @param wadlGenerator the generator, must not be null
276          * @param callback the callback, can be null

278         public WadlGeneratorControl(WadlGenerator wadlGenerator,

290     private static class CallbackList extends ArrayList<Callback> implements Callback {
291 
292         private static final long serialVersionUID = 1L;
293 
294         /**
295          * Callback all registered {@link Callback} items.
296          */
297         public void callback() {
298             for (Callback callback : this) {
299                 callback.callback();
300             }
301         }

303         /**

305          *
306          * @param e the element to append, can be null.
307          * @return true if the element was appended to the list, otherwise null.
308          */
309         @Override

311             return e != null ? super.add(e) : false;
312         }
