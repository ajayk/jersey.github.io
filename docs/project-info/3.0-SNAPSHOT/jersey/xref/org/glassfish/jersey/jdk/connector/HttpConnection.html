66       * Input buffer that is used by {@link TransportFilter} when SSL is turned on.
67       * The size cannot be smaller than a maximal size of a SSL packet, which is 16kB for payload + header, because
68       * {@link SslFilter} does not have its own buffer for buffering incoming
69       * data and therefore the entire SSL packet must fit into {@link SslFilter}

102                    CookieManager cookieManager,

104                    ScheduledExecutorService scheduler,

114     synchronized void connect() {
115         if (state != State.CREATED) {
116             throw new IllegalStateException(
117                     "Cannot try to establish connection if the connection is in other than CREATED state, current state: "
118                             + state);
119         }
120 
121         changeState(State.CONNECTING);
122         scheduleConnectTimeout();
123         filterChain.connect(new InetSocketAddress(uri.getHost(), Utils.getPort(uri)), null);
124     }
125 
126     synchronized void send(final HttpRequest httpRequest) {
127         if (state != State.IDLE) {
128             throw new IllegalStateException(
129                     "Http request cannot be sent over a connection that is in other state than IDLE. Current state: " + state);
130         }
131 
132         cancelIdleTimeout();
133 
134         this.httpRequest = httpRequest;
135         // clean state left by previous request
136         httResponse = null;
137         error = null;
138         persistentConnection = true;
139         changeState(State.SENDING_REQUEST);
140 
141         addRequestHeaders();
142 
143         filterChain.write(httpRequest, new CompletionHandler<HttpRequest>() {
144             @Override
145             public void failed(Throwable throwable) {
146                 handleError(throwable);
147             }
148 
149             @Override
150             public void completed(HttpRequest result) {
151                 handleHeaderSent();
152             }
153         });
154     }
155 
156     synchronized void close() {
157         if (state == State.CLOSED) {
158             return;
159         }
160 
161         cancelAllTimeouts();
162         filterChain.close();
163         changeState(State.CLOSED);
164     }

166     private synchronized void handleHeaderSent() {
167         if (state != State.SENDING_REQUEST) {
168             return;
169         }
170 
171         scheduleResponseTimeout();
172 
173         if (httpRequest.getBodyMode() == HttpRequest.BodyMode.NONE
174                 || httpRequest.getBodyMode() == HttpRequest.BodyMode.BUFFERED) {
175             changeState(State.RECEIVING_HEADER);
176         } else {
177             ChunkedBodyOutputStream bodyStream = (ChunkedBodyOutputStream) httpRequest.getBodyStream();
178             bodyStream.setCloseListener(() -> {
179                 synchronized (HttpConnection.this) {
180                     if (state != State.SENDING_REQUEST) {
181                         return;
182                     }
183 
184                     changeState(State.RECEIVING_HEADER);
185                 }
186             });
187         }
188     }

190     private void addRequestHeaders() {
191         Map<String, List<String>> cookies;
192         try {
193             cookies = cookieManager.get(httpRequest.getUri(), httpRequest.getHeaders());
194         } catch (IOException e) {
195             handleError(e);
196             return;
197         }
198 
199         // unfortunately CookieManager returns ""Cookie" -> empty list" pair if the cookie is not set
200         cookies.entrySet().stream().filter(cookieHeader -> cookieHeader.getValue() != null && !cookieHeader.getValue().isEmpty())
201                 .forEach(cookieHeader -> httpRequest.getHeaders().put(cookieHeader.getKey(), cookieHeader.getValue()));
202     }
203 
204     private void processResponseHeaders(HttpResponse response) throws IOException {
205         cookieManager.put(httpRequest.getUri(), httResponse.getHeaders());
206         List<String> connectionValues = response.getHeader(Constants.CONNECTION);
207         if (connectionValues != null) {
208             connectionValues.stream().filter(connectionValue -> connectionValue.equalsIgnoreCase(Constants.CONNECTION_CLOSE))
209                     .forEach(connectionValue -> persistentConnection = false);
210         }
211     }

213     protected Filter<HttpRequest, HttpResponse, HttpRequest, HttpResponse> createFilterChain(URI uri,
214                                                                                              ConnectorConfiguration
215                                                                                                      configuration) {
216         boolean secure = Constants.HTTPS.equals(uri.getScheme());
217         Filter<ByteBuffer, ByteBuffer, ?, ?> socket;
218 
219         if (secure) {
220             SSLContext sslContext = configuration.getSslContext();
221             TransportFilter transportFilter = new TransportFilter(SSL_INPUT_BUFFER_SIZE, configuration.getThreadPoolConfig(),
222                     configuration.getContainerIdleTimeout());
223 
224             if (sslContext == null) {
225                 sslContext = SslConfigurator.getDefaultContext();
226 
227             }
228 
229             socket = new SslFilter(transportFilter, sslContext, uri.getHost(), configuration.getHostnameVerifier());
230         } else {
231             socket = new TransportFilter(INPUT_BUFFER_SIZE, configuration.getThreadPoolConfig(),
232                     configuration.getContainerIdleTimeout());
233         }
234 
235         int maxHeaderSize = configuration.getMaxHeaderSize();
236         HttpFilter httpFilter = new HttpFilter(socket, maxHeaderSize, maxHeaderSize + INPUT_BUFFER_SIZE);
237 
238         ConnectorConfiguration.ProxyConfiguration proxyConfiguration = configuration.getProxyConfiguration();
239         if (proxyConfiguration.isConfigured()) {
240             ProxyFilter proxyFilter = new ProxyFilter(httpFilter, proxyConfiguration);
241             return new ConnectionFilter(proxyFilter);
242         }
243 
244         return new ConnectionFilter(httpFilter);
245     }

247     private void changeState(State newState) {
248         State old = state;
249         state = newState;
250 
251         if (LOGGER.isLoggable(Level.FINEST)) {
252             LOGGER.finest(LocalizationMessages.CONNECTION_CHANGING_STATE(uri.getHost(), uri.getPort(), old, newState));
253         }
254 
255         stateListener.onStateChanged(this, old, newState);
256     }

258     private void scheduleResponseTimeout() {
259         if (configuration.getResponseTimeout() == 0) {
260             return;
261         }
262 
263         responseTimeout = scheduler.schedule(() -> {
264             synchronized (HttpConnection.this) {
265                 if (state != State.RECEIVING_HEADER && state != State.RECEIVING_BODY) {
266                     return;
267                 }
268 
269                 responseTimeout = null;
270                 changeState(State.RESPONSE_TIMEOUT);
271                 close();
272             }
273         }, configuration.getResponseTimeout(), TimeUnit.MILLISECONDS);
274     }
275 
276     private void cancelResponseTimeout() {
277         if (responseTimeout != null) {
278             responseTimeout.cancel(true);
279             responseTimeout = null;
280         }
281     }

283     private void scheduleConnectTimeout() {
284         if (configuration.getConnectTimeout() == 0) {
285             return;

309         if (configuration.getConnectionIdleTimeout() == 0) {

333     private void cancelAllTimeouts() {
334         cancelConnectTimeout();
335         cancelIdleTimeout();

340         cancelAllTimeouts();

359     private synchronized void handleResponseRead() {
360         cancelResponseTimeout();
361         changeState(State.RECEIVED);
362         if (persistentConnection) {
363             changeStateToIdle();
364         } else {
365             changeState(State.CLOSED);
366         }
367     }

369     private class ConnectionFilter extends Filter<HttpRequest, HttpResponse, HttpRequest, HttpResponse> {
370 
371         ConnectionFilter(Filter<HttpRequest, HttpResponse, ?, ?> downstreamFilter) {
372             super(downstreamFilter);
373         }
374 
375         @Override
376         boolean processRead(HttpResponse response) {
377             synchronized (HttpConnection.this) {
378                 if (state != State.RECEIVING_HEADER && state != State.SENDING_REQUEST) {
379                     return false;
380                 }
381 
382                 if (state == State.SENDING_REQUEST) {
383                     // great we received response header so fast that we did not even switch into "receiving header" state,
384                     // do it now to complete the formal lifecycle
385                     // this happens when write completion listener is overtaken by "read event"
386                     changeState(State.RECEIVING_HEADER);
387                 }
388 
389                 httResponse = response;
390 
391                 try {
392                     processResponseHeaders(response);
393                 } catch (IOException e) {
394                     handleError(e);
395                     return false;
396                 }
397 
398                 if (response.getHasContent()) {
399                     AsynchronousBodyInputStream bodyStream = httResponse.getBodyStream();
400                     changeState(State.RECEIVING_BODY);
401                     bodyStream.setStateChangeLister(new AsynchronousBodyInputStream.StateChangeLister() {
402                         @Override
403                         public void onError(Throwable t) {
404                             handleError(t);
405                         }
406 
407                         @Override
408                         public void onAllDataRead() {
409                             handleResponseRead();
410                         }
411                     });
412 
413                 } else {
414                     handleResponseRead();
415                 }
416                 return false;
417             }
418         }
419 
420         @Override
421         void processConnect() {
422             synchronized (HttpConnection.this) {
423                 if (state != State.CONNECTING) {
424                     return;
425                 }
426 
427                 downstreamFilter.startSsl();
428             }
429         }
430 
431         @Override
432         void processSslHandshakeCompleted() {
433             synchronized (HttpConnection.this) {
434                 if (state != State.CONNECTING) {
435                     return;
436                 }
437 
438                 cancelConnectTimeout();
439                 changeStateToIdle();
440             }
441         }
442 
443         @Override
444         void processConnectionClosed() {
445             synchronized (HttpConnection.this) {
446                 cancelAllTimeouts();
447                 changeState(State.CLOSED_BY_SERVER);
448                 HttpConnection.this.close();
449             }
450         }
451 
452         @Override
453         void processError(Throwable t) {
454             handleError(t);
455         }
456 
457         @Override
458         void write(HttpRequest data, CompletionHandler<HttpRequest> completionHandler) {
459             downstreamFilter.write(data, completionHandler);
460         }
461     }
