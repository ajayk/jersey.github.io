71  /**
72   * Represents reader interceptor chain executor for both client and server side.
73   * It constructs wrapped interceptor chain and invokes it. At the end of the chain
74   * a {@link MessageBodyReader message body reader} execution interceptor is inserted,
75   * which finally reads an entity from the output stream provided by the chain.
76   *
77   * @author Miroslav Fuksa
78   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
79   */
80  public final class ReaderInterceptorExecutor extends InterceptorExecutor<ReaderInterceptor>
81          implements ReaderInterceptorContext, ServiceLocatorSupplier {

95      /**
96       * Constructs a new executor to read given type from provided {@link InputStream entityStream}.
97       *
98       * @param rawType            raw Java entity type.
99       * @param type               generic Java entity type.
100      * @param annotations        array of annotations on the declaration of the artifact
101      *                           that will be initialized with the produced instance. E.g. if the message
102      *                           body is to be converted into a method parameter, this will be the annotations
103      *                           on that parameter returned by {@code Method.getParameterAnnotations}.
104      * @param mediaType          media type of the HTTP entity.
105      * @param headers            mutable message headers.
106      * @param propertiesDelegate request-scoped properties delegate.
107      * @param inputStream        entity input stream.
108      * @param workers            {@link org.glassfish.jersey.message.MessageBodyWorkers Message body workers}.
109      * @param readerInterceptors Reader interceptor that are to be used to intercept the reading of an entity.
110      *                           The interceptors will be executed in the same order as given in this parameter.
111      * @param translateNce       if {@code true}, the {@link javax.ws.rs.core.NoContentException} thrown by a selected message
112      *                           body
113      *                           reader will be translated into a {@link javax.ws.rs.BadRequestException} as required by
114      * @param serviceLocator Service locator.
115      */
116     public ReaderInterceptorExecutor(final Class<?> rawType, final Type type,
117                                      final Annotation[] annotations,
118                                      final MediaType mediaType,
119                                      final MultivaluedMap<String, String> headers,
120                                      final PropertiesDelegate propertiesDelegate,
121                                      final InputStream inputStream,
122                                      final MessageBodyWorkers workers,
123                                      final Iterable<ReaderInterceptor> readerInterceptors,
124                                      final boolean translateNce,
125                                      final ServiceLocator serviceLocator) {
126 
127         super(rawType, type, annotations, mediaType, propertiesDelegate);
128         this.headers = headers;
129         this.inputStream = inputStream;
130         this.workers = workers;
131         this.translateNce = translateNce;
132         this.serviceLocator = serviceLocator;
133 
134         final List<ReaderInterceptor> effectiveInterceptors = Lists.newArrayList(readerInterceptors);
135         effectiveInterceptors.add(new TerminalReaderInterceptor());
136 
137         this.interceptors = effectiveInterceptors.iterator();
138         this.processedCount = 0;
139     }

141     /**
142      * Starts the interceptor chain execution.
143      *
144      * @return an entity read from the stream.
145      */
146     @Override
147     @SuppressWarnings("unchecked")
148     public Object proceed() throws IOException {
149         if (!interceptors.hasNext()) {
150             throw new ProcessingException(LocalizationMessages.ERROR_INTERCEPTOR_READER_PROCEED());
151         }
152         final ReaderInterceptor interceptor = interceptors.next();
153         traceBefore(interceptor, MsgTraceEvent.RI_BEFORE);
154         try {
155             return interceptor.aroundReadFrom(this);
156         } finally {
157             processedCount++;
158             traceAfter(interceptor, MsgTraceEvent.RI_AFTER);
159         }
160     }

192     /**
193      * Terminal reader interceptor which choose the appropriate {@link MessageBodyReader}
194      * and reads the entity from the input stream. The order of actions is the following: <br>
195      * 1. choose the appropriate {@link MessageBodyReader} <br>
196      * 3. reads the entity from the output stream <br>
197      */
198     private class TerminalReaderInterceptor implements ReaderInterceptor {
199 
200         @Override
201         @SuppressWarnings("unchecked")
202         public Object aroundReadFrom(final ReaderInterceptorContext context) throws IOException, WebApplicationException {
203             processedCount--; //this is not regular interceptor -> count down
204 
205             traceBefore(null, MsgTraceEvent.RI_BEFORE);
206             try {
207                 final TracingLogger tracingLogger = getTracingLogger();
208                 if (tracingLogger.isLogEnabled(MsgTraceEvent.MBR_FIND)) {
209                     tracingLogger.log(MsgTraceEvent.MBR_FIND,
210                             context.getType().getName(),
211                             (context.getGenericType() instanceof Class
212                                     ? ((Class) context.getGenericType()).getName() : context.getGenericType()),
213                             String.valueOf(context.getMediaType()), java.util.Arrays.toString(context.getAnnotations()));
214                 }
215 
216                 final MessageBodyReader bodyReader = workers.getMessageBodyReader(
217                         context.getType(),
218                         context.getGenericType(),
219                         context.getAnnotations(),
220                         context.getMediaType(),
221                         ReaderInterceptorExecutor.this);
222 
223                 final EntityInputStream input = new EntityInputStream(context.getInputStream());
224 
225                 if (bodyReader == null) {
226                     if (input.isEmpty() && !context.getHeaders().containsKey(HttpHeaders.CONTENT_TYPE)) {
227                         return null;
228                     } else {
229                         LOGGER.log(Level.FINE, LocalizationMessages.ERROR_NOTFOUND_MESSAGEBODYREADER(context.getMediaType(),
230                                 context.getType(), context.getGenericType()));
231                         throw new MessageBodyProviderNotFoundException(LocalizationMessages.ERROR_NOTFOUND_MESSAGEBODYREADER(
232                                 context.getMediaType(), context.getType(), context.getGenericType()));
233                     }
234                 }
235                 Object entity = invokeReadFrom(context, bodyReader, input);
236 
237                 if (bodyReader instanceof CompletableReader) {
238                     entity = ((CompletableReader) bodyReader).complete(entity);
239                 }
240                 return entity;
241             } finally {
242                 clearLastTracedInterceptor();
243                 traceAfter(null, MsgTraceEvent.RI_AFTER);
244             }
245         }
246 
247         @SuppressWarnings("unchecked")
248         private Object invokeReadFrom(final ReaderInterceptorContext context, final MessageBodyReader reader,
249                                       final EntityInputStream input) throws WebApplicationException, IOException {
250 
251             final TracingLogger tracingLogger = getTracingLogger();
252             final long timestamp = tracingLogger.timestamp(MsgTraceEvent.MBR_READ_FROM);
253             final InputStream stream = new UnCloseableInputStream(input, reader);
254 
255             try {
256                 return reader.readFrom(context.getType(), context.getGenericType(), context.getAnnotations(),
257                         context.getMediaType(), context.getHeaders(), stream);
258             } catch (final NoContentException ex) {
259                 if (translateNce) {
260                     throw new BadRequestException(ex);
261                 } else {
262                     throw ex;
263                 }
264             } finally {
265                 tracingLogger.logDuration(MsgTraceEvent.MBR_READ_FROM, timestamp, reader);
266             }
267         }
268     }

270     /**
271      * {@link javax.ws.rs.ext.MessageBodyReader}s should not close the given {@link java.io.InputStream stream}. This input
272      * stream makes sure that the stream is not closed even if MBR tries to do it.
273      */
274     private static class UnCloseableInputStream extends InputStream {
275 
276         private final InputStream original;
277         private final MessageBodyReader reader;
278 
279         private UnCloseableInputStream(final InputStream original, final MessageBodyReader reader) {
280             this.original = original;
281             this.reader = reader;
282         }
283 
284         @Override
285         public int read() throws IOException {
286             return original.read();
287         }
288 
289         @Override
290         public int read(final byte[] b) throws IOException {
291             return original.read(b);
292         }
293 
294         @Override
295         public int read(final byte[] b, final int off, final int len) throws IOException {
296             return original.read(b, off, len);
297         }
298 
299         @Override
300         public long skip(final long l) throws IOException {
301             return original.skip(l);
302         }
303 
304         @Override
305         public int available() throws IOException {
306             return original.available();
307         }
308 
309         @Override
310         public synchronized void mark(final int i) {
311             original.mark(i);
312         }
313 
314         @Override
315         public synchronized void reset() throws IOException {
316             original.reset();
317         }
318 
319         @Override
320         public boolean markSupported() {
321             return original.markSupported();
322         }
323 
324         @Override
325         public void close() throws IOException {
326             if (LOGGER.isLoggable(Level.FINE)) {
327                 LOGGER.log(Level.FINE, LocalizationMessages.MBR_TRYING_TO_CLOSE_STREAM(reader.getClass()));
328             }
329         }
330 
331         private InputStream unwrap() {
332             return original;
333         }
334     }

336     /**
337      * Make the {@link InputStream} able to close.
338      * <p/>
339      * The purpose of this utility method is to undo effect of {@link ReaderInterceptorExecutor.UnCloseableInputStream}.
340      *
341      * @param inputStream Potential {@link ReaderInterceptorExecutor.UnCloseableInputStream} to undo its effect
342      * @return Input stream that is possible to close
343      */
344     public static InputStream closeableInputStream(InputStream inputStream) {
345         if (inputStream instanceof UnCloseableInputStream) {
346             return ((UnCloseableInputStream) inputStream).unwrap();
347         } else {
348             return inputStream;
349         }
350     }
