141     /**
142      * Call back interface that the created external grammar can use
143      * to allow other parts of the code to attach the correct grammar information.
144      */
145     interface Resolver {
146 
147         /**
148          * Resolve a Class type to a QName.
149          *
150          * @param type The type of the class.
151          * @return The schema type of the class if defined, null if not.
152          */
153         QName resolve(Class type);
154     }
155 
156     /**
157      * And internal storage object to store the grammar definitions and
158      * any type resolvers that are created along the way.
159      */
160     class ExternalGrammarDefinition {
161 
162         // final public field to make a property was thinking about encapsulation
163         // but decided code much simpler without
164         public final Map<String, ApplicationDescription.ExternalGrammar>
165                 map = new HashMap<String, ApplicationDescription.ExternalGrammar>();
166 
167         private List<Resolver> typeResolvers = new ArrayList<Resolver>();
168 
169         public void addResolver(Resolver resolver) {
170             assert !typeResolvers.contains(resolver) : "Already in list";
171             typeResolvers.add(resolver);
172         }
173 
174         /**
175          * @param type the class to map
176          * @return The resolved qualified name if one is defined.
177          */
178         public QName resolve(Class type) {
179             QName name = null;
180             found:
181             for (Resolver resolver : typeResolvers) {
182                 name = resolver.resolve(type);
183                 if (name != null) {
184                     break found;
185                 }
186             }
187             return name;
188         }
189     }
190 
191     /**
192      * Perform any post create functions such as generating grammars.

194      */
