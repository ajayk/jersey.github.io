136     private static ServiceLocator _createLocator(final String name, final ServiceLocator parent, final Binder... binders) {
137         // Passing null as service locator generator would force HK2 to find appropriate one.
138         final ServiceLocator result = factory.create(name, parent, null, ServiceLocatorFactory.CreatePolicy.DESTROY);
139 
140         result.setNeutralContextClassLoader(false);
141         ServiceLocatorUtilities.enablePerThreadScope(result);
142 
143         // HK2 Immediate Scope is commented out due to JERSEY-2979 and other issues
144         // ServiceLocatorUtilities.enableImmediateScope(result);
145 
146         for (final Binder binder : binders) {
147             bind(result, binder);
148         }
149         return result;
150     }
151 
152     private static void bind(final ServiceLocator locator, final Binder binder) {
153         final DynamicConfigurationService dcs = locator.getService(DynamicConfigurationService.class);
154         final DynamicConfiguration dc = dcs.createDynamicConfiguration();
155 
156         locator.inject(binder);
157         binder.bind(dc);
158 
159         dc.commit();
160     }

162     /**
163      * Get the class by contract or create and inject a new instance.
164      *
165      * @param <T>            instance type.
166      * @param serviceLocator HK2 service locator.
167      * @param clazz          class of the instance to be provider.
168      * @return instance of the class either provided as a service or created and injected  by HK2.
169      */
170     public static <T> T getOrCreate(final ServiceLocator serviceLocator, final Class<T> clazz) {
171         try {
172             final T component = serviceLocator.getService(clazz);
173             return component == null ? serviceLocator.createAndInitialize(clazz) : component;
174         } catch (final MultiException e) {
175 
176             // Look for WebApplicationException and return it if found. MultiException is thrown when *Param field is
177             // annotated and value cannot be provided (for example fromString(String) method can throw unchecked
178             // exception.
179             //
180             // see InvalidParamTest
181             // see JERSEY-1117
182             for (final Throwable t : e.getErrors()) {
183                 if (WebApplicationException.class.isAssignableFrom(t.getClass())) {
184                     throw (WebApplicationException) t;
185                 }
186             }
187 
188             throw e;
189         }
190     }

287         if (factory.find(locator.getName()) != null) {
