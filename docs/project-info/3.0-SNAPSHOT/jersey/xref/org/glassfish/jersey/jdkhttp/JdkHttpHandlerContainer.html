111     @Override
112     public void handle(final HttpExchange exchange) throws IOException {
113         /**
114          * This is a URI that contains the path, query and fragment components.
115          */
116         URI exchangeUri = exchange.getRequestURI();
117 
118         /**
119          * The base path specified by the HTTP context of the HTTP handler. It
120          * is in decoded form.
121          */
122         String decodedBasePath = exchange.getHttpContext().getPath();
123 
124         // Ensure that the base path ends with a '/'
125         if (!decodedBasePath.endsWith("/")) {
126             if (decodedBasePath.equals(exchangeUri.getPath())) {
127                 /**
128                  * This is an edge case where the request path does not end in a
129                  * '/' and is equal to the context path of the HTTP handler.
130                  * Both the request path and base path need to end in a '/'
131                  * Currently the request path is modified.
132                  *
133                  * TODO support redirection in accordance with resource configuration feature.
134                  */
135                 exchangeUri = UriBuilder.fromUri(exchangeUri)
136                         .path("/").build();
137             }
138             decodedBasePath += "/";
139         }
140 
141         /*
142          * The following is madness, there is no easy way to get the complete
143          * URI of the HTTP request!!
144          *
145          * TODO this is missing the user information component, how can this be obtained?
146          */
147         final boolean isSecure = exchange instanceof HttpsExchange;
148         final String scheme = isSecure ? "https" : "http";
149 
150         final URI baseUri = getBaseUri(exchange, decodedBasePath, scheme);
151         final URI requestUri = getRequestUri(exchange, baseUri);
152 
153         final ResponseWriter responseWriter = new ResponseWriter(exchange);
154         final ContainerRequest requestContext = new ContainerRequest(baseUri, requestUri,
155                 exchange.getRequestMethod(), getSecurityContext(exchange.getPrincipal(), isSecure),
156                 new MapPropertiesDelegate());
157         requestContext.setEntityStream(exchange.getRequestBody());
158         requestContext.getHeaders().putAll(exchange.getRequestHeaders());
159         requestContext.setWriter(responseWriter);
160         try {
161             appHandler.handle(requestContext);
162         } finally {
163             // if the response was not committed yet by the JerseyApplication
164             // then commit it and log warning
165             responseWriter.closeAndLogWarning();
166         }
167     }

169     private URI getBaseUri(final HttpExchange exchange, final String decodedBasePath, final String scheme) {
170         final URI baseUri;
171         try {
172             final List<String> hostHeader = exchange.getRequestHeaders().get("Host");
173             if (hostHeader != null) {
174                 baseUri = new URI(scheme + "://" + hostHeader.get(0) + decodedBasePath);
175             } else {
176                 final InetSocketAddress addr = exchange.getLocalAddress();
177                 baseUri = new URI(scheme, null, addr.getHostName(), addr.getPort(),
178                         decodedBasePath, null, null);
179             }
180         } catch (final URISyntaxException ex) {
181             throw new IllegalArgumentException(ex);
182         }
183         return baseUri;
184     }
185 
186     private URI getRequestUri(final HttpExchange exchange, final URI baseUri) {
187         try {
188             return new URI(getServerAddress(baseUri) + exchange.getRequestURI());
189         } catch (URISyntaxException ex) {
190             throw new IllegalArgumentException(ex);
191         }
192     }
193 
194     private String getServerAddress(final URI baseUri) throws URISyntaxException {
195         return new URI(baseUri.getScheme(), null,  baseUri.getHost(), baseUri.getPort(), null, null, null).toString();
196     }

265     private static final class ResponseWriter implements ContainerResponseWriter {
266 
267         private final HttpExchange exchange;
268         private final AtomicBoolean closed;
269 
270         /**
271          * Creates a new ResponseWriter for given {@link HttpExchange HTTP Exchange}.
272          *
273          * @param exchange Exchange of the {@link HttpServer JDK Http Server}
274          */
275         ResponseWriter(final HttpExchange exchange) {
276             this.exchange = exchange;
277             this.closed = new AtomicBoolean(false);
278         }
279 
280         @Override
281         public OutputStream writeResponseStatusAndHeaders(final long contentLength, final ContainerResponse context)
282                 throws ContainerException {
283             final MultivaluedMap<String, String> responseHeaders = context.getStringHeaders();
284             final Headers serverHeaders = exchange.getResponseHeaders();
285             for (final Map.Entry<String, List<String>> e : responseHeaders.entrySet()) {
286                 for (final String value : e.getValue()) {
287                     serverHeaders.add(e.getKey(), value);
288                 }
289             }
290 
291             try {
292                 if (context.getStatus() == Response.Status.NO_CONTENT.getStatusCode()) {
293                     // Work around bug in LW HTTP server
294                     // http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6886436
295                     exchange.sendResponseHeaders(context.getStatus(), -1);
296                 } else {
297                     exchange.sendResponseHeaders(context.getStatus(),
298                             getResponseLength(contentLength));
299                 }
300             } catch (final IOException ioe) {
301                 throw new ContainerException(LocalizationMessages.ERROR_RESPONSEWRITER_WRITING_HEADERS(), ioe);
302             }
303 
304             return exchange.getResponseBody();
305         }
306 
307         private long getResponseLength(final long contentLength) {
308             if (contentLength == 0) {
309                 return -1;
310             }
311             if (contentLength < 0) {
312                 return 0;
313             }
314             return contentLength;
315         }
316 
317         @Override
318         public boolean suspend(final long timeOut, final TimeUnit timeUnit, final TimeoutHandler timeoutHandler) {
319             throw new UnsupportedOperationException("Method suspend is not supported by the container.");
320         }
321 
322         @Override
323         public void setSuspendTimeout(final long timeOut, final TimeUnit timeUnit) throws IllegalStateException {
324             throw new UnsupportedOperationException("Method setSuspendTimeout is not supported by the container.");
325         }
326 
327         @Override
328         public void failure(final Throwable error) {
329             try {
330                 exchange.sendResponseHeaders(Response.Status.INTERNAL_SERVER_ERROR.getStatusCode(), getResponseLength(0));
331             } catch (final IOException e) {
332                 LOGGER.log(Level.WARNING, LocalizationMessages.ERROR_RESPONSEWRITER_SENDING_FAILURE_RESPONSE(), e);
333             } finally {
334                 commit();
335                 rethrow(error);
336             }
337         }
338 
339         @Override
340         public boolean enableResponseBuffering() {
341             return true;
342         }
343 
344         @Override
345         public void commit() {
346             if (closed.compareAndSet(false, true)) {
347                 exchange.close();
348             }
349         }
350 
351         /**
352          * Rethrow the original exception as required by JAX-RS, 3.3.4
353          *
354          * @param error throwable to be re-thrown
355          */
356         private void rethrow(final Throwable error) {
357             if (error instanceof RuntimeException) {
358                 throw (RuntimeException) error;
359             } else {
360                 throw new ContainerException(error);
361             }
362         }
363 
364         /**
365          * Commits the response and logs a warning message.
366          *
367          * This method should be called by the container at the end of the
368          * handle method to make sure that the ResponseWriter was committed.
369          */
370         private void closeAndLogWarning() {
371             if (closed.compareAndSet(false, true)) {
372                 exchange.close();
373                 LOGGER.log(Level.WARNING, LocalizationMessages.ERROR_RESPONSEWRITER_RESPONSE_UNCOMMITED());
374             }
375         }
376     }
