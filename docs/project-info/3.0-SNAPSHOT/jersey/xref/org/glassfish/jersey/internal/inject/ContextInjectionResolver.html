72  /**
73   * Injection resolver for {@link Context @Context} injection annotation.
74   *
75   * @author Marek Potociar (marek.potociar at oracle.com)
76   * @author Jakub Podlesak (jakub.podlesak at oracle.com)
77   */
78  @Singleton
79  public class ContextInjectionResolver implements InjectionResolver<Context> {
80  
81      /**
82       * Context injection resolver HK2 binder.
83       */
84      public static final class Binder extends AbstractBinder {
85  
86          @Override
87          protected void configure() {
88              // @Context
89              bind(ContextInjectionResolver.class).to(new TypeLiteral<InjectionResolver<Context>>() {
90              }).in(Singleton.class);
91          }
92      }
93  
94      @Inject
95      private ServiceLocator serviceLocator;
96  
97      private final Cache<Injectee, ActiveDescriptor<?>> descriptorCache
98              = new Cache<Injectee, ActiveDescriptor<?>>(new Computable<Injectee, ActiveDescriptor<?>>() {
99  
100                 @Override
101                 public ActiveDescriptor<?> compute(Injectee key) {
102                     return serviceLocator.getInjecteeDescriptor(key);
103                 }
104             });
105 
106     @Override
107     public Object resolve(Injectee injectee, ServiceHandle<?> root) {
108         Type requiredType = injectee.getRequiredType();
109         boolean isHk2Factory = ReflectionHelper.isSubClassOf(requiredType, Factory.class);
110         Injectee newInjectee;
111 
112         if (isHk2Factory) {
113             newInjectee = getFactoryInjectee(injectee, ReflectionHelper.getTypeArgument(requiredType, 0));
114         } else {
115             newInjectee = foreignRequestScopedInjecteeCache.compute(injectee);
116         }
117 
118         ActiveDescriptor<?> ad = descriptorCache.compute(newInjectee);
119 
120         if (ad != null) {
121             final ServiceHandle handle = serviceLocator.getServiceHandle(ad, newInjectee);
122 
123             if (isHk2Factory) {
124                 return asFactory(handle);
125             } else {
126                 return handle.getService();
127             }
128         }
129         return null;
130     }
131 
132     private Factory asFactory(final ServiceHandle handle) {
133         return new Factory() {
134             @Override
135             public Object provide() {
136                 return handle.getService();
137             }
138 
139             @Override
140             public void dispose(Object instance) {
141                 //not used
142             }
143         };
144     }
145 
146     private Injectee getFactoryInjectee(final Injectee injectee, final Type requiredType) {
147         return new RequiredTypeOverridingInjectee(injectee, requiredType);
148     }
149 
150     private static class RequiredTypeOverridingInjectee extends InjecteeImpl {
151 
152         private static final long serialVersionUID = -3740895548611880187L;
153 
154         private RequiredTypeOverridingInjectee(final Injectee injectee, final Type requiredType) {
155             super(injectee);
156             setRequiredType(requiredType);
157         }
158     }
159 
160     private static class DescriptorOverridingInjectee extends InjecteeImpl {
161 
162         private static final long serialVersionUID = -3740895548611880189L;
163 
164         private DescriptorOverridingInjectee(final Injectee injectee, final ActiveDescriptor descriptor) {
165             super(injectee);
166             setInjecteeDescriptor(descriptor);
167         }
168     }
169 
170     @Override
171     public boolean isConstructorParameterIndicator() {
172         return true;
173     }
174 
175     @Override
176     public boolean isMethodParameterIndicator() {
177         return false;
178     }

181     private final Cache<Injectee, Injectee> foreignRequestScopedInjecteeCache =
182             new Cache<Injectee, Injectee>(new Computable<Injectee, Injectee>() {
183         @Override
184         public Injectee compute(Injectee injectee) {
185             if (injectee.getParent() != null) {
186                 if (Field.class.isAssignableFrom(injectee.getParent().getClass())) {
187                     Field f = (Field) injectee.getParent();
188                     if (foreignRequestScopedComponents.get().contains(f.getDeclaringClass())) {
189                         final Class<?> clazz = f.getType();
190                         if (serviceLocator.getServiceHandle(clazz).getActiveDescriptor().getScopeAnnotation()
191                                                                                             == RequestScoped.class) {
192                             final AbstractActiveDescriptor<Object> descriptor =
193                                     BuilderHelper.activeLink(clazz)
194                                             .to(clazz)
195                                             .in(RequestScoped.class)
196                                             .build();
197                             return new DescriptorOverridingInjectee(injectee, descriptor);
198                         }
199                     }
200                 }
201             }
202             return injectee;
203         }
204     });
205 
206     LazyValue<Set<Class<?>>> foreignRequestScopedComponents = Values.lazy(new Value<Set<Class<?>>>() {
207         @Override
208         public Set<Class<?>> get() {
209             return getForeignRequestScopedComponents();
210         }
211     });
212 
213     private Set<Class<?>> getForeignRequestScopedComponents() {
214         final List<ForeignRequestScopeBridge> scopeBridges = serviceLocator.getAllServices(ForeignRequestScopeBridge.class);
215         final Set<Class<?>> result = new HashSet<Class<?>>();
216         for (ForeignRequestScopeBridge bridge : scopeBridges) {
217             final Set<Class<?>> requestScopedComponents = bridge.getRequestScopedComponents();
218             if (requestScopedComponents != null) {
219                 result.addAll(requestScopedComponents);
220             }
221         }
222         return result;
223     }
