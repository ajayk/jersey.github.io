74  /**
75   * Http Authentication filter that provides basic and digest authentication (based on RFC 2617).

119     /**
120      * Create a new filter instance.
121      *
122      * @param mode              Mode.
123      * @param basicCredentials  Basic credentials (can be {@code null} if this filter does not work in the
124      *                          basic mode or if no default credentials are defined).
125      * @param digestCredentials Digest credentials (can be {@code null} if this filter does not work in the
126      *                          digest mode or if no default credentials are defined).
127      * @param configuration     Configuration (non-{@code null}).
128      */
129     HttpAuthenticationFilter(HttpAuthenticationFeature.Mode mode, Credentials basicCredentials,
130                              Credentials digestCredentials, Configuration configuration) {
131         int limit = getMaximumCacheLimit(configuration);
132 
133         final int uriCacheLimit = limit * 2; // 2 is chosen to estimate there will be two times URIs
134         // for basic and digest together than only digest
135         // (limit estimates digest max URI number)
136 
137         uriCache = Collections.synchronizedMap(new LinkedHashMap<String, Type>(uriCacheLimit) {
138             private static final long serialVersionUID = 1946245645415625L;
139 
140             @Override
141             protected boolean removeEldestEntry(Map.Entry<String, Type> eldest) {
142                 return size() > uriCacheLimit;
143             }
144         });
145 
146         this.mode = mode;
147         switch (mode) {
148             case BASIC_PREEMPTIVE:
149             case BASIC_NON_PREEMPTIVE:
150                 this.basicAuth = new BasicAuthenticator(basicCredentials);
151                 this.digestAuth = null;
152                 break;
153             case DIGEST:
154                 this.basicAuth = null;
155                 this.digestAuth = new DigestAuthenticator(digestCredentials, limit);
156                 break;
157             case UNIVERSAL:
158                 this.basicAuth = new BasicAuthenticator(basicCredentials);
159                 this.digestAuth = new DigestAuthenticator(digestCredentials, limit);
160                 break;
161             default:
162                 throw new IllegalStateException("Not implemented.");
163         }
164     }

166     private int getMaximumCacheLimit(Configuration configuration) {
167         int limit = ClientProperties.getValue(configuration.getProperties(),
168                 ClientProperties.DIGESTAUTH_URI_CACHE_SIZELIMIT, MAXIMUM_DIGEST_CACHE_SIZE);
169         if (limit < 1) {
170             limit = MAXIMUM_DIGEST_CACHE_SIZE;
171         }
172         return limit;
173     }

175     @Override
176     public void filter(ClientRequestContext request) throws IOException {
177         if ("true".equals(request.getProperty(REQUEST_PROPERTY_FILTER_REUSED))) {
178             return;
179         }
180 
181         if (request.getHeaders().containsKey(HttpHeaders.AUTHORIZATION)) {
182             return;
183         }
184 
185         Type operation = null;
186         if (mode == HttpAuthenticationFeature.Mode.BASIC_PREEMPTIVE) {
187             basicAuth.filterRequest(request);
188             operation = Type.BASIC;
189         } else if (mode == HttpAuthenticationFeature.Mode.BASIC_NON_PREEMPTIVE) {
190             // do nothing
191         } else if (mode == HttpAuthenticationFeature.Mode.DIGEST) {
192             if (digestAuth.filterRequest(request)) {
193                 operation = Type.DIGEST;
194             }
195         } else if (mode == HttpAuthenticationFeature.Mode.UNIVERSAL) {
196 
197             Type lastSuccessfulMethod = uriCache.get(getCacheKey(request));
198             if (lastSuccessfulMethod != null) {
199                 request.setProperty(REQUEST_PROPERTY_OPERATION, lastSuccessfulMethod);
200                 if (lastSuccessfulMethod == Type.BASIC) {
201                     basicAuth.filterRequest(request);
202                     operation = Type.BASIC;
203                 } else if (lastSuccessfulMethod == Type.DIGEST) {
204                     if (digestAuth.filterRequest(request)) {
205                         operation = Type.DIGEST;
206                     }
207                 }
208             }
209         }
210 
211         if (operation != null) {
212             request.setProperty(REQUEST_PROPERTY_OPERATION, operation);
213         }
214     }

216     @Override
217     public void filter(ClientRequestContext request, ClientResponseContext response) throws IOException {
218         if ("true".equals(request.getProperty(REQUEST_PROPERTY_FILTER_REUSED))) {
219             return;
220         }
221 
222         Type result = null; // which authentication is requested: BASIC or DIGEST
223         boolean authenticate;
224 
225         if (response.getStatus() == Response.Status.UNAUTHORIZED.getStatusCode()) {
226             String authString = response.getHeaders().getFirst(HttpHeaders.WWW_AUTHENTICATE);
227             if (authString != null) {
228                 final String upperCaseAuth = authString.trim().toUpperCase();
229                 if (upperCaseAuth.startsWith("BASIC")) {
230                     result = Type.BASIC;
231                 } else if (upperCaseAuth.startsWith("DIGEST")) {
232                     result = Type.DIGEST;
233                 } else {
234                     // unknown authentication -> this filter cannot authenticate with this method
235                     return;
236                 }
237             }
238             authenticate = true;
239         } else {
240             authenticate = false;
241         }
242 
243         if (mode == HttpAuthenticationFeature.Mode.BASIC_PREEMPTIVE) {
244             // do nothing -> 401 will be returned to the client
245         } else if (mode == HttpAuthenticationFeature.Mode.BASIC_NON_PREEMPTIVE) {
246             if (authenticate && result == Type.BASIC) {
247                 basicAuth.filterResponseAndAuthenticate(request, response);
248             }
249         } else if (mode == HttpAuthenticationFeature.Mode.DIGEST) {
250             if (authenticate && result == Type.DIGEST) {
251                 digestAuth.filterResponse(request, response);
252             }
253         } else if (mode == HttpAuthenticationFeature.Mode.UNIVERSAL) {
254             Type operation = (Type) request.getProperty(REQUEST_PROPERTY_OPERATION);
255             if (operation != null) {
256                 updateCache(request, !authenticate, operation);
257             }
258 
259             if (authenticate) {
260                 boolean success = false;
261 
262                 // now we have the challenge response and we can authenticate
263                 if (result == Type.BASIC) {
264                     success = basicAuth.filterResponseAndAuthenticate(request, response);
265                 } else if (result == Type.DIGEST) {
266                     success = digestAuth.filterResponse(request, response);
267                 }
268                 updateCache(request, success, result);
269             }
270         }
271     }

273     private String getCacheKey(ClientRequestContext request) {
274         return request.getUri().toString() + ":" + request.getMethod();
275     }
276 
277     private void updateCache(ClientRequestContext request, boolean success, Type operation) {
278         String cacheKey = getCacheKey(request);
279         if (success) {
280             this.uriCache.put(cacheKey, operation);
281         } else {
282             this.uriCache.remove(cacheKey);
283         }
284     }

286     /**
287      * Repeat the {@code request} with provided {@code newAuthorizationHeader}
288      * and update the {@code response} with newest response data.
289      *
290      * @param request                Request context.
291      * @param response               Response context (will be updated with the new response data).
292      * @param newAuthorizationHeader {@code Authorization} header that should be added to the new request.
293      * @return {@code true} is the authentication was successful ({@code true} if 401 response code was not returned;
294      * {@code false} otherwise).
295      */
296     static boolean repeatRequest(ClientRequestContext request, ClientResponseContext response, String newAuthorizationHeader) {
297         Client client = request.getClient();
298 
299         String method = request.getMethod();
300         MediaType mediaType = request.getMediaType();
301         URI lUri = request.getUri();
302 
303         WebTarget resourceTarget = client.target(lUri);
304 
305         Invocation.Builder builder = resourceTarget.request(mediaType);
306 
307         MultivaluedMap<String, Object> newHeaders = new MultivaluedHashMap<String, Object>();
308 
309         for (Map.Entry<String, List<Object>> entry : request.getHeaders().entrySet()) {
310             if (HttpHeaders.AUTHORIZATION.equals(entry.getKey())) {
311                 continue;
312             }
313             newHeaders.put(entry.getKey(), entry.getValue());
314         }
315 
316         newHeaders.add(HttpHeaders.AUTHORIZATION, newAuthorizationHeader);
317         builder.headers(newHeaders);
318 
319         builder.property(REQUEST_PROPERTY_FILTER_REUSED, "true");
320 
321         Invocation invocation;
322         if (request.getEntity() == null) {
323             invocation = builder.build(method);
324         } else {
325             invocation = builder.build(method,
326                     Entity.entity(request.getEntity(), request.getMediaType()));
327         }
328         Response nextResponse = invocation.invoke();
329 
330         if (nextResponse.hasEntity()) {
331             response.setEntityStream(nextResponse.readEntity(InputStream.class));
332         }
333         MultivaluedMap<String, String> headers = response.getHeaders();
334         headers.clear();
335         headers.putAll(nextResponse.getStringHeaders());
336         response.setStatus(nextResponse.getStatus());
337 
338         return response.getStatus() != Response.Status.UNAUTHORIZED.getStatusCode();
339     }

344     static class Credentials {
345 
346         private final String username;
347         private final byte[] password;
348 
349         /**
350          * Create a new credentials from username and password as byte array.
351          *
352          * @param username Username.
353          * @param password Password as byte array.
354          */
355         Credentials(String username, byte[] password) {
356             this.username = username;
357             this.password = password;
358         }
359 
360         /**
361          * Create a new credentials from username and password as {@link String}.
362          *
363          * @param username Username.
364          * @param password {@code String} password.
365          */
366         Credentials(String username, String password) {
367             this.username = username;
368             this.password = password == null ? null : password.getBytes(CHARACTER_SET);
369         }

371         /**
372          * Return username.
373          *
374          * @return username.
375          */
376         String getUsername() {
377             return username;
378         }

380         /**
381          * Return password as byte array.
382          *
383          * @return Password string in byte array representation.
384          */
385         byte[] getPassword() {
386             return password;
387         }

390     private static Credentials extractCredentials(ClientRequestContext request, Type type) {
391         String usernameKey = null;
392         String passwordKey = null;
393         if (type == null) {
394             usernameKey = HttpAuthenticationFeature.HTTP_AUTHENTICATION_USERNAME;
395             passwordKey = HttpAuthenticationFeature.HTTP_AUTHENTICATION_PASSWORD;
396         } else if (type == Type.BASIC) {
397             usernameKey = HttpAuthenticationFeature.HTTP_AUTHENTICATION_BASIC_USERNAME;
398             passwordKey = HttpAuthenticationFeature.HTTP_AUTHENTICATION_BASIC_PASSWORD;
399         } else if (type == Type.DIGEST) {
400             usernameKey = HttpAuthenticationFeature.HTTP_AUTHENTICATION_DIGEST_USERNAME;
401             passwordKey = HttpAuthenticationFeature.HTTP_AUTHENTICATION_DIGEST_PASSWORD;
402         }
403 
404         String userName = (String) request.getProperty(usernameKey);
405         if (userName != null && !userName.equals("")) {
406             byte[] pwdBytes;
407             Object password = request.getProperty(passwordKey);
408             if (password instanceof byte[]) {
409                 pwdBytes = ((byte[]) password);
410             } else if (password instanceof String) {
411                 pwdBytes = ((String) password).getBytes(CHARACTER_SET);
412             } else {
413                 throw new RequestAuthenticationException(
414                         LocalizationMessages.AUTHENTICATION_CREDENTIALS_REQUEST_PASSWORD_UNSUPPORTED());
415             }
416             return new Credentials(userName, pwdBytes);
417         }
418         return null;
419     }

421     /**
422      * Get credentials actual for the current request. Priorities in credentials selection are the following:
423      * <ol>
424      * <li>Basic/digest specific credentials defined in the request properties</li>
425      * <li>Common credentials defined in the request properties</li>
426      * <li>{@code defaultCredentials}</li>
427      * </ol>
428      *
429      * @param request            Request from which credentials should be extracted.
430      * @param defaultCredentials Default credentials (can be {@code null}).
431      * @param type               Type of requested credentials.
432      * @return Credentials or {@code null} if no credentials are found and {@code defaultCredentials} are {@code null}.
433      * @throws RequestAuthenticationException in case the {@code username} or {@code password} is invalid.
434      */
435     static Credentials getCredentials(ClientRequestContext request, Credentials defaultCredentials, Type type) {
436         Credentials commonCredentials = extractCredentials(request, type);
437 
438         if (commonCredentials != null) {
439             return commonCredentials;
440         } else {
441             Credentials specificCredentials = extractCredentials(request, null);
442 
443             return specificCredentials != null ? specificCredentials : defaultCredentials;
444         }
445     }
