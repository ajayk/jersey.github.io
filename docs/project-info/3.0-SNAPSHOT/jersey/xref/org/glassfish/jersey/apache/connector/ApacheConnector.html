127 /**
128  * A {@link Connector} that utilizes the Apache HTTP Client to send and receive
129  * HTTP request and responses.
130  * <p/>
131  * The following properties are only supported at construction of this class:
132  * <ul>
133  * <li>{@link ApacheClientProperties#CONNECTION_MANAGER}</li>
134  * <li>{@link ApacheClientProperties#REQUEST_CONFIG}</li>
135  * <li>{@link ApacheClientProperties#CREDENTIALS_PROVIDER}</li>
136  * <li>{@link ApacheClientProperties#DISABLE_COOKIES}</li>
137  * <li>{@link ClientProperties#PROXY_URI}</li>
138  * <li>{@link ClientProperties#PROXY_USERNAME}</li>
139  * <li>{@link ClientProperties#PROXY_PASSWORD}</li>
140  * <li>{@link ClientProperties#REQUEST_ENTITY_PROCESSING} - default value is {@link RequestEntityProcessing#CHUNKED}</li>
141  * <li>{@link ApacheClientProperties#PREEMPTIVE_BASIC_AUTHENTICATION}</li>
142  * </ul>
143  * <p>
144  * This connector uses {@link RequestEntityProcessing#CHUNKED chunked encoding} as a default setting. This can
145  * be overridden by the {@link ClientProperties#REQUEST_ENTITY_PROCESSING}. By default the
146  * {@link ClientProperties#CHUNKED_ENCODING_SIZE} property is only supported by using default connection manager. If custom
147  * connection manager needs to be used then chunked encoding size can be set by providing a custom
148  * {@link org.apache.http.HttpClientConnection} (via custom {@link org.apache.http.impl.conn.ManagedHttpClientConnectionFactory})
149  * and overriding {@code createOutputStream} method.
150  * </p>
151  * <p>
152  * Using of authorization is dependent on the chunk encoding setting. If the entity
153  * buffering is enabled, the entity is buffered and authorization can be performed
154  * automatically in response to a 401 by sending the request again. When entity buffering
155  * is disabled (chunked encoding is used) then the property
156  * {@link org.glassfish.jersey.apache.connector.ApacheClientProperties#PREEMPTIVE_BASIC_AUTHENTICATION} must
157  * be set to {@code true}.
158  * </p>
159  * <p>
160  * If a {@link org.glassfish.jersey.client.ClientResponse} is obtained and an
161  * entity is not read from the response then
162  * {@link org.glassfish.jersey.client.ClientResponse#close()} MUST be called
163  * after processing the response to release connection-based resources.
164  * </p>
165  * <p>
166  * Client operations are thread safe, the HTTP connection may
167  * be shared between different threads.
168  * </p>
169  * <p>
170  * If a response entity is obtained that is an instance of {@link Closeable}
171  * then the instance MUST be closed after processing the entity to release
172  * connection-based resources.
173  * </p>
174  * <p>
175  * The following methods are currently supported: HEAD, GET, POST, PUT, DELETE, OPTIONS, PATCH and TRACE.
176  * </p>
177  *
178  * @author jorgeluisw@mac.com
179  * @author Paul Sandoz
180  * @author Pavel Bucek (pavel.bucek at oracle.com)
181  * @author Arul Dhesiaseelan (aruld at acm.org)
182  * @see ApacheClientProperties#CONNECTION_MANAGER
183  */
184 @SuppressWarnings("deprecation")

202     /**
203      * Create the new Apache HTTP Client connector.
204      *
205      * @param client JAX-RS client instance for which the connector is being created.
206      * @param config client configuration.
207      */
208     ApacheConnector(final Client client, final Configuration config) {
209         final Object connectionManager = config.getProperties().get(ApacheClientProperties.CONNECTION_MANAGER);
210         if (connectionManager != null) {
211             if (!(connectionManager instanceof HttpClientConnectionManager)) {
212                 LOGGER.log(
213                         Level.WARNING,
214                         LocalizationMessages.IGNORING_VALUE_OF_PROPERTY(
215                                 ApacheClientProperties.CONNECTION_MANAGER,
216                                 connectionManager.getClass().getName(),
217                                 HttpClientConnectionManager.class.getName())
218                 );
219             }
220         }
221 
222         Object reqConfig = config.getProperties().get(ApacheClientProperties.REQUEST_CONFIG);
223         if (reqConfig != null) {
224             if (!(reqConfig instanceof RequestConfig)) {
225                 LOGGER.log(
226                         Level.WARNING,
227                         LocalizationMessages.IGNORING_VALUE_OF_PROPERTY(
228                                 ApacheClientProperties.REQUEST_CONFIG,
229                                 reqConfig.getClass().getName(),
230                                 RequestConfig.class.getName())
231                 );
232                 reqConfig = null;
233             }
234         }
235 
236         final SSLContext sslContext = client.getSslContext();
237         final HttpClientBuilder clientBuilder = HttpClientBuilder.create();
238 
239         clientBuilder.setConnectionManager(getConnectionManager(client, config, sslContext));
240         clientBuilder.setConnectionManagerShared(
241                 PropertiesHelper.getValue(config.getProperties(), ApacheClientProperties.CONNECTION_MANAGER_SHARED, false, null));
242         clientBuilder.setSslcontext(sslContext);
243 
244         final RequestConfig.Builder requestConfigBuilder = RequestConfig.custom();
245 
246         final Object credentialsProvider = config.getProperty(ApacheClientProperties.CREDENTIALS_PROVIDER);
247         if (credentialsProvider != null && (credentialsProvider instanceof CredentialsProvider)) {
248             clientBuilder.setDefaultCredentialsProvider((CredentialsProvider) credentialsProvider);
249         }
250 
251         final Object proxyUri;
252         proxyUri = config.getProperty(ClientProperties.PROXY_URI);
253         if (proxyUri != null) {
254             final URI u = getProxyUri(proxyUri);
255             final HttpHost proxy = new HttpHost(u.getHost(), u.getPort(), u.getScheme());
256             final String userName;
257             userName = ClientProperties.getValue(config.getProperties(), ClientProperties.PROXY_USERNAME, String.class);
258             if (userName != null) {
259                 final String password;
260                 password = ClientProperties.getValue(config.getProperties(), ClientProperties.PROXY_PASSWORD, String.class);
261 
262                 if (password != null) {
263                     final CredentialsProvider credsProvider = new BasicCredentialsProvider();
264                     credsProvider.setCredentials(
265                             new AuthScope(u.getHost(), u.getPort()),
266                             new UsernamePasswordCredentials(userName, password)
267                     );
268                     clientBuilder.setDefaultCredentialsProvider(credsProvider);
269                 }
270             }
271             clientBuilder.setProxy(proxy);
272         }
273 
274         final Boolean preemptiveBasicAuthProperty = (Boolean) config.getProperties()
275                 .get(ApacheClientProperties.PREEMPTIVE_BASIC_AUTHENTICATION);
276         this.preemptiveBasicAuth = (preemptiveBasicAuthProperty != null) ? preemptiveBasicAuthProperty : false;
277 
278         final boolean ignoreCookies = PropertiesHelper.isProperty(config.getProperties(), ApacheClientProperties.DISABLE_COOKIES);
279 
280         if (reqConfig != null) {
281             final RequestConfig.Builder reqConfigBuilder = RequestConfig.copy((RequestConfig) reqConfig);
282             if (ignoreCookies) {
283                 reqConfigBuilder.setCookieSpec(CookieSpecs.IGNORE_COOKIES);
284             }
285             requestConfig = reqConfigBuilder.build();
286         } else {
287             if (ignoreCookies) {
288                 requestConfigBuilder.setCookieSpec(CookieSpecs.IGNORE_COOKIES);
289             }
290             requestConfig = requestConfigBuilder.build();
291         }
292 
293         if (requestConfig.getCookieSpec() == null || !requestConfig.getCookieSpec().equals(CookieSpecs.IGNORE_COOKIES)) {
294             this.cookieStore = new BasicCookieStore();
295             clientBuilder.setDefaultCookieStore(cookieStore);
296         } else {
297             this.cookieStore = null;
298         }
299         clientBuilder.setDefaultRequestConfig(requestConfig);
300         this.client = clientBuilder.build();
301     }

303     private HttpClientConnectionManager getConnectionManager(final Client client,
304                                                              final Configuration config,
305                                                              final SSLContext sslContext) {
306         final Object cmObject = config.getProperties().get(ApacheClientProperties.CONNECTION_MANAGER);
307 
308         // Connection manager from configuration.
309         if (cmObject != null) {
310             if (cmObject instanceof HttpClientConnectionManager) {
311                 return (HttpClientConnectionManager) cmObject;
312             } else {
313                 LOGGER.log(
314                         Level.WARNING,
315                         LocalizationMessages.IGNORING_VALUE_OF_PROPERTY(
316                                 ApacheClientProperties.CONNECTION_MANAGER,
317                                 cmObject.getClass().getName(),
318                                 HttpClientConnectionManager.class.getName())
319                 );
320             }
321         }
322 
323         // Create custom connection manager.
324         return createConnectionManager(
325                 client,
326                 config,
327                 sslContext,
328                 false);
329     }

331     private HttpClientConnectionManager createConnectionManager(
332             final Client client,
333             final Configuration config,
334             final SSLContext sslContext,
335             final boolean useSystemProperties) {
336 
337         final String[] supportedProtocols = useSystemProperties ? split(
338                 System.getProperty("https.protocols")) : null;
339         final String[] supportedCipherSuites = useSystemProperties ? split(
340                 System.getProperty("https.cipherSuites")) : null;
341 
342         HostnameVerifier hostnameVerifier = client.getHostnameVerifier();
343 
344         final LayeredConnectionSocketFactory sslSocketFactory;
345         if (sslContext != null) {
346             sslSocketFactory = new SSLConnectionSocketFactory(
347                     sslContext, supportedProtocols, supportedCipherSuites, hostnameVerifier);
348         } else {
349             if (useSystemProperties) {
350                 sslSocketFactory = new SSLConnectionSocketFactory(
351                         (SSLSocketFactory) SSLSocketFactory.getDefault(),
352                         supportedProtocols, supportedCipherSuites, hostnameVerifier);
353             } else {
354                 sslSocketFactory = new SSLConnectionSocketFactory(
355                         SSLContexts.createDefault(),
356                         hostnameVerifier);
357             }
358         }
359 
360         final Registry<ConnectionSocketFactory> registry = RegistryBuilder.<ConnectionSocketFactory>create()
361                 .register("http", PlainConnectionSocketFactory.getSocketFactory())
362                 .register("https", sslSocketFactory)
363                 .build();
364 
365         final Integer chunkSize = ClientProperties.getValue(config.getProperties(),
366                 ClientProperties.CHUNKED_ENCODING_SIZE, ClientProperties.DEFAULT_CHUNK_SIZE, Integer.class);
367 
368         final PoolingHttpClientConnectionManager connectionManager =
369                 new PoolingHttpClientConnectionManager(registry, new ConnectionFactory(chunkSize));
370 
371         if (useSystemProperties) {
372             String s = System.getProperty("http.keepAlive", "true");
373             if ("true".equalsIgnoreCase(s)) {
374                 s = System.getProperty("http.maxConnections", "5");
375                 final int max = Integer.parseInt(s);
376                 connectionManager.setDefaultMaxPerRoute(max);
377                 connectionManager.setMaxTotal(2 * max);
378             }
379         }
380 
381         return connectionManager;
382     }

411     private static URI getProxyUri(final Object proxy) {
412         if (proxy instanceof URI) {
413             return (URI) proxy;
414         } else if (proxy instanceof String) {
415             return URI.create((String) proxy);
416         } else {
417             throw new ProcessingException(LocalizationMessages.WRONG_PROXY_URI_TYPE(ClientProperties.PROXY_URI));
418         }

421     @Override
422     public ClientResponse apply(final ClientRequest clientRequest) throws ProcessingException {
423         final HttpUriRequest request = getUriHttpRequest(clientRequest);
424         final Map<String, String> clientHeadersSnapshot = writeOutBoundHeaders(clientRequest.getHeaders(), request);
425 
426         try {
427             final CloseableHttpResponse response;
428             final HttpClientContext context = HttpClientContext.create();
429             if (preemptiveBasicAuth) {
430                 final AuthCache authCache = new BasicAuthCache();
431                 final BasicScheme basicScheme = new BasicScheme();
432                 authCache.put(getHost(request), basicScheme);
433                 context.setAuthCache(authCache);
434             }
435             response = client.execute(getHost(request), request, context);
436             HeaderUtils.checkHeaderChanges(clientHeadersSnapshot, clientRequest.getHeaders(), this.getClass().getName());
437 
438             final Response.StatusType status = response.getStatusLine().getReasonPhrase() == null
439                     ? Statuses.from(response.getStatusLine().getStatusCode())
440                     : Statuses.from(response.getStatusLine().getStatusCode(), response.getStatusLine().getReasonPhrase());
441 
442             final ClientResponse responseContext = new ClientResponse(status, clientRequest);
443             final List<URI> redirectLocations = context.getRedirectLocations();
444             if (redirectLocations != null && !redirectLocations.isEmpty()) {
445                 responseContext.setResolvedRequestUri(redirectLocations.get(redirectLocations.size() - 1));
446             }
447 
448             final Header[] respHeaders = response.getAllHeaders();
449             final MultivaluedMap<String, String> headers = responseContext.getHeaders();
450             for (final Header header : respHeaders) {
451                 final String headerName = header.getName();
452                 List<String> list = headers.get(headerName);
453                 if (list == null) {
454                     list = new ArrayList<>();
455                 }
456                 list.add(header.getValue());
457                 headers.put(headerName, list);
458             }
459 
460             final HttpEntity entity = response.getEntity();
461 
462             if (entity != null) {
463                 if (headers.get(HttpHeaders.CONTENT_LENGTH) == null) {
464                     headers.add(HttpHeaders.CONTENT_LENGTH, String.valueOf(entity.getContentLength()));
465                 }
466 
467                 final Header contentEncoding = entity.getContentEncoding();
468                 if (headers.get(HttpHeaders.CONTENT_ENCODING) == null && contentEncoding != null) {
469                     headers.add(HttpHeaders.CONTENT_ENCODING, contentEncoding.getValue());
470                 }
471             }
472 
473             try {
474                 responseContext.setEntityStream(new HttpClientResponseInputStream(getInputStream(response)));
475             } catch (final IOException e) {
476                 LOGGER.log(Level.SEVERE, null, e);
477             }
478 
479             return responseContext;
480         } catch (final Exception e) {
481             throw new ProcessingException(e);
482         }
483     }

517     private HttpUriRequest getUriHttpRequest(final ClientRequest clientRequest) {
518         final RequestConfig.Builder requestConfigBuilder = RequestConfig.copy(requestConfig);
519 
520         final int connectTimeout = clientRequest.resolveProperty(ClientProperties.CONNECT_TIMEOUT, -1);
521         final int socketTimeout = clientRequest.resolveProperty(ClientProperties.READ_TIMEOUT, -1);
522 
523         if (connectTimeout >= 0) {
524             requestConfigBuilder.setConnectTimeout(connectTimeout);
525         }
526         if (socketTimeout >= 0) {
527             requestConfigBuilder.setSocketTimeout(socketTimeout);
528         }
529 
530         final Boolean redirectsEnabled =
531                 clientRequest.resolveProperty(ClientProperties.FOLLOW_REDIRECTS, requestConfig.isRedirectsEnabled());
532         requestConfigBuilder.setRedirectsEnabled(redirectsEnabled);
533 
534         final Boolean bufferingEnabled = clientRequest.resolveProperty(ClientProperties.REQUEST_ENTITY_PROCESSING,
535                 RequestEntityProcessing.class) == RequestEntityProcessing.BUFFERED;
536         final HttpEntity entity = getHttpEntity(clientRequest, bufferingEnabled);
537 
538         return RequestBuilder
539                 .create(clientRequest.getMethod())
540                 .setUri(clientRequest.getUri())
541                 .setConfig(requestConfigBuilder.build())
542                 .setEntity(entity)
543                 .build();
544     }

546     private HttpEntity getHttpEntity(final ClientRequest clientRequest, final boolean bufferingEnabled) {
547         final Object entity = clientRequest.getEntity();
548 
549         if (entity == null) {
550             return null;
551         }
552 
553         final AbstractHttpEntity httpEntity = new AbstractHttpEntity() {
554             @Override
555             public boolean isRepeatable() {
556                 return false;
557             }
558 
559             @Override
560             public long getContentLength() {
561                 return -1;
562             }
563 
564             @Override
565             public InputStream getContent() throws IOException, IllegalStateException {
566                 if (bufferingEnabled) {
567                     final ByteArrayOutputStream buffer = new ByteArrayOutputStream(512);
568                     writeTo(buffer);
569                     return new ByteArrayInputStream(buffer.toByteArray());
570                 } else {
571                     return null;
572                 }
573             }
574 
575             @Override
576             public void writeTo(final OutputStream outputStream) throws IOException {
577                 clientRequest.setStreamProvider(new OutboundMessageContext.StreamProvider() {
578                     @Override
579                     public OutputStream getOutputStream(final int contentLength) throws IOException {
580                         return outputStream;
581                     }
582                 });
583                 clientRequest.writeEntity();
584             }
585 
586             @Override
587             public boolean isStreaming() {
588                 return false;
589             }
590         };
591 
592         if (bufferingEnabled) {
593             try {
594                 return new BufferedHttpEntity(httpEntity);
595             } catch (final IOException e) {
596                 throw new ProcessingException(LocalizationMessages.ERROR_BUFFERING_ENTITY(), e);
597             }
598         } else {
599             return httpEntity;
600         }
601     }

603     private static Map<String, String> writeOutBoundHeaders(final MultivaluedMap<String, Object> headers,
604                                                             final HttpUriRequest request) {
605         final Map<String, String> stringHeaders = HeaderUtils.asStringHeadersSingleValue(headers);
606 
607         for (final Map.Entry<String, String> e : stringHeaders.entrySet()) {
608             request.addHeader(e.getKey(), e.getValue());
609         }
610         return stringHeaders;
611     }

625     private static InputStream getInputStream(final CloseableHttpResponse response) throws IOException {
626 
627         final InputStream inputStream;
628 
629         if (response.getEntity() == null) {
630             inputStream = new ByteArrayInputStream(new byte[0]);
631         } else {
632             final InputStream i = response.getEntity().getContent();
633             if (i.markSupported()) {
634                 inputStream = i;
635             } else {
636                 inputStream = new BufferedInputStream(i, ReaderWriter.BUFFER_SIZE);
637             }
638         }
639 
640         return new FilterInputStream(inputStream) {
641             @Override
642             public void close() throws IOException {
643                 response.close();
644                 super.close();
645             }
646         };
647     }

649     private static class ConnectionFactory extends ManagedHttpClientConnectionFactory {
