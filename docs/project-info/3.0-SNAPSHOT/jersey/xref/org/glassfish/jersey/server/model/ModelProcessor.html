41  package org.glassfish.jersey.server.model;
42  
43  import javax.ws.rs.ConstrainedTo;
44  import javax.ws.rs.RuntimeType;
45  import javax.ws.rs.core.Configuration;
46  
47  import org.glassfish.jersey.spi.Contract;
48  
49  /**
50   * Contract for a model processors that processes {@link ResourceModel resource models} during application initialization
51   * and {@link Resource resource} returned by sub resource locators. Even though {@link ModelProcessor model processors} can
52   * completely change the resource model, the standard use case it to enhance the current resource model by
53   * additional methods and resources (like for example adding OPTIONS http methods for every URI endpoint).
54   * <p/>
55   * More model processors can be registered. These providers will be execute in the chain so that each model
56   * processor will be executed with resource model processed by the previous model processor. The first model
57   * processor in the chain will be invoked with the initial resource model from which the application was initiated.
58   * <p/>
59   * Model processors implementations can define {@link javax.annotation.Priority binding priority}
60   * to define the order in which they are executed (processors with a lower priority is invoked
61   * before processor with a higher priority). The highest possible priority (Integer.MAX_VALUE) is used for
62   * model processor which enhance resource models by the default OPTIONS method defined by JAX-RS specification and therefore
63   * this priority should not be used.
64   * <p/>
65   * Note that if model processor adds a resources that are intended to be supportive resources like
66   * {@code OPTIONS} method providing information about the resource, it should properly define the
67   * {@link org.glassfish.jersey.server.model.ResourceMethod#isExtended() extended} flag of such a new method.
68   * See {@link org.glassfish.jersey.server.model.ExtendedResource} for more information.
69   *
70   * @author Miroslav Fuksa
71   *
72   */
73  @Contract
74  @ConstrainedTo(RuntimeType.SERVER)
75  public interface ModelProcessor {
76      /**
77       * Process {@code resourceModel} and return the processed model. Returning input {@code resourceModel} will cause
78       * no effect on the final resource model.
79       *
80       * @param resourceModel Input resource model to be processed.
81       * @param configuration Runtime configuration.
82       * @return Processed resource model containing root resources. Non root resources will be ignored.
83       */
84      public ResourceModel processResourceModel(ResourceModel resourceModel, Configuration configuration);
85  
86      /**
87       * Process {@code subResourceModel} which was returned a sub resource locator.
88       * <p/>
89       * The {@code subResourceModel} contains only one {@link Resource resource} representing model that should be processed
90       * by further matching. The method must return also exactly one resource in the model. Returning input
91       * {@code subResourceModel} instance will cause no effect on the final sub resource model.
92       *
93       * @param subResourceModel {@link Resource Sub resource} which is based on sub resource returned from sub resource locator.
94       * @param configuration Runtime configuration.
95       * @return Processed resource model with one {@link Resource resource} which should be used for handling sub resource.
96       */
97      public ResourceModel processSubResource(ResourceModel subResourceModel, Configuration configuration);
98  }
